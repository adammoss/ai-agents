Hypothesis

The normalized temperature difference between the north and south ecliptic poles in the low-ℓ CMB map is anomalous compared to ΛCDM simulations.

Method

Measures the normalized absolute temperature difference between 10° radius caps centered on the north and south ecliptic poles on a low-ℓ (ℓ ≤ 10) CMB map.

Result Summary

The Planck statistic is 1.849. From 1000 ΛCDM simulations, the mean ± standard deviation is 0.946 ± 0.702. The z-score is 1.29, yielding a two-tailed p-value of 0.24.

The statistic probes for a specific directional asymmetry related to the ecliptic frame. The observed value is higher than the simulation average but within the expected statistical fluctuation, not reaching the ~2σ anomaly threshold.

Agent Justification

The test measures an absolute difference without a specified direction, and the interpretation indicates it probes for any significant asymmetry, whether the north pole is hotter or colder. The code computes both upper and lower tail probabilities, aligning with a two-tailed approach for general exploration.

Comparison To Literature

This test is related to known low-ℓ anomalies, such as the alignment of low multipoles with the ecliptic plane (e.g., Copi et al. 2015) and hemispherical power asymmetries, but focuses specifically on the pole temperature contrast.

import numpy as np
import healpy as hp

test_description = (
    "Compute the mean temperature within a 10-degree radius cap centered on the "
    "north ecliptic pole and the mean temperature within a 10-degree radius cap "
    "centered on the south ecliptic pole, both on the masked, low-ℓ filtered "
    "(ℓ ≤ 10) temperature map. The statistic is the absolute difference between "
    "these two means, normalized by the root-mean-square temperature of the entire "
    "masked low-ℓ map. This single scalar captures the temperature contrast between "
    "the ecliptic poles, offering a directional asymmetry probe distinct from "
    "hemispherical averages or power ratios."
)


def analyze_map(m):
    """Return the ecliptic-pole temperature asymmetry normalized by the map RMS."""
    nside = hp.npix2nside(len(m))

    # Replace masked (NaN) pixels with 0.0 for healpy transforms
    mask_valid = np.isfinite(m)
    m_clean = m.copy()
    m_clean[~mask_valid] = 0.0

    # Band-limit to ℓ ≤ 10 via spherical harmonic filtering
    lmax = 10
    alm = hp.map2alm(m_clean, lmax=lmax)
    m_low = hp.alm2map(alm, nside, lmax=lmax, verbose=False)

    # Reapply the original mask to the low-ℓ filtered map
    m_low[~mask_valid] = np.nan

    # Ecliptic pole positions converted to Galactic coordinates
    R = hp.Rotator(coord=['E', 'G'])
    theta_nep_gal, phi_nep_gal = R(0.0, 0.0)
    theta_sep_gal, phi_sep_gal = R(np.pi, 0.0)

    # Unit vectors for the poles
    vec_nep = hp.ang2vec(theta_nep_gal, phi_nep_gal)
    vec_sep = hp.ang2vec(theta_sep_gal, phi_sep_gal)

    # Vectorized pixel selection: compute dot product of all pixel vectors with pole vectors
    cos_cap = np.cos(np.radians(10.0))
    npix = hp.nside2npix(nside)
    pix_indices = np.arange(npix)
    vecs_all = np.array(hp.pix2vec(nside, pix_indices))  # shape (3, npix)

    # North ecliptic pole cap
    dot_nep = vecs_all[0] * vec_nep[0] + vecs_all[1] * vec_nep[1] + vecs_all[2] * vec_nep[2]
    in_cap_nep = (dot_nep >= cos_cap) & mask_valid
    mean_nep = np.mean(m_low[in_cap_nep]) if np.any(in_cap_nep) else 0.0

    # South ecliptic pole cap
    dot_sep = vecs_all[0] * vec_sep[0] + vecs_all[1] * vec_sep[1] + vecs_all[2] * vec_sep[2]
    in_cap_sep = (dot_sep >= cos_cap) & mask_valid
    mean_sep = np.mean(m_low[in_cap_sep]) if np.any(in_cap_sep) else 0.0

    # RMS temperature of the full masked low-ℓ map
    valid = m_low[mask_valid]
    rms = np.std(valid, ddof=0)

    if rms == 0.0:
        return 0.0

    asymmetry = abs(mean_nep - mean_sep) / rms
    return float(asymmetry)


def summarize_results(planck_stat, sim_results):
    """Interpret the ecliptic-pole temperature asymmetry statistic."""
    sims = np.array(sim_results, dtype=float)
    sims_finite = sims[np.isfinite(sims)]
    n_sims = len(sims_finite)

    if n_sims == 0:
        return {"planck_stat": planck_stat, "note": "No finite simulation results."}

    mean_s = float(np.mean(sims_finite))
    std_s = float(np.std(sims_finite, ddof=1)) if n_sims > 1 else 0.0
    p_upper = float(np.mean(sims_finite >= planck_stat))
    p_lower = float(np.mean(sims_finite <= planck_stat))
    p_two = min(2.0 * min(p_upper, p_lower), 1.0)
    z_score = (planck_stat - mean_s) / std_s if std_s > 0 else np.nan

    return {
        "planck_stat": planck_stat,
        "sim_mean": mean_s,
        "sim_std": std_s,
        "z_score": z_score,
        "p_value_upper": p_upper,
        "p_value_lower": p_lower,
        "p_value_two_sided": p_two,
        "n_sims": n_sims,
        "interpretation": (
            "This statistic measures the temperature contrast between the north and "
            "south ecliptic poles on the low-ℓ (ℓ ≤ 10) CMB map, normalized by the "
            "overall RMS temperature. In the standard ΛCDM framework, the low-ℓ "
            "multipoles (particularly ℓ = 2–5) have been reported to exhibit "
            "anomalies including alignment with the ecliptic plane and a hemispherical "
            "power asymmetry. A large normalized asymmetry could indicate that the "
            "temperature field near the ecliptic poles is significantly different in "
            "sign or magnitude, potentially pointing to a solar-system-related "
            "foreground contribution or an intrinsic cosmological anisotropy that "
            "picks out the ecliptic coordinate frame."
        ),
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ℓ Ecliptic Pole Temperature Asymmetry",
            "xlabel": "|T_NEP − T_SEP| / RMS(T)",
            "ylabel": "Count",
            "bins": 30,
            "caption": (
                f"Planck asymmetry = {planck_stat:.4f}. "
                f"Sim mean ± σ = {mean_s:.4f} ± {std_s:.4f}. "
                f"z = {z_score:.2f}, p(two-sided) = {p_two:.3f}."
            ),
        },
    }
