Hypothesis

The mean absolute Gaussian curvature of the low-ℓ CMB temperature map (Planck) differs from the expected value under Gaussian simulations.

Method

The mean absolute Gaussian curvature ⟨|K|⟩ of the band-limited (ℓ ≤ 30) CMB temperature map is computed at nside=256. For each pixel, first and second partial derivatives of T(θ,φ) are estimated via spherical harmonic relations and finite differences, then K = (T_θθ·T_φφ − T_θφ²)/(1 + T_θ² + T_φ²)² is evaluated. The spatial average of |K| quantifies the intrinsic curvature of the large-scale temperature surface.

Result Summary

Planck ⟨|K|⟩ = 6.59×10². Simulations yield mean ± std = 4.11×10² ± 6.48×10² (n=1000). The Planck value ranks at the 85.4th percentile of simulations (z = 0.38, two-sided p = 0.292).

The Planck mean absolute Gaussian curvature is higher than the simulation mean but well within the broad distribution of Gaussian realisations. The large scatter in simulations (std comparable to the mean) reflects sensitivity to sample variance at low ℓ. No evidence for non-Gaussianity or anomalous large-scale curvature structure is found.

Agent Justification

The test is exploratory, aiming to detect any anomaly in large-scale temperature structure that could increase or decrease curvature. The code computes both upper and lower p-values, and the description does not specify a predicted direction of deviation.

Comparison To Literature

No prior studies applying Gaussian curvature of the CMB temperature surface as an anomaly statistic were found in the literature. Related topological and morphological analyses of the CMB (e.g., Minkowski functionals) are well-established but operate on different scalar quantities.

import numpy as np
import healpy as hp

test_description = (
    "Low-ℓ Mean Absolute Gaussian Curvature. "
    "Compute the band-limited temperature map (ℓ ≤ 30) at nside=256. "
    "For each pixel, estimate first and second partial derivatives of the "
    "temperature field with respect to spherical coordinates (θ, φ) using "
    "analytic derivatives of the spherical harmonic expansion and finite "
    "differences. The Gaussian curvature K = (T_θθ·T_φφ − T_θφ²)/(1+T_θ²+T_φ²)² "
    "is computed at each pixel, and the mean of |K| over unmasked pixels is returned. "
    "This scalar quantifies the average intrinsic curvature of the temperature "
    "surface, which can deviate from Gaussianity due to anomalies in the "
    "large-scale temperature field."
)


def analyze_map(m):
    """Return the mean absolute Gaussian curvature of the low-ℓ CMB temperature surface."""
    lmax = 30
    nside = hp.npix2nside(len(m))
    valid = np.isfinite(m)
    m_filled = np.where(valid, m, 0.0)
    
    # Get alm coefficients
    alm_T = hp.map2alm(m_filled, lmax=lmax)
    
    # Get temperature map and its first derivatives (θ, φ)
    # hp.alm2map_der1 returns (map, dtheta, dphi)
    T_map, dT_theta, dT_phi = hp.alm2map_der1(alm_T, nside, lmax=lmax)
    
    # Get spherical coordinates
    theta, phi = hp.pix2ang(nside, np.arange(len(m)))
    sin_t = np.sin(theta)
    cos_t = np.cos(theta)
    sin_t_safe = np.where(sin_t > 1e-15, sin_t, 1e-15)
    
    # First derivatives
    T_theta = dT_theta  # ∂T/∂θ
    T_phi = dT_phi  # ∂T/∂φ (Note: this is dT/dφ, not sinθ·dT/dφ)
    
    # Compute second derivatives using spherical harmonic relations
    # For each l,m: the Laplacian eigenvalue is -l(l+1)
    alm_lap = alm_T.copy()
    for l in range(1, lmax + 1):
        idx = hp.Alm.getidx(lmax, l, np.arange(l + 1))
        alm_lap[idx] *= -l * (l + 1.0)
    
    # Reconstruct Laplacian map
    laplacian_T = hp.alm2map(alm_lap, nside, lmax=lmax)
    
    # We need T_φφ and T_θθ. Use the spherical Laplacian relation:
    # ∇²T = (1/sinθ)∂/∂θ(sinθ ∂T/∂θ) + (1/sin²θ)∂²T/∂φ²
    # Which can be rearranged to:
    # T_θθ = ∇²T - cotθ·T_θ - (1/sin²θ)·T_φφ
    
    # For T_φφ: In spherical harmonics, ∂²/∂φ² corresponds to -m²
    alm_phi2 = np.zeros_like(alm_T)
    for l in range(1, lmax + 1):
        idx = hp.Alm.getidx(lmax, l, np.arange(l + 1))
        alm_phi2[idx] = alm_T[idx] * (-(np.arange(l + 1) ** 2))
    
    T_phiphi = hp.alm2map(alm_phi2, nside, lmax=lmax)
    
    # Solve for T_θθ
    T_thetatheta = laplacian_T - (cos_t / sin_t_safe) * T_theta - T_phiphi / (sin_t_safe ** 2)
    
    # For T_θφ: Use finite differences on the temperature map
    # Group pixels by latitude (constant θ)
    order = np.argsort(theta)
    theta_sorted = theta[order]
    
    # Find boundaries between latitude rings
    diff_theta = np.diff(theta_sorted)
    ring_start = np.concatenate([[0], np.where(diff_theta > 1e-10)[0] + 1, [len(m)]])
    
    T_thetaphi = np.zeros(len(m))
    
    for i in range(len(ring_start) - 1):
        ring_indices = order[ring_start[i]:ring_start[i + 1]]
        nphi = len(ring_indices)
        if nphi < 2:  # Need at least 2 points for finite differences
            continue
            
        # Skip poles (θ ≈ 0 or θ ≈ π)
        th = theta[ring_indices[0]]
        if th < 0.01 or th > np.pi - 0.01:
            continue
            
        dphi = 2.0 * np.pi / nphi
        
        # Compute ∂/∂φ(T_θ) using central differences
        # Index shifting for periodic boundary
        idx_right = (np.arange(nphi) + 1) % nphi
        idx_left = (np.arange(nphi) - 1) % nphi
        
        T_theta_ring = T_theta[ring_indices]
        T_theta_right = T_theta[ring_indices[idx_right]]
        T_theta_left = T_theta[ring_indices[idx_left]]
        
        T_thetaphi[ring_indices] = (T_theta_right - T_theta_left) / (2.0 * dphi)
    
    # Gaussian curvature: K = (T_θθ·T_φφ − (T_θφ)²) / (1 + T_θ² + T_φ²)²
    denom = (1.0 + T_theta ** 2 + T_phi ** 2) ** 2
    K = (T_thetatheta * T_phiphi - T_thetaphi ** 2) / denom
    
    # Consider only valid (unmasked) pixels
    valid_final = valid & np.isfinite(K)
    
    # Return mean absolute Gaussian curvature
    return float(np.mean(np.abs(K[valid_final])))


def summarize_results(planck_stat, sim_results):
    """Interpret the Low-ℓ Mean Absolute Gaussian Curvature statistic."""
    sims = np.asarray(sim_results, dtype=float)
    sims = sims[np.isfinite(sims)]
    n = len(sims)
    if n == 0:
        return {"planck_stat": planck_stat, "note": "No finite simulation results."}
    
    mean_s = float(np.mean(sims))
    std_s = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    median_s = float(np.median(sims))
    rank = int(np.sum(sims <= planck_stat))
    p_upper = float(np.mean(sims >= planck_stat))
    p_lower = float(np.mean(sims <= planck_stat))
    p_two = min(2.0 * min(p_upper, p_lower), 1.0)
    z_score = (planck_stat - mean_s) / std_s if std_s > 0 else 0.0
    pct_rank = rank / n * 100 if n > 0 else 50.0
    
    if pct_rank < 2.5 or pct_rank > 97.5:
        sig_desc = "highly anomalous (<2.5% or >97.5% of sims)"
    elif pct_rank < 5 or pct_rank > 95:
        sig_desc = "marginally anomalous (<5% or >95% of sims)"
    elif pct_rank < 25 or pct_rank > 75:
        sig_desc = "noteworthy (outside IQR of sims)"
    else:
        sig_desc = "typical (within IQR of sims)"
    
    if planck_stat < mean_s * 0.8:
        curv_desc = "significantly lower than expected"
    elif planck_stat > mean_s * 1.2:
        curv_desc = "significantly higher than expected"
    else:
        curv_desc = "consistent with simulations"
    
    return {
        "planck_stat": planck_stat,
        "mean_abs_curvature": planck_stat,
        "sim_mean": mean_s,
        "sim_std": std_s,
        "sim_median": median_s,
        "z_score": z_score,
        "rank": rank,
        "p_value_upper": p_upper,
        "p_value_lower": p_lower,
        "p_value_two_sided": p_two,
        "n_sims": n,
        "percentile_rank": pct_rank,
        "curvature_description": curv_desc,
        "significance_description": sig_desc,
        "interpretation": (
            f"Mean |K| = {planck_stat:.6e}. {curv_desc} "
            f"(sim mean = {mean_s:.6e} ± {std_s:.6e}). "
            f"{sig_desc}. z = {z_score:.2f}, two-sided p = {p_two:.4f}."
        ),
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ℓ Mean Absolute Gaussian Curvature",
            "xlabel": "⟨|K|⟩",
            "ylabel": "Count",
            "bins": 30,
            "caption": (
                f"Planck ⟨|K|⟩ = {planck_stat:.6e}. "
                f"Sim mean ± std = {mean_s:.6e} ± {std_s:.6e}. "
                f"z = {z_score:.2f}, P(two-sided) = {p_two:.4f}."
            ),
        },
    }
