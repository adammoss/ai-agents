Hypothesis

The Planck CMB temperature map exhibits an anomaly in the low-ℓ Laplacian spatial correlation amplitude, as indicated by the fitted amplitude A being significantly different from the distribution expected from ΛCDM Gaussian simulations.

Method

Compute the Laplacian (∇²T) of the low-ℓ CMB temperature map (ℓ ≤ 30) on an nside=256 sphere. The normalized two-point spatial correlation function |C(θ)| of this Laplacian field is calculated for angular separations 10°–170° in 5° steps and fitted with an exponential decay model A·exp(−θ/ξ), where A is the amplitude at θ=10° and ξ the coherence length. The scalar test statistic is the fitted amplitude A.

Result Summary

Planck amplitude A = 0.2244 lies at −1.00σ from the ΛCDM simulation mean (0.2311 ± 0.0067). Two-tailed p-value = 0.302 (percentile = 15.1%). Anomaly strength classified as negligible.

The Laplacian spatial correlation amplitude in the Planck data is fully consistent with Gaussian ΛCDM expectations. The approximately one-sigma deficit is statistically unremarkable and does not indicate any anomalous spatial structure in the low-ℓ temperature gradient field.

Agent Justification

The test is designed for general exploration of anomalies in the Laplacian correlation amplitude without a predefined directional preference. The analysis code explicitly computes a two-sided p-value by taking the minimum of the upper and lower tail probabilities and scaling by two, confirming that both higher and lower values of A are considered potential anomalies. This approach aligns with a two-tailed hypothesis test.

Comparison To Literature

The CMB low-ℓ regime has been extensively studied for anomalies such as quadrupole–octupole alignment, power suppression, and hemispherical asymmetries. The Laplacian of the temperature field amplifies small-scale gradient features but has not been commonly used as a correlation statistic. While related work exists on low-ℓ ellipticity (Gurzadyan et al. 2003) and extended foreground effects (Abramo et al. 2006), no directly comparable Laplacian correlation amplitude analysis was found in the literature.

import numpy as np
import healpy as hp
from scipy.optimize import curve_fit

test_description = (
    "Low-ℓ Laplacian Spatial Correlation Amplitude: "
    "Compute the Laplacian (∇²T) of the low-ℓ temperature map (ℓ ≤ 30). "
    "Calculate the normalized two-point spatial correlation function C(θ) of "
    "this Laplacian field for θ from 10° to 170° in steps of 5°. Fit a single "
    "exponential decay model A·exp(-θ/ξ) to |C(θ)|, where A is the amplitude "
    "and ξ is the coherence length. The scalar statistic is the fitted amplitude A."
)


def analyze_map(m):
    mask = np.isfinite(m)
    n_valid = int(np.sum(mask))
    if n_valid < 100:
        return 0.0

    lmax = 30
    clean = np.where(mask, m, 0.0)

    # Spherical harmonic analysis of the masked low-ℓ field
    alm = hp.map2alm(clean, lmax=lmax, iter=0)
    cl = hp.alm2cl(alm, lmax=lmax)

    # Laplacian power spectrum: ∇²T has alm coefficients -ℓ(ℓ+1) a_ℓm,
    # so its power spectrum is [ℓ(ℓ+1)]² C_ℓ
    ell = np.arange(lmax + 1, dtype=float)
    cl_lap = (ell * (ell + 1)) ** 2 * cl

    # Angular separations in degrees
    theta_deg = np.arange(10, 171, 5, dtype=float)
    cos_theta = np.cos(np.deg2rad(theta_deg))
    n_theta = len(theta_deg)

    # Legendre polynomials P_ℓ(cos θ) via recurrence
    Pl = np.empty((lmax + 1, n_theta))
    Pl[0] = 1.0
    if lmax >= 1:
        Pl[1] = cos_theta
    for l in range(1, lmax):
        Pl[l + 1] = ((2 * l + 1) * cos_theta * Pl[l] - l * Pl[l - 1]) / (l + 1)

    # Correlation function: ξ(θ) = Σ_ℓ (2ℓ+1)/(4π) C_ℓ^lap P_ℓ(cos θ)
    w = (2 * ell + 1) / (4.0 * np.pi) * cl_lap
    xi = w @ Pl          # shape (n_theta,)
    xi0 = w.sum()        # ξ(0), since P_ℓ(1) = 1

    if xi0 <= 0 or not np.isfinite(xi0):
        return 0.0

    # Normalized correlation and its absolute value
    abs_c = np.abs(xi / xi0)

    # Fit |C(θ)| = A * exp(-θ_deg / ξ_coh)
    def model(th, A, xi_coh):
        return A * np.exp(-th / xi_coh)

    a0 = float(abs_c[0])
    if a0 < 1e-15:
        return 0.0

    try:
        popt, _ = curve_fit(
            model, theta_deg, abs_c,
            p0=[a0, 30.0],
            bounds=([0.0, 0.5], [5.0, 500.0]),
            maxfev=5000,
        )
        return float(popt[0])
    except Exception:
        return float(a0)


def summarize_results(planck_stat, sim_results):
    """Summarize the Low-ℓ Laplacian Spatial Correlation Amplitude test."""
    sim = np.array(sim_results, dtype=float)
    sim = sim[np.isfinite(sim)]
    n_sims = len(sim)

    if n_sims == 0:
        return {
            "planck_stat": planck_stat,
            "p_value": float("nan"),
            "n_sims": 0,
            "interpretation": "No simulation results available.",
            "plot_spec": {
                "kind": "histogram",
                "title": "Low-ℓ Laplacian Spatial Correlation Amplitude",
                "xlabel": "Fitted amplitude A",
                "ylabel": "Count",
                "bins": 40,
                "caption": "No simulation data.",
            },
        }

    mean_sim = float(np.mean(sim))
    std_sim = float(np.std(sim))

    p_upper = float(np.mean(sim >= planck_stat))
    p_lower = float(np.mean(sim <= planck_stat))
    p_value = float(min(2.0 * min(p_upper, p_lower), 1.0))

    rank_below = int(np.sum(sim < planck_stat))
    percentile = 100.0 * rank_below / n_sims
    deviation_sigma = (planck_stat - mean_sim) / std_sim if std_sim > 0 else 0.0

    if p_value <= 0.02:
        strength = "strong"
    elif p_value <= 0.10:
        strength = "moderate"
    else:
        strength = "negligible"

    if planck_stat > mean_sim + 2 * std_sim:
        direction = (
            "Planck shows significantly higher Laplacian correlation amplitude "
            "than ΛCDM simulations, suggesting anomalous spatial coherence in "
            "the low-ℓ temperature gradient field."
        )
    elif planck_stat < mean_sim - 2 * std_sim:
        direction = (
            "Planck shows significantly lower Laplacian correlation amplitude "
            "than ΛCDM simulations, suggesting unusually rapid decorrelation "
            "of the low-ℓ temperature gradient field."
        )
    else:
        direction = (
            "The Laplacian correlation amplitude is consistent with "
            "ΛCDM Gaussian expectations."
        )

    interp = (
        f"Fitted amplitude A = {planck_stat:.6f}. "
        f"Sim mean = {mean_sim:.6f} ± {std_sim:.6f} "
        f"({deviation_sigma:+.2f}σ). Percentile = {percentile:.1f}%, "
        f"two-sided p = {p_value:.4f}. "
        f"Strength: {strength}. {direction}"
    )

    caption = (
        "Histogram of the fitted exponential amplitude A for the normalized "
        "Laplacian spatial correlation |C(θ)| ≈ A·exp(-θ/ξ) of the low-ℓ "
        "(ℓ ≤ 30) CMB temperature field. The Laplacian ∇²T amplifies "
        "small-scale gradient structure, and A characterizes the overall "
        "correlation strength. Red line marks the Planck value."
    )

    return {
        "planck_stat": planck_stat,
        "p_value": p_value,
        "p_value_upper": p_upper,
        "p_value_lower": p_lower,
        "rank_below": rank_below,
        "n_sims": n_sims,
        "percentile": percentile,
        "sim_mean": mean_sim,
        "sim_std": std_sim,
        "deviation_sigma": deviation_sigma,
        "anomaly_strength": strength,
        "interpretation": interp,
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ℓ Laplacian Spatial Correlation Amplitude",
            "xlabel": "Fitted amplitude A",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": caption,
        },
    }
