Hypothesis

The skewness of the Hessian determinant is zero, indicating a symmetric distribution of curvature extrema consistent with a Gaussian random field.

Method

Measures the skewness of the determinant of the Hessian matrix of the low-ℓ (ℓ≤32) CMB temperature map smoothed to 5 degrees. This probes asymmetry in local curvature extrema, sensitive to non-Gaussian features in hot/cold spot morphology.

Result Summary

Planck skewness = -13.82. Simulated mean ± std = -16.17 ± 5.78. z-score = 0.41, two-tailed p-value = 0.80.

The observed skewness is typical and well within the range expected from Gaussian simulations. The distribution of curvature extrema shows no significant asymmetry, consistent with a Gaussian random field.

Agent Justification

The test is exploratory, aimed at detecting any non-Gaussian features in the low-ℓ temperature map, with no prior directional preference for positive or negative skewness. The analysis code computes both upper and lower p-values and interprets deviations in either direction as anomalies, supporting a two-tailed approach.

import numpy as np
import healpy as hp

test_description = (
    "Low-ℓ Hessian Determinant Skewness. "
    "Compute the Hessian matrix of the low-ℓ (ℓ ≤ 32) temperature map smoothed "
    "to 5 degrees FWHM. For each pixel i, calculate the determinant of the Hessian "
    "H_i = [[∂²T/∂θ², ∂²T/(∂θ∂φ)], [∂²T/(∂θ∂φ), ∂²T/∂φ²]] using finite "
    "differences on the HEALPix grid. The test statistic is the skewness of the "
    "distribution of H_i over all unmasked pixels. This scalar measures the "
    "asymmetry of local curvature extrema in the temperature field, which is "
    "sensitive to non-Gaussian features in the shape of hot and cold spots."
)


def analyze_map(m):
    """Return the skewness of the Hessian determinant of the low-ℓ smoothed map."""
    nside = hp.npix2nside(len(m))
    lmax = 32
    valid = np.isfinite(m)

    if np.sum(valid) < 10:
        return 0.0

    # Replace masked pixels with 0 for harmonic transforms
    m_filled = np.where(valid, m, 0.0)

    # Smooth to 5° FWHM Gaussian beam via harmonic space
    fwhm_rad = np.radians(5.0)
    alm = hp.map2alm(m_filled, lmax=lmax)
    alm_smoothed = hp.smoothalm(alm, fwhm=fwhm_rad)

    # Get smoothed map and first derivatives via alm2map_der1
    nside_int = int(nside)
    derivs = hp.alm2map_der1(alm_smoothed, nside_int, lmax=lmax)
    m_smooth = derivs[0]           # T
    dT_dtheta = derivs[1]          # ∂T/∂θ
    dT_dphi_sin = derivs[2]        # (1/sinθ) ∂T/∂φ

    npix = hp.nside2npix(nside_int)
    theta, phi = hp.pix2ang(nside_int, np.arange(npix))
    sin_theta = np.sin(theta)
    cos_theta = np.cos(theta)

    # Protect sinθ from zero for division
    safe_sin = np.where(np.abs(sin_theta) > 1e-10, sin_theta, 1e-10)
    inv_sin2 = 1.0 / (safe_sin ** 2)

    # Get pixel neighbors for finite differences
    # hp.get_all_neighbours returns [SW, S, SE, W, E, NW, N, NE] = indices 0..7
    neighbors = hp.get_all_neighbours(nside_int, np.arange(npix))
    arange = np.arange(npix)
    n_north = np.where(neighbors[6] >= 0, neighbors[6], arange)  # N
    n_south = np.where(neighbors[1] >= 0, neighbors[1], arange)  # S
    n_east  = np.where(neighbors[4] >= 0, neighbors[4], arange)  # E
    n_west  = np.where(neighbors[3] >= 0, neighbors[3], arange)  # W
    n_ne    = np.where(neighbors[7] >= 0, neighbors[7], arange)  # NE
    n_nw    = np.where(neighbors[5] >= 0, neighbors[5], arange)  # NW
    n_se    = np.where(neighbors[2] >= 0, neighbors[2], arange)  # SE
    n_sw    = np.where(neighbors[0] >= 0, neighbors[0], arange)  # SW

    # Pixel angular spacing (radians)
    dtheta = hp.nside2resol(nside_int)

    # ------------------------------------------------------------------
    # Second derivatives via central finite differences on the smoothed map
    # ------------------------------------------------------------------

    # ∂²T/∂θ² : north-south neighbors (constant φ, varying θ)
    d2T_dtheta2 = (m_smooth[n_north] - 2.0 * m_smooth + m_smooth[n_south]) / (dtheta ** 2)

    # ∂²T/(∂θ∂φ) : diagonal neighbors
    # NE ≈ (θ+dθ, φ+dφ), NW ≈ (θ+dθ, φ-dφ), etc.
    d2T_dtheta_dphi = (m_smooth[n_ne] - m_smooth[n_nw]
                       - m_smooth[n_se] + m_smooth[n_sw]) / (4.0 * dtheta ** 2)

    # ∂²T/∂φ² : east-west neighbors with sin²θ correction
    # In the equatorial band Δφ ≈ dθ/sinθ, so
    # ∂²T/∂φ² ≈ (T_E + T_W - 2T) * sin²θ / dθ²
    d2T_dphi2 = (m_smooth[n_east] - 2.0 * m_smooth + m_smooth[n_west]) * inv_sin2 / (dtheta ** 2)

    # ------------------------------------------------------------------
    # Covariant Hessian on the unit sphere (with Christoffel corrections)
    # ------------------------------------------------------------------
    cot_theta = cos_theta / safe_sin

    # H_θθ = ∇_θ∇_θ T = ∂²T/∂θ²  (Γ^θ_θθ = 0)
    H_tt = d2T_dtheta2

    # H_θφ = ∇_θ∇_φ T = ∂²T/(∂θ∂φ) - cot(θ) (1/sinθ ∂T/∂φ)
    # Note: we use (1/sinθ ∂T/∂φ) directly from alm2map_der1
    H_tp = d2T_dtheta_dphi - cot_theta * dT_dphi_sin

    # H_φφ = ∇_φ∇_φ T = ∂²T/∂φ² + sinθ cosθ ∂T/∂θ
    H_pp = d2T_dphi2 + sin_theta * cos_theta * dT_dtheta

    # Determinant of the 2×2 covariant Hessian
    det_H = H_tt * H_pp - H_tp ** 2

    # ------------------------------------------------------------------
    # Skewness of the Hessian determinant over unmasked pixels
    # ------------------------------------------------------------------
    # Exclude near-pole pixels (sinθ < 0.1) where φ-derivatives are ill-defined
    good = valid & (sin_theta > 0.1)
    valid_det = det_H[good]

    if len(valid_det) < 10:
        return 0.0

    mean_d = np.mean(valid_det)
    var_d = np.var(valid_det)

    if var_d < 1e-30:
        return 0.0

    skew = float(np.mean((valid_det - mean_d) ** 3) / (var_d ** 1.5))

    if not np.isfinite(skew):
        return 0.0
    return skew


def summarize_results(planck_stat, sim_results):
    """Interpret the Low-ℓ Hessian Determinant Skewness."""
    sims = np.asarray(sim_results, dtype=float)
    sims = sims[np.isfinite(sims)]
    n = len(sims)
    if n == 0:
        return {"planck_stat": planck_stat, "note": "No finite simulation results."}

    mean_s = float(np.mean(sims))
    std_s = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    median_s = float(np.median(sims))
    rank = int(np.sum(sims <= planck_stat))
    p_upper = float(np.mean(sims >= planck_stat))
    p_lower = float(np.mean(sims <= planck_stat))
    p_two = min(2.0 * min(p_upper, p_lower), 1.0)
    z_score = (planck_stat - mean_s) / std_s if std_s > 0 else 0.0
    pct_rank = rank / n * 100 if n > 0 else 50.0

    if pct_rank < 2.5 or pct_rank > 97.5:
        sig_desc = "highly anomalous (<2.5% or >97.5% of sims)"
    elif pct_rank < 5 or pct_rank > 95:
        sig_desc = "marginally anomalous (<5% or >95% of sims)"
    elif pct_rank < 25 or pct_rank > 75:
        sig_desc = "noteworthy (outside IQR of sims)"
    else:
        sig_desc = "typical (within IQR of sims)"

    if planck_stat > mean_s + std_s:
        curv_desc = (
            "positive curvature determinants dominate, suggesting hot-spot peaks "
            "are more concentrated/sharper than cold-spot troughs, or saddle points "
            "are biased toward negative curvature configurations"
        )
    elif planck_stat < mean_s - std_s:
        curv_desc = (
            "negative curvature determinants dominate, suggesting cold-spot troughs "
            "are more concentrated/sharper than hot-spot peaks, or saddle points "
            "are biased toward positive curvature configurations"
        )
    else:
        curv_desc = (
            "curvature determinant distribution is consistent with "
            "Gaussian random field expectations (symmetric hot/cold spot morphology)"
        )

    return {
        "planck_stat": planck_stat,
        "hessian_det_skewness": planck_stat,
        "sim_mean": mean_s,
        "sim_std": std_s,
        "sim_median": median_s,
        "z_score": z_score,
        "rank": rank,
        "p_value_upper": p_upper,
        "p_value_lower": p_lower,
        "p_value_two_sided": p_two,
        "n_sims": n,
        "percentile_rank": pct_rank,
        "curvature_asymmetry": curv_desc,
        "significance_description": sig_desc,
        "interpretation": (
            f"Hessian determinant skewness = {planck_stat:.4f}. "
            f"A Gaussian random field predicts zero skewness (symmetric curvature extrema). "
            f"{curv_desc.capitalize()} "
            f"Sim mean ± std = {mean_s:.4f} ± {std_s:.4f}. "
            f"{sig_desc.capitalize()}. z = {z_score:.2f}, two-sided p = {p_two:.4f}."
        ),
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ℓ Hessian Determinant Skewness",
            "xlabel": "Skewness of det(H) (dimensionless)",
            "ylabel": "Count",
            "bins": 30,
            "caption": (
                f"Planck skewness = {planck_stat:.4f}. "
                f"Sim mean ± std = {mean_s:.4f} ± {std_s:.4f}. "
                f"z = {z_score:.2f}, P(two-sided) = {p_two:.4f}."
            ),
        },
    }
