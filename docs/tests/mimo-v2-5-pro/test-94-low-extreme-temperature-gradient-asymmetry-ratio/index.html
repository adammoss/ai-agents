Hypothesis

The ratio of mean squared gradient magnitudes between the hottest 5% and coldest 5% pixels in the low-ℓ CMB temperature map deviates from 1, indicating asymmetry in gradient sharpness between positive and negative temperature extremes.

Method

The CMB temperature map is band-limited to ℓ = 2–16 via spherical harmonic synthesis. Squared gradient magnitudes |∇T|² are computed using spin-1 harmonic derivatives. The hottest and coldest 5% of pixels in the band-limited map are identified, and the statistic R_hot/R_cold is formed from their mean squared gradients. This ratio tests whether extreme-positive and extreme-negative temperature regions have equal gradient sharpness, which is expected under Gaussian random field symmetry.

Result Summary

Planck R_hot/R_cold = 0.693, with simulated mean ± std = 1.068 ± 0.406. The observed ratio falls at the 15.7th percentile of simulations (rank 157/1000), yielding z = −0.92 and a two-tailed p-value of 0.314. Cold extremes show somewhat sharper gradients than hot extremes, but the deviation is not statistically significant.

The asymmetry ratio is consistent with Gaussian random field expectations within roughly one standard deviation. While the observed value lies outside the interquartile range of simulations, the p-value is far from the 0.05 threshold, indicating no compelling departure from isotropy in gradient structure between temperature extremes.

Agent Justification

The test is designed to detect deviations from Gaussian symmetry in either direction (hot extremes having sharper gradients or cold extremes having sharper gradients). The analysis code explicitly calculates a two-sided p-value (p_two), and the interpretation describes asymmetries in both directions. There is no prior directional expectation, making a two-tailed test appropriate for general exploration.

Comparison To Literature

A 2022 study by Khan & Saha examined isotropy statistics of CMB hot and cold spot distributions using orientation-matrix methods (arXiv), but addressed spatial uniformity rather than gradient sharpness asymmetry. No directly comparable statistic combining low-ℓ gradient magnitudes with extreme-region comparisons was found in the literature.

import numpy as np
import healpy as hp

test_description = (
    "Low-ℓ Extreme-Temperature Gradient Asymmetry Ratio. "
    "Band-limit the CMB temperature map to ℓ = 2–16 using spherical harmonic synthesis, "
    "then compute the squared gradient magnitude |∇T|² at each pixel via spin-1 harmonic "
    "derivatives. Identify the hottest 5% and coldest 5% of pixels in the band-limited map "
    "(above the 95th percentile and below the 5th percentile). Compute the mean squared "
    "gradient magnitude ⟨|∇T|²⟩ over the hot pixels and separately over the cold pixels. "
    "The statistic is their ratio: R_hot / R_cold. This tests whether extreme-positive and "
    "extreme-negative regions of the low-ℓ temperature field have equal gradient sharpness; "
    "in a Gaussian random field the two should be statistically symmetric, so a deviation "
    "from unity flags tail asymmetry in the field's spatial structure."
)


def analyze_map(m):
    """Return the ratio of mean squared gradient in hot vs cold extreme regions at low ℓ."""
    nside = hp.npix2nside(len(m))
    lmax = 16
    valid = np.isfinite(m)

    # Replace masked pixels with 0.0 for spherical harmonic transforms
    m_filled = np.where(valid, m, 0.0)

    # Get a_ℓm coefficients
    alm = hp.map2alm(m_filled, lmax=lmax)

    # Band-limit to ℓ = 2–16: zero out ℓ < 2
    l_arr, m_arr = hp.Alm.getlm(lmax)
    low_mask = l_arr < 2
    alm[low_mask] = 0.0 + 0.0j

    # Reconstruct band-limited map for identifying hot/cold pixels
    m_band = hp.alm2map(alm, nside, lmax=lmax, pol=False)

    # Compute squared gradient magnitude via spin-1 derivatives
    # The eth operator for spin-0 → spin-1 has coefficient sqrt(ℓ(ℓ+1))
    ell = l_arr.astype(np.float64)
    eth_coeff = np.sqrt(ell * (ell + 1.0))
    eth_alm = alm * eth_coeff
    b_alm = np.zeros_like(alm)

    # alm2map_spin returns (Q, U) for spin-1 field
    # Q = ∂T/∂θ, U = (1/sinθ)(∂T/∂φ)
    # |∇T|² = Q² + U²
    Q_map, U_map = hp.alm2map_spin([eth_alm, b_alm], nside, spin=1, lmax=lmax)
    grad_sq = Q_map**2 + U_map**2

    # Identify hot (95th percentile) and cold (5th percentile) pixels
    # Use only valid (unmasked) pixels for percentile calculation
    band_valid = m_band[valid]
    p95 = np.percentile(band_valid, 95)
    p05 = np.percentile(band_valid, 5)

    hot_mask = valid & (m_band >= p95)
    cold_mask = valid & (m_band <= p05)

    # Mean squared gradient in each extreme region
    R_hot = np.mean(grad_sq[hot_mask])
    R_cold = np.mean(grad_sq[cold_mask])

    # Ratio; guard against division by zero
    if R_cold < 1e-30:
        result = 0.0
    else:
        result = float(R_hot / R_cold)

    if not np.isfinite(result):
        return 0.0
    return result


def summarize_results(planck_stat, sim_results):
    """Interpret the Low-ℓ Extreme-Temperature Gradient Asymmetry Ratio."""
    sims = np.asarray(sim_results, dtype=float)
    sims = sims[np.isfinite(sims)]
    n = len(sims)
    if n == 0:
        return {"planck_stat": planck_stat, "note": "No finite simulation results."}

    mean_s = float(np.mean(sims))
    std_s = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    median_s = float(np.median(sims))
    rank = int(np.sum(sims <= planck_stat))
    p_upper = float(np.mean(sims >= planck_stat))
    p_lower = float(np.mean(sims <= planck_stat))
    p_two = min(2.0 * min(p_upper, p_lower), 1.0)
    z_score = (planck_stat - mean_s) / std_s if std_s > 0 else 0.0
    pct_rank = rank / n * 100 if n > 0 else 50.0

    if pct_rank < 2.5 or pct_rank > 97.5:
        sig_desc = "highly anomalous (<2.5% or >97.5% of sims)"
    elif pct_rank < 5 or pct_rank > 95:
        sig_desc = "marginally anomalous (<5% or >95% of sims)"
    elif pct_rank < 25 or pct_rank > 75:
        sig_desc = "noteworthy (outside IQR of sims)"
    else:
        sig_desc = "typical (within IQR of sims)"

    deviation = planck_stat - 1.0
    if planck_stat > mean_s + std_s:
        asym_desc = (
            "hot extreme regions have sharper gradients than cold extremes, "
            "suggesting positive-temperature features are more spatially concentrated"
        )
    elif planck_stat < mean_s - std_s:
        asym_desc = (
            "cold extreme regions have sharper gradients than hot extremes, "
            "suggesting negative-temperature features are more spatially concentrated"
        )
    else:
        asym_desc = (
            "gradient sharpness between hot and cold extremes is consistent "
            "with Gaussian symmetry expectations"
        )

    return {
        "planck_stat": planck_stat,
        "gradient_asymmetry_ratio": planck_stat,
        "sim_mean": mean_s,
        "sim_std": std_s,
        "sim_median": median_s,
        "z_score": z_score,
        "rank": rank,
        "p_value_upper": p_upper,
        "p_value_lower": p_lower,
        "p_value_two_sided": p_two,
        "n_sims": n,
        "percentile_rank": pct_rank,
        "deviation_from_unity": deviation,
        "asymmetry_description": asym_desc,
        "significance_description": sig_desc,
        "interpretation": (
            f"Gradient asymmetry ratio R_hot/R_cold = {planck_stat:.4f}. "
            f"Deviation from unity = {deviation:+.4f}. {asym_desc}. "
            f"Sim mean ± std = {mean_s:.4f} ± {std_s:.4f}. "
            f"{sig_desc}. z = {z_score:.2f}, two-sided p = {p_two:.4f}."
        ),
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ℓ Extreme-Temperature Gradient Asymmetry Ratio",
            "xlabel": "R_hot / R_cold",
            "ylabel": "Count",
            "bins": 30,

                f"Sim mean ± std = {mean_s:.4f} ± {std_s:.4f}. "
                f"z = {z_score:.2f}, P(two-sided) = {p_two:.4f}."
            ),
        },
