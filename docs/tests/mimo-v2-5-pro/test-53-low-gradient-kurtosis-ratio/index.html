Hypothesis

The observed excess kurtosis of the gradient magnitude in the low-ℓ CMB map is significantly different from the distribution expected from Gaussian LCDM simulations.

Method

Computes the excess kurtosis of the gradient magnitude |∇T| from the CMB temperature map band-limited to multipoles ℓ ∈ [2, 20]. This statistic quantifies the "tail-heaviness" of gradient values, indicating whether sharp temperature features are concentrated in specific regions.

Result Summary

The Planck SMICA map has an excess kurtosis of 0.5825. This value is +0.99σ away from the mean of 1000 ΛCDM simulations (0.2443 ± 0.3407), placing it at the 86.6th percentile. The two-tailed p-value is 0.268.

The observed kurtosis is consistent with the distribution expected from a Gaussian ΛCDM universe. There is no statistically significant evidence for an unusual concentration of sharp temperature features (e.g., hot or cold spots) in the low-ℓ CMB texture.

Agent Justification

The test is designed to detect both anomalously high kurtosis (indicating concentrated sharp features) and anomalously low kurtosis (indicating unusually uniform textures). The code computes both one-sided p-values and combines them into a two-sided p-value, indicating no directional preference. Since we are interested in any significant deviation from the LCDM expectations, a two-tailed test is appropriate.

Comparison To Literature

The specific statistic of gradient magnitude kurtosis at low-ℓ does not appear to be a standard test in the CMB anomaly literature. Related studies examine other low-ℓ features like alignment or planarity of multipoles, but not this gradient-based metric.

import numpy as np
import healpy as hp

test_description = (
    "Compute the band-limited CMB temperature map restricted to ℓ ∈ [2, 20] "
    "via spherical harmonic filtering. Evaluate the spatial gradient ∇T(θ,φ) "
    "at each pixel. Construct the scalar field g(θ,φ) = |∇T|, the gradient "
    "magnitude. Compute the sample excess kurtosis of g over all unmasked "
    "pixels: κ_g = [Σ(g_i - ḡ)⁴ / N] / [Σ(g_i - ḡ)² / N]² - 3. This "
    "single scalar captures the tail-heaviness and outlier prevalence of "
    "gradient magnitudes, quantifying whether the low-ℓ CMB texture contains "
    "anomalously concentrated sharp features relative to Gaussian LCDM "
    "expectations. A map with a few regions of steep temperature variation "
    "surrounded by smooth terrain will yield a high κ_g, while uniform "
    "gradient textures yield κ_g near zero."
)


def analyze_map(m):
    """Return excess kurtosis of |∇T| in the low-ℓ (ℓ∈[2,20]) band-limited map."""
    nside = hp.npix2nside(len(m))
    mask = np.isfinite(m)
    n_valid = int(np.sum(mask))
    if n_valid < 500:
        return 0.0

    # Fill masked pixels with 0 for healpy compatibility
    m_filled = np.where(mask, m, 0.0)

    # Band-limit to ℓ ∈ [2, 20]
    lmax = 20
    alm = hp.map2alm(m_filled, lmax=lmax, iter=0)

    # Get (ell, m) arrays for vectorized indexing
    ell_arr, m_arr = hp.Alm.getlm(lmax)

    # Zero out ℓ=0 (monopole) and ℓ=1 (dipole)
    low_ell_mask = ell_arr < 2
    alm[low_ell_mask] = 0.0

    # Compute gradient using alm2map_der1 which returns (T, dT/dtheta, (1/sinθ)dT/dphi)
    try:
        _, dtheta, dphi_scaled = hp.alm2map_der1(alm, nside)
    except (AttributeError, TypeError):
        # Fallback: compute gradient via spin-1 transform
        factor = np.zeros(len(alm), dtype=complex)
        valid_idx = ~low_ell_mask
        factor[valid_idx] = -np.sqrt(
            ell_arr[valid_idx] * (ell_arr[valid_idx] + 1.0) / 2.0
        )
        alm_plus = factor * alm
        alm_minus = -alm_plus

        grad_maps = hp.alm2map_spin(
            [alm_plus, alm_minus], nside, lmax, lmax
        )
        dtheta = grad_maps[0]
        dphi_scaled = grad_maps[1]

    # Compute gradient magnitude |∇T| = sqrt((dT/dθ)² + ((1/sinθ)dT/dφ)²)
    grad_mag = np.sqrt(dtheta**2 + dphi_scaled**2)

    # Restrict to unmasked pixels with finite gradient values
    valid = mask & np.isfinite(grad_mag)
    g = grad_mag[valid]

    if len(g) < 100:
        return 0.0

    # Compute sample excess kurtosis: κ = m4/m2² - 3
    g_mean = np.mean(g)
    g_centered = g - g_mean

    m2 = np.mean(g_centered**2)
    m4 = np.mean(g_centered**4)

    if m2 <= 0:
        return 0.0

    kurtosis = m4 / (m2**2) - 3.0

    if not np.isfinite(kurtosis):
        return 0.0

    return float(kurtosis)


def summarize_results(planck_stat, sim_results):
    """Summarize the Low-ℓ Gradient Kurtosis Ratio test."""
    sim = np.array(sim_results, dtype=float)
    sim = sim[np.isfinite(sim)]
    n_sims = len(sim)

    if n_sims == 0:
        return {
            "planck_stat": planck_stat,
            "p_value": float("nan"),
            "n_sims": 0,
            "interpretation": "No simulation results available.",
            "plot_spec": {
                "kind": "histogram",
                "title": "Low-ℓ Gradient Kurtosis Ratio",
                "xlabel": "Excess kurtosis of |∇T|",
                "ylabel": "Count",
                "bins": 50,
                "caption": "No simulation data.",
            },
        }

    mean_sim = float(np.mean(sim))
    std_sim = float(np.std(sim))

    p_value_upper = float(np.mean(sim >= planck_stat))
    p_value_lower = float(np.mean(sim <= planck_stat))

    p_value_two_sided = 2.0 * min(p_value_upper, p_value_lower)
    p_value_two_sided = min(p_value_two_sided, 1.0)

    deviation_sigma = (planck_stat - mean_sim) / std_sim if std_sim > 0 else 0.0

    rank_above = int(np.sum(sim >= planck_stat))
    rank_below = int(np.sum(sim <= planck_stat))
    percentile = 100.0 * rank_below / n_sims

    if p_value_two_sided <= 0.01:
        strength = "strong"
    elif p_value_two_sided <= 0.05:
        strength = "moderate"
    else:
        strength = "negligible"

    interp = (
        f"Excess kurtosis of gradient magnitude = {planck_stat:.4f}. "
        f"Sim mean = {mean_sim:.4f} ± {std_sim:.4f} "
        f"({deviation_sigma:+.2f}σ). Percentile = {percentile:.1f}%, "
        f"two-sided p = {p_value_two_sided:.4f}. Strength: {strength}. "
    )

    if planck_stat > mean_sim + 2 * std_sim:
        interp += (
            "The Planck map shows unusually heavy-tailed gradient magnitudes, "
            "suggesting concentrated sharp features (e.g., the cold spot or "
            "other hot/cold anomalies) in the low-ℓ CMB texture relative to "
            "Gaussian LCDM expectations."
        )
    elif planck_stat < mean_sim - 2 * std_sim:
        interp += (
            "The Planck map shows unusually light-tailed gradient magnitudes, "
            "suggesting an unusually uniform low-ℓ CMB texture compared to "
            "Gaussian LCDM expectations."
        )
    else:
        interp += (
            "The gradient kurtosis is consistent with Gaussian LCDM expectations. "
            "No significant anomaly detected in the concentration of sharp "
            "temperature features at low multipoles."
        )

    caption = (
        "Histogram of excess kurtosis of gradient magnitude |∇T| in the "
        "low-ℓ (ℓ∈[2,20]) band-limited CMB temperature map across FFP "
        "simulations. Higher values indicate more concentrated sharp features "
        "(heavier tails in the gradient distribution). The red line marks the "
        "Planck observed value."
    )

    return {
        "planck_stat": planck_stat,
        "p_value_one_sided_upper": p_value_upper,
        "p_value_one_sided_lower": p_value_lower,
        "p_value_two_sided": p_value_two_sided,
        "rank_above": rank_above,
        "rank_below": rank_below,
        "n_sims": n_sims,
        "percentile": percentile,
        "sim_mean": mean_sim,
        "sim_std": std_sim,
        "deviation_sigma": deviation_sigma,
        "anomaly_strength": strength,
        "interpretation": interp,
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ℓ Gradient Kurtosis Ratio",
            "xlabel": "Excess kurtosis of |∇T|",
            "ylabel": "Number of simulations",
            "bins": 50,
            "caption": caption,
        },
    }
