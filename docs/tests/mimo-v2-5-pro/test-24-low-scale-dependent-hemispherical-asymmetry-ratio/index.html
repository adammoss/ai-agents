Hypothesis

The ratio of dipole amplitudes of local-variance maps between ℓ=11–30 and ℓ=2–10 deviates from the distribution expected under the isotropic ΛCDM model.

Method

For the SMICA CMB temperature map with a Galactic mask, the angular power spectrum is split into ℓ=2–10 and ℓ=11–30 bands. A Local Variance Estimator (LVE) with a 15° disc creates local-variance maps for each band, and a dipole is fit to each. The test statistic is the ratio of the dipole amplitude for ℓ=11–30 to that for ℓ=2–10.

Result Summary

Planck SMICA yields a ratio of 2.627. From 1000 isotropic ΛCDM simulations, the mean ratio is 2.373 ± 2.496. The two-tailed p-value is 0.578 (71.1st percentile).

The observed ratio is fully consistent with the distribution from isotropic simulations. There is no significant evidence that the hemispherical power asymmetry changes scale between the two multipole ranges.

Agent Justification

The test is exploratory, aiming to detect any scale-dependent change in the hemispherical power asymmetry without a prior directional prediction. The analysis code explicitly computes a two-sided p-value, and the interpretation checks for both significantly higher and lower ratios, indicating a two-tailed framework.

Comparison To Literature

The hemispherical power asymmetry is a known anomaly at low ℓ (e.g., Eriksen et al. 2004, *ApJ*; Planck 2016 results. VII). This test specifically probes scale dependence using a LVE-based ratio, a refinement of standard asymmetry statistics. The null result here aligns with some studies showing no strong scale dependence (e.g., arXiv:2309.03966), though other works report mild scale-dependent trends (e.g., arXiv:2309.06866).

import numpy as np
import healpy as hp

test_description = (
    "Low-ℓ Scale-Dependent Hemispherical Asymmetry Ratio: For the SMICA "
    "temperature map with Galactic mask applied, compute the angular power "
    "spectrum for multipoles ℓ=2–10 and ℓ=11–30 separately. For each "
    "multipole range, use the Local Variance Estimator (LVE) method with a "
    "disc radius of 15° to create a map of the local variance. Fit a dipole "
    "to each of the two local-variance maps. The scalar statistic is the "
    "ratio of the dipole amplitude of the ℓ=11–30 local-variance map to the "
    "dipole amplitude of the ℓ=2–10 local-variance map. This tests for a "
    "scale-dependent change in the hemispherical power asymmetry signal."
)


def analyze_map(m):
    """Compute the ratio of dipole amplitudes of LVE local-variance maps
    between the ℓ=11–30 and ℓ=2–10 multipole ranges."""

    def top_hat_beam(alpha_rad, lmax):
        """Beam window function for a top-hat disc of half-angle alpha_rad."""
        cos_a = np.cos(alpha_rad)
        denom = 1.0 - cos_a
        beam = np.zeros(lmax + 1)
        beam[0] = 1.0
        if denom < 1e-15:
            beam[:] = 1.0
            return beam
        P = np.zeros(lmax + 2)
        P[0] = 1.0
        P[1] = cos_a
        for l in range(1, lmax + 1):
            P[l + 1] = ((2 * l + 1) * cos_a * P[l] - l * P[l - 1]) / (l + 1)
        for l in range(1, lmax + 1):
            beam[l] = (P[l - 1] - P[l + 1]) / ((2 * l + 1) * denom)
        return beam

    def local_variance(m_band, mask_binary, alpha_rad, lmax_smooth):
        """Compute local variance map using LVE with top-hat disc smoothing."""
        m_filled = np.where(mask_binary > 0.5, m_band, 0.0)

        beam = top_hat_beam(alpha_rad, lmax_smooth)

        s_mask = hp.smoothing(mask_binary.astype(np.float64), beam_window=beam, lmax=lmax_smooth, iter=0)
        s_m = hp.smoothing(m_filled, beam_window=beam, lmax=lmax_smooth, iter=0)
        s_m2 = hp.smoothing(m_filled ** 2, beam_window=beam, lmax=lmax_smooth, iter=0)

        safe_s = np.maximum(np.abs(s_mask), 1e-10)
        local_mean = s_m / safe_s
        local_var = s_m2 / safe_s - local_mean ** 2

        return local_var

    def fit_dipole_amplitude(m, mask_binary):
        """Fit a dipole (monopole + ℓ=1) to the map on valid pixels.
        Returns the dipole amplitude sqrt(Bx² + By² + Bz²)."""
        valid = mask_binary > 0.5
        m_v = m[valid]
        nside = hp.npix2nside(len(m))
        theta, phi = hp.pix2ang(nside, np.arange(len(m)))
        th, ph = theta[valid], phi[valid]

        x = np.sin(th) * np.cos(ph)
        y = np.sin(th) * np.sin(ph)
        z = np.cos(th)

        A = np.column_stack([np.ones_like(x), x, y, z])
        coeffs, _, _, _ = np.linalg.lstsq(A, m_v, rcond=None)

        return float(np.sqrt(coeffs[1] ** 2 + coeffs[2] ** 2 + coeffs[3] ** 2))

    # Main logic
    nside = hp.npix2nside(len(m))

    valid = np.isfinite(m)
    m_filled = np.where(valid, m, 0.0)
    mask_binary = np.where(valid, 1.0, 0.0)

    # Extract alm up to lmax=30
    lmax_alm = 30
    alm = hp.map2alm(m_filled, lmax=lmax_alm)

    # Band-pass filtering in alm space
    size = hp.Alm.getsize(lmax_alm)
    l_vals = np.empty(size, dtype=int)
    idx = 0
    for l in range(lmax_alm + 1):
        for m_idx in range(l + 1):
            l_vals[idx] = l
            idx += 1

    low_sel = (l_vals >= 2) & (l_vals <= 10)
    high_sel = (l_vals >= 11) & (l_vals <= lmax_alm)

    alm_low = np.zeros_like(alm)
    alm_low[low_sel] = alm[low_sel]

    alm_high = np.zeros_like(alm)
    alm_high[high_sel] = alm[high_sel]

    # Reconstruct band-limited maps
    m_low = hp.alm2map(alm_low, nside=nside)
    m_high = hp.alm2map(alm_high, nside=nside)

    # LVE local variance with 15 deg top-hat disc
    alpha_rad = np.radians(15.0)
    lmax_smooth = min(64, 3 * nside)

    var_low = local_variance(m_low, mask_binary, alpha_rad, lmax_smooth)
    var_high = local_variance(m_high, mask_binary, alpha_rad, lmax_smooth)

    # Fit dipole to each local-variance map
    dip_low = fit_dipole_amplitude(var_low, mask_binary)
    dip_high = fit_dipole_amplitude(var_high, mask_binary)

    if dip_low < 1e-30:
        return 0.0

    return float(dip_high / dip_low)


def summarize_results(planck_stat, sim_results):
    """Summarize the Low-ℓ Scale-Dependent Hemispherical Asymmetry Ratio test."""
    sim = np.array(sim_results, dtype=float)
    sim = sim[np.isfinite(sim)]
    n_sims = len(sim)

    if n_sims == 0:
        return {
            "planck_stat": planck_stat,
            "p_value": float("nan"),
            "n_sims": 0,
            "interpretation": "No simulation results to compare.",
            "plot_spec": {
                "kind": "histogram",
                "title": "Low-ell Scale-Dependent Hemispherical Asymmetry Ratio",
                "xlabel": "Dipole amplitude ratio (ell=11-30 / ell=2-10)",
                "ylabel": "Number of simulations",
                "bins": 40,
                "caption": "No simulation data available.",
            },
        }

    mean_sim = float(np.mean(sim))
    std_sim = float(np.std(sim))

    p_upper = float(np.mean(sim >= planck_stat))
    p_lower = float(np.mean(sim <= planck_stat))
    p_value = float(min(2.0 * min(p_upper, p_lower), 1.0))

    rank_below = int(np.sum(sim < planck_stat))
    percentile = 100.0 * rank_below / n_sims

    if p_value <= 0.02:
        strength = "strong"
    elif p_value <= 0.10:
        strength = "moderate"
    else:
        strength = "negligible"

    interp = (
        f"Scale-dependent hemispherical asymmetry ratio = {planck_stat:.4f}. "
        f"Simulation mean = {mean_sim:.4f} +/- {std_sim:.4f}. "
        f"Percentile = {percentile:.1f}%, two-sided p = {p_value:.4f}. "
        f"Anomaly strength: {strength}. "
    )

    if planck_stat > mean_sim + 2 * std_sim:
        interp += (
            "Planck shows a significantly higher ratio, suggesting the "
            "hemispherical power asymmetry is stronger at smaller angular "
            "scales (ell=11-30) relative to larger scales (ell=2-10) compared "
            "to isotropic LCDM simulations."
        )
    elif planck_stat < mean_sim - 2 * std_sim:
        interp += (
            "Planck shows a significantly lower ratio, suggesting the "
            "hemispherical power asymmetry is weaker at smaller angular "
            "scales (ell=11-30) relative to larger scales (ell=2-10) compared "
            "to isotropic LCDM simulations."
        )
    else:
        interp += (
            "The ratio is consistent with isotropic LCDM expectations, "
            "showing no significant scale-dependent hemispherical asymmetry."
        )

    caption = (
        "Histogram of the ratio of dipole amplitudes of local-variance "
        "maps between the ell=11-30 and ell=2-10 multipole ranges, computed "
        "using the Local Variance Estimator with a 15 deg disc radius. This "
        "statistic probes whether the hemispherical power asymmetry "
        "changes with angular scale: a ratio significantly above the "
        "simulation mean indicates stronger asymmetry at smaller scales, "
        "while a ratio below indicates stronger asymmetry at larger "
        "scales. The red line marks the Planck SMICA value."
    )

    return {
        "planck_stat": planck_stat,
        "p_value": p_value,
        "p_value_upper": p_upper,
        "p_value_lower": p_lower,
        "rank_below": rank_below,
        "n_sims": n_sims,
        "percentile": percentile,
        "sim_mean": mean_sim,
        "sim_std": std_sim,
        "anomaly_strength": strength,
        "interpretation": interp,
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ell Scale-Dependent Hemispherical Asymmetry Ratio",
            "xlabel": "Dipole amplitude ratio (ell=11-30 / ell=2-10)",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": caption,
        },
    }
