Hypothesis

The low-ℓ average bispectrum amplitude of the Planck CMB temperature map is consistent with the expected distribution from ΛCDM Gaussian simulations, indicating no significant anomaly in phase coupling among low-ℓ modes.

Method

For the Planck CMB temperature map limited to ℓ ≤ 30, compute the mean absolute bispectrum ⟨|B|⟩ over all valid mode triples (satisfying triangle and parity conditions) to quantify average phase coupling (non-Gaussianity) among low-ℓ modes.

Result Summary

Planck ⟨|B|⟩ = 4580.76 μK³. ΛCDM Gaussian simulations (1000 runs): mean 8231.45 ± 6708.49 μK³, yielding a z-score of -0.54σ and a two-tailed p-value of 0.748.

The Planck statistic is consistent with the simulated distribution, indicating no significant departure from Gaussian expectations for low-ℓ phase coupling.

Agent Justification

The test description notes that a higher average bispectrum amplitude suggests stronger departure from Gaussianity, but the implementation in the code considers both directions of deviation. Specifically, the `summarize_results` function calculates two-sided p-values by comparing the Planck statistic to both higher and lower extremes of the simulation distribution, and provides interpretations for cases where the statistic is significantly higher or lower. This approach is appropriate for general exploration without a strong prior directional prediction, making the test two-tailed.

Comparison To Literature

Previous web searches did not find direct references to this exact statistic; bispectrum analyses in CMB literature (e.g., lensing bispectrum studies) are common, but low-ℓ average amplitude is not a standard metric.

import numpy as np
import healpy as hp

test_description = (
    "Low-ℓ Average Bispectrum Amplitude: "
    "For the temperature map band-limited to multipoles ℓ ≤ 30, compute the "
    "bispectrum B(ℓ₁, ℓ₂, ℓ₃) for all triples satisfying the triangle condition "
    "(|ℓ₁ - ℓ₂| ≤ ℓ₃ ≤ ℓ₁ + ℓ₂ and ℓ₁ + ℓ₂ + ℓ₃ even). The bispectrum is "
    "defined as the product of three spherical harmonic coefficients averaged "
    "over the sky. Take the absolute value of each triple and compute the mean "
    "over all valid triples to obtain a single scalar. This statistic quantifies "
    "the average level of phase coupling (non-Gaussianity) among low-ℓ modes. "
    "A higher value suggests stronger departure from Gaussianity."
)


def analyze_map(m):
    mask = np.isfinite(m)
    n_valid = int(np.sum(mask))
    if n_valid < 100:
        return 0.0

    lmax = 30
    clean = np.where(mask, m, 0.0)

    # Compute alm from the full-resolution map for accuracy
    alm = hp.map2alm(clean, lmax=lmax, iter=0)

    # Use low nside for per-ℓ map construction (lmax=30 needs nside >= 31)
    nside_low = min(32, hp.get_nside(m))
    npix_low = 12 * nside_low ** 2

    # Build per-ℓ maps at low nside via alm2map
    n_alm = hp.Alm.getsize(lmax)
    per_l_maps = np.empty((lmax + 1, npix_low), dtype=np.float64)
    for ell in range(lmax + 1):
        alm_ell = np.zeros_like(alm)
        idx0 = hp.Alm.getidx(lmax, ell, 0)
        idx1 = hp.Alm.getidx(lmax, ell, ell) + 1
        alm_ell[idx0:idx1] = alm[idx0:idx1]
        per_l_maps[ell] = hp.alm2map(alm_ell, nside_low, lmax=lmax, verbose=False)

    # Enumerate valid unordered triples (l1 <= l2 <= l3) and compute mean |B|
    abs_sum = 0.0
    count = 0
    inv_npix = 1.0 / npix_low

    for l1 in range(lmax + 1):
        m1 = per_l_maps[l1]
        for l2 in range(l1, lmax + 1):
            m12 = m1 * per_l_maps[l2]
            l3_min = l2
            l3_max = min(l1 + l2, lmax)
            if l3_max < l3_min:
                continue
            parity = (l1 + l2) & 1
            l3_start = l3_min if (l3_min & 1) == parity else l3_min + 1
            for l3 in range(l3_start, l3_max + 1, 2):
                B = np.dot(m12, per_l_maps[l3]) * inv_npix
                abs_sum += abs(B)
                count += 1

    if count == 0:
        return 0.0

    return float(abs_sum / count)


def summarize_results(planck_stat, sim_results):
    """Summarize the Low-ℓ Average Bispectrum Amplitude test."""
    sim = np.array(sim_results, dtype=float)
    sim = sim[np.isfinite(sim)]
    n_sims = len(sim)

    if n_sims == 0:
        return {
            "planck_stat": planck_stat,
            "p_value": float("nan"),
            "n_sims": 0,
            "interpretation": "No simulation results available.",
            "plot_spec": {
                "kind": "histogram",
                "title": "Low-ℓ Average Bispectrum Amplitude",
                "xlabel": "⟨|B|⟩ (μK³)",
                "ylabel": "Count",
                "bins": 40,
                "caption": "No simulation data.",
            },
        }

    mean_sim = float(np.mean(sim))
    std_sim = float(np.std(sim))

    p_upper = float(np.mean(sim >= planck_stat))
    p_lower = float(np.mean(sim <= planck_stat))
    p_value = float(min(2.0 * min(p_upper, p_lower), 1.0))

    rank_below = int(np.sum(sim < planck_stat))
    percentile = 100.0 * rank_below / n_sims
    deviation_sigma = (planck_stat - mean_sim) / std_sim if std_sim > 0 else 0.0

    if p_value <= 0.02:
        strength = "strong"
    elif p_value <= 0.10:
        strength = "moderate"
    else:
        strength = "negligible"

    if planck_stat > mean_sim + 2 * std_sim:
        direction = (
            "Planck shows significantly higher average bispectrum "
            "amplitude than ΛCDM simulations, indicating stronger phase coupling "
            "(non-Gaussianity) among low-ℓ modes than expected."
        )
    elif planck_stat < mean_sim - 2 * std_sim:
        direction = (
            "Planck shows significantly lower average bispectrum "
            "amplitude than ΛCDM simulations, indicating unusually weak phase "
            "coupling among low-ℓ modes."
        )
    else:
        direction = (
            "The average bispectrum amplitude is consistent with "
            "ΛCDM Gaussian expectations."
        )

    interp = (
        f"⟨|B|⟩ = {planck_stat:.4f} μK³. "
        f"Sim mean = {mean_sim:.4f} ± {std_sim:.4f} μK³ "
        f"({deviation_sigma:+.2f}σ). Percentile = {percentile:.1f}%, "
        f"two-sided p = {p_value:.4f}. "
        f"Strength: {strength}. {direction}"
    )

    caption = (
        "Histogram of the average bispectrum amplitude ⟨|B|⟩ for the low-ℓ "
        "(ℓ ≤ 30) CMB temperature field. The bispectrum B(ℓ₁,ℓ₂,ℓ₃) measures "
        "phase coupling between three multipole modes. For a Gaussian field, "
        "B fluctuates around zero; systematically larger |B| indicates "
        "non-Gaussianity. Red line marks the Planck value."
    )

    return {
        "planck_stat": planck_stat,
        "p_value": p_value,
        "p_value_upper": p_upper,
        "p_value_lower": p_lower,
        "rank_below": rank_below,
        "n_sims": n_sims,
        "percentile": percentile,
        "sim_mean": mean_sim,
        "sim_std": std_sim,
        "deviation_sigma": deviation_sigma,
        "anomaly_strength": strength,
        "interpretation": interp,
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ℓ Average Bispectrum Amplitude",
            "xlabel": "⟨|B|⟩ (μK³)",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": caption,
        },
    }
