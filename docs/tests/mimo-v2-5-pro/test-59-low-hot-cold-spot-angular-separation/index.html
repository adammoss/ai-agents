Hypothesis

The observed great-circle angular distance between the hottest and coldest pixels in the ℓ ≤ 30 filtered CMB temperature map differs from the expected ~90° value predicted for a statistically isotropic Gaussian random field.

Method

Apply a low-pass filter to retain only multipoles ℓ ≤ 30 from the Planck SMICA temperature map. Identify the pixels with maximum and minimum temperatures, then compute the great-circle angular distance between them. In a statistically isotropic Gaussian random field, the expected separation is approximately 90°; anomalous deviations could indicate preferred axes or non-Gaussian correlations.

Result Summary

Observed angular separation = 59.15°, compared to simulation mean = 87.33° ± 30.84°. Z-score = −0.91σ, percentile = 18.5%, two-sided p-value = 0.37. Anomaly strength: negligible.

The angular separation between the hottest and coldest spots at large angular scales (ℓ ≤ 30) is fully consistent with ΛCDM predictions. The observed separation lies comfortably within the expected simulation distribution. No anomalous preferred axis or non-Gaussian behavior is detected from this statistic alone.

Agent Justification

The test is designed to detect any anomalous deviation from the expected angular separation, whether it is significantly smaller or larger. The description mentions both possibilities: a small separation could indicate a preferred axis, while a large separation is also noted as unusual. The analysis code computes both one-sided p-values and a two-sided p-value, confirming it evaluates departures in both directions without a strong prior directional prediction. Therefore, a two-tailed test is appropriate for this exploratory investigation.

Comparison To Literature

import numpy as np
import healpy as hp

test_description = (
    "Apply a low-pass filter to retain only multipoles ℓ ≤ 30 from the CMB temperature map. "
    "Identify the pixel with the maximum temperature (hot spot) and the pixel with the minimum "
    "temperature (cold spot) in the filtered map. Compute the great-circle angular distance "
    "between these two pixels. In a statistically isotropic Gaussian random field, the expected "
    "separation is approximately 90°; anomalous deviations could indicate preferred axes or "
    "non-Gaussian correlations."
)


def analyze_map(m):
    """Return the great-circle angular distance (degrees) between the hottest and coldest spots in the ℓ ≤ 30 filtered map."""
    # Identify valid (unmasked) pixels
    mask = np.isfinite(m)
    n_valid = int(mask.sum())
    if n_valid < 10:
        return 90.0

    nside = hp.npix2nside(len(m))

    # Replace NaN with 0 for healpy transforms
    m_filled = np.where(mask, m, 0.0)

    # Get spherical harmonic coefficients, ℓ ≤ 30
    alm = hp.map2alm(m_filled, lmax=30)

    # Reconstruct the band-limited map
    m_filtered = hp.alm2map(alm, nside=nside, lmax=30, verbose=False)

    # Find hot and cold spots only among valid pixels
    valid_indices = np.where(mask)[0]
    valid_temps = m_filtered[valid_indices]

    hot_pixel = valid_indices[np.argmax(valid_temps)]
    cold_pixel = valid_indices[np.argmin(valid_temps)]

    # Convert pixel indices to unit vectors and compute angular separation
    vec_hot = hp.pix2vec(nside, hot_pixel)
    vec_cold = hp.pix2vec(nside, cold_pixel)

    # Dot product gives cosine of angular distance
    cos_angle = vec_hot[0] * vec_cold[0] + vec_hot[1] * vec_cold[1] + vec_hot[2] * vec_cold[2]
    cos_angle = np.clip(cos_angle, -1.0, 1.0)
    angular_distance_deg = float(np.degrees(np.arccos(cos_angle)))

    if not np.isfinite(angular_distance_deg):
        return 90.0

    return angular_distance_deg


def summarize_results(planck_stat, sim_results):
    """Summarize the Low-ℓ Hot-Cold Spot Angular Separation test."""
    sim = np.array(sim_results, dtype=float)
    sim = sim[np.isfinite(sim)]
    n_sims = len(sim)

    if n_sims == 0:
        return {
            "planck_stat": planck_stat,
            "p_value_two_sided": float("nan"),
            "n_sims": 0,
            "interpretation": "No simulation results available.",
            "plot_spec": {
                "kind": "histogram",
                "title": "Low-ℓ Hot–Cold Spot Angular Separation",
                "xlabel": "Angular separation (degrees)",
                "ylabel": "Count",
                "bins": 50,
                "caption": "No simulation data.",
            },
        }

    mean_sim = float(np.mean(sim))
    std_sim = float(np.std(sim))
    p_upper = float(np.mean(sim >= planck_stat))
    p_lower = float(np.mean(sim <= planck_stat))
    p_two = min(2.0 * min(p_upper, p_lower), 1.0)
    rank_below = int(np.sum(sim <= planck_stat))
    percentile = 100.0 * rank_below / n_sims
    dev_sigma = (planck_stat - mean_sim) / std_sim if std_sim > 0 else 0.0

    strength = (
        "strong" if p_two <= 0.01
        else "moderate" if p_two <= 0.05
        else "negligible"
    )

    interp = (
        f"Hot–cold spot angular separation = {planck_stat:.2f}°. "
        f"Simulation mean = {mean_sim:.2f}° ± {std_sim:.2f}° ({dev_sigma:+.2f}σ). "
        f"Percentile = {percentile:.1f}%, two-sided p = {p_two:.4f}. "
        f"Strength: {strength}. "
    )

    if planck_stat < mean_sim - 2 * std_sim:
        interp += (
            "The hot and cold spots are significantly closer together than expected from "
            "ΛCDM. This could indicate a preferred direction or a coherent dipole-like "
            "structure in the low-ℓ CMB, potentially linked to hemispherical asymmetry "
            "or the axis of evil alignment anomaly."
        )
    elif planck_stat > mean_sim + 2 * std_sim:
        interp += (
            "The hot and cold spots are significantly farther apart than expected from "
            "ΛCDM, which is unusual but less commonly discussed as an anomaly."
        )
    else:
        interp += (
            "The angular separation between the hottest and coldest spots at large "
            "angular scales (ℓ ≤ 30) is consistent with ΛCDM predictions. No "
            "anomalous preferred axis is detected from this statistic alone."
        )

    return {
        "planck_stat": planck_stat,
        "p_value_one_sided_upper": p_upper,
        "p_value_one_sided_lower": p_lower,
        "p_value_two_sided": p_two,
        "rank_below": rank_below,
        "n_sims": n_sims,
        "percentile": percentile,
        "sim_mean": mean_sim,
        "sim_std": std_sim,
        "deviation_sigma": dev_sigma,
        "anomaly_strength": strength,
        "interpretation": interp,
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ℓ Hot–Cold Spot Angular Separation",
            "xlabel": "Angular separation (degrees)",
            "ylabel": "Number of simulations",
            "bins": 50,

                "Histogram of the great-circle angular distance between the hottest and "
                "coldest pixels in band-limited (ℓ ≤ 30) CMB temperature maps across "
                "ΛCDM simulations. In an isotropic Gaussian field the expected separation "
                "is ~90°. The red line marks the Planck observed value."
            ),
        },
