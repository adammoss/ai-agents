Hypothesis

The observed ratio of cold-pixel galactic-plane concentration to geometric expectation for low-ℓ (ℓ=2–5) CMB temperature maps is not equal to 1.0, indicating an unexpected spatial association—either concentration or avoidance—between large-scale cold spots and the galactic plane.

Method

Test measures the concentration of the coldest 10% of pixels in a band-limited CMB map (ℓ=2–5) within ±20° of the galactic plane, normalized by the geometric area fraction. A ratio near 1.0 indicates no spatial association.

Result Summary

Observed ratio = 0.468. Monte Carlo simulations (1000 ΛCDM realizations) yield mean 0.425 ± 0.272. The Planck value lies at 0.16σ from the mean. Two-tailed p-value = 0.836.

The galactic-plane fraction of low-ℓ cold spots is consistent with ΛCDM expectations. No significant spatial association—either concentration or avoidance—between large-scale cold features and the galactic plane is detected.

Agent Justification

The test description and code are designed to detect any deviation from isotropy, whether excess concentration (ratio > 1) or avoidance (ratio < 1). The `summarize_results` function computes both one-sided p-values and uses a two-sided p-value for significance, considering both directions in its interpretation. While the description emphasizes concentration, the code explicitly handles avoidance, and no strong prior directional preference is mandated by the test's purpose as an anomaly probe.

Comparison To Literature

The test combines two known anomaly probes: (1) low-ℓ CMB power asymmetries or alignments (e.g., Schwarz et al. 2016, 1603.02564; Planck Collaboration 2020, 1807.06205) and (2) galactic contamination or foreground alignment at large scales. However, the specific statistic of a normalized cold-pixel fraction in a band-limited map is not a standard metric in the literature.

import numpy as np
import healpy as hp

test_description = (
    "Construct a low-ell map using only multipoles ell=2 through ell=5. "
    "Identify the coldest 10 percent of pixels and compute the fraction lying "
    "within |b| <= 20 degrees of the galactic plane. Normalize by the geometric "
    "expectation for the galactic-plane band area fraction to obtain a ratio. "
    "Values near 1.0 indicate no galactic-plane concentration; values "
    "significantly above 1.0 indicate that low-ell cold features preferentially "
    "sit near the galactic equator, probing unexpected spatial association with "
    "the galactic coordinate system."
)


def analyze_map(m):
    """Return the ratio of cold-pixel galactic-plane fraction to geometric expectation."""
    mask = np.isfinite(m)
    n_valid = int(np.sum(mask))
    if n_valid < 500:
        return 1.0

    nside = hp.npix2nside(len(m))
    m_filled = np.where(mask, m, 0.0)

    # Band-limit to ell = 2-5
    lmax = 5
    alm = hp.map2alm(m_filled, lmax=lmax, iter=0)
    # Zero out ell = 0 and 1 (monopole and dipole)
    for l in range(2):
        for i in range(2 * l + 1):
            idx = l * l + i
            alm[idx] = 0.0
    m_low = hp.alm2map(alm, nside, lmax=lmax, verbose=False)

    # Get temperatures of valid pixels only
    valid_temps = m_low[mask]
    if len(valid_temps) < 500:
        return 1.0

    # 10th percentile threshold: coldest 10%
    threshold = np.percentile(valid_temps, 10.0)

    # Cold pixels: valid pixels with temperature <= threshold
    cold_mask = mask & (m_low <= threshold)
    n_cold = int(np.sum(cold_mask))
    if n_cold == 0:
        return 1.0

    # Get galactic coordinates of cold pixels
    # (assuming map is in galactic coordinates)
    cold_pixels = np.where(cold_mask)[0]
    theta, phi = hp.pix2ang(nside, cold_pixels)

    # Galactic latitude: b = 90 - theta (degrees)
    b_deg = 90.0 - np.degrees(theta)
    n_cold_in_plane = int(np.sum(np.abs(b_deg) <= 20.0))

    # Fraction of cold pixels in the galactic plane band
    frac_in_plane = n_cold_in_plane / n_cold

    # Geometric expectation: area fraction of band |b| <= 20 deg on the sphere
    # = [cos(70 deg) - cos(110 deg)] / 2 = sin(20 deg)
    f_geo = np.sin(np.radians(20.0))  # ≈ 0.342

    stat = float(frac_in_plane / f_geo)
    if not np.isfinite(stat):
        return 1.0
    return stat


def summarize_results(planck_stat, sim_results):
    """Summarize the Low-ell Cold-Spot Galactic-Plane Fraction test."""
    sim = np.array(sim_results, dtype=float)
    sim = sim[np.isfinite(sim)]
    n_sims = len(sim)

    if n_sims == 0:
        return {
            "planck_stat": planck_stat,
            "p_value": float("nan"),
            "n_sims": 0,
            "interpretation": "No simulation results available.",
            "plot_spec": {
                "kind": "histogram",
                "title": "Low-ℓ Cold-Spot Galactic-Plane Fraction Ratio",
                "xlabel": "Fraction in |b|≤20° / f_geo",
                "ylabel": "Count",
                "bins": 50,
                "caption": "No simulation data.",
            },
        }

    mean_sim = float(np.mean(sim))
    std_sim = float(np.std(sim))

    p_value_upper = float(np.mean(sim >= planck_stat))
    p_value_lower = float(np.mean(sim <= planck_stat))
    p_value_two_sided = min(2.0 * min(p_value_upper, p_value_lower), 1.0)

    rank_above = int(np.sum(sim >= planck_stat))
    rank_below = int(np.sum(sim <= planck_stat))
    percentile = 100.0 * rank_below / n_sims
    deviation_sigma = (planck_stat - mean_sim) / std_sim if std_sim > 0 else 0.0

    if p_value_two_sided <= 0.01:
        strength = "strong"
    elif p_value_two_sided <= 0.05:
        strength = "moderate"
    else:
        strength = "negligible"

    interp = (
        f"Low-ℓ cold-spot galactic-plane fraction ratio = {planck_stat:.4f}. "
        f"Sim mean = {mean_sim:.4f} ± {std_sim:.4f} "
        f"({deviation_sigma:+.2f}σ). Percentile = {percentile:.1f}%, "
        f"two-sided p = {p_value_two_sided:.4f}. Strength: {strength}. "
    )

    if planck_stat > mean_sim + 2 * std_sim:
        interp += (
            "The low-ℓ cold spots in the Planck map show a significantly higher "
            "concentration near the galactic plane than expected from ΛCDM "
            "simulations. This could indicate residual galactic foreground "
            "contamination at large angular scales, or a genuine cosmological "
            "anomaly where cold large-scale structures are aligned with the "
            "galactic coordinate system."
        )
    elif planck_stat < mean_sim - 2 * std_sim:
        interp += (
            "The low-ℓ cold spots in the Planck map show a significantly lower "
            "concentration near the galactic plane than expected from ΛCDM "
            "simulations, suggesting the large-scale cold features preferentially "
            "avoid the galactic equator."
        )
    else:
        interp += (
            "The galactic-plane fraction of low-ℓ cold spots is consistent with "
            "ΛCDM expectations. No significant spatial association between "
            "large-scale cold features and the galactic plane is detected."
        )

    caption = (
        "Histogram of the normalized cold-pixel galactic-plane fraction for the "
        "ℓ=2–5 band-limited CMB temperature map across ΛCDM simulations. "
        "Values near 1.0 indicate no preferential concentration of cold pixels "
        "near the galactic plane. Values above 1.0 indicate excess cold pixels "
        "near |b|≤20°. The red line marks the Planck observed value."
    )

    return {
        "planck_stat": planck_stat,
        "p_value_one_sided_upper": p_value_upper,
        "p_value_one_sided_lower": p_value_lower,
        "p_value_two_sided": p_value_two_sided,
        "rank_above": rank_above,
        "rank_below": rank_below,
        "n_sims": n_sims,
        "percentile": percentile,
        "sim_mean": mean_sim,
        "sim_std": std_sim,
        "deviation_sigma": deviation_sigma,
        "anomaly_strength": strength,
        "interpretation": interp,
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ℓ Cold-Spot Galactic-Plane Fraction Ratio",
            "xlabel": "Fraction in |b|≤20° / f_geo",
            "ylabel": "Number of simulations",
            "bins": 50,
            "caption": caption,
        },
    }
