Hypothesis

The squared temperature dipole amplitude D in the Planck CMB map is larger than expected from Gaussian ΛCDM simulations.

Method

Computes T² on the celestial sphere, subtracts its mean, then extracts the rms dipole amplitude D = sqrt(|a_{1,-1}|² + |a_{1,0}|² + |a_{1,1}|²) from the ℓ=1 spherical harmonic coefficients. This probes dipolar non-Gaussianity in the squared temperature field, distinct from direct temperature dipole modulations.

Result Summary

D = 1911 μK² in Planck, compared to FFP simulation mean of 1342 ± 677 μK². Planck sits at the 81.4th percentile (+0.84σ), with one-sided p = 0.186 across 1000 simulations. Anomaly strength is negligible.

The squared-temperature dipole amplitude is fully consistent with Gaussian ΛCDM expectations. No significant preferred direction in T² is detected, indicating no evidence for the specific non-Gaussian coupling that an anomalous T² dipole would signal.

Agent Justification

The test is designed to detect an unusually large dipole amplitude in the squared temperature field, as indicated by the p-value computation (sim >= planck_stat) and the interpretation in the code which focuses on an unusually large dipole amplitude. There is a clear directional prediction for an excess.

Comparison To Literature

The T² dipole statistic is not a standard probe in the CMB anomalies literature. Related work includes dipolar modulation tests (e.g., Eriksen et al. 2004; Hoftuft et al. 2009), multipole vector analyses (Copi, Huterer & Starkman 2004), and various low-ℓ alignment studies, but none of these specifically extract the dipole of the squared temperature field. Searches for direct prior use of this particular statistic returned no relevant results, so it appears to occupy an unexplored niche.

import numpy as np
import healpy as hp

test_description = (
    "Compute the squared temperature map T² = T^2 on the sphere, subtract its "
    "mean, then compute the spherical harmonic coefficients a_{ℓm} for ℓ=1. "
    "The statistic is the rms amplitude of the dipole: D = sqrt(|a_{1,-1}|^2 + "
    "|a_{1,0}|^2 + |a_{1,1}|^2). This measures the strength of a dipolar pattern "
    "in the squared temperature field, which would indicate a specific form of "
    "non-Gaussianity distinct from dipolar modulations in the temperature itself."
)


def analyze_map(m):
    """Compute the rms dipole amplitude of the mean-subtracted squared temperature field."""
    mask = np.isfinite(m)
    n_valid = int(np.sum(mask))
    if n_valid < 500:
        return 0.0

    # Squared temperature for valid pixels
    T2 = np.where(mask, m ** 2, np.nan)

    # Mean of T² over valid pixels
    T2_mean = np.nanmean(T2)

    # Mean-subtracted T², with masked pixels set to 0 (i.e., filled with the mean)
    T2_sub = np.where(mask, T2 - T2_mean, 0.0)

    # Compute a_{ℓm} for ℓ=1 only
    alm = hp.map2alm(T2_sub, lmax=1, iter=0)

    # alm ordering: (a_{0,0}, a_{1,0}, a_{1,1}) — but for lmax=1, indices are
    #   0 -> (l=0, m=0)
    #   1 -> (l=1, m=0)
    #   2 -> (l=1, m=1)
    # The ℓ=1 coefficients are alm[1] (m=0) and alm[2] (m=1).
    # |a_{1,-1}|^2 = |a_{1,1}|^2 by the real-field symmetry, so:
    # D^2 = |a_{1,0}|^2 + 2*|a_{1,1}|^2
    a10 = alm[1]
    a11 = alm[2]

    D_sq = np.abs(a10) ** 2 + 2.0 * np.abs(a11) ** 2
    D = np.sqrt(D_sq)

    if not np.isfinite(D):
        return 0.0

    return float(D)


def summarize_results(planck_stat, sim_results):
    """Summarize the Low-ℓ Squared-Temperature Dipole Amplitude test."""
    sim = np.array(sim_results, dtype=float)
    sim = sim[np.isfinite(sim)]
    n_sims = len(sim)

    if n_sims == 0:
        return {
            "planck_stat": planck_stat,
            "p_value": float("nan"),
            "n_sims": 0,
            "interpretation": "No simulation results available.",
            "plot_spec": {
                "kind": "histogram",
                "title": "Low-ℓ Squared-Temperature Dipole Amplitude",
                "xlabel": "D (rms dipole of T²)",
                "ylabel": "Count",
                "bins": 40,
                "caption": "No simulation data.",
            },
        }

    mean_sim = float(np.mean(sim))
    std_sim = float(np.std(sim))

    # One-sided p-value (upper tail): how often do sims exceed Planck?
    p_value = float(np.mean(sim >= planck_stat))

    rank_below = int(np.sum(sim < planck_stat))
    percentile = 100.0 * rank_below / n_sims
    deviation_sigma = (planck_stat - mean_sim) / std_sim if std_sim > 0 else 0.0

    if p_value <= 0.01:
        strength = "strong"
    elif p_value <= 0.05:
        strength = "moderate"
    else:
        strength = "negligible"

    if p_value <= 0.05:
        direction = (
            "Planck's squared-temperature field has an unusually large dipole "
            "amplitude compared to Gaussian ΛCDM simulations, suggesting a "
            "preferred direction in T² that may indicate dipolar non-Gaussianity, "
            "a modulation coupling, or a residual systematic aligned with a "
            "specific axis."
        )
    else:
        direction = (
            "The dipole amplitude of the squared-temperature field is consistent "
            "with Gaussian ΛCDM expectations. No significant preferred direction "
            "in T² is detected."
        )

    interp = (
        f"D = {planck_stat:.4f} μK². "
        f"Sim mean = {mean_sim:.4f} ± {std_sim:.4f} "
        f"({deviation_sigma:+.2f}σ). Percentile = {percentile:.1f}%, "
        f"one-sided p = {p_value:.4f}. "
        f"Strength: {strength}. {direction}"
    )

    caption = (
        "Histogram of the rms dipole amplitude D of the mean-subtracted squared "
        "temperature field T² across FFP simulations. D = sqrt(|a_{1,-1}|² + "
        "|a_{1,0}|² + |a_{1,1}|²) measures the strength of a dipolar pattern in "
        "T², a probe of non-Gaussianity distinct from the usual temperature dipole. "
        "Red line marks the Planck value."
    )

    return {
        "planck_stat": planck_stat,
        "p_value": p_value,
        "rank_below": rank_below,
        "n_sims": n_sims,
        "percentile": percentile,
        "sim_mean": mean_sim,
        "sim_std": std_sim,
        "deviation_sigma": deviation_sigma,
        "anomaly_strength": strength,
        "interpretation": interp,
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ℓ Squared-Temperature Dipole Amplitude",
            "xlabel": "D (rms dipole of T² in μK²)",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": caption,
        },
    }
