Hypothesis

The Shannon entropy of the gradient magnitude for the low-ℓ CMB temperature map deviates from the characteristic level produced by a statistically isotropic Gaussian LCDM model.

Method

Band-limiting the Planck temperature map to ℓ ≤ 30, this test computes the gradient magnitude at each pixel, forms its histogram, and calculates the Shannon entropy as a scalar statistic to measure the complexity/disorder of the large-scale gradient texture.

Result Summary

Planck's observed entropy is 3.9948, compared to a simulation mean ± std of 4.0096 ± 0.0825. The z-score is -0.18, yielding a two-tailed p-value of 0.756.

The result shows no significant deviation from the ΛCDM expectation. The low-ℓ gradient texture has an entropy level completely consistent with the statistical isotropy and Gaussianity of the standard model.

Agent Justification

The test description and analysis code are designed to detect any anomaly in the complexity or disorder of the large-scale temperature gradient texture, with no prior directional preference. The code explicitly computes both upper and lower p-values and uses a two-sided p-value for interpretation, indicating that deviations in either higher or lower entropy are considered potentially anomalous. Thus, a two-tailed test is appropriate for this exploratory analysis.

Comparison To Literature

This test is a novel combination of gradient fields and information-theoretic entropy. While the analysis of large-angle gradients is known (e.g., Bernui et al., 2014, using gradient vectors), applying Shannon entropy to the gradient magnitude distribution is a distinct approach.

import numpy as np
import healpy as hp

test_description = (
    "Band-limit the temperature map to l <= 30. Compute the gradient magnitude map at each "
    "pixel using central finite differences on the sphere. Normalize the histogram of these "
    "magnitudes to form a probability distribution. Calculate the Shannon entropy of this "
    "distribution as the scalar test statistic. This measures the complexity and disorder in "
    "the large-scale temperature gradient texture. A statistically isotropic Gaussian LCDM "
    "field produces a characteristic gradient entropy level; deviations indicate anomalous "
    "structure in the low-l temperature pattern."
)


def analyze_map(m):
    """Return Shannon entropy of the gradient magnitude histogram for l <= 30 map."""
    valid = np.isfinite(m)
    if not np.any(valid):
        return 0.0

    m_clean = np.where(valid, m, 0.0)
    nside = hp.npix2nside(len(m_clean))
    lmax = 30

    # Band-limit to l <= 30
    alm = hp.map2alm(m_clean, lmax=lmax)
    m_low = hp.alm2map(alm, nside=nside, lmax=lmax, verbose=False)

    # Angular step for central differences
    step = hp.nside2resol(nside)

    # Get unit vectors for all pixels
    x, y, z = hp.pix2vec(nside, np.arange(hp.nside2npix(nside)))

    # Compute theta and phi of each pixel
    theta, phi = hp.pix2ang(nside, np.arange(hp.nside2npix(nside)))

    # Neighbors in +theta and +phi directions (unit vectors)
    sin_t = np.sin(theta)
    cos_t = np.cos(theta)
    sin_p = np.sin(phi)
    cos_p = np.cos(phi)

    # +theta direction unit vector: (cos(theta)cos(phi), cos(theta)sin(phi), -sin(theta))
    e_theta_x = cos_t * cos_p
    e_theta_y = cos_t * sin_p
    e_theta_z = -sin_t

    # +phi direction unit vector: (-sin(phi), cos(phi), 0)
    e_phi_x = -sin_p
    e_phi_y = cos_p
    e_phi_z = np.zeros_like(phi)

    # Neighbor pixel indices
    x_theta = x + step * e_theta_x
    y_theta = y + step * e_theta_y
    z_theta = z + step * e_theta_z
    x_phi = x + step * e_phi_x
    y_phi = y + step * e_phi_y
    z_phi = z + step * e_phi_z

    # Renormalize to unit sphere
    r_theta = np.sqrt(x_theta**2 + y_theta**2 + z_theta**2)
    x_theta /= r_theta
    y_theta /= r_theta
    z_theta /= r_theta
    r_phi = np.sqrt(x_phi**2 + y_phi**2 + z_phi**2)
    x_phi /= r_phi
    y_phi /= r_phi
    z_phi /= r_phi

    pix_theta = hp.vec2pix(nside, x_theta, y_theta, z_theta)
    pix_phi = hp.vec2pix(nside, x_phi, y_phi, z_phi)

    npix = hp.nside2npix(nside)
    all_indices = np.arange(npix)

    # Build neighbor arrays (down = -theta, left = -phi) using shifting
    # For proper central differences: dT/dtheta ≈ (T[+theta] - T[-theta]) / (2*step)
    # We need the -theta and -phi neighbors too
    # Instead, reuse the + neighbors and compute one-sided or use the trick:
    # T[+theta] at pixel i = T[pix_theta[i]], but T[-theta] at pixel i = T[i'] where pix_theta[i'] = i
    # That's harder. Let's just compute both sides explicitly.

    x_mtheta = x - step * e_theta_x
    y_mtheta = y - step * e_theta_y
    z_mtheta = z - step * e_theta_z
    x_mphi = x - step * e_phi_x
    y_mphi = y - step * e_phi_y
    z_mphi = z - step * e_phi_z

    r_mtheta = np.sqrt(x_mtheta**2 + y_mtheta**2 + z_mtheta**2)
    x_mtheta /= r_mtheta
    y_mtheta /= r_mtheta
    z_mtheta /= r_mtheta
    r_mphi = np.sqrt(x_mphi**2 + y_mphi**2 + z_mphi**2)
    x_mphi /= r_mphi
    y_mphi /= r_mphi
    z_mphi /= r_mphi

    pix_mtheta = hp.vec2pix(nside, x_mtheta, y_mtheta, z_mtheta)
    pix_mphi = hp.vec2pix(nside, x_mphi, y_mphi, z_mphi)

    # Central differences
    dT_dtheta = (m_low[pix_theta] - m_low[pix_mtheta]) / (2.0 * step)
    dT_dphi = (m_low[pix_phi] - m_low[pix_mphi]) / (2.0 * step)

    # Gradient magnitude
    grad_mag = np.sqrt(dT_dtheta**2 + dT_dphi**2)

    # Only keep pixels where the original mask was valid
    # Also check that gradient is finite
    grad_valid = valid & np.isfinite(grad_mag)
    g = grad_mag[grad_valid]

    if len(g) < 10:
        return 0.0

    # Build histogram and compute Shannon entropy
    n_bins = 100
    hist, _ = np.histogram(g, bins=n_bins, density=False)
    p = hist.astype(float)
    p = p / p.sum()

    # Shannon entropy: H = -sum(p * log(p)) for p > 0
    mask = p > 0
    entropy = -np.sum(p[mask] * np.log(p[mask]))

    return float(entropy)


def summarize_results(planck_stat, sim_results):
    """Interpret the Low-l Gradient Magnitude Shannon Entropy statistic."""
    sims = np.asarray(sim_results, dtype=float)
    sims = sims[np.isfinite(sims)]
    n = len(sims)

    if n == 0:
        return {"planck_stat": planck_stat, "note": "No finite simulation results."}

    mean_s = float(np.mean(sims))
    std_s = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    median_s = float(np.median(sims))

    # Rank of Planck among simulations (0 = most negative entropy anomaly)
    rank = int(np.sum(sims <= planck_stat))
    p_upper = float(np.mean(sims >= planck_stat))
    p_lower = float(np.mean(sims <= planck_stat))
    p_two = min(2.0 * min(p_upper, p_lower), 1.0)

    z_score = (planck_stat - mean_s) / std_s if std_s > 0 else 0.0

    if abs(z_score) < 1.0:
        direction = "consistent with LCDM"
    elif planck_stat > mean_s:
        direction = "higher gradient entropy (more complex/disordered low-l texture)"
    else:
        direction = "lower gradient entropy (more ordered/structured low-l texture)"

    return {
        "planck_stat": planck_stat,
        "planck_entropy": planck_stat,
        "sim_mean": mean_s,
        "sim_std": std_s,
        "sim_median": median_s,
        "z_score": z_score,
        "rank": rank,
        "p_value_upper": p_upper,
        "p_value_lower": p_lower,
        "p_value_two_sided": p_two,
        "n_sims": n,
        "direction": direction,
        "interpretation": (
            f"The gradient magnitude Shannon entropy measures the complexity "
            f"and disorder of the large-scale (l<=30) temperature gradient texture. "
            f"Higher entropy means a wider, more uniform distribution of gradient "
            f"strengths (more varied texture); lower entropy means gradients are "
            f"concentrated in a narrower range (more homogeneous or structured). "
            f"Planck entropy = {planck_stat:.4f}. "
            f"Sim mean +/- std = {mean_s:.4f} +/- {std_s:.4f}. "
            f"z = {z_score:.2f}, two-sided p = {p_two:.4f}. "
            f"Direction: {direction}."
        ),
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ℓ Gradient Magnitude Shannon Entropy",
            "xlabel": "Shannon Entropy H",
            "ylabel": "Count",
            "bins": 30,
            "caption": (
                f"Planck H = {planck_stat:.4f}. "
                f"Sim mean ± std = {mean_s:.4f} ± {std_s:.4f}. "
                f"z = {z_score:.2f}, P(two-sided) = {p_two:.4f}."
            ),
        },
    }
