Hypothesis

The Pearson correlation coefficient between low-ℓ CMB temperature and |sin(ecliptic latitude)| is not zero; i.e., there is a linear association.

Method

The Planck SMICA temperature map is low-pass filtered to ℓ_max = 30, and the Pearson correlation coefficient is computed between the filtered temperature values and |sin(β)|, where β is ecliptic latitude, over unmasked pixels. This quantifies whether low-ℓ CMB temperature is systematically enhanced or suppressed near the ecliptic plane. The observed statistic is compared to 1000 ΛCDM simulations processed identically.

Result Summary

The observed Pearson coefficient is 0.0653, with a simulation mean of 0.0032 and standard deviation of 0.1818. The z-score is 0.34, yielding a two-tailed p-value of 0.732. The observed value lies well within the expected ΛCDM distribution.

There is no detectable linear association between low-ℓ CMB temperature and distance from the ecliptic plane. The observed correlation is entirely consistent with statistical fluctuations in an isotropic Gaussian sky, showing no evidence of ecliptic-plane–related temperature modulation at large angular scales.

import numpy as np
import healpy as hp

test_description = (
    "Low-ℓ Temperature–Ecliptic-Latitude Cross-Correlation Coefficient: "
    "Low-pass filter the Planck SMICA temperature map to ℓ_max = 30. "
    "Compute the absolute ecliptic latitude |sin(β)| for each pixel, where β is the ecliptic latitude. "
    "Compute the Pearson correlation coefficient between the filtered temperature values and |sin(β)| "
    "over all pixels outside the Planck common mask. The resulting scalar measures the linear association "
    "between low-ℓ temperature and distance from the ecliptic plane."
)

# Module-level cache for ecliptic latitude
_ecliptic_cache = {}


def _get_ecliptic_sin_beta(nside):
    """Compute and cache |sin(ecliptic latitude)| for given nside."""
    if nside not in _ecliptic_cache:
        npix = hp.nside2npix(nside)
        theta_gal, phi_gal = hp.pix2ang(nside, np.arange(npix))
        rot = hp.Rotator(coord='GE')
        theta_ecl, phi_ecl = rot(theta_gal, phi_gal)
        # Ecliptic latitude β = π/2 - θ_ecl, so |sin(β)| = |cos(θ_ecl)|
        sin_beta_abs = np.abs(np.cos(theta_ecl))
        _ecliptic_cache[nside] = sin_beta_abs
    return _ecliptic_cache[nside]


def analyze_map(m):
    """Return Pearson correlation between low-ℓ temperature and |sin(ecliptic latitude)|."""
    lmax = 30

    finite_mask = np.isfinite(m)
    if not np.any(finite_mask):
        return 0.0

    nside = hp.npix2nside(len(m))

    # Replace NaN with 0 for healpy transforms
    m_clean = np.where(finite_mask, m, 0.0)

    # Low-pass filter to ℓ_max = 30
    alm = hp.map2alm(m_clean, lmax=lmax)
    T_filtered = hp.alm2map(alm, nside, lmax=lmax, verbose=False)

    # Get |sin(ecliptic latitude)| - compute inline to avoid namespace issues
    npix = hp.nside2npix(nside)
    theta_gal, phi_gal = hp.pix2ang(nside, np.arange(npix))
    rot = hp.Rotator(coord='GE')
    theta_ecl, phi_ecl = rot(theta_gal, phi_gal)
    sin_beta_abs = np.abs(np.cos(theta_ecl))

    # Pearson correlation over valid (unmasked) pixels
    valid = finite_mask
    n_valid = np.sum(valid)
    if n_valid < 3:
        return 0.0

    T_valid = T_filtered[valid]
    sb_valid = sin_beta_abs[valid]

    # Vectorized Pearson correlation
    mean_T = np.mean(T_valid)
    mean_sb = np.mean(sb_valid)
    dT = T_valid - mean_T
    dsb = sb_valid - mean_sb
    cov = np.dot(dT, dsb)
    norm_T = np.sqrt(np.dot(dT, dT))
    norm_sb = np.sqrt(np.dot(dsb, dsb))

    if norm_T < 1e-30 or norm_sb < 1e-30:
        return 0.0

    result = cov / (norm_T * norm_sb)
    return float(result)


def summarize_results(planck_stat, sim_results):
    """Interpret temperature–ecliptic-latitude correlation relative to simulations."""
    sims = np.array(sim_results, dtype=float)
    sims_finite = sims[np.isfinite(sims)]
    n_sims = len(sims_finite)

    if n_sims == 0:
        return {"planck_stat": planck_stat, "note": "No finite simulation results."}

    mean_s = float(np.mean(sims_finite))
    std_s = float(np.std(sims_finite, ddof=1)) if n_sims > 1 else 0.0
    p_value = float(np.mean(sims_finite >= planck_stat))
    p_value_lower = float(np.mean(sims_finite <= planck_stat))
    z_score = (planck_stat - mean_s) / std_s if std_s > 0 else np.nan

    return {
        "planck_stat": planck_stat,
        "sim_mean": mean_s,
        "sim_std": std_s,
        "z_score": z_score,
        "p_value_upper": p_value,
        "p_value_lower": p_value_lower,
        "p_value_two_sided": min(p_value, p_value_lower),
        "n_sims": n_sims,
        "interpretation": (
            "A significantly negative correlation indicates that low-ℓ CMB temperature "
            "is preferentially suppressed near the ecliptic plane (low |sin(β)|), "
            "which is anomalous for isotropic Gaussian fluctuations. "
            "A significantly positive correlation would indicate enhancement near the ecliptic plane."
        ),
        "expected_random": "Near 0 with spread ~1/sqrt(N_pixels) for ΛCDM",
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ℓ Temperature–Ecliptic Latitude Correlation",
            "xlabel": "Pearson correlation coefficient",
            "ylabel": "Count",
            "bins": 30,
            "caption": (
                f"Planck stat = {planck_stat:.6f}. "
                f"Sim mean ± σ = {mean_s:.6f} ± {std_s:.6f}. "
                f"z = {z_score:.2f}, p(two-sided) = {min(p_value, p_value_lower):.3f}. "
                f"Negative values indicate temperature suppression near ecliptic plane; "
                f"positive values indicate enhancement."
            ),
        },
