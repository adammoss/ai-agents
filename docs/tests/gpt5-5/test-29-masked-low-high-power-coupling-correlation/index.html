Hypothesis

The Planck CMB temperature map has an anomalously strong correlation between the low-multipole temperature field and the smoothed small-scale power field compared with LCDM simulations.

Method

Correlates the masked low-ℓ temperature field, L(n) from ℓ=2–20, with a 5°-smoothed local small-scale power map, H(n)^2 from ℓ=40–128, using a masked-pixel Pearson r after monopole/dipole removal.

Result Summary

Planck gives r=0.0256, while 1000 LCDM simulations have mean r=0.0016 and std=0.0400. The result is only 0.60σ from the simulation mean; the two-tailed p-value is about 0.55, and the |r|-tail p-value is about 0.52.

The positive sign means small-scale power is slightly higher in hotter low-ℓ regions, but the amplitude is entirely typical of LCDM simulations and not close to a 2σ anomaly.

Agent Justification

The statistic is the signed Pearson correlation r, but the anomaly criterion is unusually large |r|. Both positive and negative correlations would indicate low-high power coupling; the sign only indicates whether enhanced small-scale power aligns with hot or cold low-l regions.

Comparison To Literature

Related to Planck isotropy/local-variance and dipolar-modulation tests, which found established large-scale anomalies such as low variance and hemispherical power asymmetry but no broad failure of Gaussian ΛCDM statistics (Planck Collaboration 2015, https://arxiv.org/abs/1506.07135; Planck Collaboration 2019, https://arxiv.org/abs/1906.02552). This exact low-ℓ vs smoothed high-ℓ power Pearson statistic is not a standard published test.

import math
import numpy as np


test_description = "Masked Low-High Power-Coupling Correlation: remove the masked-sky monopole and dipole on finite unmasked pixels, fill masked pixels with zero before healpy spherical-harmonic transforms, form L(n) from multipoles 2 <= ell <= 20 and H(n) from 40 <= ell <= 128, form P(n)=H(n)^2, smooth P with a 5 degree FWHM Gaussian, and return the masked-pixel Pearson correlation coefficient between L and the smoothed P after subtracting masked means."

_LHPC_L_MIN = 2
_LHPC_L_MAX = 20
_LHPC_H_MIN = 40
_LHPC_H_MAX = 128
_LHPC_FWHM_RAD = math.radians(5.0)
_LHPC_CHUNK = 262144
_LHPC_EPS = 1.0e-30

_LHPC_STATE = {
    "geom_cache": {},
    "md_cache": {},
    "beam_cache": {},
}


def _lhpc_infer_nside(npix):
    try:
        n = int(npix)
        if n <= 0 or n % 12 != 0:
            return -1
        ns = int(round(math.sqrt(n / 12.0)))
        if ns <= 0 or 12 * ns * ns != n:
            return -1
        return ns
    except Exception:
        return -1


def _lhpc_alm_size(lmax):
    lmax = int(lmax)
    if lmax < 0:
        return 0
    return int((lmax + 1) * (lmax + 2) // 2)


def _lhpc_alm_index(lmax, ell, mm):
    return int(mm * (2 * lmax + 1 - mm) // 2 + ell)


def _lhpc_get_geometry(nside, npix, state):
    key = str(int(nside)) + ":" + str(int(npix))
    cache = state.get("geom_cache")
    if not isinstance(cache, dict):
        cache = {}
        state["geom_cache"] = cache
    item = cache.get(key)
    if item is not None:
        return item

    import healpy as hp
    pix = np.arange(int(npix), dtype=np.int64)
    vec = hp.pix2vec(int(nside), pix, nest=False)
    item = {
        "x": np.asarray(vec[0], dtype=np.float64).ravel(),
        "y": np.asarray(vec[1], dtype=np.float64).ravel(),
        "z": np.asarray(vec[2], dtype=np.float64).ravel(),
    }
    if len(cache) > 3:
        cache.clear()
    cache[key] = item
    return item


def _lhpc_make_mask_info(nside, npix, valid):
    v = np.asarray(valid, dtype=np.bool_).ravel()
    nvalid = int(np.count_nonzero(v))
    packed = np.packbits(v)
    mask_bytes = packed.tobytes()
    key = str(int(nside)) + ":" + str(int(npix)) + ":" + str(nvalid) + ":" + str(hash(mask_bytes))
    return {"key": key, "bytes": mask_bytes, "nvalid": nvalid}


def _lhpc_get_md_solver(nside, npix, valid, mask_info, geom, state):
    cache = state.get("md_cache")
    if not isinstance(cache, dict):
        cache = {}
        state["md_cache"] = cache
    key = mask_info["key"]
    item = cache.get(key)
    if item is not None and item.get("mask_bytes") == mask_info["bytes"]:
        return item

    x = geom["x"]
    y = geom["y"]
    z = geom["z"]
    idx = np.flatnonzero(np.asarray(valid, dtype=np.bool_)).astype(np.int64, copy=False)

    s0 = sx = sy = sz = 0.0
    sxx = sxy = sxz = syy = syz = szz = 0.0
    for start in range(0, int(idx.size), _LHPC_CHUNK):
        ii = idx[start:start + _LHPC_CHUNK]
        xx = x[ii]
        yy = y[ii]
        zz = z[ii]
        s0 += float(ii.size)
        sx += float(np.sum(xx))
        sy += float(np.sum(yy))
        sz += float(np.sum(zz))
        sxx += float(np.dot(xx, xx))
        sxy += float(np.dot(xx, yy))
        sxz += float(np.dot(xx, zz))
        syy += float(np.dot(yy, yy))
        syz += float(np.dot(yy, zz))
        szz += float(np.dot(zz, zz))

    G = np.array([[s0,  sx,  sy,  sz],
                  [sx, sxx, sxy, sxz],
                  [sy, sxy, syy, syz],
                  [sz, sxz, syz, szz]], dtype=np.float64)
    try:
        solver = np.linalg.pinv(G, rcond=1.0e-12)
    except Exception:
        solver = np.zeros((4, 4), dtype=np.float64)

    item = {
        "idx": idx,
        "solver": np.asarray(solver, dtype=np.float64),
        "mask_bytes": mask_info["bytes"],
    }
    if len(cache) > 6:
        cache.clear()
    cache[key] = item
    return item


def _lhpc_remove_monopole_dipole(arr, md_entry, geom, npix):
    x = geom["x"]
    y = geom["y"]
    z = geom["z"]
    idx = md_entry["idx"]
    clean = np.zeros(int(npix), dtype=np.float64)
    if int(idx.size) == 0:
        return clean

    rhs = np.zeros(4, dtype=np.float64)
    for start in range(0, int(idx.size), _LHPC_CHUNK):
        ii = idx[start:start + _LHPC_CHUNK]
        vals = np.asarray(arr[ii], dtype=np.float64)
        rhs[0] += float(np.sum(vals))
        rhs[1] += float(np.dot(x[ii], vals))
        rhs[2] += float(np.dot(y[ii], vals))
        rhs[3] += float(np.dot(z[ii], vals))

    try:
        coeff = np.asarray(md_entry["solver"].dot(rhs), dtype=np.float64).ravel()
    except Exception:
        coeff = np.zeros(4, dtype=np.float64)
    if coeff.size != 4 or not np.all(np.isfinite(coeff)):
        coeff = np.zeros(4, dtype=np.float64)

    for start in range(0, int(idx.size), _LHPC_CHUNK):
        ii = idx[start:start + _LHPC_CHUNK]
        clean[ii] = (np.asarray(arr[ii], dtype=np.float64)
                     - coeff[0] - coeff[1] * x[ii] - coeff[2] * y[ii] - coeff[3] * z[ii])
    bad = ~np.isfinite(clean)
    if np.any(bad):
        clean[bad] = 0.0
    return clean


def _lhpc_map2alm(finite_map, hp, lmax):
    import warnings
    lmax = int(lmax)
    fmap = np.asarray(finite_map, dtype=np.float64).ravel()
    if not np.all(np.isfinite(fmap)):
        fmap = np.array(fmap, dtype=np.float64, copy=True)
        fmap[~np.isfinite(fmap)] = 0.0
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        try:
            try:
                alm = hp.map2alm(fmap, lmax=lmax, iter=0, pol=False, use_weights=False)
            except TypeError:
                alm = hp.map2alm(fmap, lmax=lmax, iter=0, pol=False)
        except Exception:
            raise
    alm = np.asarray(alm, dtype=np.complex128).ravel()
    bad = ~np.isfinite(alm)
    if np.any(bad):
        alm = np.array(alm, dtype=np.complex128, copy=True)
        alm[bad] = 0.0 + 0.0j
    return alm


def _lhpc_alm2map(alm, hp, nside, lmax):
    import warnings
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        try:
            out = hp.alm2map(alm, int(nside), lmax=int(lmax), pol=False, verbose=False)
        except TypeError:
            out = hp.alm2map(alm, int(nside), lmax=int(lmax), pol=False)
    if np.ma.isMaskedArray(out):
        out = out.filled(0.0)
    out = np.asarray(out, dtype=np.float64).ravel()
    bad = ~np.isfinite(out)
    if np.any(bad):
        out = np.array(out, dtype=np.float64, copy=True)
        out[bad] = 0.0
    return out


def _lhpc_band_alm_to_lmax(alm_in, lmax_in, ell_min, ell_max, lmax_out):
    lmax_in = int(lmax_in)
    lmax_out = int(max(0, lmax_out))
    out = np.zeros(_lhpc_alm_size(lmax_out), dtype=np.complex128)
    hi = min(int(ell_max), lmax_in, lmax_out)
    lo_min = max(0, int(ell_min))
    if hi < lo_min:
        return out
    alm_in = np.asarray(alm_in, dtype=np.complex128).ravel()
    for mm in range(0, hi + 1):
        lo = max(lo_min, mm)
        if lo <= hi:
            i0 = _lhpc_alm_index(lmax_in, lo, mm)
            i1 = _lhpc_alm_index(lmax_in, hi, mm) + 1
            j0 = _lhpc_alm_index(lmax_out, lo, mm)
            j1 = _lhpc_alm_index(lmax_out, hi, mm) + 1
            n = min(i1 - i0, j1 - j0, int(alm_in.size) - i0, int(out.size) - j0)
            if n > 0:
                out[j0:j0 + n] = alm_in[i0:i0 + n]
    return out


def _lhpc_get_beam(lmax, fwhm_rad, state):
    lmax = int(lmax)
    key = str(lmax) + ":" + ("%.12g" % float(fwhm_rad))
    cache = state.get("beam_cache")
    if not isinstance(cache, dict):
        cache = {}
        state["beam_cache"] = cache
    beam = cache.get(key)
    if beam is not None:
        return beam
    sigma = float(fwhm_rad) / math.sqrt(8.0 * math.log(2.0))
    ell = np.arange(lmax + 1, dtype=np.float64)
    beam = np.exp(-0.5 * ell * (ell + 1.0) * sigma * sigma)
    beam[~np.isfinite(beam)] = 0.0
    if beam.size > 0:
        beam[0] = 1.0
    if len(cache) > 8:
        cache.clear()
    cache[key] = beam
    return beam


def _lhpc_apply_l_filter_inplace(alm, lmax, fl):
    lmax = int(lmax)
    fl = np.asarray(fl, dtype=np.float64).ravel()
    hi = min(lmax, int(fl.size) - 1)
    if hi < 0:
        return alm
    for mm in range(0, hi + 1):
        lo = mm
        i0 = _lhpc_alm_index(lmax, lo, mm)
        i1 = _lhpc_alm_index(lmax, hi, mm) + 1
        n = min(i1 - i0, int(alm.size) - i0, hi - lo + 1)
        if n > 0:
            alm[i0:i0 + n] *= fl[lo:lo + n]
    return alm


def _lhpc_gaussian_smooth_map(finite_map, hp, nside, lmax, state):
    alm = _lhpc_map2alm(finite_map, hp, int(lmax))
    expected = _lhpc_alm_size(int(lmax))
    if int(alm.size) > expected:
        alm = alm[:expected]
    beam = _lhpc_get_beam(int(lmax), _LHPC_FWHM_RAD, state)
    _lhpc_apply_l_filter_inplace(alm, int(lmax), beam)
    return _lhpc_alm2map(alm, hp, int(nside), int(lmax))


def _lhpc_masked_pearson(a, b, idx):
    idx = np.asarray(idx, dtype=np.int64).ravel()
    if int(idx.size) < 3:
        return 0.0

    count = 0
    sum_a = 0.0
    sum_b = 0.0
    for start in range(0, int(idx.size), _LHPC_CHUNK):
        ii = idx[start:start + _LHPC_CHUNK]
        aa = np.asarray(a[ii], dtype=np.float64)
        bb = np.asarray(b[ii], dtype=np.float64)
        good = np.isfinite(aa) & np.isfinite(bb)
        if np.any(good):
            sum_a += float(np.sum(aa[good]))
            sum_b += float(np.sum(bb[good]))
            count += int(np.count_nonzero(good))
    if count < 3:
        return 0.0

    mean_a = sum_a / float(count)
    mean_b = sum_b / float(count)
    if not np.isfinite(mean_a) or not np.isfinite(mean_b):
        return 0.0

    num = 0.0
    saa = 0.0
    sbb = 0.0
    for start in range(0, int(idx.size), _LHPC_CHUNK):
        ii = idx[start:start + _LHPC_CHUNK]
        aa = np.asarray(a[ii], dtype=np.float64)
        bb = np.asarray(b[ii], dtype=np.float64)
        good = np.isfinite(aa) & np.isfinite(bb)
        if np.any(good):
            da = aa[good] - mean_a
            db = bb[good] - mean_b
            num += float(np.dot(da, db))
            saa += float(np.dot(da, da))
            sbb += float(np.dot(db, db))

    if not np.isfinite(num) or not np.isfinite(saa) or not np.isfinite(sbb):
        return 0.0
    den = math.sqrt(max(saa, 0.0) * max(sbb, 0.0))
    if not np.isfinite(den) or den <= _LHPC_EPS:
        return 0.0
    r = float(num / den)
    if not np.isfinite(r):
        return 0.0
    if r > 1.0:
        r = 1.0
    elif r < -1.0:
        r = -1.0
    return float(r)


def analyze_map(m):
    try:
        if np.ma.isMaskedArray(m):
            arr = np.asarray(m.filled(np.nan), dtype=np.float64).ravel()
        else:
            arr = np.asarray(m, dtype=np.float64).ravel()
    except Exception:
        try:
            arr = np.asarray(np.real(m), dtype=np.float64).ravel()
        except Exception:
            return 0.0

    npix = int(arr.size)
    nside = _lhpc_infer_nside(npix)
    if nside <= 0:
        return 0.0

    valid = np.isfinite(arr) & (np.abs(arr) < 1.0e25)
    mask_info = _lhpc_make_mask_info(nside, npix, valid)
    if int(mask_info["nvalid"]) < 16:
        return 0.0

    lmax_possible = max(0, 3 * int(nside) - 1)
    lmax_transform = min(_LHPC_H_MAX, lmax_possible)
    if lmax_transform < _LHPC_H_MIN:
        return 0.0
    lmax_low = min(_LHPC_L_MAX, lmax_transform)
    if lmax_low < _LHPC_L_MIN:
        return 0.0

    try:
        import healpy as hp

        geom = _lhpc_get_geometry(nside, npix, _LHPC_STATE)
        md_entry = _lhpc_get_md_solver(nside, npix, valid, mask_info, geom, _LHPC_STATE)
        if int(md_entry["idx"].size) < 16:
            return 0.0

        # clean is finite and contains zero in masked pixels, so no NaN values
        # are passed to healpy spherical-harmonic routines.
        clean = _lhpc_remove_monopole_dipole(arr, md_entry, geom, npix)
        alm = _lhpc_map2alm(clean, hp, lmax_transform)
        expected = _lhpc_alm_size(lmax_transform)
        if int(alm.size) < expected:
            return 0.0
        if int(alm.size) > expected:
            alm = alm[:expected]

        alm_l = _lhpc_band_alm_to_lmax(alm, lmax_transform, _LHPC_L_MIN, _LHPC_L_MAX, lmax_low)
        alm_h = _lhpc_band_alm_to_lmax(alm, lmax_transform, _LHPC_H_MIN, _LHPC_H_MAX, lmax_transform)

        l_map = _lhpc_alm2map(alm_l, hp, nside, lmax_low)
        h_map = _lhpc_alm2map(alm_h, hp, nside, lmax_transform)
        if int(l_map.size) != npix or int(h_map.size) != npix:
            return 0.0

        with np.errstate(over="ignore", invalid="ignore"):
            p_map = h_map * h_map
        p_map = np.asarray(p_map, dtype=np.float64).ravel()
        bad = ~np.isfinite(p_map)
        if np.any(bad):
            p_map = np.array(p_map, dtype=np.float64, copy=True)
            p_map[bad] = 0.0

        p_smooth = _lhpc_gaussian_smooth_map(p_map, hp, nside, lmax_transform, _LHPC_STATE)
        if int(p_smooth.size) != npix:
            return 0.0

        r = _lhpc_masked_pearson(l_map, p_smooth, md_entry["idx"])
        if not np.isfinite(r):
            return 0.0
        if r > 1.0:
            r = 1.0
        elif r < -1.0:
            r = -1.0
        return float(r)
    except Exception:
        return 0.0


def summarize_results(planck_stat, sim_results, _desc=test_description):
    import numpy as np

    def _as_float(x):
        try:
            v = float(x)
            if np.isfinite(v):
                return v
            return np.nan
        except Exception:
            return np.nan

    ps = _as_float(planck_stat)
    try:
        sims_all = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            iterator = iter(sim_results)
        except Exception:
            iterator = []
        for item in iterator:
            v = _as_float(item)
            if np.isfinite(v):
                vals.append(v)
        sims_all = np.asarray(vals, dtype=np.float64)
    sims = sims_all[np.isfinite(sims_all)]
    n = int(sims.size)

    out = {
        "test_description": _desc,
        "statistic": "r = masked-pixel Pearson correlation between L(n), the ell=2..20 temperature field, and smoothed P(n)=H(n)^2, where H uses ell=40..128; r is signed and lies in [-1, 1].",
        "primary_tail": "two-sided in |r|; unusually large absolute correlation indicates stronger low-high power coupling than in LCDM simulations, while the sign records whether enhanced small-scale power aligns with hot or cold low-l regions",
        "harmonic_ranges": {"L_lmin": _LHPC_L_MIN, "L_lmax": _LHPC_L_MAX, "H_lmin": _LHPC_H_MIN, "H_lmax": _LHPC_H_MAX},
        "power_smoothing_fwhm_deg": 5.0,
        "monopole_dipole_fit": "least-squares fit of [1, x, y, z] on finite unmasked pixels only, subtracted before harmonic filtering",
        "mask_and_transform_handling": "non-finite pixels are ignored in masked fits/correlations and are filled with 0.0 before every healpy spherical-harmonic transform",
        "n_sims": n,
        "plot_spec": {
            "kind": "histogram",
            "title": "Masked low-high power-coupling correlation",
            "xlabel": "r = corr[L, smooth(H^2)] on unmasked pixels",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "Simulation distribution of the signed masked-pixel correlation between the low-l field L and the 5 degree smoothed local small-scale power H^2. Large |r| is the primary anomaly criterion; the sign gives the coupling direction."
        }
    }

    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic supplied to summarize_results is not finite."
        return out

    ps_c = float(np.minimum(1.0, np.maximum(-1.0, ps)))
    out["planck_r"] = ps_c
    out["planck_abs_r"] = float(abs(ps_c))
    if ps_c > 0.0:
        out["planck_coupling_direction"] = "positive: hotter low-l regions tend to coincide with higher smoothed small-scale power"
    elif ps_c < 0.0:
        out["planck_coupling_direction"] = "negative: colder low-l regions tend to coincide with higher smoothed small-scale power"
    else:
        out["planck_coupling_direction"] = "zero or numerically indistinguishable from zero"

    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    sims_c = np.minimum(1.0, np.maximum(-1.0, sims))
    mean = float(np.mean(sims_c))
    median = float(np.median(sims_c))
    std = float(np.std(sims_c, ddof=1)) if n > 1 else 0.0
    if n > 1:
        q = np.percentile(sims_c, [1, 5, 16, 50, 84, 95, 99])
        q_abs = np.percentile(np.abs(sims_c), [50, 90, 95, 99])
    else:
        q = np.repeat(sims_c[0], 7)
        q_abs = np.repeat(abs(sims_c[0]), 4)

    n_lt = int(np.count_nonzero(sims_c < ps_c))
    n_le = int(np.count_nonzero(sims_c <= ps_c))
    n_eq = int(np.count_nonzero(sims_c == ps_c))
    n_gt = int(np.count_nonzero(sims_c > ps_c))
    n_ge = int(np.count_nonzero(sims_c >= ps_c))
    abs_ps = abs(ps_c)
    abs_sims = np.abs(sims_c)
    n_abs_gt = int(np.count_nonzero(abs_sims > abs_ps))
    n_abs_ge = int(np.count_nonzero(abs_sims >= abs_ps))

    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_two_sided_value = float(min(1.0, 2.0 * min(p_upper, p_lower)))
    p_abs = float((n_abs_ge + 1.0) / (n + 1.0))

    out.update({
        "sim_mean": mean,
        "sim_median": median,
        "sim_std": std,
        "sim_min": float(np.min(sims_c)),
        "sim_max": float(np.max(sims_c)),
        "sim_percentile_1": float(q[0]),
        "sim_percentile_5": float(q[1]),
        "sim_percentile_16": float(q[2]),
        "sim_percentile_50": float(q[3]),
        "sim_percentile_84": float(q[4]),
        "sim_percentile_95": float(q[5]),
        "sim_percentile_99": float(q[6]),
        "sim_abs_r_median": float(q_abs[0]),
        "sim_abs_r_percentile_90": float(q_abs[1]),
        "sim_abs_r_percentile_95": float(q_abs[2]),
        "sim_abs_r_percentile_99": float(q_abs[3]),
        "zscore_vs_sims": float((ps_c - mean) / std) if std > 0.0 and np.isfinite(std) else None,
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "n_sim_abs_gt_abs_planck": n_abs_gt,
        "n_sim_abs_ge_abs_planck": n_abs_ge,
        "fraction_sims_ge_planck_upper_tail_including_ties": float(n_ge / n),
        "fraction_sims_le_planck_lower_tail_including_ties": float(n_le / n),
        "fraction_sims_abs_ge_abs_planck": float(n_abs_ge / n),
        "empirical_p_upper_tail_plus_one": p_upper,
