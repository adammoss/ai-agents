Hypothesis

The Planck SMICA masked low-l excess kurtosis K is anomalously different from the LCDM simulation distribution, either unusually high or unusually low.

Method

Excess kurtosis K was computed on unmasked pixels of masked, monopole/dipole-subtracted Planck SMICA and 1000 LCDM maps after retaining only 2 <= l <= 30.

Result Summary

Planck has K = 0.255, compared with simulation mean 0.0058, median -0.0157, and std 0.203. It lies at about the 89th percentile, with z ≈ 1.23 and two-tailed p = 0.218.

The Planck value is mildly high but well inside the LCDM simulation distribution and not in either 5% tail. This is not evidence for anomalous low-l non-Gaussianity by this kurtosis statistic.

Agent Justification

The code and description consider both directions: high K indicates unusually heavy-tailed or peaked low-l structure, while low K indicates unusually platykurtic structure. Since no clear prior directional preference is specified and the summary reports both upper/lower-tail and two-sided p-values, the appropriate default is a two-tailed test.

Comparison To Literature

Broadly consistent with Planck Collaboration, “Planck 2015 results. XVI. Isotropy and statistics of the CMB” (arXiv:1506.07135), which reports skewness/kurtosis-based Gaussianity tests generally consistent with simulations, while noting specific localized features such as the Cold Spot.

test_description = "Masked Low-l Excess Kurtosis: remove the best-fit monopole and dipole on the unmasked pixels of each masked nside=256 temperature map, zero-fill the mask, transform to spherical harmonics, keep only 2 <= ell <= 30, synthesize the corresponding low-ell map, and compute the unmasked-pixel excess kurtosis K = <(T-<T>)^4>/<(T-<T>)^2>^2 - 3."


def analyze_map(m, _cache={}):
    import math
    import zlib
    import numpy as np

    LMAX = 30
    ELL_MIN = 2
    TINY = 1.0e-300

    def _as_flat_float(a):
        try:
            if np.ma.isMaskedArray(a):
                arr = np.asarray(a.filled(np.nan), dtype=np.float64)
            else:
                arr = np.asarray(a, dtype=np.float64)
        except Exception:
            try:
                arr = np.asarray(np.real(a), dtype=np.float64)
            except Exception:
                arr = np.asarray([], dtype=np.float64)
        return np.ravel(arr)

    def _get_hp():
        key = "healpy_for_masked_lowl_excess_kurtosis"
        if key not in _cache:
            try:
                import healpy as hp
                _cache[key] = hp
            except Exception:
                _cache[key] = None
        return _cache.get(key)

    def _nside_from_npix(hp, npix):
        try:
            ns = int(hp.npix2nside(int(npix)))
            if ns > 0 and int(hp.nside2npix(ns)) == int(npix):
                return ns
        except Exception:
            pass
        try:
            if int(npix) > 0 and int(npix) % 12 == 0:
                ns = int(round(math.sqrt(float(npix) / 12.0)))
                if ns > 0 and 12 * ns * ns == int(npix):
                    return ns
        except Exception:
            pass
        return -1

    def _mask_signature(valid):
        v = np.asarray(valid, dtype=bool).ravel()
        try:
            crc = int(zlib.crc32(np.packbits(v).tobytes()) & 0xffffffff)
        except Exception:
            crc = int(np.count_nonzero(v) & 0xffffffff)
        return str(int(v.size)) + ":" + str(int(np.count_nonzero(v))) + ":" + str(crc)

    def _pixvecs(hp, nside):
        nside = int(nside)
        key = "pixvecs_ring_lowl_kurtosis_nside_" + str(nside)
        got = _cache.get(key)
        if got is not None:
            return got
        npix = int(hp.nside2npix(nside))
        pix = np.arange(npix, dtype=np.int64)
        try:
            x, y, z = hp.pix2vec(nside, pix, nest=False)
        except TypeError:
            x, y, z = hp.pix2vec(nside, pix)
        got = (np.asarray(x, dtype=np.float64), np.asarray(y, dtype=np.float64), np.asarray(z, dtype=np.float64))
        try:
            if int(_cache.get("pixvec_cache_count_lowl_kurtosis", 0)) < 4:
                _cache[key] = got
                _cache["pixvec_cache_count_lowl_kurtosis"] = int(_cache.get("pixvec_cache_count_lowl_kurtosis", 0)) + 1
        except Exception:
            pass
        return got

    def _fit_geometry(hp, nside, valid):
        valid = np.asarray(valid, dtype=bool).ravel()
        sig = _mask_signature(valid)
        key = "monopole_dipole_geom_lowl_kurtosis_" + str(int(nside)) + "_" + sig
        got = _cache.get(key)
        if got is not None:
            return got
        x, y, z = _pixvecs(hp, int(nside))
        if int(x.size) != int(valid.size):
            return None
        xv = x[valid]
        yv = y[valid]
        zv = z[valid]
        n = int(xv.size)
        if n < 4:
            return None
        sx = float(np.sum(xv)); sy = float(np.sum(yv)); sz = float(np.sum(zv))
        g = np.array([
            [float(n), sx, sy, sz],
            [sx, float(np.dot(xv, xv)), float(np.dot(xv, yv)), float(np.dot(xv, zv))],
            [sy, float(np.dot(yv, xv)), float(np.dot(yv, yv)), float(np.dot(yv, zv))],
            [sz, float(np.dot(zv, xv)), float(np.dot(zv, yv)), float(np.dot(zv, zv))]
        ], dtype=np.float64)
        invg = None
        try:
            invg = np.linalg.inv(g)
            if not np.all(np.isfinite(invg)):
                invg = None
        except Exception:
            invg = None
        got = {"xv": xv, "yv": yv, "zv": zv, "g": g, "invg": invg}
        try:
            if int(_cache.get("md_geom_cache_count_lowl_kurtosis", 0)) < 8:
                _cache[key] = got
                _cache["md_geom_cache_count_lowl_kurtosis"] = int(_cache.get("md_geom_cache_count_lowl_kurtosis", 0)) + 1
        except Exception:
            pass
        return got

    def _remove_monopole_dipole_zero_mask(arr, valid, hp, nside):
        arr = np.asarray(arr, dtype=np.float64).ravel()
        valid = np.asarray(valid, dtype=bool).ravel()
        out = np.zeros_like(arr, dtype=np.float64)
        if int(arr.size) != int(valid.size):
            return out
        vals = arr[valid]
        if int(vals.size) <= 0:
            return out
        geom = _fit_geometry(hp, int(nside), valid)
        if geom is None or int(vals.size) < 4:
            mu = float(np.mean(vals)) if int(vals.size) else 0.0
            resid = vals - mu
            out[valid] = np.where(np.isfinite(resid), resid, 0.0)
            return out
        rhs = np.array([
            float(np.sum(vals)),
            float(np.dot(vals, geom["xv"])),
            float(np.dot(vals, geom["yv"])),
            float(np.dot(vals, geom["zv"]))
        ], dtype=np.float64)
        try:
            if geom.get("invg", None) is not None:
                beta = np.dot(geom["invg"], rhs)
            else:
                beta = np.linalg.solve(geom["g"], rhs)
        except Exception:
            try:
                beta = np.linalg.lstsq(geom["g"], rhs, rcond=None)[0]
            except Exception:
                beta = np.array([float(np.mean(vals)), 0.0, 0.0, 0.0], dtype=np.float64)
        beta = np.where(np.isfinite(beta), beta, 0.0)
        model = beta[0] + beta[1] * geom["xv"] + beta[2] * geom["yv"] + beta[3] * geom["zv"]
        resid = vals - model
        out[valid] = np.where(np.isfinite(resid), resid, 0.0)
        return out

    def _excess_kurtosis(vals):
        vals = np.asarray(vals, dtype=np.float64).ravel()
        vals = vals[np.isfinite(vals)]
        n = int(vals.size)
        if n <= 1:
            return 0.0
        cen = vals - float(np.mean(vals))
        m2 = float(np.mean(cen * cen))
        if (not np.isfinite(m2)) or m2 <= TINY:
            return 0.0
        c2 = cen * cen
        m4 = float(np.mean(c2 * c2))
        k = m4 / (m2 * m2) - 3.0
        if not np.isfinite(k):
            k = 0.0
        return float(k)

    def _fallback_value(arr, valid):
        # Last-resort finite fallback if healpy is unavailable or the map geometry is invalid.
        # The intended statistic uses a spherical-harmonic low-ell map; this proxy only prevents
        # non-finite failures in unsupported environments.
        try:
            vals = np.asarray(arr, dtype=np.float64).ravel()[np.asarray(valid, dtype=bool).ravel()]
            return _excess_kurtosis(vals)
        except Exception:
            return 0.0

    arr = _as_flat_float(m)
    npix = int(arr.size)
    if npix <= 0:
        return 0.0
    valid = np.isfinite(arr)
    n_valid = int(np.count_nonzero(valid))
    if n_valid <= 1:
        return 0.0

    hp = _get_hp()
    if hp is None:
        return float(_fallback_value(arr, valid))
    nside = _nside_from_npix(hp, npix)
    if nside <= 0:
        return float(_fallback_value(arr, valid))
    try:
        if int(3 * int(nside) - 1) < LMAX:
            return float(_fallback_value(arr, valid))
    except Exception:
        pass

    try:
        cleaned = _remove_monopole_dipole_zero_mask(arr, valid, hp, nside)
        # Never pass NaN/inf into healpy. The mask is represented by exactly zero-valued pixels here.
        cleaned = np.where(np.isfinite(cleaned), cleaned, 0.0).astype(np.float64, copy=False)

        try:
            alm = hp.map2alm(cleaned, lmax=LMAX, iter=0, pol=False, use_weights=False)
        except TypeError:
            try:
                alm = hp.map2alm(cleaned, lmax=LMAX, iter=0, pol=False)
            except TypeError:
                alm = hp.map2alm(cleaned, lmax=LMAX, iter=0)
        alm = np.asarray(alm)
        if alm.size <= 0:
            return float(_fallback_value(arr, valid))

        # Keep only multipoles 2 <= ell <= 30. map2alm was computed only to ell=30, so only ell=0,1
        # need to be explicitly removed after the monopole/dipole-subtracted masked transform.
        try:
            filt = np.zeros(LMAX + 1, dtype=np.float64)
            filt[ELL_MIN:LMAX + 1] = 1.0
            try:
                hp.almxfl(alm, filt, mmax=LMAX, inplace=True)
                alm_f = alm
            except TypeError:
                alm_f = hp.almxfl(alm, filt, inplace=False)
        except Exception:
            alm_f = np.array(alm, copy=True)
            try:
                for ell in range(0, min(ELL_MIN, LMAX + 1)):
                    for mm in range(0, ell + 1):
                        alm_f[hp.Alm.getidx(LMAX, ell, mm)] = 0.0
            except Exception:
                pass

        try:
            low_map = hp.alm2map(alm_f, nside=int(nside), lmax=LMAX, mmax=LMAX, pol=False, verbose=False)
        except TypeError:
            try:
                low_map = hp.alm2map(alm_f, nside=int(nside), lmax=LMAX, mmax=LMAX, pol=False)
            except TypeError:
                low_map = hp.alm2map(alm_f, nside=int(nside), lmax=LMAX)
        low_map = np.asarray(low_map, dtype=np.float64).ravel()
        if int(low_map.size) != npix:
            return float(_fallback_value(arr, valid))
        vals = low_map[valid]
        k = _excess_kurtosis(vals)
        if not np.isfinite(k):
            k = 0.0
        return float(k)
    except Exception:
        return float(_fallback_value(arr, valid))


def summarize_results(planck_stat, sim_results, _desc=test_description):
    import numpy as np

    def _finite_float(x, default=np.nan):
        try:
            y = float(x)
        except Exception:
            return float(default)
        return float(y) if np.isfinite(y) else float(default)

    ps = _finite_float(planck_stat)
    try:
        sims = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            for v in sim_results:
                vals.append(_finite_float(v))
        except Exception:
            vals = []
        sims = np.asarray(vals, dtype=np.float64)
    sims = sims[np.isfinite(sims)]
    n = int(sims.size)

    out = {
        "test_description": _desc,
        "statistic": "Excess kurtosis K = <(T-<T>)^4>/<(T-<T>)^2>^2 - 3 of the masked, monopole/dipole-subtracted, 2<=ell<=30 synthesized map, evaluated on unmasked pixels only",
        "planck_excess_kurtosis": ps,
        "n_sims": n,
        "tail_of_interest": "upper tail for unusually heavy-tailed/peaked low-ell temperature structure; lower tail for unusually platykurtic low-ell structure",
        "multipole_range": "2 <= ell <= 30",
        "monopole_dipole_treatment": "best-fit monopole plus Cartesian dipole removed on unmasked pixels before zero-filling the mask and harmonic filtering",
        "mask_treatment": "input finite pixels define the binary analysis mask; masked NaN pixels are zero-filled for harmonic transforms, while kurtosis averages use only unmasked pixels",
        "plot_spec": {
            "kind": "histogram",
            "title": "Masked low-l excess kurtosis",
            "xlabel": "K = excess kurtosis of 2<=ell<=30 masked low-l map",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "LCDM simulation distribution of the masked low-l excess-kurtosis statistic. Planck is compared to identically processed simulations; larger K indicates heavier tails or a more peaked unmasked low-l temperature distribution."
        }
    }

    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_le = int(np.count_nonzero(sims <= ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_ge = int(np.count_nonzero(sims >= ps))
    n_eq = int(n - n_lt - n_gt)

    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_two = float(min(1.0, 2.0 * min(p_lower, p_upper)))
    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    z = None
    if np.isfinite(std) and std > 0.0:
        z = float((ps - mean) / std)
    q = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])

    out.update({
        "sim_mean_excess_kurtosis": mean,
        "sim_median_excess_kurtosis": median,
        "sim_std_excess_kurtosis": std,
        "sim_min_excess_kurtosis": float(np.min(sims)),
        "sim_max_excess_kurtosis": float(np.max(sims)),
        "sim_percentile_1_excess_kurtosis": float(q[0]),
        "sim_percentile_5_excess_kurtosis": float(q[1]),
        "sim_percentile_16_excess_kurtosis": float(q[2]),
        "sim_percentile_50_excess_kurtosis": float(q[3]),
        "sim_percentile_84_excess_kurtosis": float(q[4]),
        "sim_percentile_95_excess_kurtosis": float(q[5]),
        "sim_percentile_99_excess_kurtosis": float(q[6]),
        "planck_zscore_vs_sims": z,
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "rank_ascending_1_is_most_platykurtic": int(n_lt + 1),
        "rank_descending_1_is_most_leptokurtic": int(n_gt + 1),
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "fraction_sims_lt_planck": float(n_lt / n),
        "fraction_sims_le_planck": float(n_le / n),
        "empirical_p_lower_tail_platykurtic_including_ties_plus_one": p_lower,
        "empirical_p_upper_tail_leptokurtic_including_ties_plus_one": p_upper,
        "empirical_p_two_sided_plus_one": p_two,
        "difference_planck_minus_sim_median": float(ps - median),
        "is_lower_tail_5pct_platykurtic": bool(p_lower <= 0.05),
        "is_lower_tail_1pct_platykurtic": bool(p_lower <= 0.01),
        "is_upper_tail_5pct_leptokurtic": bool(p_upper <= 0.05),
        "is_upper_tail_1pct_leptokurtic": bool(p_upper <= 0.01),
        "is_two_sided_5pct": bool(p_two <= 0.05),
        "is_two_sided_1pct": bool(p_two <= 0.01)
    })

    if std == 0.0:
        out["warning"] = "All finite simulation statistics are identical; check the input maps, mask, and harmonic-transform support."
    elif p_upper <= 0.01:
        out["interpretation"] = "Planck is in the highest 1% of simulations: the masked 2<=ell<=30 temperature distribution is unusually heavy-tailed or peaked by this excess-kurtosis statistic."
    elif p_upper <= 0.05:
        out["interpretation"] = "Planck is in the highest 5% of simulations: the masked 2<=ell<=30 temperature distribution has elevated excess kurtosis by this statistic."
    elif p_lower <= 0.01:
        out["interpretation"] = "Planck is in the lowest 1% of simulations: the masked 2<=ell<=30 temperature distribution is unusually platykurtic by this statistic."
    elif p_lower <= 0.05:
        out["interpretation"] = "Planck is in the lowest 5% of simulations: the masked 2<=ell<=30 temperature distribution has unusually low excess kurtosis by this statistic."
    else:
        out["interpretation"] = "Planck is not in either 5% tail for the masked low-l excess-kurtosis statistic."
    return out
