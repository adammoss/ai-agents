Hypothesis

The masked, monopole/dipole-subtracted low-ℓ Planck map has an anomalous Galactic 180° half-turn relationship compared with simulations, meaning its correlation r between T(n) and T(Rπn) is unusually different from the simulated expectation, either too positive or too negative.

Method

Monopole/dipole-subtracted masked maps were filtered to 2≤ℓ≤32, rotated by 180° in Galactic longitude, and compared using the area-weighted Pearson correlation on mutually unmasked pixels.

Result Summary

Planck gives r=-0.108, a mild half-turn anti-correlation. Simulations have mean r=0.00495 and std=0.12865 from 1000 realizations. The two-tailed p-value is 0.408, with z≈-0.88 and Planck at about the 20th percentile.

The Planck value is well within the simulated ΛCDM distribution and is not close to the 5% two-tailed anomaly threshold. The statistic is non-degenerate.

Agent Justification

The statistic is a signed Pearson correlation where positive values indicate half-turn symmetry and negative values indicate anti-correlation. The description and code do not pre-specify that only positive correlation or only negative anti-correlation is anomalous; both tails are reported and interpreted. Therefore there is no clear directional preference, so a two-tailed test is appropriate.

Comparison To Literature

Prior searches did not identify a directly comparable published CMB test of this specific masked low-ℓ Galactic half-turn Pearson correlation, so no quantitative literature comparison is available.

test_description = "Masked Low-l Galactic Half-Turn Correlation: fit and subtract the monopole and dipole on unmasked pixels, reconstruct only 2<=ell<=32, rotate by 180 degrees in Galactic longitude, and return the area-weighted Pearson correlation on pixels unmasked both before and after the rotation."


def analyze_map(m):
    import math
    import warnings
    import numpy as np

    # Convert input to a flat float array while preserving NaNs as the mask.
    try:
        if np.ma.isMaskedArray(m):
            a = np.asarray(m.filled(np.nan), dtype=np.float64).ravel()
        else:
            a = np.asarray(m, dtype=np.float64).ravel()
    except Exception:
        try:
            a = np.asarray(np.real(m), dtype=np.float64).ravel()
        except Exception:
            return 0.0

    npix = int(a.size)
    if npix <= 0:
        return 0.0

    valid = np.isfinite(a)
    nvalid = int(np.count_nonzero(valid))
    if nvalid < 4:
        return 0.0

    # Infer HEALPix nside from the one-dimensional map length.
    try:
        if npix % 12 != 0:
            nside = -1
        else:
            nside = int(round(math.sqrt(npix / 12.0)))
            if nside <= 0 or 12 * nside * nside != npix:
                nside = -1
    except Exception:
        nside = -1

    # Non-HEALPix fallback only prevents crashes on malformed probes; the real
    # statistic is the HEALPix branch below.
    if nside <= 0:
        vals = a[valid]
        if vals.size < 3:
            return 0.0
        shift = int(vals.size // 2)
        if shift <= 0:
            return 0.0
        x0 = vals
        y0 = np.roll(vals, shift)
        good = np.isfinite(x0) & np.isfinite(y0)
        if int(np.count_nonzero(good)) < 3:
            return 0.0
        x0 = x0[good]
        y0 = y0[good]
        x0 = x0 - float(np.mean(x0))
        y0 = y0 - float(np.mean(y0))
        vx = float(np.dot(x0, x0))
        vy = float(np.dot(y0, y0))
        if vx <= 0.0 or vy <= 0.0 or (not np.isfinite(vx)) or (not np.isfinite(vy)):
            return 0.0
        r0 = float(np.dot(x0, y0) / np.sqrt(vx * vy))
        if not np.isfinite(r0):
            return 0.0
        return float(max(-1.0, min(1.0, r0)))

    lmax = min(32, max(0, 3 * int(nside) - 1))
    if lmax < 2:
        return 0.0

    try:
        import healpy as hp
    except Exception:
        return 0.0

    try:
        # Cache geometry and half-turn permutation across simulations of the
        # same nside.  This uses only vectorized HEALPix calls.
        g = globals()
        cache = g.get("_MLGHTC_GEOM_CACHE")
        if not isinstance(cache, dict):
            cache = {}
            g["_MLGHTC_GEOM_CACHE"] = cache
        key = (int(nside), int(npix))
        geom = cache.get(key)
        if geom is None:
            pix = np.arange(npix, dtype=np.int64)
            x, y, z = hp.pix2vec(int(nside), pix, nest=False)
            x = np.asarray(x, dtype=np.float64)
            y = np.asarray(y, dtype=np.float64)
            z = np.asarray(z, dtype=np.float64)
            # R_pi: (l,b) -> (l+180 deg,b), equivalently (x,y,z)->(-x,-y,z).
            rot = np.asarray(hp.vec2pix(int(nside), -x, -y, z, nest=False), dtype=np.int64)
            if rot.size != npix:
                return 0.0
            geom = (x, y, z, rot)
            cache[key] = geom
        else:
            x, y, z, rot = geom
            if int(np.asarray(rot).size) != npix:
                return 0.0

        x, y, z, rot = geom
        rot = np.asarray(rot, dtype=np.int64)

        # Fit monopole + Cartesian dipole components on unmasked pixels.  HEALPix
        # pixels have equal area, so area-weighted least squares is ordinary
        # least squares over the retained pixels.
        residual = np.zeros(npix, dtype=np.float64)
        tv = np.asarray(a[valid], dtype=np.float64)
        xv = np.asarray(x[valid], dtype=np.float64)
        yv = np.asarray(y[valid], dtype=np.float64)
        zv = np.asarray(z[valid], dtype=np.float64)

        try:
            one = float(nvalid)
            sx = float(np.sum(xv, dtype=np.float64))
            sy = float(np.sum(yv, dtype=np.float64))
            sz = float(np.sum(zv, dtype=np.float64))
            sxx = float(np.dot(xv, xv))
            syy = float(np.dot(yv, yv))
            szz = float(np.dot(zv, zv))
            sxy = float(np.dot(xv, yv))
            sxz = float(np.dot(xv, zv))
            syz = float(np.dot(yv, zv))
            gram = np.array([[one, sx,  sy,  sz],
                             [sx,  sxx, sxy, sxz],
                             [sy,  sxy, syy, syz],
                             [sz,  sxz, syz, szz]], dtype=np.float64)
            rhs = np.array([float(np.sum(tv, dtype=np.float64)),
                            float(np.dot(tv, xv)),
                            float(np.dot(tv, yv)),
                            float(np.dot(tv, zv))], dtype=np.float64)
            try:
                beta = np.linalg.solve(gram, rhs)
            except Exception:
                beta = np.linalg.lstsq(gram, rhs, rcond=None)[0]
            beta = np.asarray(beta, dtype=np.float64).ravel()
            if beta.size != 4 or not np.all(np.isfinite(beta)):
                raise ValueError("bad monopole/dipole fit")
            residual[valid] = tv - (beta[0] + beta[1] * xv + beta[2] * yv + beta[3] * zv)
        except Exception:
            mu = float(np.mean(tv)) if tv.size else 0.0
            if not np.isfinite(mu):
                mu = 0.0
            residual[valid] = tv - mu

        # Never pass NaN/Inf to healpy.  Masked pixels are finite-filled only for
        # the harmonic transform; the final correlation still uses the original
        # intersection mask.
        residual = np.asarray(residual, dtype=np.float64)
        residual[~np.isfinite(residual)] = 0.0

        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            try:
                alm = hp.map2alm(residual, lmax=int(lmax), iter=0, pol=False)
            except TypeError:
                alm = hp.map2alm(residual, lmax=int(lmax), iter=0)

        fl = np.zeros(int(lmax) + 1, dtype=np.float64)
        fl[2:int(lmax) + 1] = 1.0
        try:
            alm = hp.almxfl(alm, fl, inplace=False)
        except TypeError:
            alm = hp.almxfl(alm, fl)

        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            try:
                low = hp.alm2map(alm, nside=int(nside), lmax=int(lmax), pol=False, verbose=False)
            except TypeError:
                try:
                    low = hp.alm2map(alm, int(nside), lmax=int(lmax), verbose=False)
                except TypeError:
                    try:
                        low = hp.alm2map(alm, nside=int(nside), lmax=int(lmax), pol=False)
                    except TypeError:
                        low = hp.alm2map(alm, int(nside), lmax=int(lmax))
        low = np.asarray(low, dtype=np.float64).ravel()
        if low.size != npix:
            return 0.0
        low[~np.isfinite(low)] = 0.0

        common = valid & valid[rot]
        if int(np.count_nonzero(common)) < 3:
            return 0.0

        tx = np.asarray(low[common], dtype=np.float64)
        ty = np.asarray(low[rot[common]], dtype=np.float64)
        good = np.isfinite(tx) & np.isfinite(ty)
        if int(np.count_nonzero(good)) < 3:
            return 0.0
        tx = tx[good]
        ty = ty[good]
        tx = tx - float(np.mean(tx))
        ty = ty - float(np.mean(ty))
        vx = float(np.dot(tx, tx))
        vy = float(np.dot(ty, ty))
        if vx <= 0.0 or vy <= 0.0 or (not np.isfinite(vx)) or (not np.isfinite(vy)):
            return 0.0
        r = float(np.dot(tx, ty) / np.sqrt(vx * vy))
        if not np.isfinite(r):
            return 0.0
        if r > 1.0:
            r = 1.0
        elif r < -1.0:
            r = -1.0
        return float(r)
    except Exception:
        return 0.0


def summarize_results(planck_stat, sim_results):
    import numpy as np

    desc = globals().get("test_description", "Masked Low-l Galactic Half-Turn Correlation")
    try:
        ps = float(planck_stat)
    except Exception:
        ps = np.nan
    try:
        sims = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            for item in sim_results:
                try:
                    val = float(item)
                    if np.isfinite(val):
                        vals.append(val)
                except Exception:
                    pass
        except Exception:
            pass
        sims = np.asarray(vals, dtype=np.float64)
    sims = sims[np.isfinite(sims)]
    n = int(sims.size)

    out = {
        "test_description": desc,
        "statistic": "Area-weighted Pearson correlation r between the monopole/dipole-subtracted 2<=ell<=32 map T(n) and T(R_pi n).",
        "multipole_range": "2 <= ell <= 32, truncated only if the supplied nside cannot support ell=32",
        "rotation": "Galactic longitude half-turn R_pi: (l,b) -> (l+180 deg,b)",
        "mask_handling": "Input non-finite pixels define the mask. Monopole/dipole fitting uses unmasked pixels; final correlation uses only pixels for which both n and R_pi n were originally unmasked. NaNs are replaced by zero only before healpy harmonic transforms.",
        "sign_convention": "r>0 means half-turn correlation/symmetry; r<0 means half-turn anti-correlation; r near 0 means little linear half-turn relation.",
        "planck_r": float(ps) if np.isfinite(ps) else None,
        "n_sims": n,
        "plot_spec": {
            "kind": "histogram",
            "title": "Masked low-l Galactic half-turn correlation",
            "xlabel": "r = corr[T(n), T(R_pi n)]",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "Simulation distribution of the low-l Galactic half-turn Pearson correlation, with the Planck value compared by rank/tail probabilities."
        }
    }

    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_eq = int(n - n_lt - n_gt)
    n_le = int(n_lt + n_eq)
    n_ge = int(n_gt + n_eq)

    # Plus-one empirical p-values avoid zero p-values for finite simulation sets.
    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_two_rank = float(min(1.0, 2.0 * min(p_lower, p_upper)))

    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    z = float((ps - mean) / std) if std > 0.0 and np.isfinite(std) else None
    q = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])

    planck_dev = abs(float(ps - median))
    sim_dev = np.abs(sims - median)
    n_dev_ge = int(np.count_nonzero(sim_dev >= planck_dev))
    p_dev = float((n_dev_ge + 1.0) / (n + 1.0))

    if ps > 0.0:
        signed_direction = "positive half-turn correlation"
        p_signed = p_upper
    elif ps < 0.0:
        signed_direction = "negative half-turn anti-correlation"
        p_signed = p_lower
    else:
        signed_direction = "zero correlation"
        p_signed = 1.0

    out.update({
        "planck_correlation_sign": signed_direction,
        "sim_mean_r": mean,
        "sim_median_r": median,
        "sim_std_r": std,
        "sim_min_r": float(np.min(sims)),
        "sim_max_r": float(np.max(sims)),
        "sim_percentile_1_r": float(q[0]),
        "sim_percentile_5_r": float(q[1]),
        "sim_percentile_16_r": float(q[2]),
        "sim_percentile_50_r": float(q[3]),
        "sim_percentile_84_r": float(q[4]),
        "sim_percentile_95_r": float(q[5]),
        "sim_percentile_99_r": float(q[6]),
        "planck_zscore_vs_sims": z,
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "rank_ascending_1_is_most_anticorrelated": int(n_lt + 1),
        "rank_descending_1_is_most_correlated": int(n_gt + 1),
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "fraction_sims_le_planck": float(n_le / n),
        "fraction_sims_ge_planck": float(n_ge / n),
        "empirical_p_lower_tail_anticorrelation_plus_one": p_lower,
        "empirical_p_upper_tail_correlation_plus_one": p_upper,
        "empirical_p_two_sided_rank_plus_one": p_two_rank,
        "empirical_p_in_planck_signed_direction_plus_one": float(p_signed),
        "median_centered_abs_deviation_planck": float(planck_dev),
        "n_sims_median_centered_abs_deviation_ge_planck": n_dev_ge,
        "empirical_p_median_centered_two_sided_plus_one": p_dev,
        "difference_planck_minus_sim_median_r": float(ps - median),
        "is_5pct_low_tail_anticorrelation": bool(p_lower <= 0.05),
        "is_5pct_high_tail_correlation": bool(p_upper <= 0.05),
        "is_1pct_low_tail_anticorrelation": bool(p_lower <= 0.01),
        "is_1pct_high_tail_correlation": bool(p_upper <= 0.01),
        "is_5pct_two_sided_rank": bool(p_two_rank <= 0.05),
        "is_1pct_two_sided_rank": bool(p_two_rank <= 0.01),
        "is_5pct_median_centered_two_sided": bool(p_dev <= 0.05),
        "is_1pct_median_centered_two_sided": bool(p_dev <= 0.01)
    })

    if std == 0.0:
        out["warning"] = "All finite simulation half-turn correlations are identical; check map variation, masking, harmonic filtering, and rotation indexing."
    elif p_signed <= 0.01:
        out["interpretation"] = "Planck is in the most extreme 1% of simulations in its signed half-turn-correlation direction."
    elif p_signed <= 0.05:
        out["interpretation"] = "Planck is in the most extreme 5% of simulations in its signed half-turn-correlation direction."
    elif p_two_rank <= 0.05:
        out["interpretation"] = "Planck is unusual at the 5% level by the two-sided rank tail probability for this half-turn statistic."
    elif p_dev <= 0.05:
        out["interpretation"] = "Planck has a median-centered half-turn-correlation deviation in the most extreme 5% of simulations."
    else:
        out["interpretation"] = "Planck is not in the most extreme 5% of simulations for this low-l Galactic half-turn correlation statistic."

    return out
