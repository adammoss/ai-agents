Consistent GPT5.5 · Variation · hemispherical · one-tailed, upper tail

Hypothesis

Planck/SMICA has an unusually large low-ell hemispherical variance contrast, quantified by a high value of S = max |ln(V_plus / V_minus)|, compared with LCDM simulations.

Method

Scans antipodal hemispheres on the masked, monopole/dipole-subtracted Planck SMICA temperature map after retaining 2<=ell<=64, computing S=max |ln(V_plus/V_minus)| from hemisphere variances, and compares the same statistic with 1000 LCDM simulations.

Result Summary

Planck gives S=0.408, equivalent to a larger/smaller hemispherical variance ratio of 1.50. Simulations have mean 0.315, standard deviation 0.094, and median 0.302. The upper-tail p-value is 0.162, with Planck at about the 83.8th percentile and z≈0.99.

The Planck value is above the simulation mean but well within the LCDM distribution; 162/1000 simulations have an equal or larger maximum variance contrast. It does not reach the 5% anomaly threshold.

Not recorded.

Comparison To Literature

Hemispherical power/variance asymmetry is well documented, with local-variance analyses reporting stronger anomalies, e.g. Akrami et al. (2014) found no larger Planck asymmetry among 1000 FFP simulations, at >=3.3σ, and Adhikari (2014) reported large-scale asymmetry at >=3.3σ. This implementation differs by using a direct full-hemisphere maximum log-variance ratio over 2<=ell<=64, and finds a much weaker result.

import numpy as np
import healpy as hp
import warnings

# Low-l Hemispherical Variance Max Contrast
test_description = (
    "Low-l Hemispherical Variance Max Contrast: fit and subtract the best-fit "
    "monopole and dipole on finite unmasked pixels, zero-fill the mask only for "
    "harmonic filtering, retain 2<=ell<=64, then scan one HEALPix Nside=16 "
    "hemisphere axis from each antipodal pair.  For each retained axis compute "
    "the two masked-hemisphere sample variances after subtracting each "
    "hemisphere's own mean, discard axes with either side below 30% of a full "
    "hemisphere, and return max |ln(V_plus/V_minus)|."
)

LMAX_HEMI_VAR = 64
AXIS_NSIDE_HEMI_VAR = 16
MAX_WORK_NSIDE_HEMI_VAR = 64
MIN_HEMI_FRAC_HEMI_VAR = 0.30
TINY_HEMI_VAR = np.finfo(np.float64).tiny
MAX_HEMI_FLOAT32_BYTES = 256 * 1024 * 1024
HEMI_BUILD_BLOCK_AXES = 64
HEMI_DOT_BLOCK_AXES = 128

FIT_GEOMETRY_CACHE_HEMI_VAR = {}
WORK_MASK_CACHE_HEMI_VAR = {}
AXIS_CACHE_HEMI_VAR = {}
HEMI_GEOMETRY_CACHE_HEMI_VAR = {}


def axis_vectors_hemi_var(np=np, hp=hp, axis_nside=AXIS_NSIDE_HEMI_VAR, cache=AXIS_CACHE_HEMI_VAR):
    axes = cache.get(axis_nside)
    if axes is not None:
        return axes

    npix_axis = int(hp.nside2npix(axis_nside))
    pix = np.arange(npix_axis, dtype=np.int64)
    x, y, z = hp.pix2vec(axis_nside, pix)
    x = np.asarray(x, dtype=np.float64)
    y = np.asarray(y, dtype=np.float64)
    z = np.asarray(z, dtype=np.float64)

    # Canonical representative of each antipodal pair: positive z; on the
    # equator use positive y, and finally positive x for exact tie cases.
    eps = 1.0e-14
    keep = (z > eps) | ((np.abs(z) <= eps) & (y > eps)) | ((np.abs(z) <= eps) & (np.abs(y) <= eps) & (x >= 0.0))
    axes = np.vstack((x[keep], y[keep], z[keep])).T.astype(np.float64, copy=False)

    # The lexicographic half-sphere rule should give exactly half the pixels.
    # If numerical conventions ever change, keep the statistic usable by taking
    # the most positive half in the same lexicographic ordering.
    target = npix_axis // 2
    if axes.shape[0] != target:
        order = np.lexsort((x, y, z))
        take = order[-target:]
        axes = np.vstack((x[take], y[take], z[take])).T.astype(np.float64, copy=False)

    cache[axis_nside] = axes
    return axes


def fit_geometry_hemi_var(nside, valid, np=np, hp=hp, cache=FIT_GEOMETRY_CACHE_HEMI_VAR):
    valid = np.asarray(valid, dtype=bool).ravel()
    npix = int(valid.size)
    nvalid = int(np.count_nonzero(valid))

    entries = cache.get(npix, [])
    for candidate in entries:
        if candidate["nvalid"] == nvalid and np.array_equal(candidate["valid"], valid):
            return candidate

    pix = np.flatnonzero(valid).astype(np.int64, copy=False)
    if nvalid > 0:
        x, y, z = hp.pix2vec(nside, pix)
        x = np.asarray(x, dtype=np.float64)
        y = np.asarray(y, dtype=np.float64)
        z = np.asarray(z, dtype=np.float64)

        sx = float(np.sum(x)); sy = float(np.sum(y)); sz = float(np.sum(z))
        sxx = float(np.dot(x, x)); sxy = float(np.dot(x, y)); sxz = float(np.dot(x, z))
        syy = float(np.dot(y, y)); syz = float(np.dot(y, z)); szz = float(np.dot(z, z))
        gram = np.array(
            [[float(nvalid), sx, sy, sz],
             [sx, sxx, sxy, sxz],
             [sy, sxy, syy, syz],
             [sz, sxz, syz, szz]],
            dtype=np.float64,
        )
        gram_pinv = np.linalg.pinv(gram, rcond=1.0e-12)
    else:
        x = y = z = np.empty(0, dtype=np.float64)
        gram_pinv = np.zeros((4, 4), dtype=np.float64)

    geom = {
        "valid": valid.copy(),
        "nvalid": nvalid,
        "pix": pix,
        "x": x,
        "y": y,
        "z": z,
        "gram_pinv": gram_pinv,
    }
    cache[npix] = [geom] + entries[:2]
    return geom


def work_valid_mask_hemi_var(nside_in, valid_in, nside_work, np=np, hp=hp, cache=WORK_MASK_CACHE_HEMI_VAR):
    valid_in = np.asarray(valid_in, dtype=bool).ravel()
    if int(nside_work) == int(nside_in):
        return valid_in

    nvalid = int(np.count_nonzero(valid_in))
    key = (int(nside_in), int(nside_work), int(valid_in.size))
    entries = cache.get(key, [])
    for candidate in entries:
        if candidate["nvalid"] == nvalid and np.array_equal(candidate["valid_in"], valid_in):
            return candidate["valid_work"]

    frac = hp.ud_grade(valid_in.astype(np.float64), nside_out=int(nside_work), order_in="RING", order_out="RING", power=0)
    frac = np.asarray(frac, dtype=np.float64).ravel()
    valid_work = np.isfinite(frac) & (frac >= 0.5)

    entry = {"nvalid": nvalid, "valid_in": valid_in.copy(), "valid_work": valid_work}
    cache[key] = [entry] + entries[:1]
    return valid_work


def hemi_geometry_hemi_var(
    nside,
    valid,
    np=np,
    hp=hp,
    axis_vectors_hemi_var=axis_vectors_hemi_var,
    cache=HEMI_GEOMETRY_CACHE_HEMI_VAR,
    min_frac=MIN_HEMI_FRAC_HEMI_VAR,
    max_float32_bytes=MAX_HEMI_FLOAT32_BYTES,
    build_block_axes=HEMI_BUILD_BLOCK_AXES,
):
    valid = np.asarray(valid, dtype=bool).ravel()
    npix = int(valid.size)
    nvalid = int(np.count_nonzero(valid))
    key = (int(nside), npix)

    entries = cache.get(key, [])
    for candidate in entries:
        if candidate["nvalid"] == nvalid and np.array_equal(candidate["valid"], valid):
            return candidate

    axes = axis_vectors_hemi_var()
    naxes = int(axes.shape[0])
    pix = np.flatnonzero(valid).astype(np.int64, copy=False)

    total_elements = int(naxes * max(nvalid, 0))
    store_float32 = (4 * total_elements) <= int(max_float32_bytes)
    hemi_dtype = np.float32 if store_float32 else bool
    hemi = np.empty((naxes, nvalid), dtype=hemi_dtype)

    if nvalid > 0:
        vx, vy, vz = hp.pix2vec(int(nside), pix)
        pixvec = np.vstack((np.asarray(vx, dtype=np.float64),
                            np.asarray(vy, dtype=np.float64),
                            np.asarray(vz, dtype=np.float64)))
        for start in range(0, naxes, int(build_block_axes)):
            stop = min(naxes, start + int(build_block_axes))
            dots = axes[start:stop].dot(pixvec)
            if store_float32:
                hemi[start:stop, :] = (dots >= 0.0).astype(np.float32, copy=False)
            else:
                hemi[start:stop, :] = (dots >= 0.0)
    else:
        hemi[:, :] = 0.0 if store_float32 else False

    counts_plus = np.rint(np.sum(hemi, axis=1, dtype=np.float64)).astype(np.int64)
    counts_minus = int(nvalid) - counts_plus
    min_count = int(np.ceil(float(min_frac) * (float(npix) / 2.0)))
    axis_ok = (counts_plus >= min_count) & (counts_minus >= min_count) & (counts_plus > 1) & (counts_minus > 1)

    geom = {
        "valid": valid.copy(),
        "nvalid": nvalid,
        "pix": pix,
        "hemi": hemi,
        "counts_plus": counts_plus,
        "counts_minus": counts_minus,
        "axis_ok": axis_ok,
        "min_count": min_count,
    }
    # Hemisphere matrices can be large, so keep only the current and previous
    # masks for a given nside.
    cache[key] = [geom] + entries[:1]
    return geom


def hemi_matrix_dot_hemi_var(hemi, values32, np=np, block_axes=HEMI_DOT_BLOCK_AXES):
    hemi = np.asarray(hemi)
    values32 = np.asarray(values32, dtype=np.float32)
    naxes = int(hemi.shape[0])
    if hemi.shape[1] == 0:
        return np.zeros(naxes, dtype=np.float64)

    if hemi.dtype == np.float32:
        return np.asarray(hemi.dot(values32), dtype=np.float64)

    out = np.empty(naxes, dtype=np.float64)
    for start in range(0, naxes, int(block_axes)):
        stop = min(naxes, start + int(block_axes))
        block = hemi[start:stop].astype(np.float32, copy=False)
        out[start:stop] = np.asarray(block.dot(values32), dtype=np.float64)
    return out


def alm2map_quiet_hemi_var(alm, nside, lmax, hp=hp, warnings=warnings):
    try:
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            return hp.alm2map(alm, int(nside), lmax=int(lmax), pol=False, verbose=False)
    except TypeError:
        return hp.alm2map(alm, int(nside), lmax=int(lmax), pol=False)


def analyze_map(
    m,
    np=np,
    hp=hp,
    lmax=LMAX_HEMI_VAR,
    max_work_nside=MAX_WORK_NSIDE_HEMI_VAR,
    tiny=TINY_HEMI_VAR,
    fit_geometry_hemi_var=fit_geometry_hemi_var,
    work_valid_mask_hemi_var=work_valid_mask_hemi_var,
    hemi_geometry_hemi_var=hemi_geometry_hemi_var,
    hemi_matrix_dot_hemi_var=hemi_matrix_dot_hemi_var,
    alm2map_quiet_hemi_var=alm2map_quiet_hemi_var,
):
    arr = np.asarray(m, dtype=np.float64)
    if arr.ndim != 1:
        arr = arr.ravel()

    npix_in = int(arr.size)
    if npix_in <= 0:
        return 0.0

    try:
        nside_in = int(hp.npix2nside(npix_in))
    except Exception:
        return 0.0

    valid_in = np.isfinite(arr)
    nvalid_in = int(np.count_nonzero(valid_in))
    if nvalid_in <= 1:
        return 0.0

    # Best-fit monopole/dipole on the unmasked sky; masked pixels remain a
    # finite zero fill before any healpy spherical-harmonic transform.
    try:
        fit_geom = fit_geometry_hemi_var(nside_in, valid_in)
    except Exception:
        return 0.0

    cleaned = np.zeros(npix_in, dtype=np.float64)
    pix = fit_geom["pix"]
    if pix.size > 0:
        vals = arr[pix].astype(np.float64, copy=False)
        x = fit_geom["x"]; y = fit_geom["y"]; z = fit_geom["z"]
        rhs = np.array(
            [float(np.sum(vals)), float(np.dot(x, vals)), float(np.dot(y, vals)), float(np.dot(z, vals))],
            dtype=np.float64,
        )
        coeff = fit_geom["gram_pinv"].dot(rhs)
        if not np.all(np.isfinite(coeff)):
            return 0.0
        cleaned[pix] = vals - (coeff[0] + coeff[1] * x + coeff[2] * y + coeff[3] * z)

    lmax_eff = int(min(int(lmax), max(0, 3 * int(nside_in) - 1)))
    if lmax_eff < 2:
        return 0.0

    nside_work = int(min(int(nside_in), int(max_work_nside)))
    try:
        alm = hp.map2alm(cleaned, lmax=lmax_eff, iter=0, pol=False)
        alm = np.array(alm, dtype=np.complex128, copy=True)
        alm[hp.Alm.getidx(lmax_eff, 0, 0)] = 0.0
        alm[hp.Alm.getidx(lmax_eff, 1, 0)] = 0.0
        alm[hp.Alm.getidx(lmax_eff, 1, 1)] = 0.0
        filtered = alm2map_quiet_hemi_var(alm, nside_work, lmax_eff)
    except Exception:
        return 0.0

    filtered = np.asarray(filtered, dtype=np.float64).ravel()
    expected_work_npix = int(hp.nside2npix(nside_work))
    if filtered.size != expected_work_npix:
        return 0.0
    if not np.all(np.isfinite(filtered)):
        filtered = filtered.copy()
        filtered[~np.isfinite(filtered)] = 0.0

    try:
        valid_work = work_valid_mask_hemi_var(nside_in, valid_in, nside_work)
    except Exception:
        return 0.0
    valid_work = np.asarray(valid_work, dtype=bool).ravel()
    if valid_work.size != expected_work_npix:
        return 0.0

    try:
        hemi_geom = hemi_geometry_hemi_var(nside_work, valid_work)
    except Exception:
        return 0.0

    if hemi_geom["nvalid"] <= 1 or not bool(np.any(hemi_geom["axis_ok"])):
        return 0.0

    pix_work = hemi_geom["pix"]
    values64 = filtered[pix_work].astype(np.float64, copy=False)
    if values64.size <= 1:
        return 0.0
    if not np.all(np.isfinite(values64)):
        values64 = values64.copy()
        values64[~np.isfinite(values64)] = 0.0

    values32 = values64.astype(np.float32)
    values2_32 = (values32 * values32).astype(np.float32, copy=False)

    hemi = hemi_geom["hemi"]
    try:
        sum_plus = hemi_matrix_dot_hemi_var(hemi, values32)
        sumsq_plus = hemi_matrix_dot_hemi_var(hemi, values2_32)
    except Exception:
        return 0.0

    counts_plus_i = hemi_geom["counts_plus"]
    counts_minus_i = hemi_geom["counts_minus"]
    counts_plus = counts_plus_i.astype(np.float64)
    counts_minus = counts_minus_i.astype(np.float64)

    total_sum = float(np.sum(values64, dtype=np.float64))
    total_sumsq = float(np.dot(values64, values64))
    sum_minus = total_sum - sum_plus
    sumsq_minus = total_sumsq - sumsq_plus

    cp_safe = np.maximum(counts_plus, 1.0)
    cm_safe = np.maximum(counts_minus, 1.0)
    with np.errstate(divide="ignore", invalid="ignore", over="ignore"):
        raw_plus = sumsq_plus - (sum_plus * sum_plus) / cp_safe
        raw_minus = sumsq_minus - (sum_minus * sum_minus) / cm_safe
        raw_plus = np.where(raw_plus > 0.0, raw_plus, 0.0)
        raw_minus = np.where(raw_minus > 0.0, raw_minus, 0.0)
        var_plus = raw_plus / np.maximum(counts_plus - 1.0, 1.0)
        var_minus = raw_minus / np.maximum(counts_minus - 1.0, 1.0)

    ok = hemi_geom["axis_ok"] & np.isfinite(var_plus) & np.isfinite(var_minus)
    if not bool(np.any(ok)):
        return 0.0

    pos_plus = var_plus[ok & (var_plus > 0.0)]
    pos_minus = var_minus[ok & (var_minus > 0.0)]
    if pos_plus.size == 0 and pos_minus.size == 0:
        return 0.0
    if pos_plus.size == 0:
        min_positive = float(np.min(pos_minus))
    elif pos_minus.size == 0:
        min_positive = float(np.min(pos_plus))
    else:
        min_positive = float(min(np.min(pos_plus), np.min(pos_minus)))
    if not np.isfinite(min_positive) or min_positive <= 0.0:
        return 0.0

    # A scale-relative floor keeps exact-zero numerical edge cases finite without
    # affecting ordinary CMB maps, whose hemispherical variances are positive.
    floor = max(float(tiny), min_positive * 1.0e-12)
    with np.errstate(divide="ignore", invalid="ignore", over="ignore"):
        vp = np.maximum(var_plus[ok], floor)
        vm = np.maximum(var_minus[ok], floor)
        contrast = np.abs(np.log(vp) - np.log(vm))
    contrast = contrast[np.isfinite(contrast)]
    if contrast.size == 0:
        return 0.0

    stat = float(np.max(contrast))
    if not np.isfinite(stat):
        return 0.0
    return stat


def summarize_results(planck_stat, sim_results, np=np, test_description=test_description):
    sims = np.asarray(sim_results, dtype=np.float64)
    sims = sims[np.isfinite(sims)]
    ps = float(planck_stat)

    plot_spec = {
        "kind": "histogram",
        "title": "Low-l hemispherical variance max contrast",
        "xlabel": "S = max |ln(V_plus / V_minus)|",
        "ylabel": "Number of LCDM simulations",
        "bins": 30,
        "caption": (
            "Distribution over LCDM simulations of the maximum absolute log "
            "hemispherical variance contrast after monopole/dipole removal and "
            "2<=ell<=64 filtering.  Large S means a stronger hemispherical "
            "variance asymmetry; the Planck/SMICA value should be marked on the "
            "same axis."
        ),
    }

    out = {
        "test_description": test_description,
        "planck_stat": ps,
        "tail_of_interest": "upper tail: unusually large maximum hemispherical variance contrast",
        "plot_spec": plot_spec,
    }

    if np.isfinite(ps):
        out["planck_larger_to_smaller_variance_ratio_equiv"] = float(np.exp(min(ps, 700.0)))

    n = int(sims.size)
    out["n_sims"] = n
    if n == 0 or not np.isfinite(ps):
        out["warning"] = "No finite simulation statistics, or the Planck statistic is non-finite."
        return out

    n_ge = int(np.count_nonzero(sims >= ps))
    n_le = int(np.count_nonzero(sims <= ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_lt = int(np.count_nonzero(sims < ps))
    n_eq = int(n - n_gt - n_lt)

    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_lower = float((n_le + 1.0) / (n + 1.0))
    percentile_midrank = float(100.0 * (n_lt + 0.5 * n_eq) / n)

    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    z = float((ps - mean) / std) if std > 0.0 and np.isfinite(std) else None
    q1, q5, q16, q50, q84, q95, q99 = [float(q) for q in np.percentile(sims, [1.0, 5.0, 16.0, 50.0, 84.0, 95.0, 99.0])]

    out.update({
        "sim_mean": mean,
        "sim_median": median,
        "sim_std": std,
        "sim_percentile_1": q1,
        "sim_percentile_5": q5,
        "sim_percentile_16": q16,
        "sim_percentile_50": q50,
        "sim_percentile_84": q84,
        "sim_percentile_95": q95,
        "sim_percentile_99": q99,
        "planck_zscore_vs_sims": z,
        "planck_percentile_midrank": percentile_midrank,
        "rank_descending_min": int(n_gt + 1),
        "rank_descending_max": int(n_ge),
        "rank_ascending_min": int(n_lt + 1),
        "rank_ascending_max": int(n_le),
        "rank_convention": "descending rank 1 is the largest S, i.e. the strongest hemispherical variance contrast",
        "n_sim_ge_planck": n_ge,
        "n_sim_le_planck": n_le,
        "empirical_p_upper_asymmetry": p_upper,
        "empirical_p_lower": p_lower,
        "is_upper_tail_5pct": bool(p_upper <= 0.05),
        "is_upper_tail_1pct": bool(p_upper <= 0.01),
    })

    if std == 0.0:
        out["warning"] = "All finite simulation statistics are identical; check for a degenerate statistic or inputs."
    elif p_upper <= 0.05:
        out["interpretation"] = "Planck/SMICA lies in the high-S tail, indicating an unusually large low-ell hemispherical variance contrast relative to the simulations."
    else:
        out["interpretation"] = "Planck/SMICA is not in the high-S tail at the 5% empirical level for this maximum-contrast statistic."

    return out
