Consistent GPT5.5 · Variation · direction · one-tailed, upper tail

Hypothesis

Planck’s low-ℓ CMB temperature map has an unusually large maximum normalized hemispheric mean imbalance compared with LCDM simulations.

Method

After monopole/dipole removal and common low-ℓ filtering to 2≤ℓ≤20, the statistic S is the maximum over trial hemisphere axes of the absolute difference between opposite-hemisphere mean temperatures, normalized by the low-ℓ rms.

Result Summary

Planck gives S=0.230, below the simulation mean 0.256 and median 0.252. Among 1000 ΛCDM simulations, 67.7% have S at least as large as Planck, giving an upper-tail p-value of 0.677 and z≈-0.53.

The Planck map does not show an unusually large hemispheric mean imbalance by this statistic; if anything, its value is mildly lower than typical ΛCDM simulations.

Agent Justification

The statistic S is defined as the maximum absolute hemispheric mean contrast, so the sign/direction of the imbalance is already absorbed. Larger S specifically means a stronger imbalance, and the code evaluates the empirical upper-tail probability for Planck relative to simulations.

Comparison To Literature

This is related to the broad hemispherical asymmetry literature, but most published Planck/WMAP analyses focus on hemispherical power asymmetry, local variance, or dipole modulation rather than hemispheric mean temperature contrast. Such studies often report low-ℓ power asymmetry at ≳2σ, e.g. Planck anomaly analyses and Sanyal et al. 2024 on Planck PR4 local-variance hemispherical power asymmetry, but this mean-imbalance test does not reproduce an anomaly.

import math
import numpy as np


test_description = "Large-Scale Hemispheric Mean Imbalance: remove the best-fit monopole and dipole on finite pixels, construct a common lightweight pseudo-filtered 2<=ell<=20 large-scale field, scan HEALPix Nside_axis=16 trial hemispheres, and return the maximum normalized hemispheric mean contrast."


def analyze_map(m, _state={}):
    import math
    import numpy as np

    LMIN = 2
    LMAX = 20
    NSIDE_AXIS = 16
    MAX_SAMPLE = 2048

    state = _state

    def as_float_1d(x):
        try:
            if np.ma.isMaskedArray(x):
                a = np.asarray(x.filled(np.nan), dtype=np.float64)
            else:
                a = np.asarray(x, dtype=np.float64)
        except Exception:
            try:
                a = np.asarray(np.real(x), dtype=np.float64)
            except Exception:
                a = np.asarray([], dtype=np.float64)
        return np.ravel(a)

    def finite_scalar(x, default=0.0):
        try:
            y = float(x)
        except Exception:
            return float(default)
        return y if np.isfinite(y) else float(default)

    def nside_from_npix(npix):
        try:
            npx = int(npix)
            if npx > 0 and npx % 12 == 0:
                ns = int(round((npx / 12.0) ** 0.5))
                if ns > 0 and 12 * ns * ns == npx:
                    return ns
        except Exception:
            pass
        return -1

    def ring_vectors(nside):
        key = "ring_vectors_%d" % int(nside)
        cached = state.get(key)
        if cached is not None:
            return cached
        nside = int(nside)
        if nside <= 0:
            z0 = np.zeros(0, dtype=np.float32)
            out0 = (z0, z0, z0)
            state[key] = out0
            return out0

        npix = 12 * nside * nside
        pix = np.arange(npix, dtype=np.int64)
        x = np.empty(npix, dtype=np.float32)
        y = np.empty(npix, dtype=np.float32)
        z = np.empty(npix, dtype=np.float32)
        phi = np.empty(npix, dtype=np.float64)

        ncap = 2 * nside * (nside - 1)
        nl4 = 4 * nside
        fact = 1.0 / (3.0 * float(nside) * float(nside))

        if ncap > 0:
            north = pix < ncap
            if np.any(north):
                ip = pix[north]
                ir = np.floor(0.5 * (1.0 + np.sqrt(1.0 + 2.0 * ip.astype(np.float64)))).astype(np.int64)
                ir = np.maximum(ir, 1)
                iphi = ip + 1 - 2 * ir * (ir - 1)
                z[north] = (1.0 - (ir.astype(np.float64) ** 2) * fact).astype(np.float32)
                phi[north] = (iphi.astype(np.float64) - 0.5) * (0.5 * math.pi) / ir.astype(np.float64)

            south = pix >= npix - ncap
            if np.any(south):
                ip = (npix - pix[south] - 1).astype(np.int64)
                ir = np.floor(0.5 * (1.0 + np.sqrt(1.0 + 2.0 * ip.astype(np.float64)))).astype(np.int64)
                ir = np.maximum(ir, 1)
                iphi = 4 * ir - (ip - 2 * ir * (ir - 1))
                z[south] = (-1.0 + (ir.astype(np.float64) ** 2) * fact).astype(np.float32)
                phi[south] = (iphi.astype(np.float64) - 0.5) * (0.5 * math.pi) / ir.astype(np.float64)

        equ = np.logical_and(pix >= ncap, pix < npix - ncap)
        if np.any(equ):
            ip = pix[equ] - ncap
            ir = (ip // nl4) + nside
            iphi = (ip % nl4) + 1
            fodd = 0.5 * (1.0 + ((ir + nside) & 1).astype(np.float64))
            z[equ] = (((2.0 * nside - ir.astype(np.float64)) * 2.0) / (3.0 * nside)).astype(np.float32)
            phi[equ] = (iphi.astype(np.float64) - fodd) * (0.5 * math.pi) / float(nside)

        sint = np.sqrt(np.maximum(0.0, 1.0 - z.astype(np.float64) * z.astype(np.float64)))
        x[:] = (sint * np.cos(phi)).astype(np.float32)
        y[:] = (sint * np.sin(phi)).astype(np.float32)
        out = (x, y, z)
        state[key] = out
        return out

    def axis_vectors():
        key = "axis_vectors_%d" % int(NSIDE_AXIS)
        cached = state.get(key)
        if cached is not None:
            return cached
        ax = ring_vectors(NSIDE_AXIS)
        full = np.vstack([ax[0], ax[1], ax[2]]).T.astype(np.float32, copy=False)
        eps = np.float32(1.0e-7)
        keep = np.logical_or(full[:, 2] > eps, np.logical_and(np.abs(full[:, 2]) <= eps, full[:, 1] > 0.0))
        axes = full[keep]
        if axes.size == 0:
            axes = full
        axes = np.asarray(axes, dtype=np.float32)
        state[key] = axes
        return axes

    def fit_monopole_dipole(a, valid, x, y, z):
        n = int(np.count_nonzero(valid))
        if n <= 0:
            return np.zeros(4, dtype=np.float64)
        vals = np.asarray(a[valid], dtype=np.float64)
        if vals.size < 4:
            c = np.zeros(4, dtype=np.float64)
            c[0] = finite_scalar(np.mean(vals), 0.0)
            return c
        xv = x[valid].astype(np.float64, copy=False)
        yv = y[valid].astype(np.float64, copy=False)
        zv = z[valid].astype(np.float64, copy=False)
        sx = float(np.sum(xv)); sy = float(np.sum(yv)); sz = float(np.sum(zv))
        xx = float(np.dot(xv, xv)); xy = float(np.dot(xv, yv)); xz = float(np.dot(xv, zv))
        yy = float(np.dot(yv, yv)); yz = float(np.dot(yv, zv)); zz = float(np.dot(zv, zv))
        mat = np.array([[float(n), sx, sy, sz], [sx, xx, xy, xz], [sy, xy, yy, yz], [sz, xz, yz, zz]], dtype=np.float64)
        rhs = np.array([float(np.sum(vals)), float(np.dot(vals, xv)), float(np.dot(vals, yv)), float(np.dot(vals, zv))], dtype=np.float64)
        try:
            c = np.linalg.solve(mat, rhs)
        except Exception:
            try:
                c = np.linalg.lstsq(mat, rhs, rcond=None)[0]
            except Exception:
                c = np.zeros(4, dtype=np.float64)
                c[0] = finite_scalar(np.mean(vals), 0.0)
        if not np.all(np.isfinite(c)):
            c = np.zeros(4, dtype=np.float64)
            c[0] = finite_scalar(np.mean(vals), 0.0)
        return c.astype(np.float64, copy=False)

    def choose_sample(valid):
        idx = np.flatnonzero(valid)
        n = int(idx.size)
        if n <= MAX_SAMPLE:
            return idx.astype(np.int64, copy=False)
        loc = np.linspace(0, n - 1, MAX_SAMPLE, dtype=np.int64)
        return idx[loc].astype(np.int64, copy=False)

    def basis_size(lmax):
        lm = int(lmax)
        val = (lm + 1) * (lm + 1) - LMIN * LMIN
        return int(max(0, val))

    def adaptive_lmax(nsample):
        ns = int(nsample)
        if ns < 12:
            return 0
        trial = math.floor(math.sqrt(max(1, ns // 2) + LMIN * LMIN) - 1)
        return int(max(LMIN, min(LMAX, trial)))

    def real_sph_harm_basis(x, y, z, lmax):
        lmax = int(lmax)
        n = int(len(z))
        nb = basis_size(lmax)
        if n <= 0 or nb <= 0:
            return np.zeros((n, 0), dtype=np.float32)

        zz = np.asarray(z, dtype=np.float64)
        ph = np.arctan2(np.asarray(y, dtype=np.float64), np.asarray(x, dtype=np.float64))
        s = np.sqrt(np.maximum(0.0, 1.0 - zz * zz))

        cos_m = [np.ones(n, dtype=np.float64)]
        sin_m = [np.zeros(n, dtype=np.float64)]
        c1 = np.cos(ph)
        s1 = np.sin(ph)
        for _m in range(1, lmax + 1):
            pc = cos_m[-1]
            ps = sin_m[-1]
            cos_m.append(pc * c1 - ps * s1)
            sin_m.append(ps * c1 + pc * s1)

        B = np.empty((n, nb), dtype=np.float32)
        col = 0
        log4pi = math.log(4.0 * math.pi)
        sqrt2 = math.sqrt(2.0)

        for mm in range(0, lmax + 1):
            if mm == 0:
                pmm = np.ones(n, dtype=np.float64)
            else:
                pmm = np.ones(n, dtype=np.float64)
                fact = 1.0
                for _j in range(1, mm + 1):
                    pmm *= -fact * s
                    fact += 2.0

            p_lm_minus2 = None
            p_lm_minus1 = pmm
            for ell in range(mm, lmax + 1):
                if ell == mm:
                    plm = pmm
                elif ell == mm + 1:
                    plm = zz * (2 * mm + 1) * pmm
                else:
                    plm = ((2 * ell - 1) * zz * p_lm_minus1 - (ell + mm - 1) * p_lm_minus2) / float(ell - mm)

                if ell >= LMIN:
                    lognorm = 0.5 * (math.log(2 * ell + 1) - log4pi + math.lgamma(ell - mm + 1) - math.lgamma(ell + mm + 1))
                    norm = math.exp(lognorm)
                    if mm == 0:
                        if col < B.shape[1]:
                            B[:, col] = (norm * plm).astype(np.float32)
                        col += 1
                    else:
                        fac = sqrt2 * norm
                        if col < B.shape[1]:
                            B[:, col] = (fac * plm * cos_m[mm]).astype(np.float32)
                        col += 1
                        if col < B.shape[1]:
                            B[:, col] = (fac * plm * sin_m[mm]).astype(np.float32)
                        col += 1

                p_lm_minus2 = p_lm_minus1
                p_lm_minus1 = plm

        if col != nb:
            B = B[:, :min(col, nb)].astype(np.float32, copy=False)
        B[~np.isfinite(B)] = 0.0
        return B

    def configs_equal(cfg, npix, sample_idx):
        try:
            return cfg is not None and cfg.get("npix") == int(npix) and np.array_equal(cfg.get("sample_idx"), sample_idx)
        except Exception:
            return False

    def build_config(npix, sample_idx, x, y, z):
        sample_idx = np.asarray(sample_idx, dtype=np.int64)
        nsamp = int(sample_idx.size)
        lmax = adaptive_lmax(nsamp)
        xs = x[sample_idx].astype(np.float32, copy=False)
        ys = y[sample_idx].astype(np.float32, copy=False)
        zs = z[sample_idx].astype(np.float32, copy=False)
        B = real_sph_harm_basis(xs, ys, zs, lmax)
        nb = int(B.shape[1])

        if nsamp <= 0 or nb <= 0:
            pinv = np.zeros((0, nsamp), dtype=np.float32)
        else:
            Bd = B.astype(np.float64, copy=False)
            G = (Bd.T @ Bd) / float(max(1, nsamp))
            tr = float(np.trace(G)) / float(max(1, nb)) if nb > 0 else 1.0
            ridge = max(1.0e-10, 1.0e-6 * tr)
            G.flat[::nb + 1] += ridge
            try:
                pinv = np.linalg.solve(G, Bd.T / float(max(1, nsamp))).astype(np.float32)
            except Exception:
                pinv = np.linalg.pinv(Bd, rcond=1.0e-5).astype(np.float32)

        axes = axis_vectors()
        vecs = np.vstack([xs, ys, zs]).astype(np.float32, copy=False)
        dots = axes @ vecs
        pos = dots > 0.0
        cnt_plus = np.sum(pos, axis=1).astype(np.float32)
        cnt_minus = (float(nsamp) - cnt_plus).astype(np.float32)
        good = np.logical_and(cnt_plus > 0.0, cnt_minus > 0.0)
        cfg = {"npix": int(npix), "sample_idx": sample_idx.copy(), "lmax": int(lmax),
               "B": B.astype(np.float32, copy=False), "pinv": pinv,
               "posf": pos.astype(np.float32, copy=False),
               "cnt_plus": cnt_plus, "cnt_minus": cnt_minus, "good": good,
               "xs": xs, "ys": ys, "zs": zs, "n_axes": int(axes.shape[0])}
        return cfg

    def get_config(npix, sample_idx, x, y, z):
        cfg = state.get("last_config")
        if configs_equal(cfg, npix, sample_idx):
            return cfg
        cfg = build_config(npix, sample_idx, x, y, z)
        state["last_config"] = cfg
        return cfg

    def fallback_stat(a, valid):
        try:
            vals = np.asarray(a[valid], dtype=np.float64)
            vals = vals[np.isfinite(vals)]
            n = int(vals.size)
            if n < 8:
                return 0.0
            xx = np.linspace(-1.0, 1.0, n)
            A = np.vstack([np.ones(n), xx]).T
            try:
                c = np.linalg.lstsq(A, vals, rcond=None)[0]
                vals = vals - A @ c
            except Exception:
                vals = vals - float(np.mean(vals))
            ff = np.fft.rfft(vals)
            hi = min(20, ff.size - 1)
            if hi < 2:
                return 0.0
            filt = np.zeros_like(ff)
            filt[2:hi + 1] = ff[2:hi + 1]
            low = np.fft.irfft(filt, n=n)
            sig = finite_scalar(np.sqrt(np.mean(low * low)), 0.0)
            if sig <= 0.0:
                return 0.0
            half = max(1, n // 2)
            nwin = min(128, n)
            starts = np.linspace(0, n - 1, nwin, dtype=np.int64)
            low2 = np.concatenate([low, low])
            cs = np.concatenate([np.array([0.0]), np.cumsum(low2)])
            sums = cs[starts + half] - cs[starts]
            means1 = sums / float(half)
            means2 = (float(np.sum(low)) - sums) / float(max(1, n - half))
            out = float(np.max(np.abs(means1 - means2) / sig))
            return out if np.isfinite(out) else 0.0
        except Exception:
            return 0.0

    a = as_float_1d(m)
    if a.size <= 0:
        return 0.0
    valid = np.isfinite(a)
    nvalid = int(np.count_nonzero(valid))
    if nvalid < 8:
        return 0.0

    nside = nside_from_npix(a.size)
    if nside <= 0:
        return float(fallback_stat(a, valid))

    try:
        x, y, z = ring_vectors(nside)
        if x.size != a.size:
            return float(fallback_stat(a, valid))

        coef = fit_monopole_dipole(a, valid, x, y, z)
        sample_idx = choose_sample(valid)
        if sample_idx.size < 8:
            return 0.0
        cfg = get_config(a.size, sample_idx, x, y, z)

        xs = cfg["xs"].astype(np.float64, copy=False)
        ys = cfg["ys"].astype(np.float64, copy=False)
        zs = cfg["zs"].astype(np.float64, copy=False)
        vals = a[sample_idx].astype(np.float64, copy=False) - (coef[0] + coef[1] * xs + coef[2] * ys + coef[3] * zs)
        vals = np.asarray(vals, dtype=np.float32)
        vals[~np.isfinite(vals)] = 0.0

        B = cfg["B"]
        pinv = cfg["pinv"]
        if B.shape[1] > 0 and pinv.shape[0] == B.shape[1] and pinv.shape[1] == vals.size:
            coeff = pinv @ vals
            low = B @ coeff
        else:
            low = vals - np.float32(np.mean(vals))
        low = np.asarray(low, dtype=np.float32).ravel()
        low[~np.isfinite(low)] = 0.0
        if low.size < 8:
            return 0.0

        sigma = finite_scalar(np.sqrt(np.mean(low.astype(np.float64) ** 2)), 0.0)
        if sigma <= 0.0:
            return 0.0

        total = float(np.sum(low, dtype=np.float64))
        posf = cfg["posf"]
        sum_plus = posf @ low
        cnt_plus = cfg["cnt_plus"]
        cnt_minus = cfg["cnt_minus"]
        good = cfg["good"]
        with np.errstate(divide="ignore", invalid="ignore"):
            mean_plus = sum_plus.astype(np.float64) / cnt_plus.astype(np.float64)
            mean_minus = (total - sum_plus.astype(np.float64)) / cnt_minus.astype(np.float64)
            diff = np.abs(mean_plus - mean_minus) / float(sigma)
        if diff.size == 0 or not np.any(good):
            return 0.0
        best = finite_scalar(np.max(diff[good]), 0.0)
        return float(best) if np.isfinite(best) else 0.0
    except Exception:
        return float(fallback_stat(a, valid))


def summarize_results(planck_stat, sim_results, _desc=test_description):
    import numpy as np

    def finite_scalar(x, default=np.nan):
        try:
            y = float(x)
        except Exception:
            return float(default)
        return y if np.isfinite(y) else float(default)

    def finite_array(x):
        try:
            arr = np.asarray(x, dtype=np.float64).ravel()
        except Exception:
            vals = []
            try:
                for u in x:
                    y = finite_scalar(u, np.nan)
                    if np.isfinite(y):
                        vals.append(y)
            except Exception:
                vals = []
            arr = np.asarray(vals, dtype=np.float64)
        return arr[np.isfinite(arr)]

    ps = finite_scalar(planck_stat, np.nan)
    sims = finite_array(sim_results)
    n = int(sims.size)
    out = {
        "test_description": _desc,
        "statistic": "S = max over HEALPix Nside_axis=16 trial hemispheres of |mean(T_low,+)-mean(T_low,-)| / rms(T_low). Antipodal duplicate axes are omitted because they give the same contrast.",
        "tail_of_interest": "Upper tail: larger S means a stronger large-scale hemispheric mean imbalance.",
        "multipole_range": "Common lightweight pseudo-filter targeting 2 <= ell <= 20 on a deterministic finite-pixel sample.",
        "mask_handling": "np.isfinite defines unmasked pixels; monopole/dipole are fit on finite pixels; masked NaNs are never used as finite temperatures.",
        "planck_S": float(ps) if np.isfinite(ps) else None,
        "n_sims": n,
        "plot_spec": {
            "kind": "histogram",
            "title": "Large-scale hemispheric mean imbalance",
            "xlabel": "S = maximum normalized hemispheric mean contrast",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "Simulation distribution of S. The Planck anomaly significance is estimated by the empirical upper-tail rank among the LCDM simulations."
        }
    }
    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_eq = int(n - n_lt - n_gt)
    n_le = int(n_lt + n_eq)
    n_ge = int(n_gt + n_eq)

    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_two = float(min(1.0, 2.0 * min(p_upper, p_lower)))

    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    zscore = float((ps - mean) / std) if std > 0.0 and np.isfinite(std) else None
    q = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])

    out.update({
        "sim_mean_S": mean,
        "sim_median_S": median,
        "sim_std_S": std,
        "sim_min_S": float(np.min(sims)),
        "sim_max_S": float(np.max(sims)),
        "sim_percentile_1_S": float(q[0]),
        "sim_percentile_5_S": float(q[1]),
        "sim_percentile_16_S": float(q[2]),
        "sim_percentile_50_S": float(q[3]),
        "sim_percentile_84_S": float(q[4]),
        "sim_percentile_95_S": float(q[5]),
        "sim_percentile_99_S": float(q[6]),
        "planck_zscore_vs_sims": zscore,
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "rank_descending_1_is_largest_imbalance": int(n_gt + 1),
        "rank_ascending_1_is_smallest_imbalance": int(n_lt + 1),
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "fraction_sims_ge_planck": float(n_ge / n),
        "empirical_p_upper_tail_large_imbalance_plus_one": p_upper,
        "empirical_p_lower_tail_small_imbalance_plus_one": p_lower,
        "empirical_p_two_sided_plus_one": p_two,
        "difference_planck_minus_sim_median_S": float(ps - median),
        "is_upper_tail_5pct_large_imbalance": bool(p_upper <= 0.05),
        "is_upper_tail_1pct_large_imbalance": bool(p_upper <= 0.01)
    })

    if std == 0.0:
        out["warning"] = "All finite simulation S values are identical; check mask propagation and map variation."
    elif p_upper <= 0.01:
        out["interpretation"] = "Planck lies in the highest 1% of simulations for this hemispheric mean-imbalance statistic."
    elif p_upper <= 0.05:
        out["interpretation"] = "Planck lies in the highest 5% of simulations for this hemispheric mean-imbalance statistic."
    else:
        out["interpretation"] = "Planck is not in the 5% upper tail for this hemispheric mean-imbalance statistic."
    return out
