Consistent GPT5.5 · Variation · parity · one-tailed, upper tail

Hypothesis

Planck has an unusually large positive parity-split axis concentration contrast, i.e. odd low-l multipole axes are more mutually aligned than even low-l multipole axes, giving S = C_odd - C_even higher than expected under LCDM.

Method

Uses masked Planck SMICA pseudo-a_lm to lmax=31, derives angular-momentum power-tensor axes for each multipole, compares odd- and even-parity axis concentration via S = C_odd - C_even.

Result Summary

Planck gives S = -0.0701, while LCDM simulations have mean -0.00131 and std 0.0852. For the pre-specified upper-tail hypothesis, p = 0.788 and z = -0.80; Planck is at about the 21st percentile, not in either 5% tail.

The measured contrast is negative, indicating stronger even- than odd-parity axis concentration by this statistic. This is opposite to the hypothesised odd-axis excess and fully consistent with the simulation ensemble.

Agent Justification

The statistic is explicitly signed so that positive S indicates stronger odd-parity axis concentration relative to even parity, and the code labels the tail of interest as the upper tail with larger positive S representing the anomaly.

Comparison To Literature

Parity-split low-l directional alignments are related to prior work using power/alignment tensors; Aluri, Ralston & Weltman (2017, https://arxiv.org/abs/1703.07070) reported odd-multipole alignment evidence at roughly ≥2σ and no unusual even alignments, while Planck 2018 isotropy analyses (https://arxiv.org/abs/1906.02552) discuss persistent large-angle anomalies. This specific odd-minus-even concentration contrast does not reproduce an anomaly.

test_description = "Low-l Parity-Split Axis Concentration Contrast: compute pseudo-a_lm through lmax=31 on the masked, zero-filled map; form each multipole's angular-momentum power tensor; extract sign-free principal axes; and return S = C_odd - C_even where C_parity is the largest eigenvalue of the mean n_l n_l^T over odd/even low-l multipoles."


def analyze_map(m, _cache={}):
    import math
    import numpy as np

    LMAX = 31

    def as_float_array(x):
        try:
            if np.ma.isMaskedArray(x):
                arr = np.asarray(x.filled(np.nan), dtype=np.float64)
            else:
                arr = np.asarray(x, dtype=np.float64)
        except Exception:
            try:
                arr = np.asarray(np.real(x), dtype=np.float64)
            except Exception:
                arr = np.asarray([], dtype=np.float64)
        return np.ravel(arr)

    def get_healpy():
        if "healpy" not in _cache:
            try:
                import healpy as hp
                _cache["healpy"] = hp
            except Exception:
                _cache["healpy"] = None
        return _cache.get("healpy")

    def nside_from_npix(hp, npix):
        try:
            ns = int(hp.npix2nside(int(npix)))
            if ns > 0 and int(hp.nside2npix(ns)) == int(npix):
                return ns
        except Exception:
            pass
        try:
            if int(npix) > 0 and int(npix) % 12 == 0:
                ns = int(round(math.sqrt(float(npix) / 12.0)))
                if ns > 0 and 12 * ns * ns == int(npix):
                    return ns
        except Exception:
            pass
        return -1

    def fallback_axis_for_l(ell):
        # Deterministic paired axes: ell=2+2k and ell=3+2k share an axis, so
        # zero-power or pathological maps have no artificial odd-even contrast.
        try:
            ell_i = int(ell)
            k = (ell_i - 2) // 2 if (ell_i % 2 == 0) else (ell_i - 3) // 2
            k = int(k) % 15
            z = 1.0 - 2.0 * (float(k) + 0.5) / 15.0
            phi = 2.39996322972865332 * float(k)
            r = math.sqrt(max(0.0, 1.0 - z * z))
            v = np.array([r * math.cos(phi), r * math.sin(phi), z], dtype=np.float64)
            nr = float(np.sqrt(np.dot(v, v)))
            if np.isfinite(nr) and nr > 0.0:
                return v / nr
        except Exception:
            pass
        return np.array([1.0, 0.0, 0.0], dtype=np.float64)

    def orientation_concentration(axes):
        M = np.zeros((3, 3), dtype=np.float64)
        n_used = 0
        for axis in axes:
            try:
                v = np.asarray(axis, dtype=np.float64).ravel()
                if int(v.size) != 3:
                    continue
                nr = float(np.sqrt(np.dot(v, v)))
                if (not np.isfinite(nr)) or nr <= 0.0:
                    continue
                v = v / nr
                M += np.outer(v, v)
                n_used += 1
            except Exception:
                continue
        if n_used <= 0:
            return float(1.0 / 3.0)
        M /= float(n_used)
        M = 0.5 * (M + M.T)
        try:
            val = float(np.linalg.eigvalsh(M)[-1])
        except Exception:
            val = float(1.0 / 3.0)
        if not np.isfinite(val):
            val = float(1.0 / 3.0)
        return float(min(1.0, max(0.0, val)))

    def angmom_ops(ell):
        ell_i = int(ell)
        key = ("angmom_ops", ell_i)
        got = _cache.get(key)
        if got is not None:
            return got
        dim = 2 * ell_i + 1
        mvals = np.arange(-ell_i, ell_i + 1, dtype=np.float64)
        Jp = np.zeros((dim, dim), dtype=np.complex128)
        for mv in range(-ell_i, ell_i):
            coef = math.sqrt(float(ell_i * (ell_i + 1) - mv * (mv + 1)))
            Jp[mv + 1 + ell_i, mv + ell_i] = coef
        Jm = Jp.conjugate().T
        Jx = 0.5 * (Jp + Jm)
        Jy = (Jp - Jm) / (2.0j)
        Jz = np.diag(mvals).astype(np.complex128)
        got = (Jx, Jy, Jz)
        _cache[key] = got
        return got

    def full_alm_vector(hp, alm, ell):
        ell_i = int(ell)
        a = np.zeros(2 * ell_i + 1, dtype=np.complex128)
        try:
            val0 = alm[int(hp.Alm.getidx(LMAX, ell_i, 0))]
            a[ell_i] = complex(float(np.real(val0)), 0.0)
            for mm in range(1, ell_i + 1):
                val = complex(alm[int(hp.Alm.getidx(LMAX, ell_i, mm))])
                a[ell_i + mm] = val
                a[ell_i - mm] = ((-1.0) if (mm % 2) else 1.0) * np.conjugate(val)
        except Exception:
            return np.zeros(2 * ell_i + 1, dtype=np.complex128)
        return a

    def axis_from_alm_vector(ell, a, total_power):
        ell_i = int(ell)
        try:
            a = np.asarray(a, dtype=np.complex128).ravel()
            if int(a.size) != 2 * ell_i + 1:
                return fallback_axis_for_l(ell_i)
            power = float(np.real(np.vdot(a, a)))
            floor = 1.0e-300
            if np.isfinite(total_power) and float(total_power) > 0.0:
                floor = max(floor, 1.0e-30 * float(total_power))
            if (not np.isfinite(power)) or power <= floor:
                return fallback_axis_for_l(ell_i)
            Jx, Jy, Jz = angmom_ops(ell_i)
            Ja0 = np.dot(Jx, a)
            Ja1 = np.dot(Jy, a)
            Ja2 = np.dot(Jz, a)
            Ja = (Ja0, Ja1, Ja2)
            denom = float(ell_i * (ell_i + 1)) * power
            if (not np.isfinite(denom)) or denom <= 0.0:
                return fallback_axis_for_l(ell_i)
            Q = np.empty((3, 3), dtype=np.float64)
            for ii in range(3):
                for jj in range(3):
                    Q[ii, jj] = float(np.real(np.vdot(Ja[ii], Ja[jj])) / denom)
            Q = 0.5 * (Q + Q.T)
            if not np.all(np.isfinite(Q)):
                return fallback_axis_for_l(ell_i)
            w, v = np.linalg.eigh(Q)
            axis = np.asarray(v[:, int(np.argmax(w))], dtype=np.float64)
            nr = float(np.sqrt(np.dot(axis, axis)))
            if (not np.isfinite(nr)) or nr <= 0.0:
                return fallback_axis_for_l(ell_i)
            return axis / nr
        except Exception:
            return fallback_axis_for_l(ell_i)

    def stat_from_alm(hp, alm):
        avecs = []
        total_power = 0.0
        for ell in range(2, LMAX + 1):
            a = full_alm_vector(hp, alm, ell)
            p = float(np.real(np.vdot(a, a))) if a.size else 0.0
            if np.isfinite(p) and p > 0.0:
                total_power += p
            avecs.append((ell, a))
        odd_axes = []
        even_axes = []
        for ell, a in avecs:
            axis = axis_from_alm_vector(ell, a, total_power)
            if ell % 2:
                odd_axes.append(axis)
            else:
                even_axes.append(axis)
        stat = float(orientation_concentration(odd_axes) - orientation_concentration(even_axes))
        if not np.isfinite(stat):
            stat = 0.0
        return float(min(1.0, max(-1.0, stat)))

    def fallback_value(arr, valid):
        # Finite, data-dependent fallback used only if healpy or a valid HEALPix
        # geometry is unavailable.  It avoids collapsing every map to a constant.
        try:
            vals = np.asarray(arr, dtype=np.float64).ravel()
            mask = np.asarray(valid, dtype=bool).ravel()
            if int(mask.size) != int(vals.size):
                mask = np.isfinite(vals)
            vals = vals[mask]
            vals = vals[np.isfinite(vals)]
            n = int(vals.size)
            if n < 8:
                return 0.0
            vals = vals - float(np.mean(vals))
            rms = float(np.sqrt(np.mean(vals * vals)))
            if (not np.isfinite(rms)) or rms <= 0.0:
                return 0.0
            vals = vals / rms
            x = (2.0 * math.pi / float(n)) * (np.arange(n, dtype=np.float64) + 0.5)
            odd_axes = []
            even_axes = []
            for ell in range(2, LMAX + 1):
                f = float(ell)
                v = np.array([
                    float(np.dot(vals, np.cos(f * x))),
                    float(np.dot(vals, np.sin(f * x))),
                    float(np.dot(vals, np.cos((f + 0.6180339887498949) * x + 0.17 * f)))
                ], dtype=np.float64)
                nr = float(np.sqrt(np.dot(v, v)))
                axis = (v / nr) if (np.isfinite(nr) and nr > 1.0e-100) else fallback_axis_for_l(ell)
                if ell % 2:
                    odd_axes.append(axis)
                else:
                    even_axes.append(axis)
            s = float(orientation_concentration(odd_axes) - orientation_concentration(even_axes))
            return float(s) if np.isfinite(s) else 0.0
        except Exception:
            return 0.0

    arr = as_float_array(m)
    npix = int(arr.size)
    if npix <= 0:
        return 0.0
    valid = np.isfinite(arr)
    if int(np.count_nonzero(valid)) <= 0:
        return 0.0

    hp = get_healpy()
    if hp is None:
        return float(fallback_value(arr, valid))
    nside = nside_from_npix(hp, npix)
    if nside <= 0:
        return float(fallback_value(arr, valid))

    # Pseudo-a_lm of the masked map: all masked/non-finite pixels are zero-filled
    # before the healpy transform; no NaN values are ever passed to healpy.
    mp = np.zeros(npix, dtype=np.float64)
    try:
        mp[valid] = arr[valid]
    except Exception:
        return float(fallback_value(arr, valid))

    try:
        try:
            alm = hp.map2alm(mp, lmax=LMAX, iter=0, pol=False, use_weights=False)
        except TypeError:
            alm = hp.map2alm(mp, lmax=LMAX, iter=0, use_weights=False)
        if isinstance(alm, (tuple, list)):
            alm = alm[0]
        alm = np.asarray(alm, dtype=np.complex128).ravel()
        required = int((LMAX + 1) * (LMAX + 2) // 2)
        if int(alm.size) >= required and np.all(np.isfinite(alm[:required])):
            stat = stat_from_alm(hp, alm)
            if np.isfinite(stat):
                return float(stat)
    except Exception:
        pass
    return float(fallback_value(arr, valid))


def summarize_results(planck_stat, sim_results):
    import numpy as np

    desc = "Low-l Parity-Split Axis Concentration Contrast: compute pseudo-a_lm through lmax=31 on the masked, zero-filled map; form each multipole's angular-momentum power tensor; extract sign-free principal axes; and return S = C_odd - C_even where C_parity is the largest eigenvalue of the mean n_l n_l^T over odd/even low-l multipoles."

    def finite_float(x, default=np.nan):
        try:
            y = float(x)
        except Exception:
            return float(default)
        return float(y) if np.isfinite(y) else float(default)

    ps = finite_float(planck_stat)
    try:
        sims = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            for v in sim_results:
                vals.append(finite_float(v))
        except Exception:
            vals = []
        sims = np.asarray(vals, dtype=np.float64)
    sims = sims[np.isfinite(sims)]
    n = int(sims.size)

    out = {
        "test_description": desc,
        "statistic": "S = C_odd - C_even, C_parity = largest eigenvalue of mean_l n_l n_l^T for angular-momentum principal axes",
        "planck_S_odd_minus_even": ps,
        "n_sims": n,
        "multipoles_odd": "ell = 3,5,...,31",
        "multipoles_even": "ell = 2,4,...,30",
        "tail_of_interest": "upper tail: larger positive S means stronger mutual alignment among odd-parity low-l axes relative to even-parity axes",
        "masked_pixel_handling": "non-finite pixels are treated as masked and zero-filled before healpy.map2alm; NaNs are never passed to healpy",
        "axis_sign_treatment": "axis signs are ignored via outer products n_l n_l^T",
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-l parity-split axis concentration contrast",
            "xlabel": "S = C_odd - C_even",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "LCDM simulation distribution of S. Positive values indicate stronger mutual alignment among odd multipole axes (ell=3,5,...,31) than among even axes (ell=2,4,...,30); Planck is processed with the identical mask and pseudo-a_lm pipeline."
        }
    }

    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_le = int(np.count_nonzero(sims <= ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_ge = int(np.count_nonzero(sims >= ps))
    n_eq = int(n - n_lt - n_gt)

    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_two = float(min(1.0, 2.0 * min(p_upper, p_lower)))

    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    z = None
    if np.isfinite(std) and std > 0.0:
        z = float((ps - mean) / std)
    q = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])

    if ps > 0.0:
        direction = "positive: Planck odd-parity axes are more concentrated than Planck even-parity axes by this statistic"
    elif ps < 0.0:
        direction = "negative: Planck even-parity axes are more concentrated than Planck odd-parity axes by this statistic"
    else:
        direction = "zero: no odd-even concentration contrast in the returned scalar statistic"

    out.update({
        "planck_contrast_direction": direction,
        "sim_mean_S": mean,
        "sim_median_S": median,
        "sim_std_S": std,
        "sim_min_S": float(np.min(sims)),
        "sim_max_S": float(np.max(sims)),
        "sim_percentile_1_S": float(q[0]),
        "sim_percentile_5_S": float(q[1]),
        "sim_percentile_16_S": float(q[2]),
        "sim_percentile_50_S": float(q[3]),
        "sim_percentile_84_S": float(q[4]),
        "sim_percentile_95_S": float(q[5]),
        "sim_percentile_99_S": float(q[6]),
        "fraction_sims_positive_S": float(np.count_nonzero(sims > 0.0) / n),
        "planck_zscore_vs_sims": z,
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "rank_ascending_1_is_lowest_S": int(n_lt + 1),
        "rank_descending_1_is_highest_odd_excess": int(n_gt + 1),
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "fraction_sims_lt_planck": float(n_lt / n),
        "fraction_sims_le_planck": float(n_le / n),
        "fraction_sims_gt_planck": float(n_gt / n),
        "fraction_sims_ge_planck": float(n_ge / n),
        "empirical_p_upper_tail_high_odd_excess_including_ties_plus_one": p_upper,
        "empirical_p_lower_tail_even_excess_or_low_odd_excess_including_ties_plus_one": p_lower,
        "empirical_p_two_sided_plus_one": p_two,
        "difference_planck_minus_sim_median": float(ps - median),
        "is_upper_tail_5pct_high_odd_excess": bool(p_upper <= 0.05),
        "is_upper_tail_1pct_high_odd_excess": bool(p_upper <= 0.01),
        "is_lower_tail_5pct_even_excess_or_low_odd_excess": bool(p_lower <= 0.05),
        "is_lower_tail_1pct_even_excess_or_low_odd_excess": bool(p_lower <= 0.01),
        "is_two_sided_5pct": bool(p_two <= 0.05),
        "is_two_sided_1pct": bool(p_two <= 0.01)
    })

    if std == 0.0:
        out["warning"] = "All finite simulation statistics are identical; check the input maps, mask, and healpy/pseudo-a_lm support."
    elif p_upper <= 0.01:
        out["interpretation"] = "Planck lies in the highest 1% of simulations for S: an exceptionally large odd-parity axis concentration excess over the even-parity axes."
    elif p_upper <= 0.05:
        out["interpretation"] = "Planck lies in the highest 5% of simulations for S: an unusually large odd-parity axis concentration excess over the even-parity axes."
    elif p_lower <= 0.01:
        out["interpretation"] = "Planck lies in the lowest 1% of simulations for S: the contrast is exceptionally negative or the odd-axis concentration excess is exceptionally weak."
    elif p_lower <= 0.05:
        out["interpretation"] = "Planck lies in the lowest 5% of simulations for S: the contrast is unusually negative or the odd-axis concentration excess is unusually weak."
    else:
        out["interpretation"] = "Planck is not in either 5% tail of the empirical LCDM distribution for this odd-minus-even low-l axis concentration contrast."

    return out
