Hypothesis

The Planck SMICA map has an anomalously extreme masked low-l antipodal correlation relative to LCDM simulations, indicating unusually strong parity preference of either sign.

Method

After masked-sky monopole/dipole removal, the SMICA and 1000 LCDM maps were filtered to 2 <= ell <= 30 and the signed Pearson correlation r_anti between antipodal unmasked pixel pairs was computed. Negative values indicate odd-parity antipodal anti-correlation; positive values indicate even-parity correlation.

Result Summary

Planck has r_anti = -0.1261 versus simulations with mean 0.1073 and std 0.1350, giving z ≈ -1.73, two-tailed p = 0.078, and sigma ≈ 1.76. It lies in the lower 3.9% directional tail, but not below the two-tailed 5% threshold.

The Planck map shows a relatively strong odd-parity/antipodal anti-correlation, but the pre-specified two-tailed extremeness test is only marginal and does not reach 2 sigma significance.

Agent Justification

The statistic is signed: negative values indicate odd-parity antipodal anti-correlation and positive values indicate even-parity antipodal correlation. Since either tail could represent an anomaly and no prior direction is specified, the test should check for extremeness in either direction.

Comparison To Literature

Low-l CMB parity asymmetry is a documented anomaly in WMAP/Planck analyses, e.g. Kim & Naselsky 2010 (arXiv:1002.0148) and Planck 2018 VII, “Isotropy and Statistics of the CMB” (arXiv:1906.02552). This test probes a related real-space masked antipodal-pair statistic rather than the standard even/odd power-ratio statistic.

import numpy as np

test_description = "Masked Low-l Antipodal Correlation: remove the best-fit monopole and dipole on the unmasked sky, set masked pixels to zero for the spherical-harmonic transform, keep only multipoles 2 <= ell <= 30, and compute the signed antipodal-pair correlation r_anti over pairs for which both pixels are unmasked. Negative r_anti indicates odd-parity preference; positive r_anti indicates even-parity preference."

# Mutable caches are captured by default arguments of the public functions.  This
# is robust to runners that execute registered code with separate globals/locals.
_ANTI_CORR_STATE = {
    "geom_cache": {},
    "md_cache": {},
    "anti_cache": {},
    "pair_cache": {},
}


def analyze_map(m, _state=_ANTI_CORR_STATE):
    import math
    import numpy as np

    LMIN = 2
    LMAX_REQUEST = 30
    CHUNK = 262144
    EPS = 1.0e-30

    def _infer_nside(npix):
        try:
            npix_i = int(npix)
            if npix_i <= 0 or npix_i % 12 != 0:
                return -1
            nside = int(round(math.sqrt(npix_i / 12.0)))
            if nside <= 0 or 12 * nside * nside != npix_i:
                return -1
            return nside
        except Exception:
            return -1

    def _alm_index(lmax, ell, mm):
        # healpy Alm storage for non-negative m.
        return int(mm * (2 * lmax + 1 - mm) // 2 + ell)

    def _get_geometry(nside, npix):
        key = (int(nside), int(npix))
        cache = _state.get("geom_cache")
        if not isinstance(cache, dict):
            cache = {}
            _state["geom_cache"] = cache
        item = cache.get(key)
        if item is not None:
            return item
        import healpy as hp
        pix = np.arange(int(npix), dtype=np.int64)
        x, y, z = hp.pix2vec(int(nside), pix, nest=False)
        item = (np.asarray(x, dtype=np.float64).ravel(),
                np.asarray(y, dtype=np.float64).ravel(),
                np.asarray(z, dtype=np.float64).ravel())
        if len(cache) > 2:
            cache.clear()
        cache[key] = item
        return item

    def _get_md_solver(nside, npix, valid, mask_key, mask_bytes, x, y, z):
        cache = _state.get("md_cache")
        if not isinstance(cache, dict):
            cache = {}
            _state["md_cache"] = cache
        item = cache.get(mask_key)
        if item is not None and item.get("mask_bytes") == mask_bytes:
            return item

        idx = np.flatnonzero(valid).astype(np.int64, copy=False)
        s0 = sx = sy = sz = 0.0
        sxx = sxy = sxz = syy = syz = szz = 0.0
        for start in range(0, int(idx.size), CHUNK):
            ii = idx[start:start + CHUNK]
            xx = x[ii]
            yy = y[ii]
            zz = z[ii]
            s0 += float(ii.size)
            sx += float(np.sum(xx))
            sy += float(np.sum(yy))
            sz += float(np.sum(zz))
            sxx += float(np.dot(xx, xx))
            sxy += float(np.dot(xx, yy))
            sxz += float(np.dot(xx, zz))
            syy += float(np.dot(yy, yy))
            syz += float(np.dot(yy, zz))
            szz += float(np.dot(zz, zz))
        G = np.array([[s0,  sx,  sy,  sz],
                      [sx, sxx, sxy, sxz],
                      [sy, sxy, syy, syz],
                      [sz, sxz, syz, szz]], dtype=np.float64)
        try:
            solver = np.linalg.pinv(G, rcond=1.0e-12)
        except Exception:
            solver = np.zeros((4, 4), dtype=np.float64)
        item = {"idx": idx, "solver": np.asarray(solver, dtype=np.float64), "mask_bytes": mask_bytes}
        if len(cache) > 4:
            cache.clear()
        cache[mask_key] = item
        return item

    def _remove_monopole_dipole(arr, valid, nside, npix, mask_key, mask_bytes):
        x, y, z = _get_geometry(nside, npix)
        entry = _get_md_solver(nside, npix, valid, mask_key, mask_bytes, x, y, z)
        idx = entry["idx"]
        if int(idx.size) == 0:
            return np.zeros(int(npix), dtype=np.float64), (x, y, z)

        rhs = np.zeros(4, dtype=np.float64)
        for start in range(0, int(idx.size), CHUNK):
            ii = idx[start:start + CHUNK]
            vals = np.asarray(arr[ii], dtype=np.float64)
            rhs[0] += float(np.sum(vals))
            rhs[1] += float(np.dot(x[ii], vals))
            rhs[2] += float(np.dot(y[ii], vals))
            rhs[3] += float(np.dot(z[ii], vals))
        coeff = np.asarray(entry["solver"].dot(rhs), dtype=np.float64).ravel()
        if coeff.size != 4 or not np.all(np.isfinite(coeff)):
            coeff = np.zeros(4, dtype=np.float64)

        clean = np.zeros(int(npix), dtype=np.float64)
        for start in range(0, int(idx.size), CHUNK):
            ii = idx[start:start + CHUNK]
            clean[ii] = (np.asarray(arr[ii], dtype=np.float64)
                         - coeff[0] - coeff[1] * x[ii] - coeff[2] * y[ii] - coeff[3] * z[ii])
        clean[~np.isfinite(clean)] = 0.0
        return clean, (x, y, z)

    def _get_antipode(nside, npix, x, y, z):
        key = (int(nside), int(npix))
        cache = _state.get("anti_cache")
        if not isinstance(cache, dict):
            cache = {}
            _state["anti_cache"] = cache
        item = cache.get(key)
        if item is not None:
            return item
        import healpy as hp
        anti = np.asarray(hp.vec2pix(int(nside), -x, -y, -z, nest=False), dtype=np.int64).ravel()
        if anti.size != int(npix):
            anti = np.arange(int(npix), dtype=np.int64)
        if len(cache) > 2:
            cache.clear()
        cache[key] = anti
        return anti

    def _get_pairs(nside, npix, valid, mask_key, mask_bytes, anti):
        cache = _state.get("pair_cache")
        if not isinstance(cache, dict):
            cache = {}
            _state["pair_cache"] = cache
        item = cache.get(mask_key)
        if item is not None and item.get("mask_bytes") == mask_bytes:
            return item["i"], item["j"]

        if anti.size != int(npix) or np.any(anti < 0) or np.any(anti >= int(npix)):
            i = np.zeros(0, dtype=np.int64)
            j = np.zeros(0, dtype=np.int64)
        else:
            pix = np.arange(int(npix), dtype=np.int64)
            # Deterministic one-representative choice: p < antipode(p).
            pair_mask = valid & valid[anti] & (pix < anti)
            i = np.flatnonzero(pair_mask).astype(np.int64, copy=False)
            j = anti[i].astype(np.int64, copy=False)
        item = {"i": i, "j": j, "mask_bytes": mask_bytes}
        if len(cache) > 4:
            cache.clear()
        cache[mask_key] = item
        return i, j

    def _map2alm_requested(clean, hp, nside):
        import warnings
        lmax = int(LMAX_REQUEST)
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            try:
                try:
                    alm = hp.map2alm(clean, lmax=lmax, iter=0, pol=False, use_weights=False)
                except TypeError:
                    alm = hp.map2alm(clean, lmax=lmax, iter=0, pol=False)
            except Exception:
                # Very low nside maps cannot always support ell=30.  The normal
                # execution path uses ell<=30 exactly; this fallback merely keeps
                # pathological inputs finite rather than returning NaN/inf.
                lmax_alt = min(int(LMAX_REQUEST), max(0, 3 * int(nside) - 1))
                if lmax_alt < LMIN or lmax_alt == lmax:
                    raise
                lmax = int(lmax_alt)
                try:
                    alm = hp.map2alm(clean, lmax=lmax, iter=0, pol=False, use_weights=False)
                except TypeError:
                    alm = hp.map2alm(clean, lmax=lmax, iter=0, pol=False)
        return np.asarray(alm, dtype=np.complex128).ravel(), int(lmax)

    def _alm2map_safe(alm, hp, nside, lmax):
        import warnings
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            try:
                out = hp.alm2map(alm, int(nside), lmax=int(lmax), pol=False, verbose=False)
            except TypeError:
                out = hp.alm2map(alm, int(nside), lmax=int(lmax), pol=False)
        return np.asarray(out, dtype=np.float64).ravel()

    try:
        if np.ma.isMaskedArray(m):
            arr = np.asarray(m.filled(np.nan), dtype=np.float64).ravel()
        else:
            arr = np.asarray(m, dtype=np.float64).ravel()
    except Exception:
        try:
            arr = np.asarray(np.real(m), dtype=np.float64).ravel()
        except Exception:
            return 0.0

    npix = int(arr.size)
    nside = _infer_nside(npix)
    if nside <= 0:
        return 0.0

    # The official mask is represented by NaN.  The amplitude guard also avoids
    # accidentally accepting HEALPix UNSEEN sentinel values as valid data.
    valid = np.isfinite(arr) & (np.abs(arr) < 1.0e25)
    nvalid = int(np.count_nonzero(valid))
    if nvalid < 16:
        return 0.0

    packed = np.packbits(np.asarray(valid, dtype=np.bool_))
    mask_bytes = packed.tobytes()
    mask_key = (int(nside), int(npix), nvalid, hash(mask_bytes))

    try:
        import healpy as hp

        clean, geom = _remove_monopole_dipole(arr, valid, nside, npix, mask_key, mask_bytes)
        # Never pass NaNs to healpy.  clean is finite and has zeros outside the mask.
        alm, lmax = _map2alm_requested(clean, hp, nside)
        if lmax < LMIN:
            return 0.0

        # Keep only 2 <= ell <= lmax (normally lmax=30).
        for ell in range(0, min(LMIN, lmax + 1)):
            for mm in range(ell + 1):
                k = _alm_index(lmax, ell, mm)
                if 0 <= k < alm.size:
                    alm[k] = 0.0 + 0.0j

        filt = _alm2map_safe(alm, hp, nside, lmax)
        if filt.size != npix:
            return 0.0
        filt[~np.isfinite(filt)] = 0.0

        x, y, z = geom
        anti = _get_antipode(nside, npix, x, y, z)
        i, j = _get_pairs(nside, npix, valid, mask_key, mask_bytes, anti)
        if int(i.size) <= 0:
            return 0.0

        num = 0.0
        s1 = 0.0
        s2 = 0.0
        n_used = 0
        for start in range(0, int(i.size), CHUNK):
            ii = i[start:start + CHUNK]
            jj = j[start:start + CHUNK]
            a = np.asarray(filt[ii], dtype=np.float64)
            b = np.asarray(filt[jj], dtype=np.float64)
            good = np.isfinite(a) & np.isfinite(b)
            if np.any(good):
                aa = a[good]
                bb = b[good]
                num += float(np.dot(aa, bb))
                s1 += float(np.dot(aa, aa))
                s2 += float(np.dot(bb, bb))
                n_used += int(aa.size)
        if n_used <= 0 or not (np.isfinite(num) and np.isfinite(s1) and np.isfinite(s2)):
            return 0.0
        den = math.sqrt(max(s1, 0.0) * max(s2, 0.0))
        if not np.isfinite(den) or den <= EPS:
            return 0.0
        r = float(num / den)
        if not np.isfinite(r):
            return 0.0
        # Numerical roundoff can produce values infinitesimally outside [-1, 1].
        if r > 1.0:
            r = 1.0
        elif r < -1.0:
            r = -1.0
        return float(r)
    except Exception:
        return 0.0


def summarize_results(planck_stat, sim_results, _desc=test_description):
    import numpy as np

    def _as_float(x):
        try:
            v = float(x)
            return v if np.isfinite(v) else np.nan
        except Exception:
            return np.nan

    ps = _as_float(planck_stat)
    try:
        sims_all = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            iterator = iter(sim_results)
        except Exception:
            iterator = []
        for item in iterator:
            v = _as_float(item)
            if np.isfinite(v):
                vals.append(v)
        sims_all = np.asarray(vals, dtype=np.float64)
    sims = sims_all[np.isfinite(sims_all)]
    n = int(sims.size)

    out = {
        "test_description": _desc,
        "statistic": "Signed r_anti = sum T(p) T(-p) / sqrt[sum T(p)^2 sum T(-p)^2] over antipodal pixel pairs with both pixels unmasked, after masked-sky monopole/dipole removal and ell=2..30 filtering.",
        "tail": "signed statistic; negative/lower-tail values indicate odd-parity preference, positive/upper-tail values indicate even-parity preference; two-sided and absolute-tail ranks summarize unusually strong parity preference of either sign",
        "monopole_dipole_fit": "least-squares fit of [1, x, y, z] on finite unmasked pixels only",
        "mask_and_transform_handling": "non-finite pixels are ignored in the fit and pair selection, and are filled with 0.0 before healpy spherical-harmonic transforms",
        "pair_orientation": "one representative per antipodal pair is chosen deterministically by HEALPix pixel index p < antipode(p)",
        "harmonic_range": {"lmin": 2, "lmax": 30},
        "planck_r_anti": float(ps) if np.isfinite(ps) else None,
        "n_sims": n,
        "plot_spec": {
            "kind": "histogram",
            "title": "Masked low-l antipodal correlation",
            "xlabel": "r_anti",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "Simulation distribution of the signed antipodal-pair correlation after masked-sky monopole/dipole removal and ell=2..30 filtering. Negative values favor odd parity; positive values favor even parity."
        }
    }

    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic supplied to summarize_results is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_eq = int(n - n_lt - n_gt)
    n_le = int(n_lt + n_eq)
    n_ge = int(n_gt + n_eq)

    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_two_rank = float(min(1.0, 2.0 * min(p_lower, p_upper)))

    abs_ps = abs(float(ps))
    abs_sims = np.abs(sims)
    n_abs_gt = int(np.count_nonzero(abs_sims > abs_ps))
    n_abs_eq = int(np.count_nonzero(abs_sims == abs_ps))
    n_abs_ge = int(n_abs_gt + n_abs_eq)
    p_abs = float((n_abs_ge + 1.0) / (n + 1.0))

    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    q = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99]) if n > 1 else np.array([sims[0]] * 7, dtype=np.float64)

    if ps < 0.0:
        sign_interpretation = "negative r_anti: odd-parity / antipodal anti-correlation preference"
        directional_tail = "lower"
        directional_p = p_lower
        directional_rank = int(n_lt + 1)
        directional_rank_label = "ascending rank, 1 is most negative"
    elif ps > 0.0:
        sign_interpretation = "positive r_anti: even-parity / antipodal correlation preference"
        directional_tail = "upper"
        directional_p = p_upper
        directional_rank = int(n_gt + 1)
        directional_rank_label = "descending rank, 1 is most positive"
    else:
        sign_interpretation = "r_anti is exactly zero: no signed antipodal parity preference in this statistic"
        directional_tail = "none"
        directional_p = 1.0
        directional_rank = None
        directional_rank_label = None

    out.update({
        "sim_mean": mean,
        "sim_median": median,
        "sim_std": std,
        "sim_min": float(np.min(sims)),
        "sim_max": float(np.max(sims)),
        "sim_percentile_1": float(q[0]),
        "sim_percentile_5": float(q[1]),
        "sim_percentile_16": float(q[2]),
        "sim_percentile_50": float(q[3]),
        "sim_percentile_84": float(q[4]),
        "sim_percentile_95": float(q[5]),
        "sim_percentile_99": float(q[6]),
        "sim_fraction_negative": float(np.count_nonzero(sims < 0.0) / n),
        "sim_fraction_positive": float(np.count_nonzero(sims > 0.0) / n),
        "zscore_vs_sims": float((ps - mean) / std) if std > 0.0 and np.isfinite(std) else None,
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "fraction_sims_le_planck_lower_tail_including_ties": float(n_le / n),
        "fraction_sims_ge_planck_upper_tail_including_ties": float(n_ge / n),
        "empirical_p_lower_tail_plus_one": p_lower,
        "empirical_p_upper_tail_plus_one": p_upper,
        "empirical_p_two_sided_rank_plus_one": p_two_rank,
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "lower_tail_rank_ascending_1_is_most_negative": int(n_lt + 1),
        "upper_tail_rank_descending_1_is_most_positive": int(n_gt + 1),
        "absolute_planck_r_anti": float(abs_ps),
        "n_sim_abs_ge_abs_planck": n_abs_ge,
        "fraction_sims_abs_ge_abs_planck": float(n_abs_ge / n),
        "empirical_p_abs_tail_plus_one": p_abs,
        "sign_interpretation": sign_interpretation,
        "directional_tail_for_planck_sign": directional_tail,
        "directional_empirical_p_plus_one": float(directional_p),
        "directional_rank": directional_rank,
        "directional_rank_label": directional_rank_label,
        "is_5pct_directional_tail": bool(directional_p <= 0.05),
        "is_1pct_directional_tail": bool(directional_p <= 0.01),
        "is_5pct_two_sided_rank": bool(p_two_rank <= 0.05),
        "is_1pct_two_sided_rank": bool(p_two_rank <= 0.01),
        "is_5pct_abs_tail": bool(p_abs <= 0.05),
        "is_1pct_abs_tail": bool(p_abs <= 0.01),
    })

    if directional_p <= 0.01:
        if ps < 0.0:
            out["interpretation"] = "Planck is in the lower 1% simulation tail: unusually strong odd-parity antipodal anti-correlation for this masked low-l statistic."
        elif ps > 0.0:
            out["interpretation"] = "Planck is in the upper 1% simulation tail: unusually strong even-parity antipodal correlation for this masked low-l statistic."
        else:
            out["interpretation"] = "Planck r_anti is zero; no signed parity preference."
    elif directional_p <= 0.05:
        if ps < 0.0:
            out["interpretation"] = "Planck is in the lower 5% simulation tail: stronger odd-parity antipodal anti-correlation than most simulations."
        elif ps > 0.0:
            out["interpretation"] = "Planck is in the upper 5% simulation tail: stronger even-parity antipodal correlation than most simulations."
        else:
            out["interpretation"] = "Planck r_anti is zero; no signed parity preference."
    elif p_abs <= 0.05:
        out["interpretation"] = "Planck has an unusually large absolute antipodal-correlation magnitude compared with simulations, although the sign-specific directional tail is not below 5%."
    else:
        out["interpretation"] = "Planck is not in the 5% directional simulation tail for the signed masked low-l antipodal correlation."

    if std == 0.0:
        out["warning"] = "All finite simulation r_anti values are identical; check map variation, mask handling, and healpy availability."
    return out
