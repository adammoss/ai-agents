Hypothesis

The Planck CMB map has an anomalously strong alignment between the combined low-\(\ell\) power-tensor preferred axis for \(\ell=2..5\) and the ecliptic pole, i.e. \(A=|n_{\max}\cdot n_{\rm ecl}|\) is unusually large compared with simulations.

Method

Monopole/dipole-subtracted masked Planck temperature maps were reduced to pseudo-\(a_{\ell m}\) for \(\ell\le5\). Normalized angular-momentum-dispersion tensors for \(\ell=2..5\) were summed, and the largest-eigenvalue axis was compared with the ecliptic north pole using \(A=|n_{\max}\cdot n_{\rm ecl}|\).

Result Summary

Planck gives \(A=0.0107\), corresponding to an angle of \(89.39^\circ\) from the ecliptic pole. Simulations have mean \(A=0.507\), standard deviation 0.287, and median 0.515. For the pre-specified upper-tail test, \(p=0.993\) with 1000 simulations; only 7 simulations had smaller \(A\), giving an opposite-tail/two-sided extremeness of about 0.016.

The Planck axis is almost orthogonal to the ecliptic pole, not unusually aligned with it. Thus the stated upper-tail hypothesis is not supported, although the opposite, near-ecliptic-plane orientation would be unusual under a two-sided or lower-tail post-hoc framing.

Agent Justification

The statistic is constructed so that larger values of \(A\) mean closer alignment with the ecliptic pole, and the test description/code explicitly identify large \(A\) as the anomalous direction of interest.

Comparison To Literature

Related low-\(\ell\) axis and ecliptic-plane anomalies are well documented, including multipole-vector and angular-momentum-axis studies reporting quadrupole-octopole/ecliptic associations at roughly \(\gtrsim2\sigma\) (Weeks 2004, arXiv:astro-ph/0412231; Abramo et al. 2006, arXiv:astro-ph/0604346; Planck Collaboration 2015 isotropy analysis, arXiv:1506.07135). This exact summed \(\ell=2..5\) power-tensor pole-alignment statistic is not the standard published test.

test_description = "Low-l Power-Tensor Ecliptic Alignment: for a masked temperature map, fit and subtract the monopole and dipole on unmasked pixels, compute masked pseudo-a_lm through ell=5, form and sum normalized angular-momentum dispersion tensors for ell=2..5, and return A=|n_max dot n_ecl| for the ecliptic north pole (Galactic map coordinates assumed)."


def analyze_map(m, _cache={}):
    import math
    import zlib
    import numpy as np

    LMAX = 5
    MIN_VALID = 8

    def _finish_alignment(x):
        try:
            y = float(x)
        except Exception:
            return 0.0
        if not np.isfinite(y):
            return 0.0
        if y < 0.0:
            y = 0.0
        if y > 1.0:
            y = 1.0
        return float(y)

    def _as_flat_float(x):
        try:
            if np.ma.isMaskedArray(x):
                a = np.asarray(x.filled(np.nan), dtype=np.float64)
            else:
                a = np.asarray(x, dtype=np.float64)
        except Exception:
            try:
                a = np.asarray(np.real(x), dtype=np.float64)
            except Exception:
                a = np.asarray([], dtype=np.float64)
        return np.ravel(a)

    def _mask_signature(valid):
        v = np.asarray(valid, dtype=bool).ravel()
        n = int(v.size)
        c = int(np.count_nonzero(v))
        try:
            packed = np.packbits(v)
            crc = int(zlib.crc32(packed.tobytes()) & 0xffffffff)
        except Exception:
            crc = int(c & 0xffffffff)
        return str(n) + ":" + str(c) + ":" + str(crc)

    def _get_hp():
        if "hp_loaded" not in _cache:
            _cache["hp_loaded"] = True
            try:
                import healpy as hp
                _cache["hp"] = hp
            except Exception:
                _cache["hp"] = None
        return _cache.get("hp", None)

    def _nside_from_npix(hp, npix):
        try:
            ns = int(hp.npix2nside(int(npix)))
            if ns > 0 and int(hp.nside2npix(ns)) == int(npix):
                return ns
        except Exception:
            pass
        try:
            if int(npix) > 0 and int(npix) % 12 == 0:
                ns = int(round(math.sqrt(float(npix) / 12.0)))
                if ns > 0 and 12 * ns * ns == int(npix):
                    return ns
        except Exception:
            pass
        return -1

    def _ecliptic_north():
        key = "ecliptic_north_galactic_unit_v1"
        v = _cache.get(key)
        if v is not None:
            return v
        # North ecliptic pole in Galactic coordinates, J2000: l=96.384 deg, b=29.812 deg.
        lon = math.radians(96.384)
        lat = math.radians(29.812)
        clat = math.cos(lat)
        vec = np.array([clat * math.cos(lon), clat * math.sin(lon), math.sin(lat)], dtype=np.float64)
        norm = float(np.sqrt(np.dot(vec, vec)))
        if np.isfinite(norm) and norm > 0.0:
            vec = vec / norm
        else:
            vec = np.array([0.0, 0.0, 1.0], dtype=np.float64)
        _cache[key] = vec
        return vec

    def _gram_xyz(x, y, z):
        n = float(np.asarray(x).size)
        gram = np.empty((4, 4), dtype=np.float64)
        sx = float(np.sum(x)); sy = float(np.sum(y)); sz = float(np.sum(z))
        gram[0, 0] = n
        gram[0, 1] = sx; gram[1, 0] = sx
        gram[0, 2] = sy; gram[2, 0] = sy
        gram[0, 3] = sz; gram[3, 0] = sz
        gram[1, 1] = float(np.dot(x, x))
        gram[1, 2] = float(np.dot(x, y)); gram[2, 1] = gram[1, 2]
        gram[1, 3] = float(np.dot(x, z)); gram[3, 1] = gram[1, 3]
        gram[2, 2] = float(np.dot(y, y))
        gram[2, 3] = float(np.dot(y, z)); gram[3, 2] = gram[2, 3]
        gram[3, 3] = float(np.dot(z, z))
        return gram

    def _rhs_xyz(x, y, z, vals):
        return np.array([
            float(np.sum(vals)),
            float(np.dot(x, vals)),
            float(np.dot(y, vals)),
            float(np.dot(z, vals))
        ], dtype=np.float64)

    def _geometry(hp, nside, valid):
        sig = _mask_signature(valid)
        key = "geom_low_l_power_tensor_" + str(int(nside)) + "_" + sig
        cached = _cache.get(key)
        if cached is not None:
            return cached
        pix = np.flatnonzero(np.asarray(valid, dtype=bool).ravel()).astype(np.int64, copy=False)
        if int(pix.size) < MIN_VALID:
            return None
        try:
            x, y, z = hp.pix2vec(int(nside), pix, nest=False)
        except TypeError:
            try:
                x, y, z = hp.pix2vec(int(nside), pix)
            except Exception:
                return None
        except Exception:
            return None
        x = np.asarray(x, dtype=np.float64)
        y = np.asarray(y, dtype=np.float64)
        z = np.asarray(z, dtype=np.float64)
        if int(x.size) != int(pix.size) or int(y.size) != int(pix.size) or int(z.size) != int(pix.size):
            return None
        if not (np.all(np.isfinite(x)) and np.all(np.isfinite(y)) and np.all(np.isfinite(z))):
            return None
        gram = _gram_xyz(x, y, z)
        if not np.all(np.isfinite(gram)):
            return None
        try:
            pinv = np.linalg.pinv(gram, rcond=1.0e-12)
        except TypeError:
            try:
                pinv = np.linalg.pinv(gram)
            except Exception:
                return None
        except Exception:
            return None
        pinv = np.asarray(pinv, dtype=np.float64)
        if pinv.shape != (4, 4) or not np.all(np.isfinite(pinv)):
            return None
        geom = {"pix": pix, "x": x, "y": y, "z": z, "pinv": pinv}
        try:
            ncache = int(_cache.get("geom_low_l_power_tensor_count", 0))
            if ncache < 8:
                _cache[key] = geom
                _cache["geom_low_l_power_tensor_count"] = ncache + 1
        except Exception:
            pass
        return geom

    def _spin_ops(ell):
        ell = int(ell)
        key = "spin_ops_power_tensor_l" + str(ell)
        ops = _cache.get(key)
        if ops is not None:
            return ops
        dim = 2 * ell + 1
        ms = np.arange(-ell, ell + 1, dtype=np.float64)
        lp = np.zeros((dim, dim), dtype=np.complex128)
        ll1 = float(ell * (ell + 1))
        for k in range(dim - 1):
            mv = float(-ell + k)
            coef2 = ll1 - mv * (mv + 1.0)
            if coef2 < 0.0 and coef2 > -1.0e-12:
                coef2 = 0.0
            lp[k + 1, k] = math.sqrt(max(0.0, coef2))
        lm = lp.conjugate().T
        lx = 0.5 * (lp + lm)
        ly = -0.5j * (lp - lm)
        lz = np.diag(ms.astype(np.complex128))
        mats = [lx, ly, lz]
        ops = [[None, None, None], [None, None, None], [None, None, None]]
        for i in range(3):
            for j in range(3):
                ops[i][j] = 0.5 * (mats[i].dot(mats[j]) + mats[j].dot(mats[i]))
        _cache[key] = ops
        return ops

    def _alm_vector(hp, alm, lmax, ell):
        ell = int(ell)
        av = np.zeros(2 * ell + 1, dtype=np.complex128)
        try:
            for out_i, mv in enumerate(range(-ell, ell + 1)):
                if mv < 0:
                    mp = -int(mv)
                    a = alm[hp.Alm.getidx(int(lmax), ell, mp)]
                    if mp & 1:
                        av[out_i] = -np.conjugate(a)
                    else:
                        av[out_i] = np.conjugate(a)
                else:
                    av[out_i] = alm[hp.Alm.getidx(int(lmax), ell, int(mv))]
        except Exception:
            return None
        return av

    def _tensor_from_alm(hp, alm, lmax):
        total = np.zeros((3, 3), dtype=np.float64)
        used = 0
        for ell in range(2, int(lmax) + 1):
            a = _alm_vector(hp, alm, lmax, ell)
            if a is None:
                return None
            finite = np.isfinite(a.real) & np.isfinite(a.imag)
            if not np.all(finite):
                a = np.where(finite, a, 0.0 + 0.0j)
            power = float(np.vdot(a, a).real)
            if not np.isfinite(power) or power <= 0.0:
                continue
            ops = _spin_ops(ell)
            denom = float(ell * (ell + 1)) * power
            if not np.isfinite(denom) or denom <= 0.0:
                continue
            this_t = np.zeros((3, 3), dtype=np.float64)
            for i in range(3):
                for j in range(3):
                    val = float(np.vdot(a, ops[i][j].dot(a)).real / denom)
                    if not np.isfinite(val):
                        return None
                    this_t[i, j] = val
            total += 0.5 * (this_t + this_t.T)
            used += 1
        if used <= 0:
            return None
        total = 0.5 * (total + total.T)
        if not np.all(np.isfinite(total)):
            return None
        return total

    def _alignment_from_tensor(tensor):
        try:
            t = np.asarray(tensor, dtype=np.float64)
            if t.shape != (3, 3):
                return 0.0
            t = 0.5 * (t + t.T)
            if not np.all(np.isfinite(t)):
                return 0.0
            vals, vecs = np.linalg.eigh(t)
            imax = int(np.argmax(vals))
            axis = np.asarray(vecs[:, imax], dtype=np.float64).ravel()
            if axis.size != 3 or not np.all(np.isfinite(axis)):
                return 0.0
            norm = float(np.sqrt(np.dot(axis, axis)))
            if not np.isfinite(norm) or norm <= 0.0:
                return 0.0
            axis = axis / norm
            return _finish_alignment(abs(float(np.dot(axis, _ecliptic_north()))))
        except Exception:
            return 0.0

    def _fallback(arr, valid, hp=None, nside=-1):
        # Fallback used only if healpy harmonic transforms are unavailable/fail.  It preserves
        # a non-degenerate axis-alignment statistic based on the residual temperature-power tensor.
        try:
            pix = np.flatnonzero(np.asarray(valid, dtype=bool).ravel()).astype(np.int64, copy=False)
            n = int(pix.size)
            if n < MIN_VALID:
                return 0.0
            vals = np.asarray(arr[pix], dtype=np.float64)
            vals = np.where(np.isfinite(vals), vals, 0.0)
            got_vecs = False
            x = y = z = None
            if hp is not None and int(nside) >= 1:
                try:
                    x0, y0, z0 = hp.pix2vec(int(nside), pix, nest=False)
                except TypeError:
                    try:
                        x0, y0, z0 = hp.pix2vec(int(nside), pix)
                    except Exception:
                        x0 = y0 = z0 = None
                except Exception:
                    x0 = y0 = z0 = None
                if x0 is not None:
                    x = np.asarray(x0, dtype=np.float64)
                    y = np.asarray(y0, dtype=np.float64)
                    z = np.asarray(z0, dtype=np.float64)
                    got_vecs = (int(x.size) == n and int(y.size) == n and int(z.size) == n and
                                np.all(np.isfinite(x)) and np.all(np.isfinite(y)) and np.all(np.isfinite(z)))
            if not got_vecs:
                ii = np.arange(n, dtype=np.float64)
                z = 1.0 - 2.0 * ((ii + 0.5) / float(n))
                phi = (2.39996322972865332 * ii) % (2.0 * np.pi)
                r = np.sqrt(np.maximum(0.0, 1.0 - z * z))
                x = r * np.cos(phi)
                y = r * np.sin(phi)
            gram = _gram_xyz(x, y, z)
            rhs = _rhs_xyz(x, y, z, vals)
            try:
                beta = np.dot(np.linalg.pinv(gram, rcond=1.0e-12), rhs)
            except TypeError:
                beta = np.dot(np.linalg.pinv(gram), rhs)
            beta = np.asarray(beta, dtype=np.float64).ravel()
            if beta.size < 4 or not np.all(np.isfinite(beta[:4])):
                resid = vals - float(np.mean(vals))
            else:
                resid = vals - (beta[0] + beta[1] * x + beta[2] * y + beta[3] * z)
            resid = np.where(np.isfinite(resid), resid, 0.0)
            w = resid * resid
            sw = float(np.sum(w))
            if not np.isfinite(sw) or sw <= 0.0:
                return 0.0
            c = np.empty((3, 3), dtype=np.float64)
            c[0, 0] = float(np.dot(w, x * x)) / sw
            c[0, 1] = float(np.dot(w, x * y)) / sw; c[1, 0] = c[0, 1]
            c[0, 2] = float(np.dot(w, x * z)) / sw; c[2, 0] = c[0, 2]
            c[1, 1] = float(np.dot(w, y * y)) / sw
            c[1, 2] = float(np.dot(w, y * z)) / sw; c[2, 1] = c[1, 2]
            c[2, 2] = float(np.dot(w, z * z)) / sw
            return _alignment_from_tensor(c)
        except Exception:
            return 0.0

    arr = _as_flat_float(m)
    npix = int(arr.size)
    if npix <= 0:
        return 0.0
    valid = np.isfinite(arr) & (np.abs(arr) < 1.0e25)
    if int(np.count_nonzero(valid)) < MIN_VALID:
        return 0.0

    hp = _get_hp()
    if hp is None:
        return float(_fallback(arr, valid, None, -1))
    nside = _nside_from_npix(hp, npix)
    if nside < 2:
        return float(_fallback(arr, valid, hp, nside))

    try:
        geom = _geometry(hp, nside, valid)
        if geom is None:
            return float(_fallback(arr, valid, hp, nside))
        pix = geom["pix"]
        vals = np.asarray(arr[pix], dtype=np.float64)
        vals = np.where(np.isfinite(vals), vals, 0.0)
        rhs = _rhs_xyz(geom["x"], geom["y"], geom["z"], vals)
        if not np.all(np.isfinite(rhs)):
            return float(_fallback(arr, valid, hp, nside))
        beta = np.dot(geom["pinv"], rhs)
        beta = np.asarray(beta, dtype=np.float64).ravel()
        if beta.size < 4 or not np.all(np.isfinite(beta[:4])):
            return float(_fallback(arr, valid, hp, nside))
        model = beta[0] + beta[1] * geom["x"] + beta[2] * geom["y"] + beta[3] * geom["z"]
        resid = vals - model
        resid = np.where(np.isfinite(resid), resid, 0.0)
        if not np.isfinite(float(np.dot(resid, resid))) or float(np.dot(resid, resid)) <= 0.0:
            return 0.0

        filled = np.zeros(npix, dtype=np.float64)
        filled[pix] = resid
        # Never pass NaNs/Infs to healpy.
        filled = np.where(np.isfinite(filled), filled, 0.0)

        try:
            alm = hp.map2alm(filled, lmax=LMAX, iter=0, pol=False, use_weights=False)
        except TypeError:
            try:
                alm = hp.map2alm(filled, lmax=LMAX, iter=0, pol=False)
            except TypeError:
                alm = hp.map2alm(filled, lmax=LMAX, iter=0)
        alm = np.asarray(alm, dtype=np.complex128).ravel()
        try:
            if int(alm.size) < int(hp.Alm.getsize(LMAX)):
                return float(_fallback(arr, valid, hp, nside))
        except Exception:
            if int(alm.size) <= 0:
                return float(_fallback(arr, valid, hp, nside))
        ok = np.isfinite(alm.real) & np.isfinite(alm.imag)
        alm = np.where(ok, alm, 0.0 + 0.0j)

        tensor = _tensor_from_alm(hp, alm, LMAX)
        if tensor is None:
            return float(_fallback(arr, valid, hp, nside))
        return float(_alignment_from_tensor(tensor))
    except Exception:
        return float(_fallback(arr, valid, hp, nside))


def summarize_results(planck_stat, sim_results, _desc=test_description):
    import math
    import numpy as np

    def _ff(x, default=np.nan):
        try:
            y = float(x)
        except Exception:
            return float(default)
        if np.isfinite(y):
            return float(y)
        return float(default)

    def _angle_deg(a):
        aa = _ff(a)
        if not np.isfinite(aa):
            return None
        aa = min(1.0, max(0.0, aa))
        return float(math.degrees(math.acos(aa)))

    ps = _ff(planck_stat)
    try:
        sims = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            for v in sim_results:
                vals.append(_ff(v))
        except Exception:
            vals = []
        sims = np.asarray(vals, dtype=np.float64)
    sims = sims[np.isfinite(sims)]
    n = int(sims.size)

    out = {
        "test_description": _desc,
        "statistic": "A = |n_max . n_ecl|, where n_max is the largest-eigenvalue axis of sum_{ell=2}^5 T_ij^(ell)",
        "planck_stat": ps,
        "planck_alignment_angle_degrees": _angle_deg(ps),
        "n_sims": n,
        "statistic_range": "0 to 1",
        "tail_of_interest": "upper tail; larger A means the combined low-ell power-tensor axis is closer to the ecliptic north pole",
        "higher_is_more_anomalous": True,
        "ecliptic_north_assumption": "n_ecl is the J2000 ecliptic north pole expressed in standard Galactic HEALPix map coordinates (l=96.384 deg, b=29.812 deg)",
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-ell power-tensor alignment with the ecliptic pole",
            "xlabel": "A = |n_max . n_ecl|",
            "ylabel": "Number of simulations",
            "bins": 30,
            "caption": "Distribution of the alignment of the summed ell=2..5 angular-momentum-dispersion preferred axis with the ecliptic north pole. Values near 1 indicate unusually close alignment."
        }
    }

    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_le = int(np.count_nonzero(sims <= ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_ge = int(np.count_nonzero(sims >= ps))
    n_eq = int(n - n_lt - n_gt)

    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_two = float(min(1.0, 2.0 * min(p_upper, p_lower)))

    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    zscore = None
    if np.isfinite(std) and std > 0.0:
        zscore = float((ps - mean) / std)
    qs = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])
    sim_angles = np.degrees(np.arccos(np.clip(sims, 0.0, 1.0)))
    angle_qs = np.percentile(sim_angles, [1, 5, 50, 95, 99])

    out.update({
        "sim_mean": mean,
        "sim_median": median,
        "sim_std": std,
        "sim_min": float(np.min(sims)),
        "sim_max": float(np.max(sims)),
        "sim_percentile_1": float(qs[0]),
        "sim_percentile_5": float(qs[1]),
        "sim_percentile_16": float(qs[2]),
        "sim_percentile_50": float(qs[3]),
        "sim_percentile_84": float(qs[4]),
        "sim_percentile_95": float(qs[5]),
        "sim_percentile_99": float(qs[6]),
        "sim_alignment_angle_percentile_1_degrees": float(angle_qs[0]),
        "sim_alignment_angle_percentile_5_degrees": float(angle_qs[1]),
        "sim_alignment_angle_median_degrees": float(angle_qs[2]),
        "sim_alignment_angle_percentile_95_degrees": float(angle_qs[3]),
        "sim_alignment_angle_percentile_99_degrees": float(angle_qs[4]),
        "planck_zscore_vs_sims": zscore,
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "rank_descending_1_is_most_aligned": int(n_gt + 1),
        "rank_ascending_1_is_least_aligned": int(n_lt + 1),
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "fraction_sims_gt_planck": float(n_gt / n),
        "fraction_sims_ge_planck": float(n_ge / n),
        "empirical_p_upper_tail_including_ties_plus_one": p_upper,
        "empirical_p_lower_tail_including_ties_plus_one": p_lower,
        "empirical_p_two_sided_extremeness": p_two,
        "is_upper_tail_5pct": bool(p_upper <= 0.05),
        "is_upper_tail_1pct": bool(p_upper <= 0.01)
    })

    if std == 0.0:
        out["warning"] = "All finite simulation statistics are identical; check for degenerate inputs or missing harmonic-transform support."
    elif p_upper <= 0.01:
        out["interpretation"] = "The Planck map lies in the highest 1% tail: its low-ell power-tensor axis is very closely aligned with the ecliptic pole relative to the simulations."
    elif p_upper <= 0.05:
        out["interpretation"] = "The Planck map lies in the highest 5% tail: its low-ell power-tensor axis is unusually aligned with the ecliptic pole relative to the simulations."
    else:
        out["interpretation"] = "The Planck alignment is not in the highest 5% tail of the simulation distribution for this statistic."
    return out
