Hypothesis

The SMICA CMB map has an anomalously large low-\(\ell\) odd-parity excess relative to even-parity power, i.e. \(S=\ln(P_{\rm odd}/P_{\rm even})\) is higher than expected under LCDM simulations.

Method

Monopole/dipole-subtracted SMICA and LCDM masked temperature maps are converted to masked pseudo-spectra for 2≤ℓ≤30; the statistic is S=ln(Podd/Peven), with positive S testing an odd-multipole excess.

Result Summary

SMICA gives S=0.2057. Simulations have mean -0.0165, median -0.0103 and standard deviation 0.1231. The SMICA value is at the 96.5th percentile, with 35/1000 simulations at least as high; the upper-tail p-value is 0.035, about 1.81σ.

The data show a high odd/even low-ℓ power ratio relative to LCDM simulations. This is suggestive and passes a 5% one-tailed criterion, but it is below a 2σ-strength anomaly and uses only 1000 simulations.

Agent Justification

The test description and code explicitly identify large positive \(S\) values as the anomaly of interest, and the summary reports the upper-tail probability for an odd-parity excess.

Comparison To Literature

Low-ℓ parity asymmetry is a known CMB anomaly theme in WMAP/Planck isotropy analyses, e.g. Planck Collaboration, A&A 571, A23 (2014) and A&A 641, A7 (2020), but no directly matching study of this exact masked pseudo-spectrum log-ratio implementation was identified in the provided/prior searches.

import numpy as np
import healpy as hp

# Masked Low-l Parity Log-Ratio
test_description = (
    "Masked Low-l Parity Log-Ratio: fit and subtract the monopole and dipole "
    "on finite unmasked pixels, set masked pixels to zero, compute the masked "
    "pseudo-spectrum for 2<=ell<=30, and return ln(P_odd/P_even)."
)

# Use non-private names and capture them as default arguments so analyze_map is
# robust to registration systems that keep only the function object or execute
# submitted code with separate globals/locals.
LMAX_PARITY = 30
ODD_ELL_PARITY = np.arange(3, 30, 2, dtype=np.int64)    # 3,5,...,29 (14 multipoles)
EVEN_ELL_PARITY = np.arange(2, 31, 2, dtype=np.int64)   # 2,4,...,30 (15 multipoles)
ELL_PARITY = np.arange(LMAX_PARITY + 1, dtype=np.float64)
DL_FACTOR_PARITY = ELL_PARITY * (ELL_PARITY + 1.0) / (2.0 * np.pi)
TINY_PARITY = np.finfo(np.float64).tiny
GEOMETRY_CACHE_PARITY = {}


def analyze_map(
    m,
    np=np,
    hp=hp,
    lmax=LMAX_PARITY,
    odd_ell=ODD_ELL_PARITY,
    even_ell=EVEN_ELL_PARITY,
    dl_factor=DL_FACTOR_PARITY,
    tiny=TINY_PARITY,
    geometry_cache=GEOMETRY_CACHE_PARITY,
):
    arr = np.asarray(m, dtype=np.float64)
    if arr.ndim != 1:
        arr = arr.ravel()

    npix = int(arr.size)
    if npix <= 0:
        return 0.0

    # Masked pixels arrive as NaN.  They are excluded from the monopole/dipole
    # fit and are never passed to healpy.
    valid = np.isfinite(arr)
    nvalid = int(np.count_nonzero(valid))

    # Cache the HEALPix directions and normal-equation pseudo-inverse for the
    # finite-pixel mask.  Most calls use the same mask for Planck and simulations.
    geom = None
    entries = geometry_cache.get(npix, [])
    for candidate in entries:
        if candidate["nvalid"] == nvalid and np.array_equal(candidate["valid"], valid):
            geom = candidate
            break

    if geom is None:
        try:
            nside = int(hp.npix2nside(npix))
        except Exception:
            return 0.0

        pix = np.flatnonzero(valid).astype(np.int64, copy=False)
        if nvalid > 0:
            x, y, z = hp.pix2vec(nside, pix)
            x = np.asarray(x, dtype=np.float64)
            y = np.asarray(y, dtype=np.float64)
            z = np.asarray(z, dtype=np.float64)

            sx = float(np.sum(x)); sy = float(np.sum(y)); sz = float(np.sum(z))
            sxx = float(np.dot(x, x)); sxy = float(np.dot(x, y)); sxz = float(np.dot(x, z))
            syy = float(np.dot(y, y)); syz = float(np.dot(y, z)); szz = float(np.dot(z, z))
            gram = np.array(
                [[float(nvalid), sx, sy, sz],
                 [sx, sxx, sxy, sxz],
                 [sy, sxy, syy, syz],
                 [sz, sxz, syz, szz]],
                dtype=np.float64,
            )
            gram_pinv = np.linalg.pinv(gram, rcond=1.0e-12)
        else:
            x = y = z = np.empty(0, dtype=np.float64)
            gram_pinv = None

        geom = {
            "valid": valid.copy(),
            "nvalid": nvalid,
            "pix": pix,
            "x": x,
            "y": y,
            "z": z,
            "gram_pinv": gram_pinv,
        }
        geometry_cache[npix] = [geom] + entries[:3]

    cleaned = np.zeros(npix, dtype=np.float64)
    if nvalid > 0:
        pix = geom["pix"]
        vals = arr[pix].astype(np.float64, copy=False)
        x = geom["x"]; y = geom["y"]; z = geom["z"]
        rhs = np.array(
            [float(np.sum(vals)), float(np.dot(x, vals)),
             float(np.dot(y, vals)), float(np.dot(z, vals))],
            dtype=np.float64,
        )
        coeff = geom["gram_pinv"].dot(rhs)
        cleaned[pix] = vals - (coeff[0] + coeff[1] * x + coeff[2] * y + coeff[3] * z)

    # healpy must receive a finite map; masked pixels have been filled with zero.
    try:
        cl = hp.anafast(cleaned, lmax=lmax, iter=0, pol=False)
    except Exception:
        return 0.0
    cl = np.asarray(cl, dtype=np.float64)
    if cl.size <= lmax:
        return 0.0

    dl = dl_factor * cl[:lmax + 1]
    p_odd = float(np.mean(dl[odd_ell]))
    p_even = float(np.mean(dl[even_ell]))

    if not np.isfinite(p_odd) or not np.isfinite(p_even):
        return 0.0

    # Auto pseudo-spectra should be non-negative; the floor only handles exact
    # zero maps or tiny numerical negatives and guarantees a finite scalar.
    p_odd = max(p_odd, tiny)
    p_even = max(p_even, tiny)
    stat = float(np.log(p_odd) - np.log(p_even))
    if not np.isfinite(stat):
        return 0.0
    return stat


def summarize_results(planck_stat, sim_results, np=np, test_description=test_description):
    sims = np.asarray(sim_results, dtype=np.float64)
    sims = sims[np.isfinite(sims)]
    ps = float(planck_stat)

    plot_spec = {
        "kind": "histogram",
        "title": "Masked low-ell parity log-ratio",
        "xlabel": "S = ln(P_odd / P_even)",
        "ylabel": "Number of LCDM simulations",
        "bins": 30,
        "caption": (
            "LCDM simulation distribution of the masked low-ell parity log-ratio. "
            "Large positive values are the odd-parity-excess tail; the SMICA/Planck "
            "value should be marked on the same axis."
        ),
    }

    out = {
        "test_description": test_description,
        "planck_stat": ps,
        "tail_of_interest": "upper / positive S (odd-parity excess relative to even power)",
        "plot_spec": plot_spec,
    }

    n = int(sims.size)
    out["n_sims"] = n
    if n == 0 or not np.isfinite(ps):
        out["warning"] = "No finite simulation statistics, or Planck statistic is non-finite."
        return out

    n_ge = int(np.count_nonzero(sims >= ps))
    n_le = int(np.count_nonzero(sims <= ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_lt = int(np.count_nonzero(sims < ps))
    n_eq = int(n - n_gt - n_lt)

    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_two_sided = float(min(1.0, 2.0 * min(p_upper, p_lower)))
    percentile_midrank = float(100.0 * (n_lt + 0.5 * n_eq) / n)

    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    z = float((ps - mean) / std) if std > 0.0 and np.isfinite(std) else None
    q02_5, q16, q84, q97_5 = [float(q) for q in np.percentile(sims, [2.5, 16.0, 84.0, 97.5])]

    out.update({
        "sim_mean": mean,
        "sim_median": median,
        "sim_std": std,
        "sim_percentile_2p5": q02_5,
        "sim_percentile_16": q16,
        "sim_percentile_84": q84,
        "sim_percentile_97p5": q97_5,
        "planck_zscore_vs_sims": z,
        "planck_percentile_midrank": percentile_midrank,
        "rank_ascending_min": int(n_lt + 1),
        "rank_ascending_max": int(n_le),
        "rank_descending_min": int(n_gt + 1),
        "rank_descending_max": int(n_ge),
        "n_sim_ge_planck": n_ge,
        "n_sim_le_planck": n_le,
        "empirical_p_upper_odd_excess": p_upper,
        "empirical_p_lower": p_lower,
        "empirical_p_two_sided": p_two_sided,
        "is_upper_tail_5pct": bool(p_upper <= 0.05),
        "is_upper_tail_1pct": bool(p_upper <= 0.01),
    })

    if std == 0.0:
        out["warning"] = "All finite simulation statistics are identical; check for a degenerate statistic or inputs."
    elif ps > median:
        out["interpretation"] = "Planck/SMICA lies above the simulation median; use empirical_p_upper_odd_excess for the odd-parity-excess tail."
    else:
        out["interpretation"] = "Planck/SMICA is not above the simulation median; no positive odd-parity-excess tail is indicated."

    return out
