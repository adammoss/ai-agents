Hypothesis

The Planck low-l (2 <= ell <= 20) temperature map has an anomalous ecliptic hemispherical skewness contrast, meaning S = gamma_South - gamma_North is unusually different from the LCDM simulation expectation.

Method

After masked monopole/dipole removal, the map was filtered to 2 <= ell <= 20 and split into fixed ecliptic north/south hemispheres. The statistic was S = gamma_South - gamma_North, using weighted hemispherical skewness.

Result Summary

Planck gives S = -0.214, compared with LCDM simulations having mean 0.010 and std 0.303. The Planck value is z = -0.74, at the 21.3rd percentile, with two-tailed p = 0.426 from 1000 simulations.

Planck shows a mild negative contrast, i.e. slightly larger northern than southern ecliptic skewness, but this is well within the simulated LCDM distribution and far from a 2 sigma deviation.

Agent Justification

The statistic is signed, with positive values indicating excess southern skewness and negative values indicating excess northern skewness, but the test description does not specify a strong prior preference for either direction. The code evaluates both upper and lower tails and also reports two-sided p-values, so the natural anomaly hypothesis is a deviation in either direction.

Comparison To Literature

No directly matching published statistic was identified in prior searches. Related CMB anomaly literature discusses low-l alignments, ecliptic features, hemispherical asymmetry, and non-Gaussianity, but not this exact fixed-ecliptic low-l hemispherical skewness-contrast test.

test_description = "Low-l Ecliptic Hemispherical Skewness Contrast: for each masked nside=256 temperature map, remove the best-fit monopole and dipole on the unmasked sky, retain only multipoles 2 <= ell <= 20, synthesize the low-l map, and return S = gamma_South - gamma_North, the fixed ecliptic hemispherical weighted-skewness contrast."


def analyze_map(m, _cache={}):
    import math
    import zlib
    import numpy as np

    LMAX = 20
    NSIDE_OUT = 256
    MIN_PIX = 8
    EPS_VAR = 1.0e-30

    def _finish(x):
        try:
            y = float(x)
        except Exception:
            return 0.0
        if not np.isfinite(y):
            return 0.0
        if y > 1.0e6:
            return 1.0e6
        if y < -1.0e6:
            return -1.0e6
        return float(y)

    def _flat_float(x):
        try:
            if np.ma.isMaskedArray(x):
                a = np.asarray(x.filled(np.nan), dtype=np.float64)
            else:
                a = np.asarray(x, dtype=np.float64)
        except Exception:
            try:
                a = np.asarray(np.real(x), dtype=np.float64)
            except Exception:
                a = np.asarray([], dtype=np.float64)
        return np.ravel(a)

    def _get_hp():
        key = "hp_lowl_ecl_skew_contrast"
        if key not in _cache:
            try:
                import healpy as hp
                _cache[key] = hp
            except Exception:
                _cache[key] = None
        return _cache.get(key)

    def _nside_from_npix(hp, npix):
        try:
            ns = int(hp.npix2nside(int(npix)))
            if ns > 0 and int(hp.nside2npix(ns)) == int(npix):
                return ns
        except Exception:
            pass
        try:
            if int(npix) > 0 and int(npix) % 12 == 0:
                ns = int(round(math.sqrt(float(npix) / 12.0)))
                if ns > 0 and 12 * ns * ns == int(npix):
                    return ns
        except Exception:
            pass
        return -1

    def _mask_signature(valid):
        v = np.asarray(valid, dtype=bool).ravel()
        try:
            crc = int(zlib.crc32(np.packbits(v).tobytes()) & 0xffffffff)
        except Exception:
            crc = int(np.count_nonzero(v) & 0xffffffff)
        return str(int(v.size)) + ":" + str(int(np.count_nonzero(v))) + ":" + str(crc)

    def _ecliptic_north_vec_galactic():
        key = "j2000_ecliptic_north_vec_in_galactic_v1"
        vec = _cache.get(key)
        if vec is not None:
            return vec
        # J2000 north ecliptic pole in Galactic coordinates.
        lon = math.radians(96.384)
        lat = math.radians(29.812)
        clat = math.cos(lat)
        vec = np.array([clat * math.cos(lon), clat * math.sin(lon), math.sin(lat)], dtype=np.float64)
        nrm = float(np.sqrt(np.dot(vec, vec)))
        if np.isfinite(nrm) and nrm > 0.0:
            vec = vec / nrm
        else:
            vec = np.array([0.0, 0.0, 1.0], dtype=np.float64)
        _cache[key] = vec
        return vec

    def _pixvecs(hp, nside):
        nside = int(nside)
        key = "pixvecs_ring_nside_" + str(nside)
        got = _cache.get(key)
        if got is not None:
            return got
        npix = int(hp.nside2npix(nside))
        pix = np.arange(npix, dtype=np.int64)
        try:
            x, y, z = hp.pix2vec(nside, pix, nest=False)
        except TypeError:
            x, y, z = hp.pix2vec(nside, pix)
        got = (np.asarray(x, dtype=np.float64), np.asarray(y, dtype=np.float64), np.asarray(z, dtype=np.float64))
        # Only cache the small number of grids normally encountered; nside=256 is about 18 MB.
        try:
            if int(_cache.get("pixvec_cache_count_lowl_skew", 0)) < 4:
                _cache[key] = got
                _cache["pixvec_cache_count_lowl_skew"] = int(_cache.get("pixvec_cache_count_lowl_skew", 0)) + 1
        except Exception:
            pass
        return got

    def _north_bool(hp, nside):
        nside = int(nside)
        key = "ecliptic_north_bool_ring_nside_" + str(nside)
        got = _cache.get(key)
        if got is not None:
            return got
        x, y, z = _pixvecs(hp, nside)
        pole = _ecliptic_north_vec_galactic()
        dot = pole[0] * x + pole[1] * y + pole[2] * z
        got = np.asarray(dot >= 0.0, dtype=bool)
        _cache[key] = got
        return got

    def _region_info(hp, nside_in, valid):
        nside_in = int(nside_in)
        sig = _mask_signature(valid)
        key = "regions_lowl_ecl_skew_" + str(nside_in) + "_to_" + str(NSIDE_OUT) + "_" + sig
        got = _cache.get(key)
        if got is not None:
            return got
        north = _north_bool(hp, NSIDE_OUT)
        npix_out = int(hp.nside2npix(NSIDE_OUT))
        valid = np.asarray(valid, dtype=bool).ravel()
        if nside_in == NSIDE_OUT and int(valid.size) == npix_out:
            weights = None
            use = valid
            idx_n = np.flatnonzero(use & north).astype(np.int64, copy=False)
            idx_s = np.flatnonzero(use & (~north)).astype(np.int64, copy=False)
            w_n = None
            w_s = None
        else:
            try:
                frac = hp.ud_grade(valid.astype(np.float64), nside_out=NSIDE_OUT, order_in="RING", order_out="RING", power=0)
            except TypeError:
                frac = hp.ud_grade(valid.astype(np.float64), NSIDE_OUT)
            frac = np.asarray(frac, dtype=np.float64).ravel()
            if int(frac.size) != npix_out:
                return None
            frac = np.where(np.isfinite(frac), frac, 0.0)
            frac = np.clip(frac, 0.0, 1.0)
            use = frac > 1.0e-12
            idx_n = np.flatnonzero(use & north).astype(np.int64, copy=False)
            idx_s = np.flatnonzero(use & (~north)).astype(np.int64, copy=False)
            w_n = frac[idx_n].astype(np.float64, copy=False)
            w_s = frac[idx_s].astype(np.float64, copy=False)
            weights = frac
        if int(idx_n.size) < MIN_PIX or int(idx_s.size) < MIN_PIX:
            return None
        got = {"idx_n": idx_n, "idx_s": idx_s, "w_n": w_n, "w_s": w_s, "weights": weights}
        try:
            if int(_cache.get("region_cache_count_lowl_skew", 0)) < 8:
                _cache[key] = got
                _cache["region_cache_count_lowl_skew"] = int(_cache.get("region_cache_count_lowl_skew", 0)) + 1
        except Exception:
            pass
        return got

    def _dipole_fit_geometry(hp, nside, valid):
        nside = int(nside)
        sig = _mask_signature(valid)
        key = "md_fit_geom_" + str(nside) + "_" + sig
        got = _cache.get(key)
        if got is not None:
            return got
        valid = np.asarray(valid, dtype=bool).ravel()
        x, y, z = _pixvecs(hp, nside)
        if int(x.size) != int(valid.size):
            return None
        xv = x[valid]
        yv = y[valid]
        zv = z[valid]
        n = float(xv.size)
        if n < 4.0:
            return None
        g = np.array([
            [n, float(np.sum(xv)), float(np.sum(yv)), float(np.sum(zv))],
            [float(np.sum(xv)), float(np.dot(xv, xv)), float(np.dot(xv, yv)), float(np.dot(xv, zv))],
            [float(np.sum(yv)), float(np.dot(yv, xv)), float(np.dot(yv, yv)), float(np.dot(yv, zv))],
            [float(np.sum(zv)), float(np.dot(zv, xv)), float(np.dot(zv, yv)), float(np.dot(zv, zv))]
        ], dtype=np.float64)
        got = {"g": g, "xv": xv, "yv": yv, "zv": zv, "x": x, "y": y, "z": z}
        try:
            if int(_cache.get("md_geom_cache_count_lowl_skew", 0)) < 8:
                _cache[key] = got
                _cache["md_geom_cache_count_lowl_skew"] = int(_cache.get("md_geom_cache_count_lowl_skew", 0)) + 1
        except Exception:
            pass
        return got

    def _remove_monopole_dipole(arr, valid, hp, nside):
        geom = _dipole_fit_geometry(hp, nside, valid)
        if geom is None:
            out = np.zeros_like(arr, dtype=np.float64)
            vals = arr[valid]
            mu = float(np.mean(vals)) if int(vals.size) > 0 else 0.0
            out[valid] = vals - mu
            return out
        vals = np.asarray(arr[valid], dtype=np.float64)
        if int(vals.size) < 4:
            out = np.zeros_like(arr, dtype=np.float64)
            out[valid] = vals - (float(np.mean(vals)) if int(vals.size) else 0.0)
            return out
        rhs = np.array([
            float(np.sum(vals)),
            float(np.dot(vals, geom["xv"])),
            float(np.dot(vals, geom["yv"])),
            float(np.dot(vals, geom["zv"]))
        ], dtype=np.float64)
        try:
            beta = np.linalg.solve(geom["g"], rhs)
        except Exception:
            try:
                beta = np.linalg.lstsq(geom["g"], rhs, rcond=None)[0]
            except Exception:
                beta = np.array([float(np.mean(vals)), 0.0, 0.0, 0.0], dtype=np.float64)
        beta = np.where(np.isfinite(beta), beta, 0.0)
        model = beta[0] + beta[1] * geom["x"] + beta[2] * geom["y"] + beta[3] * geom["z"]
        out = np.zeros_like(arr, dtype=np.float64)
        resid_valid = arr[valid] - model[valid]
        resid_valid = np.where(np.isfinite(resid_valid), resid_valid, 0.0)
        out[valid] = resid_valid
        return out

    def _weighted_skew(vals, weights=None):
        vals = np.asarray(vals, dtype=np.float64).ravel()
        good = np.isfinite(vals)
        if weights is None:
            vals = vals[good]
            if int(vals.size) < MIN_PIX:
                return None
            mu = float(np.mean(vals))
            d = vals - mu
            var = float(np.mean(d * d))
            if (not np.isfinite(var)) or var <= EPS_VAR:
                return 0.0
            m3 = float(np.mean(d * d * d))
            if not np.isfinite(m3):
                return None
            return float(m3 / (var ** 1.5))
        w = np.asarray(weights, dtype=np.float64).ravel()
        if int(w.size) != int(vals.size):
            return None
        good = good & np.isfinite(w) & (w > 0.0)
        vals = vals[good]
        w = w[good]
        if int(vals.size) < MIN_PIX:
            return None
        sw = float(np.sum(w))
        if (not np.isfinite(sw)) or sw <= 0.0:
            return None
        mu = float(np.dot(w, vals) / sw)
        d = vals - mu
        var = float(np.dot(w, d * d) / sw)
        if (not np.isfinite(var)) or var <= EPS_VAR:
            return 0.0
        m3 = float(np.dot(w, d * d * d) / sw)
        if not np.isfinite(m3):
            return None
        return float(m3 / (var ** 1.5))

    def _raw_fallback(arr, valid, hp=None, nside=-1):
        # Used only if harmonic operations are unavailable. It still follows the same fixed
        # ecliptic north/south skewness contrast on the monopole/dipole-cleaned input map.
        try:
            arr = np.asarray(arr, dtype=np.float64).ravel()
            valid = np.asarray(valid, dtype=bool).ravel()
            if int(arr.size) != int(valid.size) or int(np.count_nonzero(valid)) < 2 * MIN_PIX:
                return 0.0
            if hp is not None and int(nside) > 0:
                cleaned = _remove_monopole_dipole(arr, valid, hp, int(nside))
                north = _north_bool(hp, int(nside))
                if int(north.size) == int(arr.size):
                    idx_n = np.flatnonzero(valid & north)
                    idx_s = np.flatnonzero(valid & (~north))
                else:
                    idx = np.flatnonzero(valid)
                    idx_n = idx[: int(idx.size) // 2]
                    idx_s = idx[int(idx.size) // 2:]
            else:
                vals = arr[valid]
                mu = float(np.mean(vals)) if int(vals.size) else 0.0
                cleaned = np.zeros_like(arr, dtype=np.float64)
                cleaned[valid] = vals - mu
                idx = np.flatnonzero(valid)
                idx_n = idx[: int(idx.size) // 2]
                idx_s = idx[int(idx.size) // 2:]
            if int(idx_n.size) < MIN_PIX or int(idx_s.size) < MIN_PIX:
                return 0.0
            gn = _weighted_skew(cleaned[idx_n], None)
            gs = _weighted_skew(cleaned[idx_s], None)
            if gn is None or gs is None:
                return 0.0
            return _finish(gs - gn)
        except Exception:
            return 0.0

    arr = _flat_float(m)
    npix = int(arr.size)
    if npix <= 0:
        return 0.0
    valid = np.isfinite(arr)
    if int(np.count_nonzero(valid)) < 2 * MIN_PIX:
        return 0.0

    hp = _get_hp()
    if hp is None:
        return float(_raw_fallback(arr, valid, None, -1))
    nside_in = _nside_from_npix(hp, npix)
    if nside_in <= 0:
        return float(_raw_fallback(arr, valid, hp, nside_in))

    try:
        regions = _region_info(hp, nside_in, valid)
        if regions is None:
            return float(_raw_fallback(arr, valid, hp, nside_in))

        cleaned = _remove_monopole_dipole(arr, valid, hp, nside_in)
        # Healpy must never receive NaN/inf values. Masked pixels are finite-filled by zero.
        cleaned = np.where(np.isfinite(cleaned), cleaned, 0.0).astype(np.float64, copy=False)

        try:
            alm = hp.map2alm(cleaned, lmax=LMAX, iter=0, pol=False, use_weights=False)
        except TypeError:
            try:
                alm = hp.map2alm(cleaned, lmax=LMAX, iter=0, pol=False)
            except TypeError:
                alm = hp.map2alm(cleaned, lmax=LMAX, iter=0)
        alm = np.asarray(alm, dtype=np.complex128).ravel()
        try:
            if int(alm.size) < int(hp.Alm.getsize(LMAX)):
                return float(_raw_fallback(arr, valid, hp, nside_in))
        except Exception:
            if int(alm.size) <= 0:
                return float(_raw_fallback(arr, valid, hp, nside_in))
        good_alm = np.isfinite(alm.real) & np.isfinite(alm.imag)
        alm = np.where(good_alm, alm, 0.0 + 0.0j)

        # Retain only ell=2..20. lmax already excludes ell>20; explicitly null ell=0,1.
        for ell in (0, 1):
            for mm in range(ell + 1):
                try:
                    alm[hp.Alm.getidx(LMAX, ell, mm)] = 0.0 + 0.0j
                except Exception:
                    pass

        try:
            low_map = hp.alm2map(alm, NSIDE_OUT, lmax=LMAX, pol=False, verbose=False)
        except TypeError:
            try:
                low_map = hp.alm2map(alm, NSIDE_OUT, lmax=LMAX, pol=False)
            except TypeError:
                low_map = hp.alm2map(alm, NSIDE_OUT, lmax=LMAX)
        low_map = np.asarray(low_map, dtype=np.float64).ravel()
        if int(low_map.size) != int(hp.nside2npix(NSIDE_OUT)):
            return float(_raw_fallback(arr, valid, hp, nside_in))
        low_map = np.where(np.isfinite(low_map), low_map, 0.0)

        idx_n = regions["idx_n"]
        idx_s = regions["idx_s"]
        gamma_n = _weighted_skew(low_map[idx_n], regions.get("w_n", None))
        gamma_s = _weighted_skew(low_map[idx_s], regions.get("w_s", None))
        if gamma_n is None or gamma_s is None:
            return float(_raw_fallback(arr, valid, hp, nside_in))
        return float(_finish(gamma_s - gamma_n))
    except Exception:
        return float(_raw_fallback(arr, valid, hp, nside_in))


def summarize_results(planck_stat, sim_results, _desc=test_description):
    import numpy as np

    def _finite_float(x, default=np.nan):
        try:
            y = float(x)
        except Exception:
            return float(default)
        return float(y) if np.isfinite(y) else float(default)

    ps = _finite_float(planck_stat)
    try:
        sims = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            for v in sim_results:
                vals.append(_finite_float(v))
        except Exception:
            vals = []
        sims = np.asarray(vals, dtype=np.float64)
    sims = sims[np.isfinite(sims)]
    n = int(sims.size)

    out = {
        "test_description": _desc,
        "statistic": "S = gamma_South - gamma_North",
        "planck_stat": ps,
        "n_sims": n,
        "higher_is_more_southern_positive_skewness_excess": True,
        "tail_of_interest": "upper tail for excess southern ecliptic low-l skewness relative to the north; lower tail for northern excess",
        "multipole_range": "2 <= ell <= 20",
        "monopole_dipole_treatment": "best-fit monopole and Cartesian dipole are removed on the unmasked sky before harmonic filtering",
        "ecliptic_coordinate_assumption": "Input HEALPix maps are in Galactic coordinates; fixed ecliptic hemispheres use the J2000 ecliptic north pole at Galactic (l,b)=(96.384,29.812) deg.",
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-l ecliptic hemispherical skewness contrast",
            "xlabel": "S = gamma_South - gamma_North",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "Distribution of the ell=2..20 low-l ecliptic hemispherical skewness contrast. Positive values indicate larger weighted skewness in the southern ecliptic hemisphere."
        }
    }

    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_le = int(np.count_nonzero(sims <= ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_ge = int(np.count_nonzero(sims >= ps))
    n_eq = int(n - n_lt - n_gt)

    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_two = float(min(1.0, 2.0 * min(p_upper, p_lower)))
    med = float(np.median(sims))
    n_abs_ge_med = int(np.count_nonzero(np.abs(sims - med) >= abs(ps - med)))
    p_abs_med = float((n_abs_ge_med + 1.0) / (n + 1.0))
    p_direction = p_upper if ps >= med else p_lower

    mean = float(np.mean(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    z = None
    if np.isfinite(std) and std > 0.0:
        z = float((ps - mean) / std)
    q = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])

    out.update({
        "sim_mean": mean,
        "sim_median": med,
        "sim_std": std,
        "sim_min": float(np.min(sims)),
        "sim_max": float(np.max(sims)),
        "sim_percentile_1": float(q[0]),
        "sim_percentile_5": float(q[1]),
        "sim_percentile_16": float(q[2]),
        "sim_percentile_50": float(q[3]),
        "sim_percentile_84": float(q[4]),
        "sim_percentile_95": float(q[5]),
        "sim_percentile_99": float(q[6]),
        "planck_zscore_vs_sims": z,
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "rank_descending_1_is_largest_southern_skewness_excess": int(n_gt + 1),
        "rank_ascending_1_is_largest_northern_skewness_excess": int(n_lt + 1),
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "fraction_sims_gt_planck": float(n_gt / n),
        "fraction_sims_ge_planck": float(n_ge / n),
        "empirical_p_upper_tail_southern_skewness_excess_including_ties_plus_one": p_upper,
        "empirical_p_lower_tail_northern_skewness_excess_including_ties_plus_one": p_lower,
        "empirical_p_two_sided_from_signed_rank": p_two,
        "empirical_p_two_sided_about_sim_median_plus_one": p_abs_med,
        "empirical_p_tail_in_planck_direction_vs_sim_median_plus_one": float(p_direction),
        "planck_direction_vs_zero": "southern ecliptic skewness exceeds northern" if ps > 0.0 else ("northern ecliptic skewness exceeds southern" if ps < 0.0 else "no skewness contrast"),
        "planck_direction_vs_sim_median": "high S" if ps > med else ("low S" if ps < med else "at simulation median"),
        "is_upper_tail_5pct": bool(p_upper <= 0.05),
        "is_upper_tail_1pct": bool(p_upper <= 0.01),
        "is_lower_tail_5pct": bool(p_lower <= 0.05),
        "is_lower_tail_1pct": bool(p_lower <= 0.01),
        "is_two_sided_signed_rank_5pct": bool(p_two <= 0.05),
        "is_two_sided_signed_rank_1pct": bool(p_two <= 0.01)
    })

    if std == 0.0:
        out["warning"] = "All finite simulation statistics are identical; check the input maps, mask, and harmonic-transform support."
    elif p_upper <= 0.01:
        out["interpretation"] = "Planck lies in the highest 1% of simulations for S, indicating an unusually large southern-minus-northern low-l skewness contrast."
    elif p_upper <= 0.05:
        out["interpretation"] = "Planck lies in the highest 5% of simulations for S, indicating enhanced southern-minus-northern low-l skewness."
    elif p_lower <= 0.01:
