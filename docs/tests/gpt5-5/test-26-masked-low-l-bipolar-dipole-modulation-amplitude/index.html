Hypothesis

The masked CMB temperature map has an anomalously large low-\(\ell\) dipole-modulation / hemispherical-power-asymmetry signal, measured by a bias-subtracted adjacent-multipole bipolar amplitude \(S\), compared with isotropic simulations.

Method

Measures a bias-subtracted adjacent-multipole bipolar amplitude S over 2<=ell<=64 on masked CMB temperature maps after monopole/dipole removal; larger S indicates stronger low-l dipole-modulation or hemispherical-power-asymmetry coupling.

Result Summary

Planck has S=0.00493. Using the preferred ensemble-calibrated reconstruction, the simulation mean is 0.00333 with std 0.00144, placing Planck at about z=1.09 and percentile 84.8. The upper-tail empirical p-value is 0.153 from 1000 simulations.

The Planck value is elevated relative to the simulation mean but well outside the upper 5% tail, so this statistic does not provide significant evidence for an anomalously large dipole-modulation signal.

Agent Justification

The statistic \(S\) is a nonnegative amplitude, and the test description and code explicitly state that larger \(S\) indicates a stronger dipole-modulation-like anomaly. The summary code evaluates upper-tail probabilities, so the directional alternative is that Planck has larger \(S\) than expected from simulations.

Comparison To Literature

Directly matching published studies of this exact masked pseudo-alm, adjacent-multipole, bias-subtracted scalar were not found. It is methodologically related to established CMB hemispherical-power-asymmetry, dipole-modulation, and BipoSH analyses discussed in Planck isotropy papers, but appears to differ in implementation and calibration.

import numpy as np

test_description = "Masked Low-l Bipolar Dipole-Modulation Amplitude: remove the best-fit monopole and dipole on the unmasked sky, compute masked pseudo-harmonic coefficients for 2<=ell<=64, form adjacent-multipole dipole-coupling coefficients B_M, and use the simulation ensemble when available to report the bias-subtracted amplitude S=sqrt(sum_M |B_M-<B_M>_sim|^2). Larger S indicates stronger low-ell dipole-modulation / hemispherical-power-asymmetry coupling."

# Keep all mutable state behind default arguments of the public functions.  Some
# runners execute user code with separate globals/locals; using defaults avoids
# NameError for cached state inside analyze_map and summarize_results.
_BIPOLAR_DM_STATE = {
    "geom_cache": {},
    "md_cache": {},
    "cg_cache": None,
    "raw_records": [],
    "call_index": 0,
}


def analyze_map(m, _state=_BIPOLAR_DM_STATE):
    import math
    import numpy as np

    LMIN = 2
    LMAX = 64
    LMAX_PAIR = 63
    NPAIR = LMAX_PAIR - LMIN + 1
    WNORM = float(sum(2 * ell + 1 for ell in range(LMIN, LMAX_PAIR + 1)))
    CHUNK = 262144
    EPS = 1.0e-30

    def _infer_nside(npix):
        try:
            npix_i = int(npix)
            if npix_i <= 0 or npix_i % 12 != 0:
                return -1
            nside = int(round(math.sqrt(npix_i / 12.0)))
            if nside <= 0 or 12 * nside * nside != npix_i:
                return -1
            return nside
        except Exception:
            return -1

    def _alm_index(lmax, ell, mm):
        # healpy Alm storage is in non-negative-m blocks.
        return int(mm * (2 * lmax + 1 - mm) // 2 + ell)

    def _get_geometry(nside, npix):
        key = (int(nside), int(npix))
        cache = _state.get("geom_cache")
        if not isinstance(cache, dict):
            cache = {}
            _state["geom_cache"] = cache
        item = cache.get(key)
        if item is not None:
            return item
        import healpy as hp
        pix = np.arange(int(npix), dtype=np.int64)
        x, y, z = hp.pix2vec(int(nside), pix, nest=False)
        item = (np.asarray(x, dtype=np.float64).ravel(),
                np.asarray(y, dtype=np.float64).ravel(),
                np.asarray(z, dtype=np.float64).ravel())
        if len(cache) > 2:
            cache.clear()
        cache[key] = item
        return item

    def _get_md_solver(nside, npix, valid, x, y, z):
        cache = _state.get("md_cache")
        if not isinstance(cache, dict):
            cache = {}
            _state["md_cache"] = cache
        packed = np.packbits(np.asarray(valid, dtype=np.bool_))
        mask_bytes = packed.tobytes()
        key = (int(nside), int(npix), int(np.count_nonzero(valid)), hash(mask_bytes))
        item = cache.get(key)
        if item is not None and item.get("mask_bytes") == mask_bytes:
            return item

        idx = np.flatnonzero(valid).astype(np.int64, copy=False)
        G = np.zeros((4, 4), dtype=np.float64)
        # Equal-area least squares on the unmasked sky for basis [1, x, y, z].
        for start in range(0, int(idx.size), CHUNK):
            ii = idx[start:start + CHUNK]
            xx = x[ii]
            yy = y[ii]
            zz = z[ii]
            n = float(ii.size)
            G[0, 0] += n
            G[0, 1] += float(np.sum(xx)); G[1, 0] = G[0, 1]
            G[0, 2] += float(np.sum(yy)); G[2, 0] = G[0, 2]
            G[0, 3] += float(np.sum(zz)); G[3, 0] = G[0, 3]
            G[1, 1] += float(np.dot(xx, xx))
            G[1, 2] += float(np.dot(xx, yy)); G[2, 1] = G[1, 2]
            G[1, 3] += float(np.dot(xx, zz)); G[3, 1] = G[1, 3]
            G[2, 2] += float(np.dot(yy, yy))
            G[2, 3] += float(np.dot(yy, zz)); G[3, 2] = G[2, 3]
            G[3, 3] += float(np.dot(zz, zz))
        try:
            solver = np.linalg.pinv(G, rcond=1.0e-12)
        except Exception:
            solver = np.zeros((4, 4), dtype=np.float64)
        item = {"idx": idx, "solver": np.asarray(solver, dtype=np.float64), "mask_bytes": mask_bytes}
        if len(cache) > 4:
            cache.clear()
        cache[key] = item
        return item

    def _remove_monopole_dipole(arr, valid, nside, npix):
        x, y, z = _get_geometry(nside, npix)
        entry = _get_md_solver(nside, npix, valid, x, y, z)
        idx = entry["idx"]
        rhs = np.zeros(4, dtype=np.float64)
        for start in range(0, int(idx.size), CHUNK):
            ii = idx[start:start + CHUNK]
            vals = np.asarray(arr[ii], dtype=np.float64)
            rhs[0] += float(np.sum(vals))
            rhs[1] += float(np.dot(x[ii], vals))
            rhs[2] += float(np.dot(y[ii], vals))
            rhs[3] += float(np.dot(z[ii], vals))
        coeff = np.asarray(entry["solver"].dot(rhs), dtype=np.float64).ravel()
        if coeff.size != 4 or not np.all(np.isfinite(coeff)):
            coeff = np.zeros(4, dtype=np.float64)
        clean = np.zeros(int(npix), dtype=np.float64)
        for start in range(0, int(idx.size), CHUNK):
            ii = idx[start:start + CHUNK]
            clean[ii] = (np.asarray(arr[ii], dtype=np.float64)
                         - coeff[0] - coeff[1] * x[ii] - coeff[2] * y[ii] - coeff[3] * z[ii])
        clean[~np.isfinite(clean)] = 0.0
        return clean

    def _get_cg_table():
        item = _state.get("cg_cache")
        if item is not None:
            return item
        table = {}
        # Closed-form Condon-Shortley Clebsch-Gordan coefficients for
        # <ell m, ell+1 (M-m) | 1 M>, with M=-1,0,1.  The statistic needs
        # m' = m-M and an extra factor (-1)^m'.
        for ell in range(LMIN, LMAX_PAIR + 1):
            mvals = np.arange(-ell, ell + 1, dtype=np.int64)
            coeff = np.zeros((3, mvals.size), dtype=np.float64)
            idx2 = np.zeros((3, mvals.size), dtype=np.int64)
            phase_lm = np.where(((ell - mvals) % 2) == 0, 1.0, -1.0)
            den0 = float((ell + 1) * (2 * ell + 1) * (2 * ell + 3))
            for mi, M in enumerate((-1, 0, 1)):
                mp = mvals - M
                phase_mp = np.where((mp % 2) == 0, 1.0, -1.0)
                if M == 0:
                    term = (ell + 1) * (ell + 1) - mvals * mvals
                    cg = phase_lm * np.sqrt(np.maximum(0.0, 3.0 * term.astype(np.float64) / den0))
                elif M == 1:
                    term = (ell + mvals + 1) * (ell + mvals + 2)
                    cg = phase_lm * np.sqrt(np.maximum(0.0, 3.0 * term.astype(np.float64) / (2.0 * den0)))
                else:  # M == -1
                    term = (ell - mvals + 1) * (ell - mvals + 2)
                    cg = phase_lm * np.sqrt(np.maximum(0.0, 3.0 * term.astype(np.float64) / (2.0 * den0)))
                coeff[mi, :] = phase_mp * cg
                idx2[mi, :] = mp + (ell + 1)
            table[ell] = (coeff, idx2)
        _state["cg_cache"] = table
        return table

    def _alm_full_row(alm, lmax, ell):
        out = np.empty(2 * ell + 1, dtype=np.complex128)
        out[ell] = complex(alm[_alm_index(lmax, ell, 0)])
        for mm in range(1, ell + 1):
            v = complex(alm[_alm_index(lmax, ell, mm)])
            out[ell + mm] = v
            out[ell - mm] = (1.0 if (mm % 2 == 0) else -1.0) * np.conj(v)
        return out

    def _compute_cl_and_cross(raw_alm):
        rows = {}
        cl = np.zeros(LMAX + 1, dtype=np.float64)
        for ell in range(LMIN, LMAX + 1):
            row = _alm_full_row(raw_alm, LMAX, ell)
            rows[ell] = row
            c = float(np.vdot(row, row).real / (2 * ell + 1))
            cl[ell] = c if np.isfinite(c) and c > 0.0 else 0.0
        cg = _get_cg_table()
        cross = np.zeros((NPAIR, 3), dtype=np.complex128)
        for ell in range(LMIN, LMAX_PAIR + 1):
            a1 = rows[ell]
            a2c = np.conj(rows[ell + 1])
            coeff, idx2 = cg[ell]
            for mi in range(3):
                cross[ell - LMIN, mi] = np.sum(a1 * a2c[idx2[mi]] * coeff[mi])
        cross[~np.isfinite(cross)] = 0.0
        return cl, cross

    def _smooth_fid_from_cl(cl):
        cfid = np.ones(LMAX + 1, dtype=np.float64)
        vals = np.asarray(cl[LMIN:LMAX + 1], dtype=np.float64)
        good = vals[np.isfinite(vals) & (vals > 0.0)]
        if good.size == 0:
            return cfid
        floor = max(float(np.median(good)) * 1.0e-12, EPS)
        logs = np.log(np.maximum(vals, floor))
        sm = np.empty_like(logs)
        n = int(logs.size)
        for i in range(n):
            lo = max(0, i - 2)
            hi = min(n, i + 3)
            sm[i] = float(np.mean(logs[lo:hi]))
        cfid[LMIN:LMAX + 1] = np.exp(sm)
        return cfid

    def _stabilize_fid(cfid):
        arr = np.asarray(cfid, dtype=np.float64).ravel()
        if arr.size < LMAX + 1:
            tmp = np.ones(LMAX + 1, dtype=np.float64)
            tmp[:arr.size] = arr
            arr = tmp
        else:
            arr = arr[:LMAX + 1].copy()
        good = arr[LMIN:LMAX + 1][np.isfinite(arr[LMIN:LMAX + 1]) & (arr[LMIN:LMAX + 1] > 0.0)]
        floor = max(float(np.median(good)) * 1.0e-12, EPS) if good.size else EPS
        arr = np.where(np.isfinite(arr) & (arr > floor), arr, floor)
        return arr

    def _B_from_cross(cross, cfid):
        cfid = _stabilize_fid(cfid)
        B = np.zeros(3, dtype=np.complex128)
        for ell in range(LMIN, LMAX_PAIR + 1):
            den = math.sqrt(max(float(cfid[ell]), EPS) * max(float(cfid[ell + 1]), EPS))
            if np.isfinite(den) and den > 0.0:
                B += np.asarray(cross[ell - LMIN], dtype=np.complex128) / den
        B /= WNORM
        B[~np.isfinite(B)] = 0.0
        return B

    try:
        if np.ma.isMaskedArray(m):
            arr = np.asarray(m.filled(np.nan), dtype=np.float64).ravel()
        else:
            arr = np.asarray(m, dtype=np.float64).ravel()
    except Exception:
        try:
            arr = np.asarray(np.real(m), dtype=np.float64).ravel()
        except Exception:
            return 0.0

    npix = int(arr.size)
    nside = _infer_nside(npix)
    if nside <= 0:
        return 0.0

    # The official mask is represented by NaN.  The large-amplitude guard also
    # avoids accidentally treating HEALPix UNSEEN values as real sky pixels.
    valid = np.isfinite(arr) & (np.abs(arr) < 1.0e25)
    if int(np.count_nonzero(valid)) < 16:
        return 0.0

    try:
        import healpy as hp
        clean = _remove_monopole_dipole(arr, valid, nside, npix)
        # Never pass NaNs to healpy.  clean is finite, with zeros outside the mask.
        raw_alm = hp.map2alm(clean, lmax=LMAX, iter=0, pol=False, use_weights=False)
        cl, cross = _compute_cl_and_cross(raw_alm)
        cfid = _smooth_fid_from_cl(cl)
        B = _B_from_cross(cross, cfid)
        scalar = float(np.sqrt(np.sum(np.abs(B) ** 2)))
        if not np.isfinite(scalar):
            scalar = 0.0
        scalar = float(max(0.0, scalar))

        # Store compact ingredients so summarize_results can reconstruct the
        # requested ensemble-calibrated S using the actual simulation mean C_l and
        # the simulation mean B_M when all maps have been evaluated.
        try:
            rec_index = int(_state.get("call_index", 0))
            _state["call_index"] = rec_index + 1
            records = _state.get("raw_records")
            if not isinstance(records, list):
                records = []
                _state["raw_records"] = records
            records.append({
                "index": rec_index,
                "cl": np.asarray(cl, dtype=np.float64).copy(),
                "cross": np.asarray(cross, dtype=np.complex128).copy(),
                "B_self": np.asarray(B, dtype=np.complex128).copy(),
                "scalar": scalar,
            })
            if len(records) > 5000:
                del records[:len(records) - 5000]
        except Exception:
            pass
        return scalar
    except Exception:
        return 0.0


def summarize_results(planck_stat, sim_results, _state=_BIPOLAR_DM_STATE, _desc=test_description):
    import math
    import numpy as np

    LMIN = 2
    LMAX = 64
    LMAX_PAIR = 63
    WNORM = float(sum(2 * ell + 1 for ell in range(LMIN, LMAX_PAIR + 1)))
    EPS = 1.0e-30

    def _as_float(x):
        try:
            v = float(x)
            return v if np.isfinite(v) else np.nan
        except Exception:
            return np.nan

    ps = _as_float(planck_stat)
    try:
        sims_all = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            for item in sim_results:
                v = _as_float(item)
                if np.isfinite(v):
                    vals.append(v)
        except Exception:
            pass
        sims_all = np.asarray(vals, dtype=np.float64)
    sims = sims_all[np.isfinite(sims_all)]
    n = int(sims.size)

    out = {
        "test_description": _desc,
        "statistic": "S = sqrt(sum_{M=-1}^{1} |B_M - <B_M>_sim|^2), where B_M is an adjacent-multipole masked pseudo-alm dipole-coupling over 2<=ell<=63. analyze_map returns a finite scalar per map; when cached ensemble components are available, this summary reconstructs the exact simulation-calibrated S values.",
        "tail": "upper; larger S means stronger low-ell dipole-modulation / hemispherical-power-asymmetry coupling",
        "mask_handling": "Non-finite pixels are excluded from the monopole/dipole fit and are filled with 0 before healpy.map2alm.",
        "harmonic_range": {"lmin": 2, "lmax": 64, "coupled_pairs_lmin": 2, "coupled_pairs_lmax": 63},
        "planck_S_returned_by_analyze_map": float(ps) if np.isfinite(ps) else None,
        "n_sims": n,
        "plot_spec": {
            "kind": "histogram",
            "title": "Masked low-l bipolar dipole-modulation amplitude",
            "xlabel": "S (adjacent-l dipole coupling amplitude)",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "Simulation distribution for the scalar returned by analyze_map. If cached harmonic components are available, the table also reports the preferred ensemble-calibrated version using the simulation mean pseudo-C_l and simulation mean B_M."
        }
    }

    def _rank_summary(value, sample):
        sample = np.asarray(sample, dtype=np.float64).ravel()
        sample = sample[np.isfinite(sample)]
        nn = int(sample.size)
        if nn <= 0 or not np.isfinite(value):
            return {}
        n_lt = int(np.count_nonzero(sample < value))
        n_gt = int(np.count_nonzero(sample > value))
        n_eq = int(nn - n_lt - n_gt)
        n_le = int(n_lt + n_eq)
        n_ge = int(n_gt + n_eq)
        p_upper = float((n_ge + 1.0) / (nn + 1.0))
        p_lower = float((n_le + 1.0) / (nn + 1.0))
        p_two = float(min(1.0, 2.0 * min(p_upper, p_lower)))
        mean = float(np.mean(sample))
        median = float(np.median(sample))
        std = float(np.std(sample, ddof=1)) if nn > 1 else 0.0
        q = np.percentile(sample, [1, 5, 16, 50, 84, 95, 99]) if nn > 1 else np.array([sample[0]] * 7)
        return {
            "sim_mean": mean,
            "sim_median": median,
            "sim_std": std,
            "sim_min": float(np.min(sample)),
            "sim_max": float(np.max(sample)),
            "sim_percentile_1": float(q[0]),
            "sim_percentile_5": float(q[1]),
            "sim_percentile_16": float(q[2]),
            "sim_percentile_50": float(q[3]),
            "sim_percentile_84": float(q[4]),
            "sim_percentile_95": float(q[5]),
            "sim_percentile_99": float(q[6]),
            "zscore_vs_sims": float((value - mean) / std) if std > 0.0 and np.isfinite(std) else None,
            "n_sim_lt_planck": n_lt,
            "n_sim_le_planck": n_le,
            "n_sim_eq_planck": n_eq,
            "n_sim_gt_planck": n_gt,
            "n_sim_ge_planck": n_ge,
            "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / nn),
            "upper_tail_rank_descending_1_is_largest": int(n_gt + 1),
            "upper_tail_midrank_descending_1_is_largest": float(n_gt + 0.5 * n_eq + 1.0),
            "fraction_sims_ge_planck_upper_tail_including_ties": float(n_ge / nn),
            "fraction_sims_gt_planck_strict_upper_tail": float(n_gt / nn),
            "empirical_p_upper_tail_plus_one": p_upper,
            "empirical_p_lower_tail_plus_one": p_lower,
            "empirical_p_two_sided_rank_plus_one": p_two,
            "is_5pct_upper_tail": bool(p_upper <= 0.05),
            "is_1pct_upper_tail": bool(p_upper <= 0.01),
            "is_5pct_two_sided_rank": bool(p_two <= 0.05),
            "is_1pct_two_sided_rank": bool(p_two <= 0.01),
        }

    if np.isfinite(ps) and n > 0:
        basic = _rank_summary(ps, sims)
        for k, v in basic.items():
            out["returned_" + k] = v
    elif not np.isfinite(ps):
        out["warning"] = "The Planck statistic supplied to summarize_results is not finite."
        return out
    else:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    def _close(a, b):
        try:
            aa = float(a); bb = float(b)
            if not (np.isfinite(aa) and np.isfinite(bb)):
                return False
            return abs(aa - bb) <= 1.0e-8 * max(1.0, abs(aa), abs(bb))
        except Exception:
            return False

    def _stabilize_fid(cfid):
        arr = np.asarray(cfid, dtype=np.float64).ravel()
        if arr.size < LMAX + 1:
            tmp = np.ones(LMAX + 1, dtype=np.float64)
            tmp[:arr.size] = arr
            arr = tmp
        else:
            arr = arr[:LMAX + 1].copy()
        good = arr[LMIN:LMAX + 1][np.isfinite(arr[LMIN:LMAX + 1]) & (arr[LMIN:LMAX + 1] > 0.0)]
        floor = max(float(np.median(good)) * 1.0e-12, EPS) if good.size else EPS
        arr = np.where(np.isfinite(arr) & (arr > floor), arr, floor)
        return arr

    def _B_from_cross(cross, cfid):
        cfid = _stabilize_fid(cfid)
        cr = np.asarray(cross, dtype=np.complex128)
        if cr.shape[0] < (LMAX_PAIR - LMIN + 1) or cr.shape[1] < 3:
            return None
        B = np.zeros(3, dtype=np.complex128)
        for ell in range(LMIN, LMAX_PAIR + 1):
            den = math.sqrt(max(float(cfid[ell]), EPS) * max(float(cfid[ell + 1]), EPS))
            if np.isfinite(den) and den > 0.0:
                B += cr[ell - LMIN, :3] / den
        B /= WNORM
        B[~np.isfinite(B)] = 0.0
        return B

    # Preferred reconstruction of the exact requested statistic, if analyze_map
    # was run in this interpreter and stored the compact harmonic ingredients.
    try:
        records0 = _state.get("raw_records", [])
        records = []
        if isinstance(records0, list):
            for r in records0:
                if isinstance(r, dict) and ("cl" in r) and ("cross" in r):
                    records.append(r)
        if n > 0 and len(records) >= n + 1:
            # Locate the block corresponding to the supplied planck_stat followed
            # by the supplied simulations.  This avoids confusion from any probe
            # calls made during registration.
            best_score = -1
            best_start = max(0, len(records) - (n + 1))
            check_js = list(range(min(n, 10)))
            if n > 15:
                check_js += list(range(max(10, n - 5), n))
            for start in range(0, len(records) - n):
                score = 0
                if _close(records[start].get("scalar", np.nan), ps):
                    score += 1000
                for j in check_js:
                    if _close(records[start + 1 + j].get("scalar", np.nan), sims[j]):
                        score += 1
                if score > best_score:
                    best_score = score
                    best_start = start
            block = records[best_start:best_start + n + 1]
            if len(block) == n + 1:
                planck_rec = block[0]
                sim_recs = block[1:]
                cls = []
                for r in sim_recs:
                    c = np.asarray(r.get("cl"), dtype=np.float64).ravel()
                    if c.size >= LMAX + 1:
                        cls.append(c[:LMAX + 1])
                if len(cls) == n:
                    cfid = _stabilize_fid(np.mean(np.vstack(cls), axis=0))
                    B_planck = _B_from_cross(planck_rec.get("cross"), cfid)
                    B_sims_list = []
                    for r in sim_recs:
                        b = _B_from_cross(r.get("cross"), cfid)
                        if b is not None:
                            B_sims_list.append(b)
                    if B_planck is not None and len(B_sims_list) == n:
                        B_sims = np.vstack(B_sims_list)
                        B_mean = np.mean(B_sims, axis=0)
                        cal_planck = float(np.sqrt(np.sum(np.abs(B_planck - B_mean) ** 2)))
                        cal_sims = np.sqrt(np.sum(np.abs(B_sims - B_mean[None, :]) ** 2, axis=1)).astype(np.float64)
                        if np.isfinite(cal_planck) and np.all(np.isfinite(cal_sims)):
