Hypothesis

The Planck CMB map has an anomalously cold fixed-scale R=5° SMHW spot, meaning its minimum standardized wavelet coefficient is more negative than expected from LCDM simulations.

Method

Planck and 1000 LCDM simulations were monopole/dipole-subtracted on the unmasked sky, filtered once with an R=5° spherical Mexican-hat wavelet, restricted to pixels with ≥90% effective mask support, standardized over valid pixels, and summarized by the minimum standardized coefficient.

Result Summary

Planck statistic = -4.595; simulations have mean ± std = -4.692 ± 0.261. Since more negative values are colder, Planck is not extreme: 61.2% of simulations are as cold or colder, giving one-tailed lower p = 0.612 and only ~0.37 simulation-standard-deviations on the warm side of the mean.

This fixed-scale extremum does not support an anomalously cold Planck SMHW spot; the measured Planck value lies well inside the LCDM simulation distribution and far from the 5% cold tail.

Agent Justification

The test is explicitly a cold-spot extremum test; the statistic is the minimum z-scored SMHW coefficient, and the code defines the tail of interest as the lower tail where more negative values are colder/more extreme.

Comparison To Literature

SMHW Cold Spot anomalies are well documented: Cruz et al. reported only ~0.2% of Gaussian simulations with such a WMAP cold spot (arXiv:astro-ph/0405341), and Cruz et al. 2006 found wavelet-space significance below 1%, including 0.69% for kurtosis (arXiv:astro-ph/0601427). This test differs by using a single fixed 5° scale and a minimum standardized coefficient, yielding no anomaly.

import numpy as np

test_description = "Fixed-Scale SMHW Cold-Spot Extremum: remove the unmasked-sky monopole and dipole from each masked nside=256 temperature map, convolve the finite sky with a spherical Mexican-hat wavelet at the single fixed scale R=5 degrees, retain only pixels whose absolute-wavelet effective mask support is at least 90%, standardize the filtered coefficients over those valid pixels, and return the minimum standardized coefficient, i.e. the most negative fixed-scale cold-spot amplitude."


def analyze_map(m):
    import numpy as _np
    try:
        import healpy as _hp
    except Exception:
        _hp = None

    _R_DEG = 5.0
    _R_RAD = float(_np.deg2rad(_R_DEG))
    _SUPPORT_THRESHOLD = 0.90
    _LMAX_CAP = 128
    _NQUAD_MIN = 1024
    _ABS_CUT = 1.0e20
    _CACHE_MAX = 4

    def _as_float_map(x):
        try:
            if _np.ma.isMaskedArray(x):
                y = _np.asarray(x.filled(_np.nan), dtype=_np.float64).ravel()
            else:
                y = _np.asarray(x, dtype=_np.float64).ravel()
            return y
        except Exception:
            try:
                y = _np.real(_np.asarray(x)).astype(_np.float64).ravel()
                return y
            except Exception:
                y = _np.asarray([], dtype=_np.float64)
                return y

    def _finite_mask(a):
        a = _np.asarray(a).ravel()
        good = _np.isfinite(a)
        try:
            with _np.errstate(invalid="ignore", over="ignore"):
                good = _np.logical_and(good, _np.abs(a) < _ABS_CUT)
        except Exception:
            pass
        good = _np.asarray(good, dtype=bool)
        return good

    def _fallback_stat(a):
        a = _as_float_map(a)
        good = _finite_mask(a)
        vals = _np.asarray(a[good], dtype=_np.float64)
        vals = vals[_np.isfinite(vals)]
        if vals.size < 2:
            return 0.0
        vals = vals - float(_np.mean(vals))
        sd = float(_np.std(vals))
        if not _np.isfinite(sd) or sd <= 0.0:
            return 0.0
        out = float(_np.min(vals) / sd)
        if not _np.isfinite(out):
            return 0.0
        return out

    def _pinv_sym(a):
        a = _np.asarray(a, dtype=_np.float64)
        try:
            a = 0.5 * (a + a.T)
            w, v = _np.linalg.eigh(a)
            scale = float(_np.max(_np.abs(w))) if w.size else 0.0
            if not _np.isfinite(scale) or scale <= 0.0:
                z = _np.zeros_like(a)
                return z
            keep = _np.abs(w) > 1.0e-12 * scale
            inv = _np.zeros_like(w)
            inv[keep] = 1.0 / w[keep]
            out = _np.dot(v * inv, v.T)
            return out
        except Exception:
            try:
                out = _np.linalg.pinv(a, rcond=1.0e-12)
                return out
            except Exception:
                z = _np.zeros_like(a)
                return z

    def _get_cache():
        if not hasattr(analyze_map, "_fixed_smhw_cache_v2"):
            analyze_map._fixed_smhw_cache_v2 = {"beam": {}, "geom": {}}
        cache = analyze_map._fixed_smhw_cache_v2
        return cache

    def _mask_key(nside, lmax, valid, pix):
        nvalid = int(pix.size)
        if nvalid <= 0:
            sample_sum = 0
            first = -1
            last = -1
        else:
            ntake = min(64, nvalid)
            take = _np.linspace(0, nvalid - 1, ntake).astype(_np.int64)
            sample = pix[take].astype(_np.int64, copy=False)
            sample_sum = int(_np.sum(sample % 1000003))
            first = int(pix[0])
            last = int(pix[-1])
        key = (int(nside), int(valid.size), int(lmax), nvalid, first, last, sample_sum)
        return key

    def _smhw_profile_from_mu(mu, R):
        mu = _np.asarray(mu, dtype=_np.float64)
        psi = _np.zeros_like(mu, dtype=_np.float64)
        den = 1.0 + mu
        ok = den > 1.0e-15
        if not _np.any(ok):
            return psi
        q = (1.0 - mu[ok]) / den[ok]
        y2 = 4.0 * q
        u2 = y2 / (R * R)
        close = _np.isfinite(u2) & (u2 < 745.0)
        if _np.any(close):
            y2c = y2[close]
            u2c = u2[close]
            norm_den = _np.sqrt(2.0 * _np.pi) * R * _np.sqrt(1.0 + 0.5 * R * R + 0.25 * R ** 4)
            norm = 1.0 / norm_den
            val = norm * (1.0 + 0.25 * y2c) ** 2 * (2.0 - u2c) * _np.exp(-0.5 * u2c)
            tmp = _np.zeros_like(q)
            tmp[close] = val
            psi[ok] = tmp
        psi[~_np.isfinite(psi)] = 0.0
        return psi

    def _compute_beams(lmax):
        cache = _get_cache()["beam"]
        key = (int(lmax), float(_R_RAD))
        got = cache.get(key)
        if got is not None:
            return got

        nquad = int(max(_NQUAD_MIN, 4 * int(lmax) + 256))
        try:
            mu, wt = _np.polynomial.legendre.leggauss(nquad)
        except Exception:
            nquad = int(max(512, 2 * int(lmax) + 128))
            mu, wt = _np.polynomial.legendre.leggauss(nquad)

        psi = _smhw_profile_from_mu(mu, _R_RAD)
        g = _np.abs(psi)
        beam = _np.zeros(int(lmax) + 1, dtype=_np.float64)
        support_beam = _np.zeros(int(lmax) + 1, dtype=_np.float64)

        p0 = _np.ones_like(mu)
        beam[0] = 2.0 * _np.pi * float(_np.sum(wt * psi * p0))
        support_beam[0] = 2.0 * _np.pi * float(_np.sum(wt * g * p0))
        if int(lmax) >= 1:
            p1 = mu.copy()
            beam[1] = 2.0 * _np.pi * float(_np.sum(wt * psi * p1))
            support_beam[1] = 2.0 * _np.pi * float(_np.sum(wt * g * p1))
            pm2 = p0
            pm1 = p1
            for ell in range(2, int(lmax) + 1):
                pell = ((2.0 * ell - 1.0) * mu * pm1 - (ell - 1.0) * pm2) / float(ell)
                beam[ell] = 2.0 * _np.pi * float(_np.sum(wt * psi * pell))
                support_beam[ell] = 2.0 * _np.pi * float(_np.sum(wt * g * pell))
                pm2 = pm1
                pm1 = pell

        beam[0] = 0.0
        if beam.size > 1:
            mx = float(_np.max(_np.abs(beam[1:])))
        else:
            mx = 0.0
        if _np.isfinite(mx) and mx > 0.0:
            beam = beam / mx

        ell = _np.arange(int(lmax) + 1, dtype=_np.float64)
        central = float(_np.sum((2.0 * ell + 1.0) * beam) / (4.0 * _np.pi))
        if _np.isfinite(central) and central < 0.0:
            beam = -beam

        norm = float(support_beam[0])
        if not _np.isfinite(norm) or norm <= 0.0:
            ell = _np.arange(int(lmax) + 1, dtype=_np.float64)
            sigma = _R_RAD
            support_beam = _np.exp(-0.5 * ell * (ell + 1.0) * sigma * sigma)
            support_beam[0] = 1.0
        else:
            support_beam = support_beam / norm
            support_beam[0] = 1.0

        beam[~_np.isfinite(beam)] = 0.0
        support_beam[~_np.isfinite(support_beam)] = 0.0
        got = {"wave": beam.astype(_np.float64), "support": support_beam.astype(_np.float64)}
        cache[key] = got
        return got

    def _map2alm_safe(mp, lmax):
        if _hp is None:
            raise RuntimeError("healpy is not available")
        mp = _np.asarray(mp, dtype=_np.float64)
        mp = _np.where(_np.isfinite(mp), mp, 0.0)
        try:
            alm = _hp.map2alm(mp, lmax=int(lmax), iter=0, pol=False, use_weights=False)
            return alm
        except TypeError:
            try:
                alm = _hp.map2alm(mp, lmax=int(lmax), iter=0, pol=False)
                return alm
            except TypeError:
                alm = _hp.map2alm(mp, lmax=int(lmax), iter=0)
                return alm

    def _alm2map_safe(alm, nside, lmax):
        if _hp is None:
            raise RuntimeError("healpy is not available")
        try:
            mp = _hp.alm2map(alm, int(nside), lmax=int(lmax), pol=False, verbose=False)
            return mp
        except TypeError:
            try:
                mp = _hp.alm2map(alm, int(nside), lmax=int(lmax), verbose=False)
                return mp
            except TypeError:
                mp = _hp.alm2map(alm, int(nside), lmax=int(lmax))
                return mp

    def _almxfl_safe(alm, fl):
        try:
            out = _hp.almxfl(alm, fl, inplace=False)
            return out
        except TypeError:
            out = _hp.almxfl(alm, fl)
            return out

    def _support_pixels(nside, valid, lmax, support_beam):
        mask_map = _np.zeros(int(valid.size), dtype=_np.float64)
        mask_map[valid] = 1.0
        alm = _map2alm_safe(mask_map, int(lmax))
        salm = _almxfl_safe(alm, support_beam)
        support = _alm2map_safe(salm, int(nside), int(lmax))
        support = _np.asarray(support, dtype=_np.float64).ravel()
        good = valid & _np.isfinite(support) & (support >= _SUPPORT_THRESHOLD)
        good = _np.asarray(good, dtype=bool)
        return {"good": good, "support": support}

    def _get_geometry(nside, valid, lmax, support_beam):
        cache = _get_cache()["geom"]
        pix = _np.flatnonzero(valid).astype(_np.int64, copy=False)
        key = _mask_key(nside, lmax, valid, pix)
        entries = cache.get(key, [])
        for geom in entries:
            try:
                if _np.array_equal(geom.get("valid"), valid):
                    return geom
            except Exception:
                pass

        nvalid = int(pix.size)
        x, y, z = _hp.pix2vec(int(nside), pix)
        x = _np.asarray(x, dtype=_np.float64)
        y = _np.asarray(y, dtype=_np.float64)
        z = _np.asarray(z, dtype=_np.float64)

        gram = _np.empty((4, 4), dtype=_np.float64)
        sx = float(_np.sum(x))
        sy = float(_np.sum(y))
        sz = float(_np.sum(z))
        gram[0, 0] = float(nvalid)
        gram[0, 1] = sx
        gram[1, 0] = sx
        gram[0, 2] = sy
        gram[2, 0] = sy
        gram[0, 3] = sz
        gram[3, 0] = sz
        gram[1, 1] = float(_np.dot(x, x))
        gram[1, 2] = float(_np.dot(x, y))
        gram[2, 1] = gram[1, 2]
        gram[1, 3] = float(_np.dot(x, z))
        gram[3, 1] = gram[1, 3]
        gram[2, 2] = float(_np.dot(y, y))
        gram[2, 3] = float(_np.dot(y, z))
        gram[3, 2] = gram[2, 3]
        gram[3, 3] = float(_np.dot(z, z))
        pinv = _pinv_sym(gram)

        sp = _support_pixels(int(nside), valid, int(lmax), support_beam)
        valid_support = sp["good"]
        geom = {
            "nside": int(nside),
            "npix": int(valid.size),
            "lmax": int(lmax),
            "valid": valid.copy(),
            "pix": pix,
            "x": x,
            "y": y,
            "z": z,
            "pinv": pinv,
            "valid_support": valid_support,
            "n_support": int(_np.count_nonzero(valid_support))
        }
        cache[key] = [geom] + entries[:max(0, _CACHE_MAX - 1)]
        return geom

    arr = _as_float_map(m)
    npix = int(arr.size)
    if npix <= 0:
        return 0.0
    valid = _finite_mask(arr)
    nvalid = int(_np.count_nonzero(valid))
    if nvalid < 8:
        val = _fallback_stat(arr)
        return float(val)
    if _hp is None:
        val = _fallback_stat(arr)
        return float(val)

    try:
        nside = int(_hp.npix2nside(npix))
        if nside <= 0 or 12 * nside * nside != npix:
            val = _fallback_stat(arr)
            return float(val)
    except Exception:
        val = _fallback_stat(arr)
        return float(val)

    lmax_full = int(3 * nside - 1)
    lmax = int(max(1, min(lmax_full, _LMAX_CAP)))

    try:
        beams = _compute_beams(lmax)
        wave_beam = beams["wave"]
        support_beam = beams["support"]
        geom = _get_geometry(nside, valid, lmax, support_beam)
        pix = geom["pix"]
        if int(pix.size) < 8:
            val = _fallback_stat(arr)
            return float(val)

        vals = _np.asarray(arr[pix], dtype=_np.float64)
        finite_vals = _np.isfinite(vals)
        if not _np.all(finite_vals):
            vals = _np.where(finite_vals, vals, 0.0)

        x = geom["x"]
        y = geom["y"]
        z = geom["z"]
        rhs = _np.array([
            float(_np.sum(vals)),
            float(_np.dot(x, vals)),
            float(_np.dot(y, vals)),
            float(_np.dot(z, vals))
        ], dtype=_np.float64)
        coeff = _np.dot(geom["pinv"], rhs)
        if not _np.all(_np.isfinite(coeff)):
            coeff = _np.zeros(4, dtype=_np.float64)

        residual_vals = vals - (coeff[0] + coeff[1] * x + coeff[2] * y + coeff[3] * z)
        residual_vals = _np.where(_np.isfinite(residual_vals), residual_vals, 0.0)
        filled = _np.zeros(npix, dtype=_np.float64)
        filled[pix] = residual_vals

        alm = _map2alm_safe(filled, lmax)
        walm = _almxfl_safe(alm, wave_beam)
        filt = _alm2map_safe(walm, nside, lmax)
        filt = _np.asarray(filt, dtype=_np.float64).ravel()

        good = _np.asarray(geom["valid_support"], dtype=bool)
        if good.size != filt.size:
            good = valid & _np.isfinite(filt)
        else:
            good = good & _np.isfinite(filt)
        if int(_np.count_nonzero(good)) < 2:
            good = valid & _np.isfinite(filt)
        if int(_np.count_nonzero(good)) < 2:
            val = _fallback_stat(arr)
            return float(val)

        coeffs = _np.asarray(filt[good], dtype=_np.float64)
        coeffs = coeffs[_np.isfinite(coeffs)]
        if coeffs.size < 2:
            val = _fallback_stat(arr)
            return float(val)
        mu = float(_np.mean(coeffs))
        sd = float(_np.std(coeffs))
        if not _np.isfinite(sd) or sd <= 0.0:
            return 0.0
        mn = float(_np.min(coeffs))
        stat = float((mn - mu) / sd)
        if not _np.isfinite(stat):
            return 0.0
        return stat
    except Exception:
        val = _fallback_stat(arr)
        return float(val)


def summarize_results(planck_stat, sim_results):
    import numpy as _np

    desc = "Fixed-Scale SMHW Cold-Spot Extremum: remove the unmasked-sky monopole and dipole, filter once with an R=5 degree spherical Mexican-hat wavelet, keep pixels with at least 90% absolute-wavelet mask support, z-score the filtered coefficients on those pixels, and use the minimum z-score. More negative values are colder/more extreme."

    def _finite_float(x, default=float("nan")):
        try:
            y = float(x)
        except Exception:
            return float(default)
        if not _np.isfinite(y):
            return float(default)
        return float(y)

    ps = _finite_float(planck_stat)
    sims = _np.asarray(sim_results, dtype=_np.float64).ravel()
    sims = sims[_np.isfinite(sims)]
    n = int(sims.size)

    out = {
        "test_description": desc,
        "planck_stat": ps,
        "n_sims": n,
        "tail_of_interest": "lower tail: more negative minimum standardized SMHW coefficient means a colder fixed-scale spot",
        "plot_spec": {
            "kind": "histogram",
            "title": "Fixed-scale R=5 deg SMHW cold-spot extremum",
            "xlabel": "min z-scored SMHW coefficient over valid pixels",
            "ylabel": "Number of simulations",
            "bins": 30,
            "caption": "Simulation distribution of the most negative standardized spherical Mexican-hat wavelet coefficient at the single fixed scale R=5 degrees. The Planck value is anomalously cold if it lies far in the left tail."
        }
    }

    if not _np.isfinite(ps):
        out["warning"] = "Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics are available."
        return out

    n_lt = int(_np.count_nonzero(sims < ps))
    n_le = int(_np.count_nonzero(sims <= ps))
    n_gt = int(_np.count_nonzero(sims > ps))
    n_ge = int(_np.count_nonzero(sims >= ps))
    n_eq = int(n - n_lt - n_gt)

    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_two = float(min(1.0, 2.0 * min(p_lower, p_upper)))
    midrank_percentile = float(100.0 * (n_lt + 0.5 * n_eq) / n)

    mean = float(_np.mean(sims))
    median = float(_np.median(sims))
    std = float(_np.std(sims, ddof=1)) if n > 1 else 0.0
    z_vs_sims = None
    if _np.isfinite(std) and std > 0.0:
        z_vs_sims = float((ps - mean) / std)
    qs = _np.percentile(sims, [0.5, 1.0, 5.0, 16.0, 50.0, 84.0, 95.0, 99.0, 99.5])

    out.update({
        "sim_mean": mean,
        "sim_median": median,
        "sim_std": std,
        "sim_min": float(_np.min(sims)),
        "sim_max": float(_np.max(sims)),
        "sim_percentile_0_5": float(qs[0]),
        "sim_percentile_1": float(qs[1]),
        "sim_percentile_5": float(qs[2]),
        "sim_percentile_16": float(qs[3]),
        "sim_percentile_50": float(qs[4]),
        "sim_percentile_84": float(qs[5]),
        "sim_percentile_95": float(qs[6]),
        "sim_percentile_99": float(qs[7]),
        "sim_percentile_99_5": float(qs[8]),
        "planck_zscore_vs_sim_distribution": z_vs_sims,
        "planck_percentile_midrank": midrank_percentile,
        "rank_ascending_1_is_coldest": int(n_lt + 1),
        "rank_ascending_tie_low": int(n_lt + 1),
        "rank_ascending_tie_high": int(n_le + 1),
        "rank_convention": "Ascending rank 1 is the most negative/coldest statistic; tie_high includes simulations tied with Planck.",
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "empirical_p_lower_cold_tail": p_lower,
        "empirical_p_upper_warm_tail": p_upper,
        "empirical_p_two_sided_extremeness": p_two,
        "is_cold_tail_5pct": bool(p_lower <= 0.05),
        "is_cold_tail_1pct": bool(p_lower <= 0.01)
    })

    if std == 0.0:
        out["warning"] = "All finite simulation statistics are identical; check for degenerate inputs or a failed filtering pipeline."
    elif p_lower <= 0.01:
        out["interpretation"] = "Planck lies in the lower 1% cold-tail of the fixed-scale SMHW extrema from the simulations."
    elif p_lower <= 0.05:
        out["interpretation"] = "Planck lies in the lower 5% cold-tail of the fixed-scale SMHW extrema from the simulations."
    else:
        out["interpretation"] = "Planck is not in the lower 5% cold-tail for this fixed-scale SMHW statistic."
    return out
