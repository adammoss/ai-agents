Consistent GPT5.5 · Variation · parity · one-tailed, upper tail

Hypothesis

SMICA has an unusually large optimized low-\(\ell\) mirror-parity power imbalance between mirror-even and mirror-odd modes compared with LCDM simulations.

Method

SMICA and 1000 matched LCDM simulations were masked, monopole/dipole-subtracted, fitted for 2<=ell<=20 coefficients, rotated over an antipodally identified nside=8 plane grid, and scored with S=max_n |ln[(P_even+epsilon)/(P_odd+epsilon)]| for l+m mirror parity.

Result Summary

SMICA S=0.8446 versus simulation mean 0.9600, median 0.9394, std 0.1711. Its midrank percentile is 26.9%; 731/1000 simulations have larger S, giving an upper-tail p=0.731, about -0.67 sigma relative to the simulation mean.

The observed optimized mirror-even/mirror-odd low-l power imbalance is not unusually large; it is below the typical simulation value and far from the upper 5% anomaly region.

Agent Justification

The statistic \(S=\max_n |\log((P_{\rm even}+\epsilon)/(P_{\rm odd}+\epsilon))|\) measures the magnitude of the imbalance, with the sign already removed. The code explicitly treats larger \(S\) as more anomalous and evaluates the upper tail, so only unusually high values support the anomaly hypothesis.

Comparison To Literature

Broad low-l parity anomalies are discussed in Planck Collaboration XVI (2016, A&A 594, A16) and VII (2020, A&A 641, A7), but the supplied/previous searches found no directly comparable published test using this exact optimized log-imbalance, mask-fitting pipeline, ell range and grid.

test_description = "Low-l Mirror-Parity Plane Log-Imbalance: after masked monopole/dipole subtraction, estimate the 2<=ell<=20 temperature multipoles with the same finite-filled masked-sky pipeline for SMICA and simulations; on a fixed antipodally identified HEALPix nside=8 grid of plane normals, split the low-l field into mirror-even and mirror-odd parts for each plane and return S=max_n |ln((P_even+epsilon)/(P_odd+epsilon))|."


def analyze_map(m, _cache={}):
    import math
    import zlib
    import numpy as np

    EPSILON = 1.0e-30
    LMAX_REQUESTED = 20
    AXIS_NSIDE = 8

    def _to_float_array(x):
        try:
            if np.ma.isMaskedArray(x):
                a = np.asarray(x.filled(np.nan), dtype=np.float64)
            else:
                a = np.asarray(x, dtype=np.float64)
        except Exception:
            try:
                a = np.asarray(np.real(x), dtype=np.float64)
            except Exception:
                a = np.asarray([], dtype=np.float64)
        return np.ravel(a)

    def _fallback_stat(x):
        a = _to_float_array(x)
        good = np.isfinite(a) & (np.abs(a) < 1.0e25)
        v = np.asarray(a[good], dtype=np.float64)
        if v.size < 2:
            return 0.0
        v = v - float(np.mean(v))
        v = np.where(np.isfinite(v), v, 0.0)
        n = int(v.size)
        ns = int(min(n, 4096))
        if ns < 2:
            return 0.0
        idx = np.linspace(0, n - 1, ns, dtype=np.int64)
        x0 = np.asarray(v[idx], dtype=np.float64)
        x0 = x0 - float(np.mean(x0))
        total = float(np.dot(x0, x0))
        if (not np.isfinite(total)) or total <= 0.0:
            return 0.0
        shifts = (0, max(1, ns // 7), max(1, ns // 3), max(1, ns // 2))
        best = 0.0
        for shift in shifts:
            y0 = np.roll(x0[::-1], int(shift))
            ep = 0.25 * float(np.dot(x0 + y0, x0 + y0))
            op = 0.25 * float(np.dot(x0 - y0, x0 - y0))
            if np.isfinite(ep) and np.isfinite(op) and ep >= 0.0 and op >= 0.0:
                with np.errstate(all="ignore"):
                    q = abs(math.log((ep + EPSILON) / (op + EPSILON)))
                if np.isfinite(q) and q > best:
                    best = float(q)
        if not np.isfinite(best):
            best = 0.0
        return float(min(max(best, 0.0), 1.0e6))

    def _mask_signature(valid):
        valid = np.asarray(valid, dtype=bool).ravel()
        try:
            packed = np.packbits(valid)
            crc = int(zlib.crc32(packed.tobytes()) & 0xffffffff)
        except Exception:
            crc = int(np.count_nonzero(valid)) & 0xffffffff
        return str(int(valid.size)) + ":" + str(int(np.count_nonzero(valid))) + ":" + str(crc)

    def _get_hp():
        if "hp_import_tried" not in _cache:
            _cache["hp_import_tried"] = True
            try:
                import healpy as hp
                _cache["hp"] = hp
            except Exception:
                _cache["hp"] = None
        return _cache.get("hp", None)

    def _nside_from_npix(hp, npix):
        try:
            nside0 = int(hp.npix2nside(int(npix)))
            if nside0 > 0 and int(hp.nside2npix(nside0)) == int(npix):
                return nside0
        except Exception:
            pass
        try:
            if int(npix) > 0 and int(npix) % 12 == 0:
                nside0 = int(round(math.sqrt(float(npix) / 12.0)))
                if nside0 > 0 and 12 * nside0 * nside0 == int(npix):
                    return nside0
        except Exception:
            pass
        return -1

    def _monopole_dipole_geometry(hp, nside, valid):
        sig = _mask_signature(valid)
        key = ("md_geom", int(nside), sig)
        cached = _cache.get(key)
        if cached is not None:
            return cached
        valid_pix = np.flatnonzero(valid).astype(np.int64, copy=False)
        if valid_pix.size < 32:
            return None
        x, y, z = hp.pix2vec(int(nside), valid_pix)
        x = np.asarray(x, dtype=np.float64)
        y = np.asarray(y, dtype=np.float64)
        z = np.asarray(z, dtype=np.float64)
        n = float(valid_pix.size)
        gram = np.empty((4, 4), dtype=np.float64)
        gram[0, 0] = n
        gram[0, 1] = float(np.sum(x));       gram[1, 0] = gram[0, 1]
        gram[0, 2] = float(np.sum(y));       gram[2, 0] = gram[0, 2]
        gram[0, 3] = float(np.sum(z));       gram[3, 0] = gram[0, 3]
        gram[1, 1] = float(np.dot(x, x))
        gram[1, 2] = float(np.dot(x, y));    gram[2, 1] = gram[1, 2]
        gram[1, 3] = float(np.dot(x, z));    gram[3, 1] = gram[1, 3]
        gram[2, 2] = float(np.dot(y, y))
        gram[2, 3] = float(np.dot(y, z));    gram[3, 2] = gram[2, 3]
        gram[3, 3] = float(np.dot(z, z))
        try:
            pinv = np.linalg.pinv(gram, rcond=1.0e-12)
        except TypeError:
            pinv = np.linalg.pinv(gram)
        except Exception:
            pinv = np.zeros((4, 4), dtype=np.float64)
            pinv[0, 0] = 1.0 / max(n, 1.0)
        geom = {"valid_pix": valid_pix, "x": x, "y": y, "z": z, "pinv": np.asarray(pinv, dtype=np.float64)}
        _cache[key] = geom
        return geom

    def _axis_vectors(hp, axis_nside):
        axis_nside = int(max(1, axis_nside))
        key = ("axes", axis_nside)
        cached = _cache.get(key)
        if cached is not None:
            return cached
        npix_axis = int(hp.nside2npix(axis_nside))
        pix = np.arange(npix_axis, dtype=np.int64)
        ax, ay, az = hp.pix2vec(axis_nside, pix)
        ax = np.asarray(ax, dtype=np.float64)
        ay = np.asarray(ay, dtype=np.float64)
        az = np.asarray(az, dtype=np.float64)
        tol = 1.0e-14
        keep = (az > tol) | ((np.abs(az) <= tol) & ((ay > tol) | ((np.abs(ay) <= tol) & (ax >= 0.0))))
        if int(np.count_nonzero(keep)) <= 0:
            keep = pix < max(1, npix_axis // 2)
        axes = np.vstack((ax[keep], ay[keep], az[keep])).T
        norm = np.sqrt(np.sum(axes * axes, axis=1))
        ok = np.isfinite(norm) & (norm > 0.0)
        if int(np.count_nonzero(ok)) <= 0:
            axes = np.array([[0.0, 0.0, 1.0]], dtype=np.float64)
        else:
            axes = np.asarray(axes[ok] / norm[ok, None], dtype=np.float64)
        _cache[key] = axes
        return axes

    def _interp_weights_for_xyz(hp, nside, x, y, z):
        zc = np.clip(np.asarray(z, dtype=np.float64), -1.0, 1.0)
        theta = np.arccos(zc)
        phi = np.mod(np.arctan2(np.asarray(y, dtype=np.float64), np.asarray(x, dtype=np.float64)), 2.0 * np.pi)
        try:
            pix4, w4 = hp.get_interp_weights(int(nside), theta, phi, nest=False)
        except TypeError:
            pix4, w4 = hp.get_interp_weights(int(nside), theta, phi)
        pix4 = np.asarray(pix4, dtype=np.int64)
        w4 = np.asarray(w4, dtype=np.float64)
        if pix4.ndim == 1:
            pix4 = pix4.reshape(4, -1)
        if w4.ndim == 1:
            w4 = w4.reshape(4, -1)
        if pix4.shape[0] != 4 and pix4.shape[-1] == 4:
            pix4 = pix4.T
        if w4.shape[0] != 4 and w4.shape[-1] == 4:
            w4 = w4.T
        return {"pix4": pix4, "w4": w4}

    def _reflection_geometry(hp, nside_eval, axis_nside):
        nside_eval = int(max(1, nside_eval))
        axis_nside = int(max(1, axis_nside))
        key = ("reflection_geom", nside_eval, axis_nside)
        cached = _cache.get(key)
        if cached is not None:
            return cached
        npix_eval = int(hp.nside2npix(nside_eval))
        pix = np.arange(npix_eval, dtype=np.int64)
        vx, vy, vz = hp.pix2vec(nside_eval, pix)
        vx = np.asarray(vx, dtype=np.float64)
        vy = np.asarray(vy, dtype=np.float64)
        vz = np.asarray(vz, dtype=np.float64)
        axes = _axis_vectors(hp, axis_nside)
        if axes is None or np.asarray(axes).size == 0:
            return None
        axes = np.asarray(axes, dtype=np.float64)
        naxes = int(axes.shape[0])
        vec = np.vstack((vx, vy, vz))
        pix_dtype = np.uint16 if npix_eval <= np.iinfo(np.uint16).max else np.uint32
        pix_chunks = []
        w_chunks = []
        target_points = 1200000
        chunk = int(max(1, min(naxes, target_points // max(1, npix_eval))))
        for i0 in range(0, naxes, chunk):
            aa = np.asarray(axes[i0:min(naxes, i0 + chunk)], dtype=np.float64)
            d = np.dot(aa, vec)
            xr = vx[None, :] - 2.0 * d * aa[:, 0:1]
            yr = vy[None, :] - 2.0 * d * aa[:, 1:2]
            zr = vz[None, :] - 2.0 * d * aa[:, 2:3]
            iw = _interp_weights_for_xyz(hp, nside_eval, xr.ravel(), yr.ravel(), zr.ravel())
            pix4 = iw["pix4"]
            w4 = iw["w4"]
            nflat = int(aa.shape[0]) * npix_eval
            if pix4.shape[0] != 4 or w4.shape[0] != 4 or pix4.shape[1] != nflat or w4.shape[1] != nflat:
                return None
            pix4 = np.clip(pix4, 0, npix_eval - 1).T.reshape(int(aa.shape[0]), npix_eval, 4)
            w4 = w4.T.reshape(int(aa.shape[0]), npix_eval, 4)
            pix_chunks.append(pix4.astype(pix_dtype, copy=False))
            w_chunks.append(w4.astype(np.float32, copy=False))
        if not pix_chunks:
            return None
        pix4_all = np.concatenate(pix_chunks, axis=0)
        w4_all = np.concatenate(w_chunks, axis=0)
        geom = {"pix4": pix4_all, "w4": w4_all, "naxes": int(pix4_all.shape[0]), "npix_eval": npix_eval}
        _cache[key] = geom
        return geom

    arr = _to_float_array(m)
    npix = int(arr.size)
    if npix <= 0:
        return 0.0
    valid = np.isfinite(arr) & (np.abs(arr) < 1.0e25)
    if int(np.count_nonzero(valid)) < 32:
        return float(_fallback_stat(arr))

    hp = _get_hp()
    if hp is None:
        return float(_fallback_stat(arr))
    nside = _nside_from_npix(hp, npix)
    if nside < 2:
        return float(_fallback_stat(arr))

    try:
        nside_eval = int(min(max(2, int(nside)), 8))
        lmax_use = int(min(LMAX_REQUESTED, 3 * int(nside) - 1, 3 * nside_eval - 1))
        if lmax_use < 2:
            return float(_fallback_stat(arr))

        md = _monopole_dipole_geometry(hp, nside, valid)
        if md is None:
            return float(_fallback_stat(arr))
        vp = md["valid_pix"]
        vals = np.asarray(arr[vp], dtype=np.float64)
        vals = np.where(np.isfinite(vals), vals, 0.0)
        rhs = np.array([
            float(np.sum(vals)),
            float(np.dot(md["x"], vals)),
            float(np.dot(md["y"], vals)),
            float(np.dot(md["z"], vals))
        ], dtype=np.float64)
        coeff = np.dot(md["pinv"], rhs)
        if coeff.shape[0] != 4 or not np.all(np.isfinite(coeff)):
            coeff = np.zeros(4, dtype=np.float64)
            coeff[0] = float(np.mean(vals)) if vals.size else 0.0
        residual = vals - (coeff[0] + coeff[1] * md["x"] + coeff[2] * md["y"] + coeff[3] * md["z"])
        residual = np.where(np.isfinite(residual), residual, 0.0)
        residual_power = float(np.dot(residual, residual))
        if (not np.isfinite(residual_power)) or residual_power <= 0.0:
            return 0.0

        filled = np.zeros(npix, dtype=np.float64)
        filled[vp] = residual
        filled = np.where(np.isfinite(filled), filled, 0.0)
        try:
            alm = hp.map2alm(filled, lmax=lmax_use, iter=0, pol=False, use_weights=False)
        except TypeError:
            try:
                alm = hp.map2alm(filled, lmax=lmax_use, iter=0, pol=False)
            except TypeError:
                alm = hp.map2alm(filled, lmax=lmax_use, iter=0)
        alm = np.asarray(alm, dtype=np.complex128)
        if alm.size <= 0:
            return 0.0
        for ell in range(0, min(2, lmax_use + 1)):
            for mm in range(ell + 1):
                try:
                    alm[hp.Alm.getidx(lmax_use, ell, mm)] = 0.0
                except Exception:
                    pass
        alm = np.where(np.isfinite(alm.real) & np.isfinite(alm.imag), alm, 0.0)
        alm_power = float(np.sum(np.abs(alm) ** 2))
        if (not np.isfinite(alm_power)) or alm_power <= 0.0:
            return 0.0

        try:
            low = hp.alm2map(alm, nside_eval, lmax=lmax_use, pol=False, verbose=False)
        except TypeError:
            try:
                low = hp.alm2map(alm, nside_eval, lmax=lmax_use, pol=False)
            except TypeError:
                low = hp.alm2map(alm, nside_eval, lmax=lmax_use)
        low = np.asarray(low, dtype=np.float64).ravel()
        low = np.where(np.isfinite(low), low, 0.0)
        if int(low.size) != int(hp.nside2npix(nside_eval)):
            return float(_fallback_stat(arr))
        total_power = float(np.dot(low, low))
        if (not np.isfinite(total_power)) or total_power <= 0.0:
            return 0.0

        rg = _reflection_geometry(hp, nside_eval, AXIS_NSIDE)
        if rg is None:
            return float(_fallback_stat(arr))
        pix4 = rg["pix4"]
        w4 = rg["w4"]
        if pix4.ndim != 3 or w4.ndim != 3 or pix4.shape != w4.shape or pix4.shape[2] != 4:
            return float(_fallback_stat(arr))

        reflected = (low[pix4[:, :, 0]] * w4[:, :, 0] +
                     low[pix4[:, :, 1]] * w4[:, :, 1] +
                     low[pix4[:, :, 2]] * w4[:, :, 2] +
                     low[pix4[:, :, 3]] * w4[:, :, 3])
        base = low.reshape(1, -1)
        with np.errstate(all="ignore"):
            even_power = 0.25 * np.sum((base + reflected) * (base + reflected), axis=1)
            odd_power = 0.25 * np.sum((base - reflected) * (base - reflected), axis=1)
        even_power = np.asarray(even_power, dtype=np.float64)
        odd_power = np.asarray(odd_power, dtype=np.float64)
        good = np.isfinite(even_power) & np.isfinite(odd_power) & (even_power >= 0.0) & (odd_power >= 0.0)
        if int(np.count_nonzero(good)) <= 0:
            return 0.0
        e = even_power[good]
        o = odd_power[good]
        with np.errstate(all="ignore"):
            q = np.abs(np.log((e + EPSILON) / (o + EPSILON)))
        q = q[np.isfinite(q)]
        if q.size <= 0:
            return 0.0
        stat = float(np.max(q))
        if not np.isfinite(stat):
            stat = 0.0
        return float(min(max(stat, 0.0), 1.0e6))
    except Exception:
        return float(_fallback_stat(arr))


def summarize_results(planck_stat, sim_results, _desc="Low-l Mirror-Parity Plane Log-Imbalance: after masked monopole/dipole subtraction, estimate the 2<=ell<=20 temperature multipoles with the same finite-filled masked-sky pipeline for SMICA and simulations; on a fixed antipodally identified HEALPix nside=8 grid of plane normals, split the low-l field into mirror-even and mirror-odd parts for each plane and return S=max_n |ln((P_even+epsilon)/(P_odd+epsilon))|."):
    import numpy as np

    EPSILON = 1.0e-30

    def _finite_float(x, default=np.nan):
        try:
            y = float(x)
        except Exception:
            return float(default)
        if np.isfinite(y):
            return float(y)
        return float(default)

    ps = _finite_float(planck_stat)
    try:
        sims = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            for v in sim_results:
                vals.append(_finite_float(v))
        except Exception:
            vals = []
        sims = np.asarray(vals, dtype=np.float64)
    sims = sims[np.isfinite(sims)]
    n = int(sims.size)

    out = {
        "test_description": _desc,
        "statistic": "S = max over antipodally identified plane normals of |log((P_even+epsilon)/(P_odd+epsilon))|",
        "planck_stat": ps,
        "n_sims": n,
        "ell_range": "2 <= ell <= 20, reduced only if the input nside is too small",
        "axis_grid_nside": 8,
        "epsilon": EPSILON,
        "higher_is_more_anomalous": True,
        "tail_of_interest": "upper tail: larger S means a stronger optimized mirror-even/mirror-odd low-l power imbalance",
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-l mirror-parity plane log-imbalance",
            "xlabel": "S = max |log(P_even/P_odd)|",
            "ylabel": "Number of simulations",
            "bins": 30,
            "caption": "Simulation distribution of the optimized low-ell mirror-parity log-imbalance; the observed SMICA value is assessed in the upper tail."
        }
    }
    if not np.isfinite(ps):
        out["warning"] = "The Planck/SMICA statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_le = int(np.count_nonzero(sims <= ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_ge = int(np.count_nonzero(sims >= ps))
    n_eq = int(n - n_lt - n_gt)
    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_two = float(min(1.0, 2.0 * min(p_upper, p_lower)))
    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    zscore = None
    if np.isfinite(std) and std > 0.0:
        zscore = float((ps - mean) / std)
    qs = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])

    out.update({
        "sim_mean": mean,
        "sim_median": median,
        "sim_std": std,
        "sim_min": float(np.min(sims)),
        "sim_max": float(np.max(sims)),
        "sim_percentile_1": float(qs[0]),
        "sim_percentile_5": float(qs[1]),
        "sim_percentile_16": float(qs[2]),
        "sim_percentile_50": float(qs[3]),
        "sim_percentile_84": float(qs[4]),
        "sim_percentile_95": float(qs[5]),
        "sim_percentile_99": float(qs[6]),
        "planck_zscore_vs_sims": zscore,
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "rank_descending_1_is_largest_S": int(n_gt + 1),
        "rank_descending_tie_low": int(n_gt + 1),
        "rank_descending_tie_high": int(n_ge + 1),
        "rank_ascending_1_is_smallest_S": int(n_lt + 1),
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "fraction_sims_gt_planck": float(n_gt / n),
        "fraction_sims_ge_planck": float(n_ge / n),
        "empirical_p_upper_tail_including_ties_plus_one": p_upper,
        "empirical_p_lower_tail_including_ties_plus_one": p_lower,
        "empirical_p_two_sided_extremeness": p_two,
        "is_upper_tail_5pct": bool(p_upper <= 0.05),
        "is_upper_tail_1pct": bool(p_upper <= 0.01)
    })
    if std == 0.0:
        out["warning"] = "All finite simulation statistics are identical; check for degenerate maps or an unavailable spherical-harmonic backend."
    elif p_upper <= 0.01:
        out["interpretation"] = "SMICA lies in the highest 1% tail for this optimized low-l mirror-parity imbalance."
    elif p_upper <= 0.05:
        out["interpretation"] = "SMICA lies in the highest 5% tail for this optimized low-l mirror-parity imbalance."
    else:
        out["interpretation"] = "SMICA is not in the highest 5% tail for this optimized low-l mirror-parity imbalance."
    return out
