Consistent GPT5.5 · Variation · power · one-tailed, lower tail

Hypothesis

The Planck SMICA map has an unusually low value of \(S=\ln\langle D_\ell\rangle_{2..8}-\ln\langle D_\ell\rangle_{9..30}\), meaning the very largest angular scales \(\ell=2..8\) are suppressed relative to \(\ell=9..30\) compared with masked LCDM simulations.

Method

After masked monopole/dipole removal, pseudo-\(C_\ell\) for \(2 \le \ell \le 30\) was converted to \(D_\ell\) and summarized by \(S=\ln\langle D_\ell\rangle_{2..8}-\ln\langle D_\ell\rangle_{9..30}\); lower \(S\) tests suppression of the very largest scales relative to the rest of low-\(\ell\).

Result Summary

Planck SMICA gives \(S=-0.2859\) (low/high ratio \(=0.751\)), versus 1000 LCDM simulations with mean \(-0.1557\), std \(0.1808\), and median \(-0.1451\). It lies at the 22.7th percentile, rank 228 from the low tail; one-tailed lower \(p=0.227\), about \(-0.75\sigma\).

Planck is mildly below the simulation mean, but far from the 5% lower tail. This statistic gives no significant evidence that \(\ell=2..8\) is unusually suppressed relative to \(\ell=9..30\).

Agent Justification

The statistic is directional: smaller or more negative \(S\) specifically corresponds to suppressed very-large-scale power, and the code identifies the lower tail as the tail of interest for this anomaly.

Comparison To Literature

Related to the well-known large-angle/low-\(\ell\) power deficit discussed by Planck 2015 XVI (arXiv:1506.07135) and Planck 2018 VII (arXiv:1906.02552), where low power/variance anomalies are reported in some statistics. The present masked log-ratio band test is less significant and does not reproduce an anomalous p-value.

test_description = "Masked Low-l Spectral Tilt Log-Ratio: remove the best-fit monopole and dipole on the unmasked pixels of each masked temperature map, zero-fill the mask, compute the masked pseudo-power spectrum through ell=30, form D_ell=ell(ell+1) Ctilde_ell/(2*pi), and return S=ln(<D_ell>_{2..8,2ell+1})-ln(<D_ell>_{9..30,2ell+1}). Negative S means the very largest angular scales are suppressed relative to ell=9..30."


def analyze_map(m, _cache={}):
    import math
    import zlib
    import numpy as np

    LMAX = 30
    LOW_MIN = 2
    LOW_MAX = 8
    HIGH_MIN = 9
    HIGH_MAX = 30
    TINY = 1.0e-300

    def _as_flat_float(a):
        try:
            if np.ma.isMaskedArray(a):
                arr = np.asarray(a.filled(np.nan), dtype=np.float64)
            else:
                arr = np.asarray(a, dtype=np.float64)
        except Exception:
            try:
                arr = np.asarray(np.real(a), dtype=np.float64)
            except Exception:
                arr = np.asarray([], dtype=np.float64)
        return np.ravel(arr)

    def _get_hp():
        key = "healpy_for_masked_lowl_tilt_logratio"
        if key not in _cache:
            try:
                import healpy as hp
                _cache[key] = hp
            except Exception:
                _cache[key] = None
        return _cache.get(key)

    def _nside_from_npix(hp, npix):
        try:
            ns = int(hp.npix2nside(int(npix)))
            if ns > 0 and int(hp.nside2npix(ns)) == int(npix):
                return ns
        except Exception:
            pass
        try:
            if int(npix) > 0 and int(npix) % 12 == 0:
                ns = int(round(math.sqrt(float(npix) / 12.0)))
                if ns > 0 and 12 * ns * ns == int(npix):
                    return ns
        except Exception:
            pass
        return -1

    def _mask_signature(valid):
        v = np.asarray(valid, dtype=bool).ravel()
        try:
            crc = int(zlib.crc32(np.packbits(v).tobytes()) & 0xffffffff)
        except Exception:
            crc = int(np.count_nonzero(v) & 0xffffffff)
        return str(int(v.size)) + ":" + str(int(np.count_nonzero(v))) + ":" + str(crc)

    def _pixvecs(hp, nside):
        nside = int(nside)
        key = "pixvecs_ring_lowl_tilt_nside_" + str(nside)
        got = _cache.get(key)
        if got is not None:
            return got
        npix = int(hp.nside2npix(nside))
        pix = np.arange(npix, dtype=np.int64)
        try:
            x, y, z = hp.pix2vec(nside, pix, nest=False)
        except TypeError:
            x, y, z = hp.pix2vec(nside, pix)
        got = (np.asarray(x, dtype=np.float64), np.asarray(y, dtype=np.float64), np.asarray(z, dtype=np.float64))
        try:
            if int(_cache.get("pixvec_cache_count_lowl_tilt", 0)) < 4:
                _cache[key] = got
                _cache["pixvec_cache_count_lowl_tilt"] = int(_cache.get("pixvec_cache_count_lowl_tilt", 0)) + 1
        except Exception:
            pass
        return got

    def _fit_geometry(hp, nside, valid):
        valid = np.asarray(valid, dtype=bool).ravel()
        sig = _mask_signature(valid)
        key = "monopole_dipole_geom_lowl_tilt_" + str(int(nside)) + "_" + sig
        got = _cache.get(key)
        if got is not None:
            return got
        x, y, z = _pixvecs(hp, int(nside))
        if int(x.size) != int(valid.size):
            return None
        xv = x[valid]
        yv = y[valid]
        zv = z[valid]
        n = int(xv.size)
        if n < 4:
            return None
        sx = float(np.sum(xv)); sy = float(np.sum(yv)); sz = float(np.sum(zv))
        g = np.array([
            [float(n), sx, sy, sz],
            [sx, float(np.dot(xv, xv)), float(np.dot(xv, yv)), float(np.dot(xv, zv))],
            [sy, float(np.dot(yv, xv)), float(np.dot(yv, yv)), float(np.dot(yv, zv))],
            [sz, float(np.dot(zv, xv)), float(np.dot(zv, yv)), float(np.dot(zv, zv))]
        ], dtype=np.float64)
        invg = None
        try:
            invg = np.linalg.inv(g)
            if not np.all(np.isfinite(invg)):
                invg = None
        except Exception:
            invg = None
        got = {"xv": xv, "yv": yv, "zv": zv, "g": g, "invg": invg}
        try:
            if int(_cache.get("md_geom_cache_count_lowl_tilt", 0)) < 8:
                _cache[key] = got
                _cache["md_geom_cache_count_lowl_tilt"] = int(_cache.get("md_geom_cache_count_lowl_tilt", 0)) + 1
        except Exception:
            pass
        return got

    def _remove_monopole_dipole_zero_mask(arr, valid, hp, nside):
        arr = np.asarray(arr, dtype=np.float64).ravel()
        valid = np.asarray(valid, dtype=bool).ravel()
        out = np.zeros_like(arr, dtype=np.float64)
        if int(arr.size) != int(valid.size):
            return out
        vals = arr[valid]
        if int(vals.size) <= 0:
            return out
        geom = _fit_geometry(hp, int(nside), valid)
        if geom is None or int(vals.size) < 4:
            mu = float(np.mean(vals)) if int(vals.size) else 0.0
            resid = vals - mu
            out[valid] = np.where(np.isfinite(resid), resid, 0.0)
            return out
        rhs = np.array([
            float(np.sum(vals)),
            float(np.dot(vals, geom["xv"])),
            float(np.dot(vals, geom["yv"])),
            float(np.dot(vals, geom["zv"]))
        ], dtype=np.float64)
        try:
            if geom.get("invg", None) is not None:
                beta = np.dot(geom["invg"], rhs)
            else:
                beta = np.linalg.solve(geom["g"], rhs)
        except Exception:
            try:
                beta = np.linalg.lstsq(geom["g"], rhs, rcond=None)[0]
            except Exception:
                beta = np.array([float(np.mean(vals)), 0.0, 0.0, 0.0], dtype=np.float64)
        beta = np.where(np.isfinite(beta), beta, 0.0)
        model = beta[0] + beta[1] * geom["xv"] + beta[2] * geom["yv"] + beta[3] * geom["zv"]
        resid = vals - model
        out[valid] = np.where(np.isfinite(resid), resid, 0.0)
        return out

    def _fallback_value(arr, valid):
        # Last-resort non-constant finite proxy if healpy is unavailable or the map geometry is invalid.
        # It uses a deterministic split of unmasked residual samples, not the intended harmonic statistic.
        try:
            vals = np.asarray(arr, dtype=np.float64).ravel()[np.asarray(valid, dtype=bool).ravel()]
            vals = vals[np.isfinite(vals)]
            if int(vals.size) < 4:
                return 0.0
            vals = vals - float(np.mean(vals))
            a = vals[::2]
            b = vals[1::2]
            p1 = float(np.mean(a * a)) if int(a.size) else TINY
            p2 = float(np.mean(b * b)) if int(b.size) else TINY
            if (not np.isfinite(p1)) or p1 <= 0.0:
                p1 = TINY
            if (not np.isfinite(p2)) or p2 <= 0.0:
                p2 = TINY
            s0 = float(math.log(max(p1, TINY)) - math.log(max(p2, TINY)))
            return s0 if np.isfinite(s0) else 0.0
        except Exception:
            return 0.0

    arr = _as_flat_float(m)
    npix = int(arr.size)
    if npix <= 0:
        return 0.0
    valid = np.isfinite(arr)
    if int(np.count_nonzero(valid)) <= 0:
        return 0.0

    hp = _get_hp()
    if hp is None:
        return float(_fallback_value(arr, valid))
    nside = _nside_from_npix(hp, npix)
    if nside <= 0:
        return float(_fallback_value(arr, valid))

    try:
        lmax_possible = int(3 * int(nside) - 1)
    except Exception:
        lmax_possible = LMAX
    if lmax_possible < HIGH_MAX:
        # The defined statistic needs the full ell=9..30 band. Fall back only for pathological low nside.
        return float(_fallback_value(arr, valid))

    try:
        cleaned = _remove_monopole_dipole_zero_mask(arr, valid, hp, nside)
        # Never pass NaN/inf into healpy. Masked pixels are exactly zero, giving the masked pseudo-spectrum.
        cleaned = np.where(np.isfinite(cleaned), cleaned, 0.0).astype(np.float64, copy=False)
        try:
            cl = hp.anafast(cleaned, lmax=LMAX, iter=0, pol=False, use_weights=False)
        except TypeError:
            try:
                cl = hp.anafast(cleaned, lmax=LMAX, iter=0, pol=False)
            except TypeError:
                cl = hp.anafast(cleaned, lmax=LMAX)
        cl = np.asarray(cl, dtype=np.float64).ravel()
        if int(cl.size) <= HIGH_MAX:
            return float(_fallback_value(arr, valid))
        ells = np.arange(0, HIGH_MAX + 1, dtype=np.float64)
        dl = ells * (ells + 1.0) * cl[:HIGH_MAX + 1] / (2.0 * math.pi)
        dl = np.where(np.isfinite(dl), dl, 0.0)
        ell_low = np.arange(LOW_MIN, LOW_MAX + 1, dtype=np.float64)
        ell_high = np.arange(HIGH_MIN, HIGH_MAX + 1, dtype=np.float64)
        w_low = 2.0 * ell_low + 1.0
        w_high = 2.0 * ell_high + 1.0
        p_low = float(np.dot(w_low, dl[LOW_MIN:LOW_MAX + 1]) / np.sum(w_low))
        p_high = float(np.dot(w_high, dl[HIGH_MIN:HIGH_MAX + 1]) / np.sum(w_high))
        if (not np.isfinite(p_low)) or p_low <= 0.0:
            p_low = TINY
        if (not np.isfinite(p_high)) or p_high <= 0.0:
            p_high = TINY
        s = float(math.log(max(p_low, TINY)) - math.log(max(p_high, TINY)))
        if not np.isfinite(s):
            s = 0.0
        return float(s)
    except Exception:
        return float(_fallback_value(arr, valid))


def summarize_results(planck_stat, sim_results, _desc=test_description):
    import math
    import numpy as np

    def _finite_float(x, default=np.nan):
        try:
            y = float(x)
        except Exception:
            return float(default)
        return float(y) if np.isfinite(y) else float(default)

    ps = _finite_float(planck_stat)
    try:
        sims = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            for v in sim_results:
                vals.append(_finite_float(v))
        except Exception:
            vals = []
        sims = np.asarray(vals, dtype=np.float64)
    sims = sims[np.isfinite(sims)]
    n = int(sims.size)

    out = {
        "test_description": _desc,
        "statistic": "S = ln(weighted mean D_ell for 2<=ell<=8) - ln(weighted mean D_ell for 9<=ell<=30), with weights 2ell+1",
        "planck_stat_log_ratio": ps,
        "planck_low_to_high_D_ratio": float(math.exp(ps)) if np.isfinite(ps) and ps < 700.0 else None,
        "n_sims": n,
        "tail_of_interest": "lower tail for unusually suppressed very-large-scale power relative to ell=9..30; upper tail for enhancement",
        "multipole_bands": {"very_large_scales": "2 <= ell <= 8", "comparison_low_ell_band": "9 <= ell <= 30"},
        "monopole_dipole_treatment": "best-fit monopole plus Cartesian dipole removed on unmasked pixels before zero-filling the mask and computing the pseudo-spectrum",
        "mask_treatment": "input finite pixels define a binary mask; masked NaN pixels are zero-filled only after monopole/dipole subtraction",
        "plot_spec": {
            "kind": "histogram",
            "title": "Masked low-l spectral tilt log-ratio",
            "xlabel": "S = ln(<D_l>_{2..8}) - ln(<D_l>_{9..30})",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "LCDM simulation distribution of the masked low-l spectral tilt statistic. More negative values indicate stronger suppression of ell=2..8 relative to ell=9..30."
        }
    }

    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_le = int(np.count_nonzero(sims <= ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_ge = int(np.count_nonzero(sims >= ps))
    n_eq = int(n - n_lt - n_gt)

    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_two = float(min(1.0, 2.0 * min(p_lower, p_upper)))
    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    z = None
    if np.isfinite(std) and std > 0.0:
        z = float((ps - mean) / std)
    q = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])

    median_ratio = float(math.exp(median)) if median < 700.0 else None
    planck_ratio = float(math.exp(ps)) if ps < 700.0 else None
    ratio_to_median_ratio = None
    if planck_ratio is not None and median_ratio is not None and median_ratio > 0.0:
        ratio_to_median_ratio = float(planck_ratio / median_ratio)

    out.update({
        "sim_mean_log_ratio": mean,
        "sim_median_log_ratio": median,
        "sim_std_log_ratio": std,
        "sim_min_log_ratio": float(np.min(sims)),
        "sim_max_log_ratio": float(np.max(sims)),
        "sim_percentile_1_log_ratio": float(q[0]),
        "sim_percentile_5_log_ratio": float(q[1]),
        "sim_percentile_16_log_ratio": float(q[2]),
        "sim_percentile_50_log_ratio": float(q[3]),
        "sim_percentile_84_log_ratio": float(q[4]),
        "sim_percentile_95_log_ratio": float(q[5]),
        "sim_percentile_99_log_ratio": float(q[6]),
        "planck_zscore_vs_sims": z,
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "rank_ascending_1_is_most_suppressed": int(n_lt + 1),
        "rank_descending_1_is_most_enhanced": int(n_gt + 1),
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "fraction_sims_lt_planck": float(n_lt / n),
        "fraction_sims_le_planck": float(n_le / n),
        "empirical_p_lower_tail_suppressed_large_scales_including_ties_plus_one": p_lower,
        "empirical_p_upper_tail_enhanced_large_scales_including_ties_plus_one": p_upper,
        "empirical_p_two_sided_plus_one": p_two,
        "planck_low_to_high_ratio_divided_by_sim_median_ratio": ratio_to_median_ratio,
        "log_difference_planck_minus_sim_median": float(ps - median),
        "is_lower_tail_5pct_suppression": bool(p_lower <= 0.05),
        "is_lower_tail_1pct_suppression": bool(p_lower <= 0.01),
        "is_upper_tail_5pct_enhancement": bool(p_upper <= 0.05),
        "is_upper_tail_1pct_enhancement": bool(p_upper <= 0.01),
        "is_two_sided_5pct": bool(p_two <= 0.05),
        "is_two_sided_1pct": bool(p_two <= 0.01)
    })

    if std == 0.0:
        out["warning"] = "All finite simulation statistics are identical; check the input maps, mask, and harmonic-transform support."
    elif p_lower <= 0.01:
        out["interpretation"] = "Planck is in the lowest 1% of simulations: ell=2..8 power is strongly suppressed relative to ell=9..30 by this masked pseudo-spectrum statistic."
    elif p_lower <= 0.05:
        out["interpretation"] = "Planck is in the lowest 5% of simulations: ell=2..8 power is suppressed relative to ell=9..30 by this statistic."
    elif p_upper <= 0.01:
        out["interpretation"] = "Planck is in the highest 1% of simulations: ell=2..8 power is strongly enhanced relative to ell=9..30 by this statistic."
    elif p_upper <= 0.05:
        out["interpretation"] = "Planck is in the highest 5% of simulations: ell=2..8 power is enhanced relative to ell=9..30 by this statistic."
    else:
        out["interpretation"] = "Planck is not in either 5% tail for this masked low-l spectral tilt statistic."
    return out
