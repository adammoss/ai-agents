Hypothesis

The CMB temperature map has an anomalously large maximum hemispherical contrast in low-ell spectral tilt, meaning one pair of opposite hemispheres shows an unusually different log variance ratio between ell=11–30 and ell=2–10 compared with simulations.

Method

Scans coarse HEALPix hemisphere axes and measures the maximum absolute north-south difference in local low-l tilt, tau=log(V_ell11-30/V_ell2-10), using masked, mean-subtracted, bandpassed CMB temperature maps.

Result Summary

Planck gives S=0.826, compared with simulations mean=0.689, std=0.210, median=0.652. The one-tailed upper p-value is 0.241 from 1000 simulations, with Planck at the 75.9th percentile and about 0.70 sigma.

The measured Planck contrast is somewhat above the simulation mean but well within the LCDM simulation distribution. It is not in the upper 5% tail, so the hypothesised anomalously large hemispherical low-l spectral-tilt contrast is not supported.

Agent Justification

The statistic is S=max_axis |tau_N−tau_S|, so the sign of the hemispherical difference is removed and only the magnitude of the contrast matters. The code explicitly evaluates the upper tail, where larger S means a stronger hemispherical spectral-tilt contrast.

import math
import numpy as np


test_description = "Low-l Hemispherical Spectral-Tilt Max Contrast: subtract the mean on finite unmasked pixels, fill masked pixels with 0.0, make bandpassed maps for ell=2..10 and ell=11..30, scan an unoriented coarse HEALPix nside=4 grid of hemisphere axes, compute the unmasked pixel variance of each band in the two opposite hemispheres, define tau=log(V_11-30/V_2-10), and return S=max_axis |tau_N-tau_S|."

_LST_LOW_MIN = 2
_LST_LOW_MAX = 10
_LST_HIGH_MIN = 11
_LST_HIGH_MAX = 30
_LST_AXIS_NSIDE = 4
_LST_CHUNK = 8192
_LST_MIN_VALID_PIX = 32
_LST_MIN_HEMI_PIX = 8
_LST_MAX_HEMI_CACHE_ENTRIES = 24000000
_LST_REL_FLOOR = 1.0e-30
_LST_ABS_FLOOR = 1.0e-300

_LST_STATE = {
    "axes_cache": {},
    "geom_cache": {},
    "hemi_cache": {},
}


def _lst_infer_nside(npix):
    try:
        n = int(npix)
        if n <= 0 or n % 12 != 0:
            return -1
        ns = int(round(math.sqrt(n / 12.0)))
        if ns <= 0 or 12 * ns * ns != n:
            return -1
        return ns
    except Exception:
        return -1


def _lst_alm_size(lmax):
    lmax = int(lmax)
    if lmax < 0:
        return 0
    return int((lmax + 1) * (lmax + 2) // 2)


def _lst_alm_index(lmax, ell, mm):
    return int(mm * (2 * lmax + 1 - mm) // 2 + ell)


def _lst_get_axes(hp, state):
    cache = state.get("axes_cache")
    if not isinstance(cache, dict):
        cache = {}
        state["axes_cache"] = cache
    key = str(int(_LST_AXIS_NSIDE))
    axes = cache.get(key)
    if axes is not None:
        return axes

    npix_axis = int(hp.nside2npix(int(_LST_AXIS_NSIDE)))
    pix = np.arange(npix_axis, dtype=np.int64)
    vec = hp.pix2vec(int(_LST_AXIS_NSIDE), pix, nest=False)
    axes_all = np.vstack((np.asarray(vec[0], dtype=np.float64),
                          np.asarray(vec[1], dtype=np.float64),
                          np.asarray(vec[2], dtype=np.float64))).T
    norm = np.sqrt(np.sum(axes_all * axes_all, axis=1))
    good = np.isfinite(norm) & (norm > 0.0)
    axes_all = axes_all[good]
    norm = norm[good]
    if axes_all.size == 0:
        axes_all = np.array([[0.0, 0.0, 1.0]], dtype=np.float64)
        norm = np.array([1.0], dtype=np.float64)
    axes_all = axes_all / norm[:, None]

    # A hemisphere axis is unoriented: axis a and -a give the same absolute
    # contrast after swapping N and S.  Keep one deterministic representative
    # from each antipodal pair for speed while retaining the same max |Delta tau|.
    tol = 1.0e-14
    x = axes_all[:, 0]
    y = axes_all[:, 1]
    z = axes_all[:, 2]
    keep = (z > tol) | ((np.abs(z) <= tol) & ((y > tol) | ((np.abs(y) <= tol) & (x >= -tol))))
    if int(np.count_nonzero(keep)) > 0:
        axes = np.asarray(axes_all[keep], dtype=np.float64)
    else:
        axes = np.asarray(axes_all, dtype=np.float64)

    if len(cache) > 4:
        cache.clear()
    cache[key] = axes
    return axes


def _lst_get_geometry(hp, nside, npix, state):
    cache = state.get("geom_cache")
    if not isinstance(cache, dict):
        cache = {}
        state["geom_cache"] = cache
    key = str(int(nside)) + ":" + str(int(npix))
    item = cache.get(key)
    if item is not None:
        return item

    pix = np.arange(int(npix), dtype=np.int64)
    vec = hp.pix2vec(int(nside), pix, nest=False)
    item = {
        "x": np.asarray(vec[0], dtype=np.float64).ravel(),
        "y": np.asarray(vec[1], dtype=np.float64).ravel(),
        "z": np.asarray(vec[2], dtype=np.float64).ravel(),
    }
    if len(cache) > 3:
        cache.clear()
    cache[key] = item
    return item


def _lst_make_mask_info(nside, npix, valid):
    v = np.asarray(valid, dtype=np.bool_).ravel()
    nvalid = int(np.count_nonzero(v))
    packed = np.packbits(v)
    mask_bytes = packed.tobytes()
    key = (str(int(nside)) + ":" + str(int(npix)) + ":" + str(nvalid) + ":" +
           str(hash(mask_bytes)) + ":axis" + str(int(_LST_AXIS_NSIDE)))
    return {"key": key, "bytes": mask_bytes, "nvalid": nvalid}


def _lst_get_hemi_entry(hp, nside, npix, valid, mask_info, state):
    cache = state.get("hemi_cache")
    if not isinstance(cache, dict):
        cache = {}
        state["hemi_cache"] = cache
    key = mask_info["key"]
    item = cache.get(key)
    if item is not None and item.get("mask_bytes") == mask_info["bytes"]:
        return item

    idx = np.flatnonzero(np.asarray(valid, dtype=np.bool_)).astype(np.int64, copy=False)
    axes = _lst_get_axes(hp, state)
    naxis = int(axes.shape[0])
    nvalid = int(idx.size)
    H = None
    counts = None

    # Cache a boolean hemisphere-membership matrix when it is modest in size.
    # This avoids recomputing axis dot products for every simulation when the
    # mask is fixed, but falls back to chunked accumulation for larger maps.
    if nvalid > 0 and naxis > 0 and naxis * nvalid <= int(_LST_MAX_HEMI_CACHE_ENTRIES):
        geom = _lst_get_geometry(hp, nside, npix, state)
        H = np.empty((naxis, nvalid), dtype=np.bool_)
        for start in range(0, nvalid, int(_LST_CHUNK)):
            end = min(nvalid, start + int(_LST_CHUNK))
            ii = idx[start:end]
            xyz = np.empty((3, int(ii.size)), dtype=np.float64)
            xyz[0, :] = geom["x"][ii]
            xyz[1, :] = geom["y"][ii]
            xyz[2, :] = geom["z"][ii]
            H[:, start:end] = (axes.dot(xyz) >= 0.0)
        counts = np.asarray(np.count_nonzero(H, axis=1), dtype=np.float64)

    item = {
        "idx": idx,
        "H": H,
        "counts": counts,
        "mask_bytes": mask_info["bytes"],
        "naxis": naxis,
    }
    if len(cache) > 4:
        cache.clear()
    cache[key] = item
    return item


def _lst_map2alm(finite_map, hp, lmax):
    import warnings
    lmax = int(lmax)
    fmap = np.asarray(finite_map, dtype=np.float64).ravel()
    if not np.all(np.isfinite(fmap)):
        fmap = np.array(fmap, dtype=np.float64, copy=True)
        fmap[~np.isfinite(fmap)] = 0.0
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        try:
            try:
                alm = hp.map2alm(fmap, lmax=lmax, iter=0, pol=False, use_weights=False)
            except TypeError:
                alm = hp.map2alm(fmap, lmax=lmax, iter=0, pol=False)
        except Exception:
            raise
    alm = np.asarray(alm, dtype=np.complex128).ravel()
    expected = _lst_alm_size(lmax)
    if int(alm.size) < expected:
        tmp = np.zeros(expected, dtype=np.complex128)
        tmp[:int(alm.size)] = alm
        alm = tmp
    elif int(alm.size) > expected:
        alm = alm[:expected]
    bad = ~np.isfinite(alm)
    if np.any(bad):
        alm = np.array(alm, dtype=np.complex128, copy=True)
        alm[bad] = 0.0 + 0.0j
    return alm


def _lst_alm2map(alm, hp, nside, lmax):
    import warnings
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        try:
            out = hp.alm2map(alm, int(nside), lmax=int(lmax), pol=False, verbose=False)
        except TypeError:
            out = hp.alm2map(alm, int(nside), lmax=int(lmax), pol=False)
    if np.ma.isMaskedArray(out):
        out = out.filled(0.0)
    out = np.asarray(out, dtype=np.float64).ravel()
    if not np.all(np.isfinite(out)):
        out = np.array(out, dtype=np.float64, copy=True)
        out[~np.isfinite(out)] = 0.0
    return out


def _lst_band_alm(alm, lmax, ell_min, ell_max):
    lmax = int(lmax)
    ell_min = int(ell_min)
    ell_max = min(int(ell_max), lmax)
    out = np.zeros(_lst_alm_size(lmax), dtype=np.complex128)
    if ell_max < ell_min:
        return out
    alm = np.asarray(alm, dtype=np.complex128).ravel()
    for mm in range(0, ell_max + 1):
        lo = max(ell_min, mm)
        hi = ell_max
        if lo <= hi:
            i0 = _lst_alm_index(lmax, lo, mm)
            i1 = _lst_alm_index(lmax, hi, mm) + 1
            if i0 < int(alm.size):
                n = min(i1 - i0, int(alm.size) - i0, int(out.size) - i0)
                if n > 0:
                    out[i0:i0 + n] = alm[i0:i0 + n]
    return out


def _lst_make_band_maps(clean_map, hp, nside):
    lmax = int(_LST_HIGH_MAX)
    alm = _lst_map2alm(clean_map, hp, lmax)
    alm_low = _lst_band_alm(alm, lmax, _LST_LOW_MIN, _LST_LOW_MAX)
    alm_high = _lst_band_alm(alm, lmax, _LST_HIGH_MIN, _LST_HIGH_MAX)
    low_map = _lst_alm2map(alm_low, hp, nside, lmax)
    high_map = _lst_alm2map(alm_high, hp, nside, lmax)
    return low_map, high_map


def _lst_variance_from_sums(sum1, sum2, count):
    sum1 = np.asarray(sum1, dtype=np.float64)
    sum2 = np.asarray(sum2, dtype=np.float64)
    count = np.asarray(count, dtype=np.float64)
    out = np.full(count.shape, np.nan, dtype=np.float64)
    good = np.isfinite(count) & (count >= float(_LST_MIN_HEMI_PIX))
    if np.any(good):
        with np.errstate(divide="ignore", invalid="ignore", over="ignore"):
            mean = sum1[good] / count[good]
            var = sum2[good] / count[good] - mean * mean
        var = np.asarray(var, dtype=np.float64)
        var[~np.isfinite(var)] = np.nan
        # Roundoff can make a true non-negative variance slightly negative.
        neg = np.isfinite(var) & (var < 0.0)
        if np.any(neg):
            var[neg] = 0.0
        out[good] = var
    return out


def _lst_chunked_hemi_sums(idx, lv, hv, hp, nside, npix, state):
    axes = _lst_get_axes(hp, state)
    geom = _lst_get_geometry(hp, nside, npix, state)
    naxis = int(axes.shape[0])
    n = int(idx.size)
    counts = np.zeros(naxis, dtype=np.float64)
    sums = np.zeros((naxis, 4), dtype=np.float64)
    if n <= 0 or naxis <= 0:
        return counts, sums

    for start in range(0, n, int(_LST_CHUNK)):
        end = min(n, start + int(_LST_CHUNK))
        ii = idx[start:end]
        l = np.asarray(lv[start:end], dtype=np.float64)
        h = np.asarray(hv[start:end], dtype=np.float64)
        vals = np.empty((int(ii.size), 4), dtype=np.float64)
        vals[:, 0] = l
        vals[:, 1] = l * l
        vals[:, 2] = h
        vals[:, 3] = h * h
        xyz = np.empty((3, int(ii.size)), dtype=np.float64)
        xyz[0, :] = geom["x"][ii]
        xyz[1, :] = geom["y"][ii]
        xyz[2, :] = geom["z"][ii]
        hemi = (axes.dot(xyz) >= 0.0)
        counts += np.asarray(np.count_nonzero(hemi, axis=1), dtype=np.float64)
        sums += np.asarray(hemi.dot(vals), dtype=np.float64)
    return counts, sums


def _lst_max_tilt_contrast(low_map, high_map, hemi_entry, hp, nside, npix, state):
    idx0 = np.asarray(hemi_entry.get("idx", []), dtype=np.int64).ravel()
    n0 = int(idx0.size)
    if n0 < int(_LST_MIN_VALID_PIX):
        return 0.0

    low = np.asarray(low_map, dtype=np.float64).ravel()
    high = np.asarray(high_map, dtype=np.float64).ravel()
    if int(low.size) != int(npix) or int(high.size) != int(npix):
        return 0.0

    lv0 = low[idx0]
    hv0 = high[idx0]
    good = np.isfinite(lv0) & np.isfinite(hv0)
    if not np.all(good):
        if int(np.count_nonzero(good)) < int(_LST_MIN_VALID_PIX):
            return 0.0
        idx = idx0[good]
        lv = np.asarray(lv0[good], dtype=np.float64)
        hv = np.asarray(hv0[good], dtype=np.float64)
        H = None
        cached_counts = None
    else:
        idx = idx0
        lv = np.asarray(lv0, dtype=np.float64)
        hv = np.asarray(hv0, dtype=np.float64)
        H = hemi_entry.get("H")
        cached_counts = hemi_entry.get("counts")

    n = int(idx.size)
    if n < int(_LST_MIN_VALID_PIX):
        return 0.0

    with np.errstate(over="ignore", invalid="ignore"):
        lv2 = lv * lv
        hv2 = hv * hv
    if (not np.all(np.isfinite(lv2))) or (not np.all(np.isfinite(hv2))):
        return 0.0

    total = np.array([float(np.sum(lv)), float(np.sum(lv2)),
                      float(np.sum(hv)), float(np.sum(hv2))], dtype=np.float64)
    if not np.all(np.isfinite(total)):
        return 0.0

    if H is not None and cached_counts is not None and int(np.asarray(H).shape[1]) == n:
        vals = np.empty((n, 4), dtype=np.float64)
        vals[:, 0] = lv
        vals[:, 1] = lv2
        vals[:, 2] = hv
        vals[:, 3] = hv2
        counts_N = np.asarray(cached_counts, dtype=np.float64).ravel()
        try:
            sums_N = np.asarray(H.dot(vals), dtype=np.float64)
        except Exception:
            counts_N, sums_N = _lst_chunked_hemi_sums(idx, lv, hv, hp, nside, npix, state)
    else:
        counts_N, sums_N = _lst_chunked_hemi_sums(idx, lv, hv, hp, nside, npix, state)

    if sums_N.ndim != 2 or int(sums_N.shape[1]) != 4:
        return 0.0
    counts_N = np.asarray(counts_N, dtype=np.float64).ravel()
    naxis = int(counts_N.size)
    if naxis <= 0 or int(sums_N.shape[0]) != naxis:
        return 0.0

    counts_S = float(n) - counts_N
    sums_S = total[None, :] - sums_N

    with np.errstate(divide="ignore", invalid="ignore", over="ignore"):
        gvar_low = total[1] / float(n) - (total[0] / float(n)) ** 2
        gvar_high = total[3] / float(n) - (total[2] / float(n)) ** 2
    if not np.isfinite(gvar_low):
        gvar_low = 0.0
    if not np.isfinite(gvar_high):
        gvar_high = 0.0
    if gvar_low < 0.0:
        gvar_low = 0.0
    if gvar_high < 0.0:
        gvar_high = 0.0
    scale = max(float(gvar_low), float(gvar_high))
    if not np.isfinite(scale) or scale <= 0.0:
        return 0.0
    floor = max(float(_LST_ABS_FLOOR), float(_LST_REL_FLOOR) * scale)

    var_low_N = _lst_variance_from_sums(sums_N[:, 0], sums_N[:, 1], counts_N)
    var_high_N = _lst_variance_from_sums(sums_N[:, 2], sums_N[:, 3], counts_N)
    var_low_S = _lst_variance_from_sums(sums_S[:, 0], sums_S[:, 1], counts_S)
    var_high_S = _lst_variance_from_sums(sums_S[:, 2], sums_S[:, 3], counts_S)

    good_axis = (counts_N >= float(_LST_MIN_HEMI_PIX)) & (counts_S >= float(_LST_MIN_HEMI_PIX))
    good_axis &= np.isfinite(var_low_N) & np.isfinite(var_high_N) & np.isfinite(var_low_S) & np.isfinite(var_high_S)
    if not np.any(good_axis):
        return 0.0

    with np.errstate(divide="ignore", invalid="ignore", over="ignore"):
        tau_N = np.log((var_high_N + floor) / (var_low_N + floor))
        tau_S = np.log((var_high_S + floor) / (var_low_S + floor))
        diff = np.abs(tau_N - tau_S)
    diff = np.asarray(diff, dtype=np.float64)
    diff[~good_axis] = np.nan
    finite = np.isfinite(diff)
    if not np.any(finite):
        return 0.0
    S = float(np.max(diff[finite]))
    if not np.isfinite(S) or S < 0.0:
        return 0.0
    return S


def analyze_map(m):
    try:
        if np.ma.isMaskedArray(m):
            arr = np.asarray(m.filled(np.nan), dtype=np.float64).ravel()
        else:
            arr = np.asarray(m, dtype=np.float64).ravel()
    except Exception:
        try:
            arr = np.asarray(np.real(m), dtype=np.float64).ravel()
        except Exception:
            return 0.0

    npix = int(arr.size)
    nside = _lst_infer_nside(npix)
    if nside <= 0:
        return 0.0

    # Need lmax=30 to realize the specified ell=11..30 band without truncation.
    if 3 * int(nside) - 1 < int(_LST_HIGH_MAX):
        return 0.0

    valid = np.isfinite(arr) & (np.abs(arr) < 1.0e25)
    nvalid = int(np.count_nonzero(valid))
    if nvalid < int(_LST_MIN_VALID_PIX):
        return 0.0

    try:
        vals = np.asarray(arr[valid], dtype=np.float64)
        mean = float(np.mean(vals))
    except Exception:
        return 0.0
    if not np.isfinite(mean):
        return 0.0

    clean = np.zeros(npix, dtype=np.float64)
    clean[valid] = arr[valid] - mean
    if not np.all(np.isfinite(clean)):
        clean[~np.isfinite(clean)] = 0.0

    try:
        import healpy as hp

        mask_info = _lst_make_mask_info(nside, npix, valid)
        if int(mask_info["nvalid"]) < int(_LST_MIN_VALID_PIX):
            return 0.0

        # clean is finite everywhere, with masked pixels filled by zero, so no
        # NaNs are ever passed into healpy transforms.
        low_map, high_map = _lst_make_band_maps(clean, hp, nside)
        if int(low_map.size) != npix or int(high_map.size) != npix:
            return 0.0

        hemi_entry = _lst_get_hemi_entry(hp, nside, npix, valid, mask_info, _LST_STATE)
        S = _lst_max_tilt_contrast(low_map, high_map, hemi_entry, hp, nside, npix, _LST_STATE)
        if not np.isfinite(S) or S < 0.0:
            return 0.0
        return float(S)
    except Exception:
        return 0.0


def summarize_results(planck_stat, sim_results, _desc=test_description,
                      _low_min=_LST_LOW_MIN, _low_max=_LST_LOW_MAX,
                      _high_min=_LST_HIGH_MIN, _high_max=_LST_HIGH_MAX,
                      _axis_nside=_LST_AXIS_NSIDE):
    import math
    import numpy as np

    def _as_float(x):
        try:
            v = float(x)
            if np.isfinite(v):
                return v
            return np.nan
        except Exception:
            return np.nan

    ps = _as_float(planck_stat)
    try:
        sims_all = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            iterator = iter(sim_results)
        except Exception:
            iterator = []
        for item in iterator:
            v = _as_float(item)
            if np.isfinite(v):
                vals.append(v)
        sims_all = np.asarray(vals, dtype=np.float64)
    sims = sims_all[np.isfinite(sims_all)]
    sims = np.maximum(sims, 0.0)
    n = int(sims.size)

    out = {
        "test_description": _desc,
        "statistic": "S = max over coarse HEALPix hemisphere axes of |tau_N - tau_S|, where tau=log(V_ell11..30/V_ell2..10) and V is the unmasked pixel variance in that hemisphere.",
        "primary_tail": "upper tail in S; larger values mean a stronger hemispherical contrast in the local low-l spectral tilt (high-band variance relative to low-band variance)",
        "low_band_ell": [int(_low_min), int(_low_max)],
        "high_band_ell": [int(_high_min), int(_high_max)],
        "axis_grid": {"healpix_nside": int(_axis_nside), "orientation": "unoriented; one representative from each antipodal pair is scanned"},
        "mask_and_transform_handling": "finite pixels define the mask and the masked-sky mean; masked/non-finite pixels are filled with 0.0 before healpy spherical-harmonic transforms and ignored in hemisphere variances",
        "n_sims": n,
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-l hemispherical spectral-tilt max contrast",
            "xlabel": "S = max |Delta log(V_11-30 / V_2-10)|",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "Simulation distribution of the maximum hemispherical contrast in local spectral tilt. Large S indicates that one pair of opposite hemispheres has unusually different high-to-low low-l variance ratios."
        }
    }

    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic supplied to summarize_results is not finite."
        return out

    ps = max(0.0, float(ps))
    out["planck_S"] = ps
    if ps < 700.0:
        out["planck_max_high_low_ratio_contrast_factor"] = float(math.exp(ps))
    else:
        out["planck_max_high_low_ratio_contrast_factor"] = None
        out["planck_ratio_factor_note"] = "exp(S) overflows double precision; S itself is the stable log contrast."

    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    if n > 1:
        q = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])
    else:
        q = np.repeat(sims[0], 7)

    n_gt = int(np.count_nonzero(sims > ps))
    n_ge = int(np.count_nonzero(sims >= ps))
    n_lt = int(np.count_nonzero(sims < ps))
    n_le = int(np.count_nonzero(sims <= ps))
    n_eq = int(np.count_nonzero(sims == ps))
    p_upper = float((n_ge + 1.0) / (n + 1.0))

    out.update({
        "sim_mean": mean,
        "sim_median": median,
        "sim_std": std,
        "sim_min": float(np.min(sims)),
        "sim_max": float(np.max(sims)),
        "sim_percentile_1": float(q[0]),
        "sim_percentile_5": float(q[1]),
        "sim_percentile_16": float(q[2]),
        "sim_percentile_50": float(q[3]),
        "sim_percentile_84": float(q[4]),
        "sim_percentile_95": float(q[5]),
        "sim_percentile_99": float(q[6]),
        "zscore_vs_sims": float((ps - mean) / std) if std > 0.0 and np.isfinite(std) else None,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "fraction_sims_ge_planck": float(n_ge / n),
        "empirical_p_upper_tail_plus_one": p_upper,
        "upper_tail_rank_descending_1_is_largest": int(n_gt + 1),
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "is_5pct_upper_tail": bool(p_upper <= 0.05),
        "is_1pct_upper_tail": bool(p_upper <= 0.01),
    })

    if p_upper <= 0.01:
        out["interpretation"] = "Planck lies in the upper 1% simulation tail for S: an unusually strong hemispherical contrast in the low-l spectral tilt for this statistic."
    elif p_upper <= 0.05:
        out["interpretation"] = "Planck lies in the upper 5% simulation tail for S: stronger hemispherical spectral-tilt contrast than most LCDM simulations."
    else:
        out["interpretation"] = "Planck is not in the upper 5% simulation tail for this low-l hemispherical spectral-tilt contrast statistic."

    warnings = []
    if n > 1 and np.isfinite(std) and std == 0.0:
        warnings.append("All finite simulation S values are identical; check map variation, healpy availability, lmax/nside compatibility, and mask handling.")
    if n > 1 and float(np.max(sims) - np.min(sims)) <= 1.0e-15:
        warnings.append("The simulation distribution is numerically collapsed to a constant.")
    if warnings:
        out["warnings"] = warnings
    return out
