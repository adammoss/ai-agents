Borderline GPT5.5 · Variation · direction · one-tailed, lower tail

Hypothesis

The Planck SMICA masked temperature map has anomalously low low-ell power, i.e. a smaller S = ln(P_low) than expected from LCDM simulations.

Method

Monopole/dipole-subtracted masked SMICA and LCDM maps were zero-filled in the mask, converted to pseudo-\(C_\ell\) up to \(\ell=30\), and summarized by \(S=\ln(P_{\rm low})\), with the lower tail testing low power.

Result Summary

Planck has \(S=7.553\) (\(P_{\rm low}=1906\,\mu{\rm K}^2\)); simulations have mean \(7.786\), median \(7.780\), and std \(0.146\). Only 42/1000 simulations are lower, giving one-tailed \(p=0.042\) (\(\approx1.7\sigma\) one-sided).

The Planck value lies in the lowest 4.2% of the LCDM simulations, supporting a low-\(\ell\) power deficit, but the significance is marginal and below a robust 2-sigma anomaly threshold.

Agent Justification

The statistic is explicitly designed to detect a low-power deficit, and the code defines the anomaly direction as the lower tail, where smaller S means lower low-ell power.

Comparison To Literature

Broad low-\(\ell\) temperature power deficits are well known from WMAP/Planck anomaly studies, including Planck Collaboration, “Planck 2018 results. VII. Isotropy and Statistics of the CMB” (A&A 641, A7; arXiv:1906.02552). However, no close literature match was identified for this exact masked log-amplitude pseudo-spectrum statistic.

test_description = "Masked Low-l Power Deficit Log-Amplitude: for each masked temperature map, remove the best-fit monopole and dipole on unmasked pixels, set masked pixels to zero, compute the cut-sky pseudo-spectrum through ell=30, and return S = ln(P_low) for P_low = (4*pi*f_sky2)^-1 sum_{ell=2}^{30} (2ell+1) Ctilde_ell. The lower tail is the low-power-deficit anomaly direction."


def analyze_map(m, _cache={}):
    import math
    import zlib
    import numpy as np

    LMAX = 30
    ELL_MIN = 2
    TINY_POWER = 1.0e-300

    def _finite_scalar(x, default=0.0):
        try:
            y = float(x)
        except Exception:
            return float(default)
        if not np.isfinite(y):
            return float(default)
        return float(y)

    def _as_flat_float(a):
        try:
            if np.ma.isMaskedArray(a):
                arr = np.asarray(a.filled(np.nan), dtype=np.float64)
            else:
                arr = np.asarray(a, dtype=np.float64)
        except Exception:
            try:
                arr = np.asarray(np.real(a), dtype=np.float64)
            except Exception:
                arr = np.asarray([], dtype=np.float64)
        return np.ravel(arr)

    def _get_hp():
        key = "healpy_for_masked_lowl_power_logamp"
        if key not in _cache:
            try:
                import healpy as hp
                _cache[key] = hp
            except Exception:
                _cache[key] = None
        return _cache.get(key)

    def _nside_from_npix(hp, npix):
        try:
            ns = int(hp.npix2nside(int(npix)))
            if ns > 0 and int(hp.nside2npix(ns)) == int(npix):
                return ns
        except Exception:
            pass
        try:
            if int(npix) > 0 and int(npix) % 12 == 0:
                ns = int(round(math.sqrt(float(npix) / 12.0)))
                if ns > 0 and 12 * ns * ns == int(npix):
                    return ns
        except Exception:
            pass
        return -1

    def _mask_signature(valid):
        v = np.asarray(valid, dtype=bool).ravel()
        try:
            crc = int(zlib.crc32(np.packbits(v).tobytes()) & 0xffffffff)
        except Exception:
            crc = int(np.count_nonzero(v) & 0xffffffff)
        return str(int(v.size)) + ":" + str(int(np.count_nonzero(v))) + ":" + str(crc)

    def _pixvecs(hp, nside):
        nside = int(nside)
        key = "pixvecs_ring_lowl_power_nside_" + str(nside)
        got = _cache.get(key)
        if got is not None:
            return got
        npix = int(hp.nside2npix(nside))
        pix = np.arange(npix, dtype=np.int64)
        try:
            x, y, z = hp.pix2vec(nside, pix, nest=False)
        except TypeError:
            x, y, z = hp.pix2vec(nside, pix)
        got = (np.asarray(x, dtype=np.float64), np.asarray(y, dtype=np.float64), np.asarray(z, dtype=np.float64))
        try:
            if int(_cache.get("pixvec_cache_count_lowl_power", 0)) < 4:
                _cache[key] = got
                _cache["pixvec_cache_count_lowl_power"] = int(_cache.get("pixvec_cache_count_lowl_power", 0)) + 1
        except Exception:
            pass
        return got

    def _fit_geometry(hp, nside, valid):
        valid = np.asarray(valid, dtype=bool).ravel()
        sig = _mask_signature(valid)
        key = "monopole_dipole_geom_lowl_power_" + str(int(nside)) + "_" + sig
        got = _cache.get(key)
        if got is not None:
            return got
        x, y, z = _pixvecs(hp, int(nside))
        if int(x.size) != int(valid.size):
            return None
        xv = x[valid]
        yv = y[valid]
        zv = z[valid]
        n = int(xv.size)
        if n < 4:
            return None
        # Normal matrix for the uniform-pixel least-squares fit to [1, x, y, z].
        g = np.array([
            [float(n), float(np.sum(xv)), float(np.sum(yv)), float(np.sum(zv))],
            [float(np.sum(xv)), float(np.dot(xv, xv)), float(np.dot(xv, yv)), float(np.dot(xv, zv))],
            [float(np.sum(yv)), float(np.dot(yv, xv)), float(np.dot(yv, yv)), float(np.dot(yv, zv))],
            [float(np.sum(zv)), float(np.dot(zv, xv)), float(np.dot(zv, yv)), float(np.dot(zv, zv))]
        ], dtype=np.float64)
        invg = None
        try:
            invg = np.linalg.inv(g)
            if not np.all(np.isfinite(invg)):
                invg = None
        except Exception:
            invg = None
        got = {"x": x, "y": y, "z": z, "xv": xv, "yv": yv, "zv": zv, "g": g, "invg": invg}
        try:
            if int(_cache.get("md_geom_cache_count_lowl_power", 0)) < 8:
                _cache[key] = got
                _cache["md_geom_cache_count_lowl_power"] = int(_cache.get("md_geom_cache_count_lowl_power", 0)) + 1
        except Exception:
            pass
        return got

    def _remove_monopole_dipole_zero_mask(arr, valid, hp, nside):
        arr = np.asarray(arr, dtype=np.float64).ravel()
        valid = np.asarray(valid, dtype=bool).ravel()
        out = np.zeros_like(arr, dtype=np.float64)
        if int(arr.size) != int(valid.size):
            return out
        vals = arr[valid]
        if int(vals.size) <= 0:
            return out
        geom = _fit_geometry(hp, int(nside), valid)
        if geom is None or int(vals.size) < 4:
            mu = float(np.mean(vals)) if int(vals.size) else 0.0
            resid = vals - mu
            out[valid] = np.where(np.isfinite(resid), resid, 0.0)
            return out
        rhs = np.array([
            float(np.sum(vals)),
            float(np.dot(vals, geom["xv"])),
            float(np.dot(vals, geom["yv"])),
            float(np.dot(vals, geom["zv"]))
        ], dtype=np.float64)
        try:
            if geom.get("invg", None) is not None:
                beta = np.dot(geom["invg"], rhs)
            else:
                beta = np.linalg.solve(geom["g"], rhs)
        except Exception:
            try:
                beta = np.linalg.lstsq(geom["g"], rhs, rcond=None)[0]
            except Exception:
                beta = np.array([float(np.mean(vals)), 0.0, 0.0, 0.0], dtype=np.float64)
        beta = np.where(np.isfinite(beta), beta, 0.0)
        model_valid = beta[0] + beta[1] * geom["xv"] + beta[2] * geom["yv"] + beta[3] * geom["zv"]
        resid = vals - model_valid
        out[valid] = np.where(np.isfinite(resid), resid, 0.0)
        # Pixels outside valid remain exactly zero, implementing the binary mask W.
        return out

    def _fallback_log_low_power(arr, valid):
        # Last-resort finite value if healpy is unavailable. This is not the primary statistic,
        # but preserves a non-constant log-amplitude based on the unmasked residual variance.
        try:
            vals = np.asarray(arr, dtype=np.float64).ravel()[np.asarray(valid, dtype=bool).ravel()]
            vals = vals[np.isfinite(vals)]
            if int(vals.size) <= 1:
                return float(math.log(TINY_POWER))
            vals = vals - float(np.mean(vals))
            p = float(np.mean(vals * vals))
            if (not np.isfinite(p)) or p <= 0.0:
                p = TINY_POWER
            return float(math.log(max(p, TINY_POWER)))
        except Exception:
            return float(math.log(TINY_POWER))

    arr = _as_flat_float(m)
    npix = int(arr.size)
    if npix <= 0:
        return float(math.log(TINY_POWER))
    valid = np.isfinite(arr)
    n_valid = int(np.count_nonzero(valid))
    if n_valid <= 0:
        return float(math.log(TINY_POWER))
    f_sky2 = float(np.mean(valid.astype(np.float64)))  # binary W: mean(W^2)=mean(W)
    if (not np.isfinite(f_sky2)) or f_sky2 <= 0.0:
        return float(math.log(TINY_POWER))

    hp = _get_hp()
    if hp is None:
        return float(_fallback_log_low_power(arr, valid))
    nside = _nside_from_npix(hp, npix)
    if nside <= 0:
        return float(_fallback_log_low_power(arr, valid))

    try:
        cleaned = _remove_monopole_dipole_zero_mask(arr, valid, hp, nside)
        # Never pass NaNs/infs into healpy. The mask is already applied by zeros outside valid.
        cleaned = np.where(np.isfinite(cleaned), cleaned, 0.0).astype(np.float64, copy=False)

        lmax_use = int(LMAX)
        try:
            max_l_by_nside = int(3 * int(nside) - 1)
            if max_l_by_nside >= ELL_MIN:
                lmax_use = min(lmax_use, max_l_by_nside)
        except Exception:
            pass
        if lmax_use < ELL_MIN:
            return float(_fallback_log_low_power(arr, valid))

        try:
            cl = hp.anafast(cleaned, lmax=lmax_use, iter=0, pol=False, use_weights=False)
        except TypeError:
            try:
                cl = hp.anafast(cleaned, lmax=lmax_use, iter=0, pol=False)
            except TypeError:
                cl = hp.anafast(cleaned, lmax=lmax_use)
        cl = np.asarray(cl, dtype=np.float64).ravel()
        if int(cl.size) <= ELL_MIN:
            return float(_fallback_log_low_power(arr, valid))
        ell_hi = min(lmax_use, int(cl.size) - 1, LMAX)
        if ell_hi < ELL_MIN:
            return float(_fallback_log_low_power(arr, valid))
        ells = np.arange(ELL_MIN, ell_hi + 1, dtype=np.float64)
        cls = cl[ELL_MIN:ell_hi + 1]
        cls = np.where(np.isfinite(cls), cls, 0.0)
        weighted_sum = float(np.dot(2.0 * ells + 1.0, cls))
        p_low = weighted_sum / (4.0 * math.pi * f_sky2)
        if (not np.isfinite(p_low)) or p_low <= 0.0:
            p_low = TINY_POWER
        s = float(math.log(max(p_low, TINY_POWER)))
        if not np.isfinite(s):
            s = float(math.log(TINY_POWER))
        return float(s)
    except Exception:
        return float(_fallback_log_low_power(arr, valid))


def summarize_results(planck_stat, sim_results, _desc=test_description):
    import numpy as np
    import math

    def _finite_float(x, default=np.nan):
        try:
            y = float(x)
        except Exception:
            return float(default)
        return float(y) if np.isfinite(y) else float(default)

    ps = _finite_float(planck_stat)
    try:
        sims = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            for v in sim_results:
                vals.append(_finite_float(v))
        except Exception:
            vals = []
        sims = np.asarray(vals, dtype=np.float64)
    sims = sims[np.isfinite(sims)]
    n = int(sims.size)

    out = {
        "test_description": _desc,
        "statistic": "S = ln(P_low), where P_low is the f_sky2-corrected cut-sky pseudo-spectrum power summed over 2 <= ell <= 30",
        "planck_stat_log_microK2": ps,
        "planck_P_low_microK2": float(math.exp(ps)) if np.isfinite(ps) and ps < 700.0 else None,
        "n_sims": n,
        "tail_of_interest": "lower tail (smaller S means lower low-ell power)",
        "multipole_range": "2 <= ell <= 30",
        "mask_normalization": "f_sky2 = mean(W^2), with the supplied finite-pixel mask treated as binary W=1 unmasked and W=0 masked",
        "monopole_dipole_treatment": "best-fit monopole plus Cartesian dipole removed on unmasked pixels before zero-filling the mask and computing the pseudo-spectrum",
        "plot_spec": {
            "kind": "histogram",
            "title": "Masked low-ell power log-amplitude",
            "xlabel": "S = ln(P_low / microK^2)",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "Distribution of log cut-sky low-ell power from the LCDM simulations. The low-power-deficit anomaly corresponds to the lower tail."
        }
    }

    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_le = int(np.count_nonzero(sims <= ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_ge = int(np.count_nonzero(sims >= ps))
    n_eq = int(n - n_lt - n_gt)

    # Plus-one empirical p-values are stable for finite simulation ensembles and handle zero counts.
    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_two = float(min(1.0, 2.0 * min(p_lower, p_upper)))
    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    z = None
    if np.isfinite(std) and std > 0.0:
        z = float((ps - mean) / std)
    q = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])

    # Ratios are easiest to interpret in linear power, while all ranking is done in S.
    median_power = float(math.exp(median)) if median < 700.0 else None
    planck_power = float(math.exp(ps)) if ps < 700.0 else None
    power_ratio_to_median = None
    if planck_power is not None and median_power is not None and median_power > 0.0:
        power_ratio_to_median = float(planck_power / median_power)

    out.update({
        "sim_mean_log_microK2": mean,
        "sim_median_log_microK2": median,
        "sim_std_log_microK2": std,
        "sim_min_log_microK2": float(np.min(sims)),
        "sim_max_log_microK2": float(np.max(sims)),
        "sim_percentile_1_log_microK2": float(q[0]),
        "sim_percentile_5_log_microK2": float(q[1]),
        "sim_percentile_16_log_microK2": float(q[2]),
        "sim_percentile_50_log_microK2": float(q[3]),
        "sim_percentile_84_log_microK2": float(q[4]),
        "sim_percentile_95_log_microK2": float(q[5]),
        "sim_percentile_99_log_microK2": float(q[6]),
        "planck_zscore_vs_sims": z,
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "rank_ascending_1_is_lowest_power": int(n_lt + 1),
        "rank_descending_1_is_highest_power": int(n_gt + 1),
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "fraction_sims_lt_planck": float(n_lt / n),
        "fraction_sims_le_planck": float(n_le / n),
        "empirical_p_lower_tail_power_deficit_including_ties_plus_one": p_lower,
        "empirical_p_upper_tail_power_excess_including_ties_plus_one": p_upper,
        "empirical_p_two_sided_plus_one": p_two,
        "planck_power_ratio_to_sim_median": power_ratio_to_median,
        "log_difference_planck_minus_sim_median": float(ps - median),
        "is_lower_tail_5pct_power_deficit": bool(p_lower <= 0.05),
        "is_lower_tail_1pct_power_deficit": bool(p_lower <= 0.01),
        "is_two_sided_5pct": bool(p_two <= 0.05),
        "is_two_sided_1pct": bool(p_two <= 0.01)
    })

    if std == 0.0:
        out["warning"] = "All finite simulation statistics are identical; check the input maps, mask, and healpy harmonic-transform support."
    elif p_lower <= 0.01:
        out["interpretation"] = "Planck is in the lowest 1% of simulations for masked low-ell power, a strong low-power deficit by this statistic."
    elif p_lower <= 0.05:
        out["interpretation"] = "Planck is in the lowest 5% of simulations for masked low-ell power, indicating a low-power deficit by this statistic."
    else:
        out["interpretation"] = "Planck is not in the 5% lower tail for this masked low-ell power statistic."
    return out
