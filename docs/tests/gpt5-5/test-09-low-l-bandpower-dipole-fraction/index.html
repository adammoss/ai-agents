Consistent GPT5.5 · Variation · power · one-tailed, upper tail

Hypothesis

The Planck SMICA map has an anomalously large fractional dipole amplitude in the squared low-ell temperature field, indicating stronger low-l bandpower asymmetry than expected under LCDM simulations.

Method

Mean-subtracted masked maps were pseudo-alm filtered to 2<=ell<=64, squared on the unmasked sky, and fit as P(n)=p0+d·n; the statistic is A_P=|d|/p0.

Result Summary

Planck SMICA gives A_P=0.3368. The 1000 LCDM simulations have mean 0.2182 and std 0.0961; 113 simulations are larger, giving an upper-tail p=0.113 and z≈1.23.

SMICA shows a higher-than-average low-ell bandpower dipole fraction, but it is not in the upper 5% tail and is not statistically anomalous under the LCDM simulations.

Agent Justification

The statistic A_P=|d|/p0 is a nonnegative dipole-amplitude measure, and the code explicitly treats larger A_P values as more anomalous, ranking SMICA in the upper tail of the simulation distribution.

Comparison To Literature

This is related to the well-documented hemispherical power asymmetry/dipole-modulation and local-variance dipole literature, including Planck 2018 isotropy results, which confirm large-angle temperature anomalies while finding broad consistency with LCDM elsewhere (Planck Collaboration 2020, A&A 641, A7; arXiv:1906.02552, https://arxiv.org/abs/1906.02552). The present squared low-l bandpower-dipole statistic is not identical to those standard estimators.

test_description = "Low-l Bandpower Dipole Fraction: remove the unmasked-sky mean, compute masked pseudo-alm coefficients with masked pixels filled by zero, keep 2<=ell<=64, reconstruct the low-l temperature map, square it on the unmasked sky, fit P(n)=p0+d.n by equal-weight least squares, and return A_P=|d|/p0."


def analyze_map(m, _cache={}):
    import math
    import zlib
    import numpy as np

    LMAX_REQUESTED = 64
    STAT_CAP = 1.0e6

    def _clean_float(x, default=0.0):
        try:
            y = float(x)
        except Exception:
            return float(default)
        if np.isfinite(y):
            return float(y)
        return float(default)

    def _finish_stat(x):
        y = _clean_float(x, 0.0)
        if y < 0.0:
            y = 0.0
        if y > STAT_CAP:
            y = STAT_CAP
        return float(y)

    def _as_flat_float(x):
        try:
            if np.ma.isMaskedArray(x):
                a = np.asarray(x.filled(np.nan), dtype=np.float64)
            else:
                a = np.asarray(x, dtype=np.float64)
        except Exception:
            try:
                a = np.asarray(np.real(x), dtype=np.float64)
            except Exception:
                a = np.asarray([], dtype=np.float64)
        return np.ravel(a)

    def _mask_sig(valid):
        v = np.asarray(valid, dtype=bool).ravel()
        n = int(v.size)
        c = int(np.count_nonzero(v))
        try:
            packed = np.packbits(v)
            crc = int(zlib.crc32(packed.tobytes()) & 0xffffffff)
        except Exception:
            crc = int(c & 0xffffffff)
        return str(n) + ":" + str(c) + ":" + str(crc)

    def _get_hp():
        if "hp_loaded" not in _cache:
            _cache["hp_loaded"] = True
            try:
                import healpy as hp
                _cache["hp"] = hp
            except Exception:
                _cache["hp"] = None
        return _cache.get("hp", None)

    def _nside_from_npix(hp, npix):
        try:
            ns = int(hp.npix2nside(int(npix)))
            if ns > 0 and int(hp.nside2npix(ns)) == int(npix):
                return ns
        except Exception:
            pass
        try:
            if int(npix) > 0 and int(npix) % 12 == 0:
                ns = int(round(math.sqrt(float(npix) / 12.0)))
                if ns > 0 and 12 * ns * ns == int(npix):
                    return ns
        except Exception:
            pass
        return -1

    def _geometry(hp, nside, valid):
        sig = _mask_sig(valid)
        key = "geom_low_l_bandpower_dipole_" + str(int(nside)) + "_" + sig
        cached = _cache.get(key)
        if cached is not None:
            return cached
        pix = np.flatnonzero(np.asarray(valid, dtype=bool)).astype(np.int64, copy=False)
        if int(pix.size) < 8:
            return None
        try:
            x, y, z = hp.pix2vec(int(nside), pix, nest=False)
        except TypeError:
            x, y, z = hp.pix2vec(int(nside), pix)
        except Exception:
            return None
        x = np.asarray(x, dtype=np.float64)
        y = np.asarray(y, dtype=np.float64)
        z = np.asarray(z, dtype=np.float64)
        if int(x.size) != int(pix.size) or int(y.size) != int(pix.size) or int(z.size) != int(pix.size):
            return None
        n = float(pix.size)
        sx = float(np.sum(x))
        sy = float(np.sum(y))
        sz = float(np.sum(z))
        gram = np.empty((4, 4), dtype=np.float64)
        gram[0, 0] = n
        gram[0, 1] = sx
        gram[1, 0] = sx
        gram[0, 2] = sy
        gram[2, 0] = sy
        gram[0, 3] = sz
        gram[3, 0] = sz
        gram[1, 1] = float(np.dot(x, x))
        gram[1, 2] = float(np.dot(x, y))
        gram[2, 1] = gram[1, 2]
        gram[1, 3] = float(np.dot(x, z))
        gram[3, 1] = gram[1, 3]
        gram[2, 2] = float(np.dot(y, y))
        gram[2, 3] = float(np.dot(y, z))
        gram[3, 2] = gram[2, 3]
        gram[3, 3] = float(np.dot(z, z))
        if not np.all(np.isfinite(gram)):
            return None
        try:
            pinv = np.linalg.pinv(gram, rcond=1.0e-12)
        except TypeError:
            pinv = np.linalg.pinv(gram)
        except Exception:
            return None
        pinv = np.asarray(pinv, dtype=np.float64)
        if pinv.shape != (4, 4) or not np.all(np.isfinite(pinv)):
            return None
        geom = {"pix": pix, "x": x, "y": y, "z": z, "pinv": pinv}
        _cache[key] = geom
        return geom

    def _zero_unwanted_ells(hp, alm, lmax):
        try:
            lmax_i = int(lmax)
            for ell in range(0, min(2, lmax_i + 1)):
                for mm in range(ell + 1):
                    alm[hp.Alm.getidx(lmax_i, ell, mm)] = 0.0 + 0.0j
        except Exception:
            pass
        return alm

    def _fallback(arr, valid):
        vals = np.asarray(arr[valid], dtype=np.float64)
        vals = vals[np.isfinite(vals)]
        n = int(vals.size)
        if n < 8:
            return 0.0
        vals = vals - float(np.mean(vals))
        power = vals * vals
        p0 = float(np.mean(power)) if n > 0 else 0.0
        if not np.isfinite(p0) or p0 <= 0.0:
            return 0.0
        i = np.arange(n, dtype=np.float64)
        zz = 1.0 - 2.0 * ((i + 0.5) / float(n))
        phi = (2.39996322972865332 * i) % (2.0 * np.pi)
        rr = np.sqrt(np.maximum(0.0, 1.0 - zz * zz))
        xx = rr * np.cos(phi)
        yy = rr * np.sin(phi)
        rhs = np.array([float(np.sum(power)), float(np.dot(xx, power)), float(np.dot(yy, power)), float(np.dot(zz, power))], dtype=np.float64)
        gram = np.empty((4, 4), dtype=np.float64)
        gram[0, 0] = float(n)
        gram[0, 1] = float(np.sum(xx))
        gram[1, 0] = gram[0, 1]
        gram[0, 2] = float(np.sum(yy))
        gram[2, 0] = gram[0, 2]
        gram[0, 3] = float(np.sum(zz))
        gram[3, 0] = gram[0, 3]
        gram[1, 1] = float(np.dot(xx, xx))
        gram[1, 2] = float(np.dot(xx, yy))
        gram[2, 1] = gram[1, 2]
        gram[1, 3] = float(np.dot(xx, zz))
        gram[3, 1] = gram[1, 3]
        gram[2, 2] = float(np.dot(yy, yy))
        gram[2, 3] = float(np.dot(yy, zz))
        gram[3, 2] = gram[2, 3]
        gram[3, 3] = float(np.dot(zz, zz))
        try:
            coeff = np.dot(np.linalg.pinv(gram), rhs)
        except Exception:
            return 0.0
        coeff = np.asarray(coeff, dtype=np.float64).ravel()
        if coeff.size < 4 or not np.all(np.isfinite(coeff[:4])):
            return 0.0
        denom = float(coeff[0])
        if not np.isfinite(denom) or denom <= 0.0:
            denom = p0
        if not np.isfinite(denom) or denom <= 0.0:
            return 0.0
        d2 = float(np.dot(coeff[1:4], coeff[1:4]))
        if not np.isfinite(d2) or d2 < 0.0:
            return 0.0
        return _finish_stat(math.sqrt(d2) / denom)

    arr = _as_flat_float(m)
    npix = int(arr.size)
    if npix <= 0:
        return 0.0
    valid = np.isfinite(arr) & (np.abs(arr) < 1.0e25)
    if int(np.count_nonzero(valid)) < 8:
        return 0.0

    hp = _get_hp()
    if hp is None:
        return float(_fallback(arr, valid))
    nside = _nside_from_npix(hp, npix)
    if nside < 1:
        return float(_fallback(arr, valid))
    lmax_use = int(min(LMAX_REQUESTED, max(0, 3 * int(nside) - 1)))
    if lmax_use < 2:
        return float(_fallback(arr, valid))

    try:
        geom = _geometry(hp, nside, valid)
        if geom is None:
            return float(_fallback(arr, valid))
        pix = geom["pix"]
        vals = np.asarray(arr[pix], dtype=np.float64)
        vals = np.where(np.isfinite(vals), vals, 0.0)
        mu = float(np.mean(vals)) if vals.size > 0 else 0.0
        if not np.isfinite(mu):
            mu = 0.0
        filled = np.zeros(npix, dtype=np.float64)
        filled[pix] = vals - mu
        filled = np.where(np.isfinite(filled), filled, 0.0)
        try:
            alm = hp.map2alm(filled, lmax=lmax_use, iter=0, pol=False, use_weights=False)
        except TypeError:
            try:
                alm = hp.map2alm(filled, lmax=lmax_use, iter=0, pol=False)
            except TypeError:
                alm = hp.map2alm(filled, lmax=lmax_use, iter=0)
        alm = np.asarray(alm, dtype=np.complex128)
        if alm.size <= 0:
            return 0.0
        okalm = np.isfinite(alm.real) & np.isfinite(alm.imag)
        alm = np.where(okalm, alm, 0.0 + 0.0j)
        alm = _zero_unwanted_ells(hp, alm, lmax_use)
        try:
            low_map = hp.alm2map(alm, int(nside), lmax=lmax_use, pol=False, verbose=False)
        except TypeError:
            try:
                low_map = hp.alm2map(alm, int(nside), lmax=lmax_use, pol=False)
            except TypeError:
                low_map = hp.alm2map(alm, int(nside), lmax=lmax_use)
        low_map = np.asarray(low_map, dtype=np.float64).ravel()
        if int(low_map.size) != npix:
            return 0.0
        tl = np.asarray(low_map[pix], dtype=np.float64)
        tl = np.where(np.isfinite(tl), tl, 0.0)
        power = tl * tl
        mean_power = float(np.mean(power)) if power.size > 0 else 0.0
        if not np.isfinite(mean_power) or mean_power <= 0.0:
            return 0.0
        rhs = np.array([float(np.sum(power)), float(np.dot(geom["x"], power)), float(np.dot(geom["y"], power)), float(np.dot(geom["z"], power))], dtype=np.float64)
        if not np.all(np.isfinite(rhs)):
            return 0.0
        coeff = np.dot(geom["pinv"], rhs)
        coeff = np.asarray(coeff, dtype=np.float64).ravel()
        if coeff.size < 4 or not np.all(np.isfinite(coeff[:4])):
            return 0.0
        p0 = float(coeff[0])
        if not np.isfinite(p0) or p0 <= 0.0:
            p0 = mean_power
        if not np.isfinite(p0) or p0 <= 0.0:
            return 0.0
        d2 = float(np.dot(coeff[1:4], coeff[1:4]))
        if not np.isfinite(d2) or d2 < 0.0:
            return 0.0
        return _finish_stat(math.sqrt(d2) / p0)
    except Exception:
        return float(_fallback(arr, valid))


def summarize_results(planck_stat, sim_results, _desc=test_description):
    import numpy as np

    def _ff(x, default=np.nan):
        try:
            y = float(x)
        except Exception:
            return float(default)
        if np.isfinite(y):
            return float(y)
        return float(default)

    ps = _ff(planck_stat)
    try:
        sims = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            for v in sim_results:
                vals.append(_ff(v))
        except Exception:
            vals = []
        sims = np.asarray(vals, dtype=np.float64)
    sims = sims[np.isfinite(sims)]
    n = int(sims.size)
    out = {
        "test_description": _desc,
        "statistic": "A_P = |d|/p0 from P(n)=T_L(n)^2 after a masked-pseudo-alm 2<=ell<=64 filter",
        "planck_stat": ps,
        "n_sims": n,
        "tail_of_interest": "upper tail; larger A_P means a stronger dipole in the low-l bandpower map",
        "higher_is_more_anomalous": True,
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-l bandpower dipole fraction",
            "xlabel": "A_P = |d|/p0",
            "ylabel": "Number of simulations",
            "bins": 30,
            "caption": "Distribution of the fractional dipole amplitude of the squared low-l temperature field. The SMICA value is compared with the upper tail of the simulations."
        }
    }
    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_le = int(np.count_nonzero(sims <= ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_ge = int(np.count_nonzero(sims >= ps))
    n_eq = int(n - n_lt - n_gt)
    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_two = float(min(1.0, 2.0 * min(p_upper, p_lower)))
    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    zscore = None
    if np.isfinite(std) and std > 0.0:
        zscore = float((ps - mean) / std)
    qs = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])
    out.update({
        "sim_mean": mean,
        "sim_median": median,
        "sim_std": std,
        "sim_min": float(np.min(sims)),
        "sim_max": float(np.max(sims)),
        "sim_percentile_1": float(qs[0]),
        "sim_percentile_5": float(qs[1]),
        "sim_percentile_16": float(qs[2]),
        "sim_percentile_50": float(qs[3]),
        "sim_percentile_84": float(qs[4]),
        "sim_percentile_95": float(qs[5]),
        "sim_percentile_99": float(qs[6]),
        "planck_zscore_vs_sims": zscore,
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "rank_descending_1_is_largest_A_P": int(n_gt + 1),
        "rank_ascending_1_is_smallest_A_P": int(n_lt + 1),
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "fraction_sims_gt_planck": float(n_gt / n),
        "fraction_sims_ge_planck": float(n_ge / n),
        "empirical_p_upper_tail_including_ties_plus_one": p_upper,
        "empirical_p_lower_tail_including_ties_plus_one": p_lower,
        "empirical_p_two_sided_extremeness": p_two,
        "is_upper_tail_5pct": bool(p_upper <= 0.05),
        "is_upper_tail_1pct": bool(p_upper <= 0.01)
    })
    if std == 0.0:
        out["warning"] = "All finite simulation statistics are identical; check for degenerate inputs or missing harmonic-transform support."
    elif p_upper <= 0.01:
        out["interpretation"] = "SMICA lies in the highest 1% tail for this low-l bandpower dipole fraction."
    elif p_upper <= 0.05:
        out["interpretation"] = "SMICA lies in the highest 5% tail for this low-l bandpower dipole fraction."
    else:
        out["interpretation"] = "SMICA is not in the highest 5% tail for this low-l bandpower dipole fraction."
    return out
