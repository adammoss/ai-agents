Hypothesis

The Planck quadrupole and octopole area normals are unusually aligned with the ecliptic pole, meaning the quadrupole and octopole planes are more parallel to the ecliptic plane than expected in LCDM simulations.

Method

Multipole-vector area normals were derived from masked low-ell Planck SMICA and 1000 LCDM simulations using the same least-squares fit; the statistic A is the mean absolute dot product of the quadrupole/octopole area normals with the ecliptic pole, with larger A indicating stronger ecliptic-plane alignment.

Result Summary

Planck gives A = 0.199, well below the simulation mean 0.495 and median 0.490. The pre-specified upper-tail p-value is 0.985, so Planck is not unusually high. It is instead low relative to simulations, with z ≈ -2.12 and a two-sided empirical rank p ≈ 0.032.

The tested hypothesis is not supported: the Planck quadrupole/octopole planes are not unusually parallel to the ecliptic plane under this statistic. If anything, this pipeline finds an opposite-direction low-A deviation.

Agent Justification

The statistic A increases with stronger ecliptic alignment, and the code explicitly evaluates the anomaly using the upper-tail simulation rank; only unusually large A values are treated as anomalous.

Comparison To Literature

Broad quadrupole-octopole and ecliptic-alignment anomalies are discussed in the CMB literature, but no directly matching study of this exact masked equal-area least-squares mean area-normal/ecliptic statistic was identified in the supplied search results.

def analyze_map(m, _state={"geom_cache": {}, "fit_cache": {}, "null_basis": None}):
    import math
    import numpy as np

    MAX_FIT_CACHE = 4
    STORE_MATRIX_NVALID_MAX = 300000
    CHUNK = 200000

    def _infer_nside(npix):
        try:
            npix_i = int(npix)
            if npix_i <= 0 or npix_i % 12 != 0:
                return -1
            nside = int(round(math.sqrt(npix_i / 12.0)))
            if nside <= 0 or 12 * nside * nside != npix_i:
                return -1
            return nside
        except Exception:
            return -1

    def _get_geometry(nside, npix):
        key = (int(nside), int(npix))
        cache = _state["geom_cache"]
        item = cache.get(key)
        if item is not None:
            return item
        import healpy as hp
        pix = np.arange(int(npix), dtype=np.int64)
        x, y, z = hp.pix2vec(int(nside), pix, nest=False)
        x = np.asarray(x, dtype=np.float64).ravel()
        y = np.asarray(y, dtype=np.float64).ravel()
        z = np.asarray(z, dtype=np.float64).ravel()
        if x.size != int(npix) or y.size != int(npix) or z.size != int(npix):
            raise ValueError("bad HEALPix geometry")
        if len(cache) > 2:
            cache.clear()
        cache[key] = (x, y, z)
        return x, y, z

    def _basis_matrix(x, y, z):
        x = np.asarray(x, dtype=np.float64)
        y = np.asarray(y, dtype=np.float64)
        z = np.asarray(z, dtype=np.float64)
        n = int(x.size)
        B = np.empty((n, 16), dtype=np.float64)
        xx = x * x
        yy = y * y
        zz = z * z
        xy = x * y
        xz = x * z
        yz = y * z

        # ell=0 and ell=1 nuisance terms
        B[:, 0] = 1.0
        B[:, 1] = x
        B[:, 2] = y
        B[:, 3] = z

        # A real homogeneous harmonic-polynomial basis for ell=2 on the unit sphere.
        B[:, 4] = 0.5 * (2.0 * zz - xx - yy)
        B[:, 5] = xz
        B[:, 6] = yz
        B[:, 7] = xx - yy
        B[:, 8] = 2.0 * xy

        # A real homogeneous harmonic-polynomial basis for ell=3 on the unit sphere.
        B[:, 9] = 0.5 * z * (2.0 * zz - 3.0 * xx - 3.0 * yy)
        common_m1 = 4.0 * zz - xx - yy
        B[:, 10] = x * common_m1
        B[:, 11] = y * common_m1
        B[:, 12] = z * (xx - yy)
        B[:, 13] = 2.0 * xy * z
        B[:, 14] = x * (xx - 3.0 * yy)
        B[:, 15] = y * (3.0 * xx - yy)
        return B

    def _gram_chunked(x, y, z, idx):
        G = np.zeros((16, 16), dtype=np.float64)
        idx = np.asarray(idx, dtype=np.int64)
        for start in range(0, int(idx.size), CHUNK):
            ii = idx[start:start + CHUNK]
            B = _basis_matrix(x[ii], y[ii], z[ii])
            G += B.T.dot(B)
        return G

    def _rhs_chunked(a, x, y, z, idx):
        rhs = np.zeros(16, dtype=np.float64)
        idx = np.asarray(idx, dtype=np.int64)
        for start in range(0, int(idx.size), CHUNK):
            ii = idx[start:start + CHUNK]
            B = _basis_matrix(x[ii], y[ii], z[ii])
            rhs += B.T.dot(np.asarray(a[ii], dtype=np.float64))
        return rhs

    def _get_fit_entry(nside, npix, valid, x, y, z):
        nvalid = int(np.count_nonzero(valid))
        packed = np.packbits(np.asarray(valid, dtype=np.bool_))
        mask_bytes = packed.tobytes()
        key = (int(nside), int(npix), int(nvalid), hash(mask_bytes))
        cache = _state["fit_cache"]
        entry = cache.get(key)
        if entry is not None and entry.get("mask_bytes") == mask_bytes:
            return entry

        if len(cache) >= MAX_FIT_CACHE:
            cache.clear()

        if nvalid <= STORE_MATRIX_NVALID_MAX:
            xv = x[valid]
            yv = y[valid]
            zv = z[valid]
            X = _basis_matrix(xv, yv, zv)
            G = X.T.dot(X)
            solver = np.linalg.pinv(G, rcond=1.0e-12)
            entry = {"mode": "matrix", "X": X, "solver": solver, "mask_bytes": mask_bytes}
        else:
            idx = np.flatnonzero(valid).astype(np.int64, copy=False)
            G = _gram_chunked(x, y, z, idx)
            solver = np.linalg.pinv(G, rcond=1.0e-12)
            entry = {"mode": "chunk", "idx": idx, "solver": solver, "mask_bytes": mask_bytes}

        cache[key] = entry
        return entry

    def _fit_low_l_coefficients(a, valid, nside, npix):
        x, y, z = _get_geometry(nside, npix)
        entry = _get_fit_entry(nside, npix, valid, x, y, z)
        if entry.get("mode") == "matrix":
            vals = np.asarray(a[valid], dtype=np.float64)
            rhs = entry["X"].T.dot(vals)
        else:
            rhs = _rhs_chunked(a, x, y, z, entry["idx"])
        coeff = np.asarray(entry["solver"].dot(rhs), dtype=np.float64).ravel()
        if coeff.size != 16 or not np.all(np.isfinite(coeff)):
            raise ValueError("bad low-l fit")
        return coeff

    def _pad_poly(p, n):
        p = np.asarray(p, dtype=np.complex128).ravel()
        out = np.zeros(int(n), dtype=np.complex128)
        out[:min(out.size, p.size)] = p[:min(out.size, p.size)]
        return out

    def _lincomb_poly(terms):
        max_len = 1
        for _, p in terms:
            max_len = max(max_len, len(p))
        out = np.zeros(max_len, dtype=np.complex128)
        for c, p in terms:
            pp = np.asarray(p, dtype=np.complex128).ravel()
            out[:pp.size] += c * pp
        return out

    def _get_null_poly_basis():
        item = _state.get("null_basis")
        if item is not None:
            return item

        # Null-cone parametrization r(zeta) = (1-zeta^2, i(1+zeta^2), 2 zeta).
        px = np.array([1.0, 0.0, -1.0], dtype=np.complex128)
        py = np.array([1.0j, 0.0, 1.0j], dtype=np.complex128)
        pz = np.array([0.0, 2.0], dtype=np.complex128)
        x2 = np.convolve(px, px)
        y2 = np.convolve(py, py)
        z2 = np.convolve(pz, pz)
        xy = np.convolve(px, py)
        xz = np.convolve(px, pz)
        yz = np.convolve(py, pz)

        b2 = []
        b2.append(0.5 * _lincomb_poly([(2.0, z2), (-1.0, x2), (-1.0, y2)]))
        b2.append(xz)
        b2.append(yz)
        b2.append(_lincomb_poly([(1.0, x2), (-1.0, y2)]))
        b2.append(2.0 * xy)
        B2 = np.vstack([_pad_poly(p, 5) for p in b2])

        inner_o0 = _lincomb_poly([(2.0, z2), (-3.0, x2), (-3.0, y2)])
        inner_m1 = _lincomb_poly([(4.0, z2), (-1.0, x2), (-1.0, y2)])
        inner_m2c = _lincomb_poly([(1.0, x2), (-1.0, y2)])
        inner_m3c = _lincomb_poly([(1.0, x2), (-3.0, y2)])
        inner_m3s = _lincomb_poly([(3.0, x2), (-1.0, y2)])

        b3 = []
        b3.append(0.5 * np.convolve(pz, inner_o0))
        b3.append(np.convolve(px, inner_m1))
        b3.append(np.convolve(py, inner_m1))
        b3.append(np.convolve(pz, inner_m2c))
        b3.append(2.0 * np.convolve(np.convolve(px, py), pz))
        b3.append(np.convolve(px, inner_m3c))
        b3.append(np.convolve(py, inner_m3s))
        B3 = np.vstack([_pad_poly(p, 7) for p in b3])

        _state["null_basis"] = (B2, B3)
        return B2, B3

    def _root_abs2(z):
        return float(z.real * z.real + z.imag * z.imag)

    def _partner_chordal_dist2(a, b):
        # Roots for one multipole vector are antipodal partners on the Riemann
        # sphere: b = -1/conj(a), with a=0 paired to infinity and vice versa.
        tiny = 1.0e-300
        huge = 1.0e150

        if a is None:
            if b is None:
                return 1.0
            rb2 = _root_abs2(b)
            if not np.isfinite(rb2):
                return 1.0
            return float(rb2 / (1.0 + rb2))

        ra = abs(a)
        if (not np.isfinite(ra)) or ra > huge:
            if b is None:
                return 0.0
            rb2 = _root_abs2(b)
            if not np.isfinite(rb2):
                return 1.0
            return float(rb2 / (1.0 + rb2))

        if ra < tiny:
            if b is None:
                return 0.0
            rb2 = _root_abs2(b)
            if not np.isfinite(rb2):
                return 0.0
            return float(1.0 / (1.0 + rb2))

        partner = -1.0 / np.conj(a)
        if b is None:
            rp2 = _root_abs2(partner)
            return float(1.0 / (1.0 + rp2))

        rb = abs(b)
        if (not np.isfinite(rb)) or rb > huge:
            rp2 = _root_abs2(partner)
            return float(1.0 / (1.0 + rp2))

        rb2 = _root_abs2(b)
        rp2 = _root_abs2(partner)
        num = _root_abs2(b - partner)
        den = (1.0 + rb2) * (1.0 + rp2)
        if den <= 0.0 or not np.isfinite(num) or not np.isfinite(den):
            return float("inf")
        return float(num / den)

    def _vector_from_root(root):
        if root is None:
            return np.array([0.0, 0.0, 1.0], dtype=np.float64)
        zeta = complex(root)
        if not (np.isfinite(zeta.real) and np.isfinite(zeta.imag)):
            return np.array([0.0, 0.0, 1.0], dtype=np.float64)
        r2 = zeta.real * zeta.real + zeta.imag * zeta.imag
        if (not np.isfinite(r2)) or r2 > 1.0e100:
            if zeta == 0:
                return np.array([0.0, 0.0, -1.0], dtype=np.float64)
            w = 1.0 / zeta
            w2 = w.real * w.real + w.imag * w.imag
            den = 1.0 + w2
            v = np.array([2.0 * w.real / den, -2.0 * w.imag / den, (1.0 - w2) / den], dtype=np.float64)
        else:
            den = 1.0 + r2
            v = np.array([2.0 * zeta.real / den, 2.0 * zeta.imag / den, (r2 - 1.0) / den], dtype=np.float64)
        n = float(np.linalg.norm(v))
        if n <= 0.0 or not np.isfinite(n):
            return np.array([0.0, 0.0, 1.0], dtype=np.float64)
        return v / n

    def _multipole_vectors_from_poly(poly, ell):
        ell = int(ell)
        target_degree = 2 * ell
        p = _pad_poly(poly, target_degree + 1)
        scale = float(np.max(np.abs(p))) if p.size else 0.0
        if scale <= 0.0 or not np.isfinite(scale):
            return None
        p = p / scale

        # Trim only high-degree coefficients.  Each trimmed high-degree
        # coefficient corresponds to a root at infinity; low-degree zeros are
        # genuine roots at zeta=0.
        tol = 1.0e-10
        hi = target_degree
        while hi > 0 and abs(p[hi]) <= tol:
            hi -= 1
        if hi == 0 and abs(p[0]) <= tol:
            return None
        n_infinity = int(target_degree - hi)

        roots = []
        if hi >= 1:
            coeff_desc = np.asarray(p[:hi + 1][::-1], dtype=np.complex128)
            try:
                rr = np.roots(coeff_desc)
            except Exception:
                return None
            for r in np.asarray(rr).ravel():
                if np.isfinite(r.real) and np.isfinite(r.imag):
                    roots.append(complex(r))
                else:
                    roots.append(None)
        roots.extend([None] * n_infinity)

        if len(roots) < target_degree:
            roots.extend([None] * (target_degree - len(roots)))
        elif len(roots) > target_degree:
            roots = roots[:target_degree]

        used = [False] * len(roots)
        vectors = []
        for i in range(len(roots)):
            if used[i]:
                continue
            used[i] = True
            best_j = None
            best_d = float("inf")
            for j in range(i + 1, len(roots)):
                if used[j]:
                    continue
                d = _partner_chordal_dist2(roots[i], roots[j])
                if d < best_d:
                    best_d = d
                    best_j = j
            if best_j is None:
                return None
            used[best_j] = True
            vectors.append(_vector_from_root(roots[i]))
            if len(vectors) == ell:
                break

        if len(vectors) != ell:
            return None
        V = np.asarray(vectors, dtype=np.float64)
        if V.shape != (ell, 3) or not np.all(np.isfinite(V)):
            return None
        norms = np.linalg.norm(V, axis=1)
        if np.any(norms <= 0.0) or not np.all(np.isfinite(norms)):
            return None
        return V / norms[:, None]

    def _fallback_perpendicular(v):
        v = np.asarray(v, dtype=np.float64).ravel()
        if v.size != 3 or not np.all(np.isfinite(v)):
            return np.array([0.0, 0.0, 1.0], dtype=np.float64)
        nv = float(np.linalg.norm(v))
        if nv <= 0.0 or not np.isfinite(nv):
            return np.array([0.0, 0.0, 1.0], dtype=np.float64)
        v = v / nv
        if abs(v[0]) < abs(v[1]) and abs(v[0]) < abs(v[2]):
            axis = np.array([1.0, 0.0, 0.0], dtype=np.float64)
        elif abs(v[1]) < abs(v[2]):
            axis = np.array([0.0, 1.0, 0.0], dtype=np.float64)
        else:
            axis = np.array([0.0, 0.0, 1.0], dtype=np.float64)
        n = np.cross(v, axis)
        nn = float(np.linalg.norm(n))
        if nn <= 0.0 or not np.isfinite(nn):
            return np.array([1.0, 0.0, 0.0], dtype=np.float64)
        return n / nn

    def _unit_area_normal(a, b):
        a = np.asarray(a, dtype=np.float64).ravel()
        b = np.asarray(b, dtype=np.float64).ravel()
        c = np.cross(a, b)
        nc = float(np.linalg.norm(c))
        if nc <= 0.0 or not np.isfinite(nc):
            return _fallback_perpendicular(a)
        return c / nc

    def _ecliptic_pole_galactic_unit_vector():
        # North ecliptic pole in Galactic coordinates (J2000).  The sign is
        # immaterial because only absolute dot products are used.
        l_deg = 96.384
        b_deg = 29.811
        l = math.radians(l_deg)
        b = math.radians(b_deg)
        cb = math.cos(b)
        e = np.array([cb * math.cos(l), cb * math.sin(l), math.sin(b)], dtype=np.float64)
        e /= np.linalg.norm(e)
        return e

    try:
        if np.ma.isMaskedArray(m):
            a = np.asarray(m.filled(np.nan), dtype=np.float64).ravel()
        else:
            a = np.asarray(m, dtype=np.float64).ravel()
    except Exception:
        try:
            a = np.asarray(np.real(m), dtype=np.float64).ravel()
        except Exception:
            return 0.0

    npix = int(a.size)
    nside = _infer_nside(npix)
    if nside <= 0:
        return 0.0

    valid = np.isfinite(a)
    nvalid = int(np.count_nonzero(valid))
    if nvalid < 16:
        return 0.0

    try:
        coeff = _fit_low_l_coefficients(a, valid, nside, npix)
        B2null, B3null = _get_null_poly_basis()
        poly2 = np.asarray(coeff[4:9], dtype=np.float64).dot(B2null)
        poly3 = np.asarray(coeff[9:16], dtype=np.float64).dot(B3null)

        v2 = _multipole_vectors_from_poly(poly2, 2)
        v3 = _multipole_vectors_from_poly(poly3, 3)
        if v2 is None or v3 is None:
            return 0.0

        n2 = _unit_area_normal(v2[0], v2[1])
        n31 = _unit_area_normal(v3[0], v3[1])
        n32 = _unit_area_normal(v3[0], v3[2])
        n33 = _unit_area_normal(v3[1], v3[2])
        e = _ecliptic_pole_galactic_unit_vector()

        vals = np.array([abs(float(np.dot(n2, e))),
                         abs(float(np.dot(n31, e))),
                         abs(float(np.dot(n32, e))),
                         abs(float(np.dot(n33, e)))], dtype=np.float64)
        vals = np.clip(vals, 0.0, 1.0)
        A = float(np.mean(vals))
        if not np.isfinite(A):
            return 0.0
        if A < 0.0:
            A = 0.0
        elif A > 1.0:
            A = 1.0
        return float(A)
    except Exception:
        return 0.0


def summarize_results(planck_stat, sim_results):
    import numpy as np

    desc = "Quadrupole-Octopole Ecliptic Area-Normal Alignment: fit ell=0..3 by equal-area least squares on unmasked HEALPix pixels, treat ell=0,1 as nuisance terms, extract ell=2,3 multipole-vector area normals, and return A, the mean absolute alignment of the quadrupole and octopole area normals with the ecliptic pole."
    try:
        ps = float(planck_stat)
    except Exception:
        ps = np.nan
    try:
        sims = np.asarray(sim_results, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            for item in sim_results:
                try:
                    v = float(item)
                    if np.isfinite(v):
                        vals.append(v)
                except Exception:
                    pass
        except Exception:
            pass
        sims = np.asarray(vals, dtype=np.float64)
    sims = sims[np.isfinite(sims)]
    n = int(sims.size)

    out = {
        "test_description": desc,
        "statistic": "A = (|n2.e| + |n3_12.e| + |n3_13.e| + |n3_23.e|)/4, with n's the quadrupole/octopole multipole-vector area normals and e the ecliptic pole.",
        "tail": "upper; larger A means the quadrupole and octopole planes are more nearly parallel to the ecliptic plane",
        "fit": "equal-area least-squares fit of real ell=0..3 harmonic polynomials on finite unmasked HEALPix pixels; ell=0,1 are nuisance terms and ell=2,3 are retained",
        "mask_handling": "Pixels with non-finite values are ignored in the least-squares fit; no healpy harmonic transform is applied to arrays containing NaNs.",
        "ecliptic_pole_galactic_degrees": {"l": 96.384, "b": 29.811},
        "planck_A": float(ps) if np.isfinite(ps) else None,
        "n_sims": n,
        "plot_spec": {
            "kind": "histogram",
            "title": "Quadrupole-octopole ecliptic area-normal alignment",
            "xlabel": "A = mean |area normal · ecliptic pole|",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "Simulation distribution of the quadrupole/octopole ecliptic area-normal alignment statistic. The anomaly is assessed by the upper-tail rank of the Planck value."
        }
    }

    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_eq = int(n - n_lt - n_gt)
    n_le = int(n_lt + n_eq)
    n_ge = int(n_gt + n_eq)

    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_two = float(min(1.0, 2.0 * min(p_upper, p_lower)))

    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    z = float((ps - mean) / std) if std > 0.0 and np.isfinite(std) else None
    q = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])

    out.update({
        "sim_mean_A": mean,
        "sim_median_A": median,
        "sim_std_A": std,
        "sim_min_A": float(np.min(sims)),
        "sim_max_A": float(np.max(sims)),
        "sim_percentile_1_A": float(q[0]),
        "sim_percentile_5_A": float(q[1]),
        "sim_percentile_16_A": float(q[2]),
        "sim_percentile_50_A": float(q[3]),

        "sim_percentile_95_A": float(q[5]),
        "sim_percentile_99_A": float(q[6]),
        "planck_zscore_vs_sims": z,
        "difference_planck_minus_sim_median_A": float(ps - median),
        "planck_percentile_midrank_low_to_high": float(100.0 * (n_lt + 0.5 * n_eq) / n),
        "upper_tail_rank_descending_1_is_largest_A": int(n_gt + 1),
        "upper_tail_midrank_descending_1_is_largest_A": float(n_gt + 0.5 * n_eq + 1.0),
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
