Anomaly GPT5.5 · Repeat · other · one-tailed, lower tail

Hypothesis

The Planck masked temperature map has an anomalously low large-angle correlation statistic S1/2 compared with ΛCDM simulations, indicating a deficit of correlations at angles above 60 degrees.

Method

Masked Planck and ΛCDM simulation temperature maps were monopole/dipole-subtracted on the unmasked sky, converted to masked pseudo-C_l for 2≤l≤64, and integrated as S_{1/2}=∫_{-1}^{1/2} C(θ)^2 dcosθ. Low values test for deficient correlations above 60°.

Result Summary

Planck S_{1/2}=1320.97 μK^4 versus simulation mean 36187.98 μK^4 and median 18176.00 μK^4. Only 6/1000 simulations were lower, giving one-tailed p=0.006, empirical plus-one p≈0.00699, and σ≈−2.51.

Planck lies in the lowest ≈1% of the ΛCDM simulation distribution for this masked estimator, indicating a statistically significant deficit of large-angle temperature correlations.

Agent Justification

Not recorded.

Comparison To Literature

This is consistent with the long-reported low large-angle CMB correlation / S_{1/2} anomaly in WMAP and Planck analyses. Planck 2018 “Isotropy and Statistics of the CMB” confirms persistence of large-scale temperature anomalies including lack of large-angle correlation; see Planck Collaboration VII, arXiv:1906.02552.

test_description = "Masked Large-Angle Correlation S1/2: for each masked temperature map, fit and subtract the best-fit monopole plus dipole on the finite unmasked sky, set masked pixels to zero, compute the masked pseudo-C_l spectrum for 2 <= l <= 64, reconstruct C(theta)=sum_l (2l+1) C_l P_l(cos theta)/(4*pi), and return S_{1/2}=int_{-1}^{1/2} C(x)^2 dx. Low values indicate a deficit of correlations above 60 degrees."


def _make_masked_s12_analyzer():
    import numpy as np
    try:
        import healpy as hp
    except Exception:
        hp = None

    LMIN = 2
    LMAX = 64
    ABS_CUT = 1.0e25
    CACHE_MAX = 4
    state = {"geom": [], "kernel": {}}

    def _as_float_1d(m):
        try:
            if np.ma.isMaskedArray(m):
                return np.asarray(m.filled(np.nan), dtype=np.float64).ravel()
            return np.asarray(m, dtype=np.float64).ravel()
        except Exception:
            try:
                return np.real(np.asarray(m)).astype(np.float64, copy=False).ravel()
            except Exception:
                return np.asarray([], dtype=np.float64)

    def _finite_mask(a):
        a = np.asarray(a).ravel()
        good = np.isfinite(a)
        # This extra guard only catches pathological finite sentinels such as hp.UNSEEN;
        # ordinary masked pixels are selected with np.isfinite as required.
        try:
            with np.errstate(invalid="ignore", over="ignore"):
                good = good & (np.abs(a) < ABS_CUT)
        except Exception:
            pass
        return np.asarray(good, dtype=bool)

    def _fallback_stat(a):
        arr = _as_float_1d(a)
        good = _finite_mask(arr)
        vals = np.asarray(arr[good], dtype=np.float64)
        vals = vals[np.isfinite(vals)]
        if vals.size < 2:
            return 0.0
        with np.errstate(all="ignore"):
            vals = vals - float(np.mean(vals))
            var = float(np.mean(vals * vals))
            stat = float(var * var)
        if (not np.isfinite(stat)) or stat < 0.0:
            return 0.0
        return float(stat)

    def _pinv_sym(a):
        try:
            a = np.asarray(a, dtype=np.float64)
            a = 0.5 * (a + a.T)
            w, v = np.linalg.eigh(a)
            scale = float(np.max(np.abs(w))) if w.size else 0.0
            if (not np.isfinite(scale)) or scale <= 0.0:
                return np.zeros_like(a, dtype=np.float64)
            keep = np.abs(w) > (1.0e-12 * scale)
            invw = np.zeros_like(w, dtype=np.float64)
            invw[keep] = 1.0 / w[keep]
            return np.dot(v * invw, v.T)
        except Exception:
            try:
                return np.linalg.pinv(np.asarray(a, dtype=np.float64), rcond=1.0e-12)
            except Exception:
                return np.zeros((4, 4), dtype=np.float64)

    def _get_kernel(lmax):
        lmax = int(lmax)
        cached = state["kernel"].get(lmax)
        if cached is not None:
            return cached
        if lmax < LMIN:
            k = np.zeros((0, 0), dtype=np.float64)
            state["kernel"][lmax] = k
            return k

        # Gauss-Legendre quadrature mapped from y in [-1,1] to x in [-1,1/2].
        # nquad >= lmax+1 exactly integrates products of Legendre polynomials up to degree 2*lmax.
        nquad = int(max(128, lmax + 1))
        y, wy = np.polynomial.legendre.leggauss(nquad)
        x = 0.75 * y - 0.25
        w = 0.75 * wy

        p = np.empty((lmax + 1, nquad), dtype=np.float64)
        p[0, :] = 1.0
        if lmax >= 1:
            p[1, :] = x
        for ell in range(2, lmax + 1):
            p[ell, :] = ((2.0 * ell - 1.0) * x * p[ell - 1, :] - (ell - 1.0) * p[ell - 2, :]) / float(ell)

        ells = np.arange(LMIN, lmax + 1, dtype=np.float64)
        pref = (2.0 * ells + 1.0) / (4.0 * np.pi)
        q = p[LMIN:lmax + 1, :] * pref[:, None]
        k = np.dot(q * w[None, :], q.T)
        k = 0.5 * (k + k.T)
        k[~np.isfinite(k)] = 0.0
        state["kernel"][lmax] = k.astype(np.float64, copy=False)
        return state["kernel"][lmax]

    def _get_geometry(nside, valid):
        valid = np.asarray(valid, dtype=bool).ravel()
        nside = int(nside)
        npix = int(valid.size)
        nvalid = int(np.count_nonzero(valid))

        geoms = state["geom"]
        for i, geom in enumerate(geoms):
            try:
                if (geom["nside"] == nside and geom["npix"] == npix and
                        geom["nvalid"] == nvalid and np.array_equal(geom["valid"], valid)):
                    if i != 0:
                        geoms.insert(0, geoms.pop(i))
                    return geoms[0]
            except Exception:
                pass

        pix = np.flatnonzero(valid).astype(np.int64, copy=False)
        if pix.size == 0 or hp is None:
            geom = {
                "nside": nside, "npix": npix, "nvalid": 0, "valid": valid.copy(),
                "pix": pix, "x": np.asarray([], dtype=np.float64),
                "y": np.asarray([], dtype=np.float64), "z": np.asarray([], dtype=np.float64),
                "pinv": np.zeros((4, 4), dtype=np.float64)
            }
            geoms.insert(0, geom)
            del geoms[CACHE_MAX:]
            return geom

        x, y, z = hp.pix2vec(nside, pix)
        x = np.asarray(x, dtype=np.float64)
        y = np.asarray(y, dtype=np.float64)
        z = np.asarray(z, dtype=np.float64)

        gram = np.empty((4, 4), dtype=np.float64)
        gram[0, 0] = float(pix.size)
        gram[0, 1] = gram[1, 0] = float(np.sum(x))
        gram[0, 2] = gram[2, 0] = float(np.sum(y))
        gram[0, 3] = gram[3, 0] = float(np.sum(z))
        gram[1, 1] = float(np.dot(x, x))
        gram[1, 2] = gram[2, 1] = float(np.dot(x, y))
        gram[1, 3] = gram[3, 1] = float(np.dot(x, z))
        gram[2, 2] = float(np.dot(y, y))
        gram[2, 3] = gram[3, 2] = float(np.dot(y, z))
        gram[3, 3] = float(np.dot(z, z))
        pinv = _pinv_sym(gram)

        geom = {
            "nside": nside, "npix": npix, "nvalid": nvalid, "valid": valid.copy(),
            "pix": pix, "x": x, "y": y, "z": z, "pinv": pinv
        }
        geoms.insert(0, geom)
        del geoms[CACHE_MAX:]
        return geom

    def _anafast(mp, lmax):
        mp = np.asarray(mp, dtype=np.float64).ravel()
        # Never pass NaN or inf values into healpy spherical-harmonic transforms.
        mp = np.where(np.isfinite(mp), mp, 0.0)
        try:
            cl = hp.anafast(mp, lmax=int(lmax), iter=0, pol=False, use_weights=False)
        except TypeError:
            try:
                cl = hp.anafast(mp, lmax=int(lmax), iter=0, pol=False)
            except TypeError:
                cl = hp.anafast(mp, lmax=int(lmax), iter=0)
        cl = np.asarray(cl, dtype=np.float64)
        if cl.ndim > 1:
            cl = np.asarray(cl[0], dtype=np.float64)
        return cl.ravel()

    def analyze_map(m):
        arr = _as_float_1d(m)
        npix = int(arr.size)
        if npix <= 0:
            return 0.0

        valid = _finite_mask(arr)
        nvalid = int(np.count_nonzero(valid))
        if nvalid < 8 or hp is None:
            return float(_fallback_stat(arr))

        try:
            nside = int(hp.npix2nside(npix))
            if nside <= 0 or 12 * nside * nside != npix:
                return float(_fallback_stat(arr))
        except Exception:
            return float(_fallback_stat(arr))

        # Requested upper limit is l=64; for very low nside inputs, cap at the usual HEALPix limit.
        lmax = int(min(LMAX, max(0, 3 * nside - 1)))
        if lmax < LMIN:
            return float(_fallback_stat(arr))

        try:
            geom = _get_geometry(nside, valid)
            pix = geom["pix"]
            if int(pix.size) < 8:
                return float(_fallback_stat(arr))

            vals = np.asarray(arr[pix], dtype=np.float64)
            vals = np.where(np.isfinite(vals), vals, 0.0)
            x = geom["x"]
            y = geom["y"]
            z = geom["z"]

            rhs = np.array([
                float(np.sum(vals)),
                float(np.dot(x, vals)),
                float(np.dot(y, vals)),
                float(np.dot(z, vals))
            ], dtype=np.float64)
            coeff = np.dot(geom["pinv"], rhs)
            if not np.all(np.isfinite(coeff)):
                coeff = np.zeros(4, dtype=np.float64)
                coeff[0] = float(np.mean(vals)) if vals.size else 0.0

            resid = vals - (coeff[0] + coeff[1] * x + coeff[2] * y + coeff[3] * z)
            resid = np.where(np.isfinite(resid), resid, 0.0)

            masked_resid = np.zeros(npix, dtype=np.float64)
            masked_resid[pix] = resid

            cl = _anafast(masked_resid, lmax)
            if cl.size < lmax + 1:
                padded = np.zeros(lmax + 1, dtype=np.float64)
                if cl.size > 0:
                    padded[:cl.size] = cl
                cl = padded
            else:
                cl = np.asarray(cl[:lmax + 1], dtype=np.float64)
            cl = np.where(np.isfinite(cl), cl, 0.0)

            clv = np.asarray(cl[LMIN:lmax + 1], dtype=np.float64)
            kernel = _get_kernel(lmax)
            with np.errstate(all="ignore"):
                stat = float(np.dot(clv, np.dot(kernel, clv)))
            if not np.isfinite(stat):
                return float(_fallback_stat(arr))
            if stat < 0.0:
                stat = 0.0
            return float(stat)
        except Exception:
            return float(_fallback_stat(arr))

    return analyze_map


analyze_map = _make_masked_s12_analyzer()


def summarize_results(planck_stat, sim_results):
    import numpy as _np

    desc = "Masked Large-Angle Correlation S1/2: for each masked temperature map, fit and subtract the best-fit monopole plus dipole on the finite unmasked sky, set masked pixels to zero, compute the masked pseudo-C_l spectrum for 2 <= l <= 64, reconstruct C(theta)=sum_l (2l+1) C_l P_l(cos theta)/(4*pi), and return S_{1/2}=int_{-1}^{1/2} C(x)^2 dx. Low values indicate a deficit of correlations above 60 degrees."

    def _finite_float(x, default=float("nan")):
        try:
            y = float(x)
        except Exception:
            return float(default)
        if not _np.isfinite(y):
            return float(default)
        return float(y)

    ps = _finite_float(planck_stat)
    try:
        sims = _np.asarray(sim_results, dtype=_np.float64).ravel()
    except Exception:
        vals = []
        try:
            for v in sim_results:
                vals.append(_finite_float(v))
        except Exception:
            vals = []
        sims = _np.asarray(vals, dtype=_np.float64).ravel()
    sims = sims[_np.isfinite(sims)]
    n = int(sims.size)

    out = {
        "test_description": desc,
        "planck_stat": ps,
        "n_sims": n,
        "tail_of_interest": "lower tail: smaller S_{1/2} means less large-angle (theta > 60 deg) temperature correlation in the masked pseudo-C_l estimate",
        "statistic_units": "microK^4, with the pseudo-spectrum normalization set by the common mask and zero fill",
        "plot_spec": {
            "kind": "histogram",
            "title": "Masked large-angle correlation S1/2",
            "xlabel": "S1/2 from masked pseudo-C_l (microK^4)",
            "ylabel": "Number of simulations",
            "bins": 30,
            "caption": "LCDM simulation distribution of S1/2 after the same mask, monopole/dipole subtraction on the unmasked sky, and l=2..64 pseudo-spectrum estimator. The anomaly corresponds to Planck lying in the left tail."
        }
    }

    if not _np.isfinite(ps):
        out["warning"] = "Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics are available."
        return out

    n_lt = int(_np.count_nonzero(sims < ps))
    n_le = int(_np.count_nonzero(sims <= ps))
    n_gt = int(_np.count_nonzero(sims > ps))
    n_ge = int(_np.count_nonzero(sims >= ps))
    n_eq = int(n - n_lt - n_gt)

    p_lower_raw = float(n_le / n)
    p_lower_strict_raw = float(n_lt / n)
    p_lower_plus_one = float((n_le + 1.0) / (n + 1.0))
    p_upper_plus_one = float((n_ge + 1.0) / (n + 1.0))
    p_two_sided = float(min(1.0, 2.0 * min(p_lower_plus_one, p_upper_plus_one)))
    rank_low = int(n_lt + 1)
    rank_high = int(n_le + 1)
    midrank_percentile = float(100.0 * (n_lt + 0.5 * n_eq) / n)

    mean = float(_np.mean(sims))
    median = float(_np.median(sims))
    std = float(_np.std(sims, ddof=1)) if n > 1 else 0.0
    z_vs_sims = None
    if _np.isfinite(std) and std > 0.0:
        z_vs_sims = float((ps - mean) / std)
    qs = _np.percentile(sims, [0.5, 1.0, 5.0, 16.0, 50.0, 84.0, 95.0, 99.0, 99.5])

    out.update({
        "sim_mean": mean,
        "sim_median": median,
        "sim_std": std,
        "sim_min": float(_np.min(sims)),
        "sim_max": float(_np.max(sims)),
        "sim_percentile_0_5": float(qs[0]),
        "sim_percentile_1": float(qs[1]),
        "sim_percentile_5": float(qs[2]),
        "sim_percentile_16": float(qs[3]),
        "sim_percentile_50": float(qs[4]),
        "sim_percentile_84": float(qs[5]),
        "sim_percentile_95": float(qs[6]),
        "sim_percentile_99": float(qs[7]),
        "sim_percentile_99_5": float(qs[8]),
        "planck_zscore_vs_sim_distribution": z_vs_sims,
        "planck_percentile_midrank": midrank_percentile,
        "rank_ascending_1_is_lowest_s12": rank_low,
        "rank_ascending_tie_low": rank_low,
        "rank_ascending_tie_high": rank_high,
        "rank_convention": "Ascending rank 1 is the smallest S1/2 (largest large-angle-correlation deficit); tie_high includes simulations tied with Planck.",
        "n_sim_lt_planck": n_lt,
        "n_sim_le_planck": n_le,
        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,
        "n_sim_ge_planck": n_ge,
        "fraction_sims_lt_planck": p_lower_strict_raw,
        "fraction_sims_le_planck": p_lower_raw,
        "empirical_p_lower_deficit_tail": p_lower_plus_one,
        "empirical_p_upper_high_correlation_tail": p_upper_plus_one,
        "empirical_p_two_sided_extremeness": p_two_sided,
        "is_lower_tail_5pct": bool(p_lower_plus_one <= 0.05),
        "is_lower_tail_1pct": bool(p_lower_plus_one <= 0.01)
    })

    if std == 0.0:
        out["warning"] = "All finite simulation S1/2 values are identical; check for degenerate inputs or a failed pseudo-spectrum pipeline."
    elif p_lower_plus_one <= 0.01:
        out["interpretation"] = "Planck lies in the lowest 1% tail of the LCDM simulation S1/2 distribution for this masked large-angle-correlation estimator."
    elif p_lower_plus_one <= 0.05:
        out["interpretation"] = "Planck lies in the lowest 5% tail of the LCDM simulation S1/2 distribution for this masked large-angle-correlation estimator."
    else:
        out["interpretation"] = "Planck is not in the lowest 5% tail of the LCDM simulation S1/2 distribution for this masked estimator."
    return out
