Hypothesis

The Planck CMB map has an anomalously high fraction of low-l temperature power concentrated in ecliptic m=0 zonal modes compared with simulations.

Method

After monopole/dipole subtraction on unmasked pixels, the test computes the masked pseudo-harmonic fraction of low-l temperature power in ecliptic m=0 modes for 2<=l<=20.

Result Summary

Planck gives F=0.1094, compared with simulations with mean 0.0993, median 0.0893, and std 0.0528. It lies at the 65.8th percentile; 342/1000 simulations have larger F. The one-tailed upper p-value is 0.342, about 0.41 sigma.

Planck has a slightly above-average ecliptic zonal power fraction, but this is common in the LCDM simulation ensemble and is far from the 5% upper-tail anomaly threshold.

Agent Justification

The statistic F is constructed so that larger values mean more low-l power lies in ecliptic zonal modes. The test specifically targets excess concentration about the ecliptic poles, and the code identifies the upper tail as “unusually high” or “excess” ecliptic zonal power.

Comparison To Literature

No directly comparable published test of this exact l=2..20 masked ecliptic m=0 power-fraction statistic was identified. Related low-l ecliptic-alignment studies, such as Weeks (2004) and Abramo et al. (2006), use multipole-vector/alignment statistics rather than this zonal power fraction.

import math
import numpy as np


test_description = "Low-l Ecliptic Zonal Power Fraction: fit and subtract the monopole and dipole on unmasked pixels, zero-fill masked pixels, compute low-l masked pseudo-harmonic power for 2<=l<=20, and return the fraction in ecliptic m=0 zonal modes."

_LMIN = 2
_LMAX = 20
_N_ZONAL_MODES = _LMAX - _LMIN + 1
_TOTAL_REAL_MODE_WEIGHT = int(sum(2 * ell + 1 for ell in range(_LMIN, _LMAX + 1)))
_FULL_SKY_EQUAL_CL_REFERENCE_F = float(_N_ZONAL_MODES) / float(_TOTAL_REAL_MODE_WEIGHT)
_STATE = {}


def _as_float_1d(m):
    try:
        if np.ma.isMaskedArray(m):
            arr = np.asarray(m.filled(np.nan), dtype=np.float64)
        else:
            arr = np.asarray(m, dtype=np.float64)
    except Exception:
        try:
            arr = np.asarray(np.real(m), dtype=np.float64)
        except Exception:
            arr = np.asarray([], dtype=np.float64)
    return np.ravel(arr)


def _finite_scalar(x, default=np.nan):
    try:
        y = float(x)
    except Exception:
        return float(default)
    if np.isfinite(y):
        return y
    return float(default)


def _get_healpy():
    if "healpy" not in _STATE:
        try:
            import healpy as hp
            _STATE["healpy"] = hp
        except Exception:
            _STATE["healpy"] = None
    return _STATE.get("healpy")


def _nside_from_npix(hp, npix):
    answer = -1
    try:
        ns = int(hp.npix2nside(int(npix)))
        if ns > 0 and int(hp.nside2npix(ns)) == int(npix):
            answer = ns
    except Exception:
        answer = -1
    if answer <= 0:
        try:
            npx = int(npix)
            if npx > 0 and npx % 12 == 0:
                guess = int(round((float(npx) / 12.0) ** 0.5))
                if guess > 0 and 12 * guess * guess == npx:
                    answer = guess
        except Exception:
            answer = -1
    return int(answer)


def _pixel_vectors(hp, nside):
    key = ("pixel_vectors", int(nside))
    if key in _STATE:
        return _STATE[key]
    npix = int(hp.nside2npix(int(nside)))
    pix = np.arange(npix, dtype=np.int64)
    try:
        x, y, z = hp.pix2vec(int(nside), pix, nest=False)
    except TypeError:
        x, y, z = hp.pix2vec(int(nside), pix)
    vecs = [np.asarray(x, dtype=np.float64), np.asarray(y, dtype=np.float64), np.asarray(z, dtype=np.float64)]
    _STATE[key] = vecs
    return vecs


def _fit_monopole_dipole(arr, valid, x, y, z):
    n_valid = int(np.count_nonzero(valid))
    if n_valid <= 0:
        coef0 = np.zeros(4, dtype=np.float64)
        return coef0

    vals = np.asarray(arr[valid], dtype=np.float64)
    if n_valid < 4:
        mu = 0.0
        if vals.size > 0:
            mu = _finite_scalar(np.mean(vals), 0.0)
        coef1 = np.array([mu, 0.0, 0.0, 0.0], dtype=np.float64)
        return coef1

    xv = np.asarray(x[valid], dtype=np.float64)
    yv = np.asarray(y[valid], dtype=np.float64)
    zv = np.asarray(z[valid], dtype=np.float64)

    n = float(n_valid)
    sx = float(np.sum(xv))
    sy = float(np.sum(yv))
    sz = float(np.sum(zv))
    sxx = float(np.dot(xv, xv))
    syy = float(np.dot(yv, yv))
    szz = float(np.dot(zv, zv))
    sxy = float(np.dot(xv, yv))
    sxz = float(np.dot(xv, zv))
    syz = float(np.dot(yv, zv))
    st = float(np.sum(vals))
    stx = float(np.dot(vals, xv))
    sty = float(np.dot(vals, yv))
    stz = float(np.dot(vals, zv))

    mat = np.array([[n, sx, sy, sz], [sx, sxx, sxy, sxz], [sy, sxy, syy, syz], [sz, sxz, syz, szz]], dtype=np.float64)
    rhs = np.array([st, stx, sty, stz], dtype=np.float64)
    coef = None
    try:
        coef = np.linalg.solve(mat, rhs)
    except Exception:
        try:
            coef = np.linalg.lstsq(mat, rhs, rcond=None)[0]
        except Exception:
            coef = None
    if coef is None or not np.all(np.isfinite(coef)):
        mu = _finite_scalar(np.mean(vals), 0.0) if vals.size > 0 else 0.0
        coef = np.array([mu, 0.0, 0.0, 0.0], dtype=np.float64)
    return np.asarray(coef, dtype=np.float64)


def _monopole_dipole_removed_map(arr, valid, x, y, z):
    npix = int(arr.size)
    cleaned = np.zeros(npix, dtype=np.float64)
    if npix <= 0:
        return cleaned
    if int(np.count_nonzero(valid)) <= 0:
        return cleaned
    coef = _fit_monopole_dipole(arr, valid, x, y, z)
    cleaned[valid] = np.asarray(arr[valid], dtype=np.float64) - (coef[0] + coef[1] * x[valid] + coef[2] * y[valid] + coef[3] * z[valid])
    cleaned[~np.isfinite(cleaned)] = 0.0
    return cleaned


def _ecliptic_costheta(hp, nside, x, y, z):
    key = ("ecliptic_costheta", int(nside))
    if key in _STATE:
        return _STATE[key]
    ct = None
    try:
        npix = int(hp.nside2npix(int(nside)))
        pix = np.arange(npix, dtype=np.int64)
        try:
            theta, phi = hp.pix2ang(int(nside), pix, nest=False)
        except TypeError:
            theta, phi = hp.pix2ang(int(nside), pix)
        rot = hp.Rotator(coord=["G", "E"])
        theta_e, phi_e = rot(theta, phi)
        ct = np.cos(np.asarray(theta_e, dtype=np.float64))
    except Exception:
        ct = None
    if ct is None or np.asarray(ct).size != np.asarray(x).size or not np.all(np.isfinite(ct)):
        try:
            l_rad = math.radians(96.384)
            b_rad = math.radians(29.811)
            cb = math.cos(b_rad)
            px = cb * math.cos(l_rad)
            py = cb * math.sin(l_rad)
            pz = math.sin(b_rad)
            ct = px * np.asarray(x, dtype=np.float64) + py * np.asarray(y, dtype=np.float64) + pz * np.asarray(z, dtype=np.float64)
        except Exception:
            ct = None
    if ct is None:
        return None
    ct = np.asarray(ct, dtype=np.float64)
    ct = np.clip(ct, -1.0, 1.0)
    _STATE[key] = ct
    return ct


def _map2alm_scalar(hp, mp, lmax):
    lmax = int(lmax)
    arr = np.asarray(mp, dtype=np.float64).ravel()
    arr = np.where(np.isfinite(arr), arr, 0.0)
    try:
        alm = hp.map2alm(arr, lmax=lmax, mmax=lmax, iter=0, pol=False, use_weights=False)
    except TypeError:
        try:
            alm = hp.map2alm(arr, lmax=lmax, mmax=lmax, iter=0, pol=False)
        except TypeError:
            alm = hp.map2alm(arr, lmax=lmax, mmax=lmax, iter=0)
    if isinstance(alm, list) or isinstance(alm, tuple):
        alm = alm[0]
    return np.asarray(alm, dtype=np.complex128).ravel()


def _alm_denominator_power(hp, alm, lmax):
    lmax = int(lmax)
    lhi = min(_LMAX, lmax)
    if lhi < _LMIN:
        return 0.0
    a = np.asarray(alm, dtype=np.complex128).ravel()
    den = 0.0
    good = False
    for ell in range(_LMIN, lhi + 1):
        for mm in range(0, ell + 1):
            try:
                idx = int(hp.Alm.getidx(lmax, ell, mm))
            except Exception:
                idx = -1
            if idx < 0 or idx >= a.size:
                continue
            z = a[idx]
            re = float(np.real(z))
            im = float(np.imag(z))
            p = re * re + im * im
            if np.isfinite(p):
                if mm == 0:
                    den += p
                else:
                    den += 2.0 * p
                good = True
    if good and np.isfinite(den) and den > 0.0:
        return float(den)
    return 0.0


def _ecliptic_zonal_numerator(mp, costheta, lmax):
    vals = np.asarray(mp, dtype=np.float64).ravel()
    vals = np.where(np.isfinite(vals), vals, 0.0)
    ct = np.asarray(costheta, dtype=np.float64).ravel()
    if vals.size <= 0 or ct.size != vals.size:
        return 0.0
    lhi = min(_LMAX, int(lmax))
    if lhi < _LMIN:
        return 0.0
    omega_pix = 4.0 * math.pi / float(vals.size)
    p_lm2 = np.ones(vals.size, dtype=np.float64)
    p_lm1 = ct.copy()
    num = 0.0
    for ell in range(2, lhi + 1):
        p_l = ((2.0 * ell - 1.0) * ct * p_lm1 - (ell - 1.0) * p_lm2) / float(ell)
        norm = math.sqrt((2.0 * ell + 1.0) / (4.0 * math.pi))
        a_l0 = omega_pix * norm * float(np.dot(vals, p_l))
        pwr = a_l0 * a_l0
        if np.isfinite(pwr):
            num += pwr
        p_lm2 = p_lm1
        p_lm1 = p_l
    if np.isfinite(num) and num >= 0.0:
        return float(num)
    return 0.0


def _fallback_fraction_from_array(arr, valid):
    try:
        a = np.asarray(arr, dtype=np.float64).ravel()
        if valid is None or np.asarray(valid).size != a.size:
            v = np.isfinite(a)
        else:
            v = np.asarray(valid, dtype=bool).ravel() & np.isfinite(a)
        vals = np.asarray(a[v], dtype=np.float64)
        vals = vals[np.isfinite(vals)]
        if vals.size < 8:
            return 0.0
        vals = vals - float(np.mean(vals))
        ss = float(np.dot(vals, vals))
        if not np.isfinite(ss) or ss <= 0.0:
            return 0.0
        ff = np.fft.rfft(vals)
        hi = min(_LMAX, int(ff.size) - 1)
        if hi < _LMIN:
            return 0.0
        powers = np.abs(ff[_LMIN:hi + 1]) ** 2
        powers = powers[np.isfinite(powers)]
        if powers.size <= 0:
            return 0.0
        den = float(np.sum(powers))
        if not np.isfinite(den) or den <= 0.0:
            return 0.0
        f = float(powers[0]) / den
        if np.isfinite(f):
            return float(min(1.0, max(0.0, f)))
    except Exception:
        pass
    return 0.0


def analyze_map(m):
    arr = _as_float_1d(m)
    if arr.size <= 0:
        return 0.0
    valid = np.isfinite(arr)
    if int(np.count_nonzero(valid)) <= 0:
        return 0.0

    hp = _get_healpy()
    if hp is None:
        return _fallback_fraction_from_array(arr, valid)

    nside = _nside_from_npix(hp, int(arr.size))
    if nside <= 0:
        return _fallback_fraction_from_array(arr, valid)

    try:
        vecs = _pixel_vectors(hp, nside)
        x = vecs[0]
        y = vecs[1]
        z = vecs[2]
        if x.size != arr.size:
            return _fallback_fraction_from_array(arr, valid)
        mp = _monopole_dipole_removed_map(arr, valid, x, y, z)
        lmax = _LMAX
        try:
            alm = _map2alm_scalar(hp, mp, lmax)
        except Exception:
            ltry = min(_LMAX, max(_LMIN, 3 * int(nside) - 1))
            if ltry == lmax:
                return _fallback_fraction_from_array(arr, valid)
            lmax = int(ltry)
            alm = _map2alm_scalar(hp, mp, lmax)
        denominator = _alm_denominator_power(hp, alm, lmax)
        if not np.isfinite(denominator) or denominator <= 0.0:
            return 0.0
        ct = _ecliptic_costheta(hp, nside, x, y, z)
        if ct is None:
            return _fallback_fraction_from_array(arr, valid)
        numerator = _ecliptic_zonal_numerator(mp, ct, lmax)
        if not np.isfinite(numerator) or numerator < 0.0:
            return 0.0
        f = float(numerator) / float(denominator)
        if not np.isfinite(f):
            return 0.0
        return float(min(1.0, max(0.0, f)))
    except Exception:
        return _fallback_fraction_from_array(arr, valid)


def _finite_array(x):
    try:
        a = np.asarray(x, dtype=np.float64).ravel()
    except Exception:
        vals = []
        try:
            for item in x:
                y = _finite_scalar(item)
                if np.isfinite(y):
                    vals.append(y)
        except Exception:
            vals = []
        a = np.asarray(vals, dtype=np.float64)
    return a[np.isfinite(a)]


def _float_or_none(x):
    y = _finite_scalar(x)
    if np.isfinite(y):
        return float(y)
    return None


def summarize_results(planck_stat, sim_results):

        "coordinate_frame": "Ecliptic zonal axis. Input HEALPix maps are treated as Galactic/RING; Y_l0 is evaluated about the ecliptic pole.",
        "mask_handling": "Only finite pixels enter the monopole/dipole fit. The cut-sky map is then zero-filled on masked pixels before harmonic analysis, giving the same binary mask weights for data and simulations.",
        "multipole_range": "2 <= l <= 20",
        "planck_F": _float_or_none(ps),
        "n_sims": n,
        "full_sky_equal_Cl_mode_count_reference_F": float(_FULL_SKY_EQUAL_CL_REFERENCE_F),
        "tail_of_interest": "Upper tail indicates unusually high low-l power in ecliptic m=0 zonal modes; lower tail indicates a zonal deficit.",
        "plot_spec": {
            "kind": "histogram",
            "title": "Low-l ecliptic zonal power fraction",
            "xlabel": "F: fraction of l=2..20 pseudo-power in ecliptic m=0 modes",
            "ylabel": "Number of simulations",
            "bins": 40,
            "caption": "Planck is compared with simulations using identical masking, monopole/dipole subtraction, and pseudo-harmonic weighting."
        }
    }
    if not np.isfinite(ps):
        out["warning"] = "The Planck statistic is not finite."
        return out
    if n <= 0:
        out["warning"] = "No finite simulation statistics were supplied."
        return out

    n_lt = int(np.count_nonzero(sims < ps))
    n_gt = int(np.count_nonzero(sims > ps))
    n_eq = int(n - n_lt - n_gt)
    n_le = int(n_lt + n_eq)
    n_ge = int(n_gt + n_eq)
    p_upper = float((n_ge + 1.0) / (n + 1.0))
    p_lower = float((n_le + 1.0) / (n + 1.0))
    p_two = float(min(1.0, 2.0 * min(p_upper, p_lower)))
    mean = float(np.mean(sims))
    median = float(np.median(sims))
    std = float(np.std(sims, ddof=1)) if n > 1 else 0.0
    zscore = None
    if np.isfinite(std) and std > 0.0:
        zscore = float((ps - mean) / std)
    q = np.percentile(sims, [1, 5, 16, 50, 84, 95, 99])

    out.update({

        "sim_std_F": std,

        "n_sim_lt_planck": n_lt,

        "n_sim_eq_planck": n_eq,
        "n_sim_gt_planck": n_gt,

        "fraction_sims_lt_planck": float(n_lt / n),

    elif p_upper <= 0.05:
        out["interpretation"] = "Planck is in the highest 5% of simulations: low-l pseudo-power is unusually concentrated in ecliptic zonal modes."

        out["interpretation"] = "Planck is not in the 5% upper or lower tail of the simulation distribution for this ecliptic zonal power fraction."
